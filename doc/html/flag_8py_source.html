61 class Flag(six.with_metaclass(_FlagMetaClass, object)):
62  """Information about a command-line flag.
63 
64  'Flag' objects define the following fields:
65  .name - the name for this flag;
66  .default - the default value for this flag;
67  .default_as_str - default value as repr'd string, e.g., "'true'" (or None);
68  .value - the most recent parsed value of this flag; set by Parse();
69  .help - a help string or None if no help is available;
70  .short_name - the single letter alias for this flag (or None);
71  .boolean - if 'true', this flag does not accept arguments;
72  .present - true if this flag was parsed from command line flags;
73  .parser - an ArgumentParser object;
74  .serializer - an ArgumentSerializer object;
75  .allow_override - the flag may be redefined without raising an error, and
76  newly defined flag overrides the old one.
77  .allow_cpp_override - the flag may be redefined in C++ without raising an
78  error, value "transfered" to C++, and the flag is
79  replaced by the C++ flag after init;
80  .allow_hide_cpp - the flag may be redefined despite hiding a C++ flag with
81  the same name;
82  .using_default_value - the flag value has not been set by user;
83  .allow_overwrite - the flag may be parsed more than once without raising
84  an error, the last set value will be used;
85 
86  The only public method of a 'Flag' object is Parse(), but it is
87  typically only called by a 'FlagValues' object. The Parse() method is
88  a thin wrapper around the 'ArgumentParser' Parse() method. The parsed
89  value is saved in .value, and the .present attribute is updated. If
90  this flag was already present, an Error is raised.
91 
92  Parse() is also called during __init__ to parse the default value and
93  initialize the .value attribute. This enables other python modules to
94  safely use flags even if the __main__ module neglects to parse the
95  command line arguments. The .present attribute is cleared after
96  __init__ parsing. If the default value is set to None, then the
97  __init__ parsing step is skipped and the .value attribute is
98  initialized to None.
99 
100  Note: The default value is also presented to the user in the help
101  string, so it is important that it be a legal value for this flag.
102  """

171

173  argument: String, value to be parsed for flag.
174  """

257  element.appendChild(_helpers.CreateXMLDOMElement(doc, 'name', self.name))

261  if self.help:

263  doc, 'meaning', self.help))

265  # command line, or as a Python object. We serialize this value in the
266  # latter case in order to remain consistent.
267  if self.serializer and not isinstance(self.default, str):
268  if self.default is not None:

270  else:
271  default_serialized = ''
272  else:
273  default_serialized = self.default
274  element.appendChild(_helpers.CreateXMLDOMElement(

276  element.appendChild(_helpers.CreateXMLDOMElement(
277  doc, 'current', self.value))

279  doc, 'type', self.flag_type()))

281  for e in self._extra_xml_dom_elements(doc):

338 class MultiFlag(Flag):
339  """A flag that can appear multiple time on the command-line.
340 
341  The value of such a flag is a list that contains the individual values
342  from all the appearances of that flag on the command-line.
343 
344  See the __doc__ for Flag for most behavior of this class. Only
345  differences in behavior are described here:
346 
347  * The default value may be either a single value or a list of values.
348  A single value is interpreted as the [value] singleton list.
349 
350  * The value of the flag is always a list, even if the option was
351  only supplied once, and even if the default value is a single
352  value
353  """
354 
355  def __init__(self, *args, **kwargs):
356  Flag.__init__(self, *args, **kwargs)
357  self.help += ';\n repeat this option to specify a list of values'

359  def parse(self, arguments):
360  """Parses one or more arguments with the installed parser.
361 
362  Args:
363  arguments: a single argument or a list of arguments (typically a
364  list of default values); a single argument is converted
365  internally into a list containing one item.
366  """
367  if not isinstance(arguments, list):
368  # Default value may be a list of values. Most other arguments
369  # will not be, so convert them into a single-item list to make
370  # processing simpler below.
371  arguments = [arguments]
372 
373  if self.present:
374  # keep a backup reference to list of previously supplied option values
375  values = self.value
376  else:
377  # "erase" the defaults with an empty list
378  values = []
379 
380  for item in arguments:
381  # have Flag superclass parse argument, overwriting self.value reference
382  Flag.Parse(self, item) # also increments self.present
383  values.append(self.value)
384 
385  # put list of option values back in the 'value' attribute
386  self.value = values

396

400  if s: s += ' '
