48 bool LaunchDataParser::ParseLaunchData(LaunchInfo* launch_info,
49  const char* launch_data,
50  int64_t data_size) {
51  int64_t start, end, equal;
52  start = 0;
53  while (start < data_size) {
54  // Move start to point to the next non-null character.
55  while (launch_data[start] == '\0' && start < data_size) {
56  start++;
57  }
58  // Move end to the next null character or end of the string,
59  // also find the first equal symbol appears.
60  end = start;
61  equal = -1;
62  while (launch_data[end] != '\0' && end < data_size) {
63  if (equal == -1 && launch_data[end] == '=') {
64  equal = end;
65  }
66  end++;
67  }
68  if (equal == -1) {
69  PrintError(L"Cannot find equal symbol in line: %hs",
70  string(launch_data + start, end - start).c_str());
71  return false;
72  } else if (start == equal) {
73  PrintError(L"Key is empty string in line: %hs",
74  string(launch_data + start, end - start).c_str());
75  return false;
76  } else {
77  string key(launch_data + start, equal - start);
78  string value(launch_data + equal + 1, end - equal - 1);
79  if (launch_info->find(key) != launch_info->end()) {
80  PrintError(L"Duplicated launch info key: %hs", key.c_str());
81  return false;
82  }
83  launch_info->insert(make_pair(key, blaze_util::CstringToWstring(value)));
84  }
85  start = end + 1;
86  }
87  return true;
88 }

90 bool LaunchDataParser::GetLaunchInfo(const wstring& binary_path,
91  LaunchInfo* launch_info) {
92  unique_ptr<ifstream> binary =
93  make_unique<ifstream>(AsAbsoluteWindowsPath(binary_path.c_str()).c_str(),
94  ios::binary | ios::in);
95  if (!binary->good()) {
96  PrintError(L"Cannot open the binary to read launch data");
97  return false;
98  }
99  int64_t data_size = ReadDataSize(binary.get());
100  if (data_size == 0) {
101  PrintError(L"No data appended, cannot launch anything!");
102  return false;
103  }
104  unique_ptr<char[]> launch_data(new char[data_size]);
105  ReadLaunchData(binary.get(), launch_data.get(), data_size);
106  if (!ParseLaunchData(launch_info, launch_data.get(), data_size)) {
107  return false;
108  }
109  return true;
110 }
