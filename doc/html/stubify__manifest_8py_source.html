66 def StubifyMobileInstall(manifest_string):
67  """Does the stubification on an XML string for mobile-install.
68 
69  Args:
70  manifest_string: the input manifest as a string.
71  Returns:
72  A tuple of (output manifest, old application class, app package)
73  Raises:
74  Exception: if something goes wrong
75  """
76  manifest, application = _ParseManifest(manifest_string)
77 
78  old_application = application.get(
79  "{%s}name" % ANDROID, "android.app.Application")
80 
81  application.set("{%s}name" % ANDROID, MOBILE_INSTALL_STUB_APPLICATION)
82  application.attrib.pop("{%s}hasCode" % ANDROID, None)
83  read_permission = manifest.findall(
84  './uses-permission[@android:name="%s"]' % READ_EXTERNAL_STORAGE,
85  namespaces={"android": ANDROID})
86 
87  if not read_permission:
88  read_permission = ElementTree.Element("uses-permission")
89  read_permission.set("{%s}name" % ANDROID, READ_EXTERNAL_STORAGE)
90  manifest.insert(0, read_permission)
91 
92  new_manifest = ElementTree.tostring(manifest)
93  app_package = manifest.get("package")
94  if not app_package:
95  raise BadManifestException("manifest tag does not have a package specified")
96  return (new_manifest, old_application, app_package)

117 def _ParseManifest(manifest_string):
118  """Parses the given manifest xml.
119 
120  Args:
121  manifest_string: the manifest as a string.
122  Returns:
123  a tuple of the manifest ElementTree and the application tag.
124  Raises:
125  BadManifestException: if the manifest is bad.
126  """
127  manifest = ElementTree.fromstring(manifest_string)
128  if manifest.tag != "manifest":
129  raise BadManifestException("invalid input manifest")
130 
131  app_list = manifest.findall("application")
132  if len(app_list) == 1:
133  # <application> element is present
134  application = app_list[0]
135  elif len(app_list) == 0: # pylint: disable=g-explicit-length-test
136  # <application> element is not present
137  application = ElementTree.Element("application")
138  manifest.insert(0, application)
139  else:
140  raise BadManifestException("multiple <application> elements present")
141  return (manifest, application)
