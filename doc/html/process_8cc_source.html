50 bool WaitableProcess::Create(const std::wstring& argv0,
51  const std::wstring& argv_rest, void* env,
52  const std::wstring& wcwd, std::wstring* error) {
53  // On Windows 7, the subprocess cannot inherit console handles via the
54  // attribute list (CreateProcessW fails with ERROR_NO_SYSTEM_RESOURCES).
55  // If we pass only invalid handles here, the Create method creates an
56  // AutoAttributeList with 0 handles and initializes the STARTUPINFOEXW as
57  // empty and disallowing handle inheritance. At the same time, CreateProcessW
58  // specifies neither CREATE_NEW_CONSOLE nor DETACHED_PROCESS, so the
59  // subprocess *does* inherit the console, which is exactly what we want, and
60  // it works on all supported Windows versions.
61  // Fixes https://github.com/bazelbuild/bazel/issues/8676
62  return Create(argv0, argv_rest, env, wcwd, INVALID_HANDLE_VALUE,
63  INVALID_HANDLE_VALUE, INVALID_HANDLE_VALUE, nullptr, true,
64  error);
65 }

77 bool WaitableProcess::Create(const std::wstring& argv0,
78  const std::wstring& argv_rest, void* env,
79  const std::wstring& wcwd, HANDLE stdin_process,
80  HANDLE stdout_process, HANDLE stderr_process,
81  LARGE_INTEGER* opt_out_start_time,
82  bool create_window, std::wstring* error) {
83  std::wstring cwd;
84  std::wstring error_msg(AsShortPath(wcwd, &cwd));
85  if (!error_msg.empty()) {
86  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
87  L"WaitableProcess::Create", argv0, error_msg);
88  return false;
89  }
90 
91  std::wstring argv0short;
92  error_msg = AsExecutablePathForCreateProcess(argv0, &argv0short);
93  if (!error_msg.empty()) {
94  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
95  L"WaitableProcess::Create", argv0, error_msg);
96  return false;
97  }
98 
99  std::wstring commandline =
100  argv_rest.empty() ? argv0short : (argv0short + L" " + argv_rest);
101  std::unique_ptr<WCHAR[]> mutable_commandline(
102  new WCHAR[commandline.size() + 1]);
103  wcsncpy(mutable_commandline.get(), commandline.c_str(),
104  commandline.size() + 1);
105  // MDSN says that the default for job objects is that breakaway is not
106  // allowed. Thus, we don't need to do any more setup here.
107  job_ = CreateJobObject(NULL, NULL);
108  if (!job_.IsValid()) {
109  DWORD err_code = GetLastError();
110  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
111  L"WaitableProcess::Create", argv0, err_code);
112  return false;
113  }
114 
115  JOBOBJECT_EXTENDED_LIMIT_INFORMATION job_info = {0};
116  job_info.BasicLimitInformation.LimitFlags =
117  JOB_OBJECT_LIMIT_KILL_ON_JOB_CLOSE;
118  if (!SetInformationJobObject(job_, JobObjectExtendedLimitInformation,
119  &job_info, sizeof(job_info))) {
120  DWORD err_code = GetLastError();
121  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
122  L"WaitableProcess::Create", argv0, err_code);
123  return false;
124  }
125 
126  ioport_ = CreateIoCompletionPort(INVALID_HANDLE_VALUE, nullptr, 0, 1);
127  if (!ioport_.IsValid()) {
128  DWORD err_code = GetLastError();
129  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
130  L"WaitableProcess::Create", argv0, err_code);
131  return false;
132  }
133  JOBOBJECT_ASSOCIATE_COMPLETION_PORT port;
134  port.CompletionKey = job_;
135  port.CompletionPort = ioport_;
136  if (!SetInformationJobObject(job_,
137  JobObjectAssociateCompletionPortInformation,
138  &port, sizeof(port))) {
139  DWORD err_code = GetLastError();
140  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
141  L"WaitableProcess::Create", argv0, err_code);
142  return false;
143  }
144 
145  std::unique_ptr<AutoAttributeList> attr_list;
146  if (!AutoAttributeList::Create(stdin_process, stdout_process, stderr_process,
147  &attr_list, &error_msg)) {
148  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
149  L"WaitableProcess::Create", L"", error_msg);
150  return false;
151  }
152 
153  // kMaxCmdline value: see lpCommandLine parameter of CreateProcessW.
154  static constexpr size_t kMaxCmdline = 32767;
155 
156  std::wstring cmd_sample = mutable_commandline.get();
157  if (cmd_sample.size() > 500) {
158  cmd_sample = cmd_sample.substr(0, 495) + L"(...)";
159  }
160  if (wcsnlen_s(mutable_commandline.get(), kMaxCmdline) == kMaxCmdline) {
161  std::wstringstream error_msg;
162  error_msg << L"command is longer than CreateProcessW's limit ("
163  << kMaxCmdline << L" characters)";
164  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
165  L"CreateProcessWithExplicitHandles", cmd_sample,
166  error_msg.str().c_str());
167  return false;
168  }
169 
170  PROCESS_INFORMATION process_info = {0};
171  STARTUPINFOEXW info;
172  attr_list->InitStartupInfoExW(&info);
173  if (!CreateProcessW(
174  /* lpApplicationName */ NULL,
175  /* lpCommandLine */ mutable_commandline.get(),
176  /* lpProcessAttributes */ NULL,
177  /* lpThreadAttributes */ NULL,
178  /* bInheritHandles */ attr_list->InheritAnyHandles() ? TRUE : FALSE,
179  /* dwCreationFlags */ (create_window ? 0 : CREATE_NO_WINDOW) |
180  CREATE_NEW_PROCESS_GROUP // So that Ctrl-Break isn't propagated
181  | CREATE_SUSPENDED // So that it doesn't start a new job itself
182  | EXTENDED_STARTUPINFO_PRESENT | CREATE_UNICODE_ENVIRONMENT,
183  /* lpEnvironment */ env,
184  /* lpCurrentDirectory */ cwd.empty() ? NULL : cwd.c_str(),
185  /* lpStartupInfo */ &info.StartupInfo,
186  /* lpProcessInformation */ &process_info)) {
187  DWORD err = GetLastError();
188 
189  std::wstring errmsg;
190  if (err == ERROR_NO_SYSTEM_RESOURCES && !IsWindows8OrGreater() &&
191  attr_list->HasConsoleHandle()) {
192  errmsg =
193  L"Unrecoverable error: host OS is Windows 7 and subprocess"
194  " got an inheritable console handle";
195  } else {
196  errmsg = GetLastErrorString(err) + L" (error: " + ToString(err) + L")";
197  }
198 
199  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__, L"CreateProcessW",
200  cmd_sample, errmsg);
201  return false;
202  }
203 
204  pid_ = process_info.dwProcessId;
205  process_ = process_info.hProcess;
206  AutoHandle thread(process_info.hThread);
207 
208  if (!AssignProcessToJobObject(job_, process_)) {
209  BOOL is_in_job = false;
210  if (IsProcessInJob(process_, NULL, &is_in_job) && is_in_job &&
211  !IsWindows8OrGreater()) {
212  // Pre-Windows 8 systems don't support nested jobs, and Bazel is already
213  // in a job. We can't create nested jobs, so just revert to
214  // TerminateProcess() and hope for the best. In batch mode, the launcher
215  // puts Bazel in a job so that will take care of cleanup once the
216  // command finishes.
217  job_ = INVALID_HANDLE_VALUE;
218  ioport_ = INVALID_HANDLE_VALUE;
219  } else {
220  DWORD err_code = GetLastError();
221  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
222  L"WaitableProcess::Create", argv0, err_code);
223  return false;
224  }
225  }
226 
227  // Now that we put the process in a new job object, we can start executing
228  // it
229  if (ResumeThread(thread) == -1) {
230  DWORD err_code = GetLastError();
231  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
232  L"WaitableProcess::Create", argv0, err_code);
233  return false;
234  }
235 
236  if (opt_out_start_time) {
237  QueryPerformanceCounter(opt_out_start_time);
238  }
239  *error = L"";
240  return true;
241 }

243 int WaitableProcess::WaitFor(int64_t timeout_msec,
244  LARGE_INTEGER* opt_out_end_time,
245  std::wstring* error) {
246  struct Defer {
247  LARGE_INTEGER* t;
248  Defer(LARGE_INTEGER* cnt) : t(cnt) {}
249  ~Defer() {
250  if (t) {
251  QueryPerformanceCounter(t);
252  }
253  }
254  } defer_query_end_time(opt_out_end_time);
255 
256  DWORD win32_timeout = timeout_msec < 0 ? INFINITE : timeout_msec;
257  int result;
258  switch (WaitForSingleObject(process_, win32_timeout)) {
259  case WAIT_OBJECT_0:
260  result = kWaitSuccess;
261  break;
262 
263  case WAIT_TIMEOUT:
264  result = kWaitTimeout;
265  break;
266 
267  // Any other case is an error and should be reported back to Bazel.
268  default:
269  DWORD err_code = GetLastError();
270  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
271  L"WaitableProcess::WaitFor", ToString(pid_),
272  err_code);
273  return kWaitError;
274  }
275 
276  // Ensure that the process is really terminated (if WaitForSingleObject
277  // above timed out, we have to explicitly kill it) and that it doesn't
278  // leave behind any subprocesses.
279  if (!Terminate(error)) {
280  return kWaitError;
281  }
282 
283  if (job_.IsValid()) {
284  // Wait for the job object to complete, signalling that all subprocesses
285  // have exited.
286  DWORD CompletionCode;
287  ULONG_PTR CompletionKey;
288  LPOVERLAPPED Overlapped;
289  while (GetQueuedCompletionStatus(ioport_, &CompletionCode,
290  &CompletionKey, &Overlapped, INFINITE) &&
291  !((HANDLE)CompletionKey == (HANDLE)job_ &&
292  CompletionCode == JOB_OBJECT_MSG_ACTIVE_PROCESS_ZERO)) {
293  // Still waiting...
294  }
295 
296  job_ = INVALID_HANDLE_VALUE;
297  ioport_ = INVALID_HANDLE_VALUE;
298  }
299 
300  // Fetch and store the exit code in case Bazel asks us for it later,
301  // because we cannot do this anymore after we closed the handle.
302  GetExitCode(error);
303 
304  if (process_.IsValid()) {
305  process_ = INVALID_HANDLE_VALUE;
306  }
307 
308  return result;
309 }

325 bool WaitableProcess::Terminate(std::wstring* error) {
326  static constexpr UINT exit_code = 130; // 128 + SIGINT, like on Linux
327 
328  if (job_.IsValid()) {
329  if (!TerminateJobObject(job_, exit_code)) {
330  DWORD err_code = GetLastError();
331  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
332  L"WaitableProcess::Terminate", ToString(pid_),
333  err_code);
334  return false;
335  }
336  } else if (process_.IsValid()) {
337  if (!TerminateProcess(process_, exit_code)) {
338  DWORD err_code = GetLastError();
339  std::wstring our_error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
340  L"WaitableProcess::Terminate",
341  ToString(pid_), err_code);
342 
343  // If the process exited, despite TerminateProcess having failed, we're
344  // still happy and just ignore the error. It might have been a race
345  // where the process exited by itself just before we tried to kill it.
346  // However, if the process is *still* running at this point (evidenced
347  // by its exit code still being STILL_ACTIVE) then something went
348  // really unexpectedly wrong and we should report that error.
349  if (GetExitCode(error) == STILL_ACTIVE) {
350  // Restore the error message from TerminateProcess - it will be much
351  // more helpful for debugging in case something goes wrong here.
352  *error = our_error;
353  return false;
354  }
355  }
356 
357  if (WaitForSingleObject(process_, INFINITE) != WAIT_OBJECT_0) {
358  DWORD err_code = GetLastError();
359  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
360  L"WaitableProcess::Terminate", ToString(pid_),
361  err_code);
362  return false;
363  }
364  }
365 
366  *error = L"";
367  return true;
368 }

370 // Escape arguments for CreateProcessW.
371 //
372 // This algorithm is based on information found in
373 // http://daviddeley.com/autohotkey/parameters/parameters.htm
374 //
375 // The following source specifies a similar algorithm:
376 // https://blogs.msdn.microsoft.com/twistylittlepassagesallalike/2011/04/23/everyone-quotes-command-line-arguments-the-wrong-way/
377 // unfortunately I found this algorithm only after creating the one below, but
378 // fortunately they seem to do the same.
379 std::wstring WindowsEscapeArg(const std::wstring& s) {
380  if (s.empty()) {
381  return L"\"\"";
382  } else {
383  bool needs_escaping = false;
384  for (const auto& c : s) {
385  if (c == ' ' || c == '"') {
386  needs_escaping = true;
387  break;
388  }
389  }
390  if (!needs_escaping) {
391  return s;
392  }
393  }
394 
395  std::wostringstream result;
396  result << L'"';
397  int start = 0;
398  for (int i = 0; i < s.size(); ++i) {
399  char c = s[i];
400  if (c == '"' || c == '\\') {
401  // Copy the segment since the last special character.
402  if (start >= 0) {
403  result << s.substr(start, i - start);
404  start = -1;
405  }
406 
407  // Handle the current special character.
408  if (c == '"') {
409  // This is a quote character. Escape it with a single backslash.
410  result << L"\\\"";
411  } else {
412  // This is a backslash (or the first one in a run of backslashes).
413  // Whether we escape it depends on whether the run ends with a quote.
414  int run_len = 1;
415  int j = i + 1;
416  while (j < s.size() && s[j] == '\\') {
417  run_len++;
418  j++;
419  }
420  if (j == s.size()) {
421  // The run of backslashes goes to the end.
422  // We have to escape every backslash with another backslash.
423  for (int k = 0; k < run_len * 2; ++k) {
424  result << L'\\';
425  }
426  break;
427  } else if (j < s.size() && s[j] == '"') {
428  // The run of backslashes is terminated by a quote.
429  // We have to escape every backslash with another backslash, and
430  // escape the quote with one backslash.
431  for (int k = 0; k < run_len * 2; ++k) {
432  result << L'\\';
433  }
434  result << L"\\\"";
435  i += run_len; // 'i' is also increased in the loop iteration step
436  } else {
437  // No quote found. Each backslash counts for itself, they must not be
438  // escaped.
439  for (int k = 0; k < run_len; ++k) {
440  result << L'\\';
441  }
442  i += run_len - 1; // 'i' is also increased in the loop iteration step
443  }
444  }
445  } else {
446  // This is not a special character. Start the segment if necessary.
447  if (start < 0) {
448  start = i;
449  }
450  }
451  }
452  // Save final segment after the last special character.
453  if (start != -1) {
454  result << s.substr(start);
455  }
456  result << L'"';
457  return result.str();
458 }
