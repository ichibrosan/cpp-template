146  def _Exec(self, adb_args):
147  """Executes the given adb command + args."""
148  args = [self._adb_path] + self._extra_adb_args + adb_args
149  # TODO(ahumesky): Because multiple instances of adb are executed in
150  # parallel, these debug logging lines will get interleaved.
151  logging.debug("Executing: %s", " ".join(args))
152 
153  # adb sometimes requires the user's home directory to access things in
154  # $HOME/.android (e.g. keys to authorize with the device). To avoid any
155  # potential problems with python picking up things in the user's home
156  # directory, HOME is not set in the environment around python and is instead
157  # passed explicitly as a flag.
158  env = {}
159  if self._user_home_dir:
160  env["HOME"] = self._user_home_dir
161 
162  # On Windows, adb requires the SystemRoot environment variable.
163  if Adb._IsHostOsWindows():
164  value = os.getenv("SYSTEMROOT")
165  if not value:
166  raise EnvvarError(("The %SYSTEMROOT% environment variable must "
167  "be set or Adb won't work"))
168  env["SYSTEMROOT"] = value
169 
170  adb = subprocess.Popen(
171  args,
172  stdin=subprocess.PIPE,
173  stdout=subprocess.PIPE,
174  stderr=subprocess.PIPE,
175  env=env)
176  stdout, stderr = adb.communicate()
177  stdout = stdout.strip()
178  stderr = stderr.strip()
179  logging.debug("adb ret: %s", adb.returncode)
180  logging.debug("adb out: %s", stdout)
181  logging.debug("adb err: %s", stderr)
182 
183  # Check these first so that the more specific error gets raised instead of
184  # the more generic AdbError.
185  stdout = six.ensure_str(stdout)
186  stderr = six.ensure_str(stderr)
187  if "device not found" in stderr:
188  raise DeviceNotFoundError()
189  elif "device unauthorized" in stderr:
190  raise DeviceUnauthorizedError()
191  elif MultipleDevicesError.CheckError(stderr):
192  # The error messages are from adb's transport.c, but something adds
193  # "error: " to the beginning, so take it off so that we don't end up
194  # printing "Error: error: ..."
195  raise MultipleDevicesError(re.sub("^error: ", "", stderr))
196  elif "INSTALL_FAILED_OLDER_SDK" in stdout:
197  raise OldSdkException()
198 
199  if adb.returncode != 0:
200  raise AdbError(args, adb.returncode, stdout, stderr)
201 
202  return adb.returncode, stdout, stderr, args

357 def UploadDexes(adb, execroot, app_dir, temp_dir, dexmanifest, full_install):
358  """Uploads dexes to the device so that the state.
359 
360  Does the minimum amount of work necessary to make the state of the device
361  consistent with what was built.
362 
363  Args:
364  adb: the Adb instance representing the device to install to
365  execroot: the execroot
366  app_dir: the directory things should be installed under on the device
367  temp_dir: a local temporary directory
368  dexmanifest: contents of the dex manifest
369  full_install: whether to do a full install
370 
371  Returns:
372  None.
373  """
374 
375  # Fetch the manifest on the device
376  dex_dir = targetpath.join(app_dir, "dex")
377  adb.Mkdir(dex_dir)
378 
379  old_manifest = None
380 
381  if not full_install:
382  logging.info("Fetching dex manifest from device...")
383  old_manifest_contents = adb.Pull(targetpath.join(dex_dir, "manifest"))
384  if old_manifest_contents:
385  old_manifest = ParseManifest(old_manifest_contents)
386  else:
387  logging.info("Dex manifest not found on device")
388 
389  if old_manifest is None:
390  # If the manifest is not found, maybe a previous installation attempt
391  # was interrupted. Wipe the slate clean. Do this also in case we do a full
392  # installation.
393  old_manifest = {}
394  adb.Delete(targetpath.join(dex_dir, "*"))
395 
396  new_manifest = ParseManifest(dexmanifest)
397  dexes_to_delete = set(old_manifest) - set(new_manifest)
398 
399  # Figure out which dexes to upload: those that are present in the new manifest
400  # but not in the old one and those whose checksum was changed
401  common_dexes = set(new_manifest).intersection(old_manifest)
402  dexes_to_upload = set(d for d in common_dexes
403  if new_manifest[d].sha256 != old_manifest[d].sha256)
404  dexes_to_upload.update(set(new_manifest) - set(old_manifest))
405 
406  if not dexes_to_delete and not dexes_to_upload:
407  # If we have nothing to do, don't bother removing and rewriting the manifest
408  logging.info("Application dexes up-to-date")
409  return
410 
411  # Delete the manifest so that we know how to get back to a consistent state
412  # if we are interrupted.
413  adb.Delete(targetpath.join(dex_dir, "manifest"))
414 
415  # Tuple of (local, remote) files to push to the device.
416  files_to_push = []
417 
418  # Sort dexes to be uploaded by the zip file they are in so that we only need
419  # to open each zip only once.
420  dexzips_in_upload = set(new_manifest[d].input_file for d in dexes_to_upload
421  if new_manifest[d].zippath != "-")
422  for i, dexzip_name in enumerate(dexzips_in_upload):
423  zip_dexes = [
424  d for d in dexes_to_upload if new_manifest[d].input_file == dexzip_name]
425  dexzip_tempdir = hostpath.join(temp_dir, "dex", str(i))
426  with zipfile.ZipFile(hostpath.join(execroot, dexzip_name)) as dexzip:
427  for dex in zip_dexes:
428  zippath = new_manifest[dex].zippath
429  dexzip.extract(zippath, dexzip_tempdir)
430  files_to_push.append((hostpath.join(dexzip_tempdir, zippath),
431  targetpath.join(dex_dir, dex)))
432 
433  # Now gather all the dexes that are not within a .zip file.
434  dexes_to_upload = set(
435  d for d in dexes_to_upload if new_manifest[d].zippath == "-")
436  for dex in dexes_to_upload:
437  files_to_push.append((new_manifest[dex].input_file, targetpath.join(
438  dex_dir, dex)))
439 
440  num_files = len(dexes_to_delete) + len(files_to_push)
441  logging.info("Updating %d dex%s...", num_files, "es" if num_files > 1 else "")
442 
443  # Delete the dexes that are not in the new manifest
444  adb.DeleteMultiple(targetpath.join(dex_dir, dex) for dex in dexes_to_delete)
445 
446  # Upload all the files.
447  upload_walltime_start = time.time()
448  fs = [adb.Push(local, remote) for local, remote in files_to_push]
449  done, not_done = futures.wait(fs, return_when=futures.FIRST_EXCEPTION)
450  upload_walltime = time.time() - upload_walltime_start
451  logging.debug("Dex upload walltime: %s seconds", upload_walltime)
452 
453  # If there is anything in not_done, then some adb call failed and we
454  # can cancel the rest.
455  if not_done:
456  for f in not_done:
457  f.cancel()
458 
459  # If any adb call resulted in an exception, re-raise it.
460  for f in done:
461  f.result()
462 
463  # If no dex upload failed, upload the manifest. If any upload failed, the
464  # exception should have been re-raised above.
465  # Call result() to raise the exception if there was one.
466  adb.PushString(dexmanifest, targetpath.join(dex_dir, "manifest")).result()

469 def Checksum(filename):
470  """Compute the SHA-256 checksum of a file."""
471  h = hashlib.sha256()
472  with open(filename, "rb") as f:
473  while True:
474  data = f.read(65536)
475  if not data:
476  break
477 
478  h.update(data)
479 
480  return h.hexdigest()

577  adb.Mkdir(targetpath.join(app_dir, "native"))

617  if not_done:
618  for f in not_done:
619  f.cancel()
620

622  for f in done:
623  f.result()

656 def SplitIncrementalInstall(adb, app_package, execroot, split_main_apk,
657  split_apks):
658  """Does incremental installation using split packages."""
659  app_dir = targetpath.join(DEVICE_DIRECTORY, app_package)
660  device_manifest_path = targetpath.join(app_dir, "split_manifest")
661  device_manifest = adb.Pull(device_manifest_path)
662  expected_timestamp = adb.Pull(targetpath.join(app_dir, "install_timestamp"))
663  actual_timestamp = adb.GetInstallTime(app_package)
664  device_checksums = {}
665  if device_manifest is not None:
666  for manifest_line in device_manifest.split("\n"):
667  if manifest_line:
668  name, checksum = manifest_line.split(" ")
669  device_checksums[name] = checksum
670 
671  install_checksums = {}
672  install_checksums["__MAIN__"] = Checksum(
673  hostpath.join(execroot, split_main_apk))
674  for apk in split_apks:
675  install_checksums[apk] = Checksum(hostpath.join(execroot, apk))
676 
677  reinstall_main = False
678  if (device_manifest is None or actual_timestamp is None or
679  actual_timestamp != expected_timestamp or
680  install_checksums["__MAIN__"] != device_checksums["__MAIN__"] or
681  set(device_checksums.keys()) != set(install_checksums.keys())):
682  # The main app is not up to date or not present or something happened
683  # with the on-device manifest. Start from scratch. Notably, we cannot
684  # uninstall a split package, so if the set of packages changes, we also
685  # need to do a full reinstall.
686  reinstall_main = True
687  device_checksums = {}
688 
689  apks_to_update = [
690  apk for apk in split_apks if
691  apk not in device_checksums or
692  device_checksums[apk] != install_checksums[apk]]
693 
694  if not apks_to_update and not reinstall_main:
695  # Nothing to do
696  return
697 
698  # Delete the device manifest so that if something goes wrong, we do a full
699  # reinstall next time
700  adb.Delete(device_manifest_path)
701 
702  if reinstall_main:
703  logging.info("Installing main APK...")
704  adb.Uninstall(app_package)
705  adb.InstallMultiple(targetpath.join(execroot, split_main_apk))
706  adb.PushString(
707  adb.GetInstallTime(app_package),
708  targetpath.join(app_dir, "install_timestamp")).result()
709 
710  logging.info("Reinstalling %s APKs...", len(apks_to_update))
711 
712  for apk in apks_to_update:
713  adb.InstallMultiple(targetpath.join(execroot, apk), app_package)
714 
715  install_manifest = [
716  six.ensure_str(name) + " " + checksum
717  for name, checksum in install_checksums.items()
718  ]
719  adb.PushString("\n".join(install_manifest),
720  targetpath.join(app_dir, "split_manifest")).result()

723 def IncrementalInstall(adb_path,
724  execroot,
725  stub_datafile,
726  output_marker,
727  adb_jobs,
728  start_type,
729  dexmanifest=None,
730  apk=None,
731  native_libs=None,
732  resource_apk=None,
733  split_main_apk=None,
734  split_apks=None,
735  user_home_dir=None,
736  extra_adb_args=None):
737  """Performs an incremental install.
738 
739  Args:
740  adb_path: Path to the adb executable.
741  execroot: Exec root.
742  stub_datafile: The stub datafile containing the app's package name.
743  output_marker: Path to the output marker file.
744  adb_jobs: The number of instances of adb to use in parallel.
745  start_type: A string describing whether/how to start the app after
746  installing it. Can be 'no', 'cold', or 'warm'.
747  dexmanifest: Path to the .dex manifest file.
748  apk: Path to the .apk file. May be None to perform an incremental install.
749  native_libs: Native libraries to install.
750  resource_apk: Path to the apk containing the app's resources.
751  split_main_apk: the split main .apk if split installation is desired.
752  split_apks: the list of split .apks to be installed.
753  user_home_dir: Path to the user's home directory.
754  extra_adb_args: Extra arguments that will always be passed to adb.
755  """
756  temp_dir = tempfile.mkdtemp()
757  try:
758  adb = Adb(adb_path, temp_dir, adb_jobs, user_home_dir, extra_adb_args)
759  app_package = GetAppPackage(hostpath.join(execroot, stub_datafile))
760  app_dir = targetpath.join(DEVICE_DIRECTORY, app_package)
761  if split_main_apk:
762  SplitIncrementalInstall(adb, app_package, execroot, split_main_apk,
763  split_apks)
764  else:
765  if not apk:
766  VerifyInstallTimestamp(adb, app_package)
767 
768  with open(hostpath.join(execroot, dexmanifest), "rb") as f:
769  dexmanifest = six.ensure_str(f.read(), "utf-8")
770  UploadDexes(adb, execroot, app_dir, temp_dir, dexmanifest, bool(apk))
771  # TODO(ahumesky): UploadDexes waits for all the dexes to be uploaded, and
772  # then UploadResources is called. We could instead enqueue everything
773  # onto the threadpool so that uploading resources happens sooner.
774  UploadResources(adb, hostpath.join(execroot, resource_apk), app_dir)
775  UploadNativeLibs(adb, native_libs, app_dir, bool(apk))
776  if apk:
777  apk_path = targetpath.join(execroot, apk)
778  adb.Install(apk_path)
779  future = adb.PushString(
780  adb.GetInstallTime(app_package),
781  targetpath.join(DEVICE_DIRECTORY, app_package, "install_timestamp"))
782  future.result()
783  else:
784  if start_type == "warm":
785  adb.StopAppAndSaveState(app_package)
786  else:
787  adb.StopApp(app_package)
788 
789  if start_type in ["cold", "warm", "debug"]:
790  logging.info("Starting application %s", app_package)
791  adb.StartApp(app_package, start_type)
792 
793  with open(output_marker, "wb") as _:
794  pass
795  except DeviceNotFoundError:
796  sys.exit("Error: Device not found")
797  except DeviceUnauthorizedError:
798  sys.exit("Error: Device unauthorized. Please check the confirmation "
799  "dialog on your device.")
800  except MultipleDevicesError as e:
801  sys.exit("Error: " + str(e) + "\nTry specifying a device serial with "
802  "\"bazel mobile-install --adb_arg=-s --adb_arg=$ANDROID_SERIAL\"")
803  except OldSdkException as e:
804  sys.exit("Error: The device does not support the API level specified in "
805  "the application's manifest. Check minSdkVersion in "
806  "AndroidManifest.xml")
807  except TimestampException as e:
808  sys.exit("Error:\n%s" % str(e))
809  except AdbError as e:
810  sys.exit("Error:\n%s" % str(e))
811  finally:
812  shutil.rmtree(temp_dir, True)
