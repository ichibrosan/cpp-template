25 bool Concatenator::Merge(const CDH *cdh, const LH *lh) {
26  if (insert_newlines_ && buffer_.get() && buffer_->data_size() &&
27  '\n' != buffer_->last_byte()) {
28  Append("\n", 1);
29  }
30  CreateBuffer();
31  if (Z_NO_COMPRESSION == lh->compression_method()) {
32  buffer_->ReadEntryContents(lh);
33  } else if (Z_DEFLATED == lh->compression_method()) {
34  if (!inflater_) {
35  inflater_.reset(new Inflater());
36  }
37  buffer_->DecompressEntryContents(cdh, lh, inflater_.get());
38  } else {
39  diag_errx(2, "%s is neither stored nor deflated", filename_.c_str());
40  }
41  return true;
42 }

44 void *Concatenator::OutputEntry(bool compress) {
45  if (!buffer_) {
46  return nullptr;
47  }
48 
49  // Allocate a contiguous buffer for the local file header and
50  // deflated data. We assume that deflate decreases the size, so if
51  // the deflater reports overflow, we just save original data.
52  size_t deflated_buffer_size =
53  sizeof(LH) + filename_.size() + buffer_->data_size();
54 
55  // Huge entry (>4GB) needs Zip64 extension field with 64-bit original
56  // and compressed size values.
57  uint8_t
58  zip64_extension_buffer[sizeof(Zip64ExtraField) + 2 * sizeof(uint64_t)];
59  bool huge_buffer = ziph::zfield_needs_ext64(buffer_->data_size());
60  if (huge_buffer) {
61  deflated_buffer_size += sizeof(zip64_extension_buffer);
62  }
63  LH *lh = reinterpret_cast<LH *>(malloc(deflated_buffer_size));
64  if (lh == nullptr) {
65  return nullptr;
66  }
67  lh->signature();
68  lh->version(20);
69  lh->bit_flag(0x0);
70  lh->last_mod_file_time(1); // 00:00:01
71  lh->last_mod_file_date(30 << 9 | 1 << 5 | 1); // 2010-01-01
72  lh->crc32(0x12345678);
73  lh->compressed_file_size32(0);
74  lh->file_name(filename_.c_str(), filename_.size());
75 
76  if (huge_buffer) {
77  // Add Z64 extension if this is a huge entry.
78  lh->uncompressed_file_size32(0xFFFFFFFF);
79  Zip64ExtraField *z64 =
80  reinterpret_cast<Zip64ExtraField *>(zip64_extension_buffer);
81  z64->signature();
82  z64->payload_size(2 * sizeof(uint64_t));
83  z64->attr64(0, buffer_->data_size());
84  lh->extra_fields(reinterpret_cast<uint8_t *>(z64), z64->size());
85  } else {
86  lh->uncompressed_file_size32(buffer_->data_size());
87  lh->extra_fields(nullptr, 0);
88  }
89 
90  uint32_t checksum;
91  uint64_t compressed_size;
92  uint16_t method;
93  if (compress) {
94  method = buffer_->CompressOut(lh->data(), &checksum, &compressed_size);
95  } else {
96  buffer_->CopyOut(lh->data(), &checksum);
97  method = Z_NO_COMPRESSION;
98  compressed_size = buffer_->data_size();
99  }
100  lh->crc32(checksum);
101  lh->compression_method(method);
102  if (huge_buffer) {
103  lh->compressed_file_size32(ziph::zfield_needs_ext64(compressed_size)
104  ? 0xFFFFFFFF
105  : compressed_size);
106  // Not sure if this has to be written in the small case, but it shouldn't
107  // hurt.
108  const_cast<Zip64ExtraField *>(lh->zip64_extra_field())
109  ->attr64(1, compressed_size);
110  } else {
111  // If original data is <4GB, the compressed one is, too.
112  lh->compressed_file_size32(compressed_size);
113  }
114  return reinterpret_cast<void *>(lh);
115 }

127 bool XmlCombiner::Merge(const CDH *cdh, const LH *lh) {
128  if (!concatenator_) {
129  concatenator_.reset(new Concatenator(filename_, false));
130  concatenator_->Append(start_tag_);
131  concatenator_->Append("\n");
132  }
133  // To ensure xml concatentation is idempotent, read in the entry being added
134  // and remove the start and end tags if they are present.
135  TransientBytes bytes_;
136  if (Z_NO_COMPRESSION == lh->compression_method()) {
137  bytes_.ReadEntryContents(lh);
138  } else if (Z_DEFLATED == lh->compression_method()) {
139  if (!inflater_) {
140  inflater_.reset(new Inflater());
141  }
142  bytes_.DecompressEntryContents(cdh, lh, inflater_.get());
143  } else {
144  diag_errx(2, "%s is neither stored nor deflated", filename_.c_str());
145  }
146  uint32_t checksum;
147  char *buf = reinterpret_cast<char *>(malloc(bytes_.data_size()));
148  // TODO(b/37631490): optimize this to avoid copying the bytes twice
149  bytes_.CopyOut(reinterpret_cast<uint8_t *>(buf), &checksum);
150  int start_offset = 0;
151  if (strncmp(buf, start_tag_.c_str(), start_tag_.length()) == 0) {
152  start_offset = start_tag_.length();
153  }
154  uint64_t end = bytes_.data_size();
155  while (end >= end_tag_.length() && std::isspace(buf[end - 1])) end--;
156  if (strncmp(buf + end - end_tag_.length(), end_tag_.c_str(),
157  end_tag_.length()) == 0) {
158  end -= end_tag_.length();
159  } else {
160  // Leave trailing whitespace alone if we didn't find a match.
161  end = bytes_.data_size();
162  }
163  concatenator_->Append(buf + start_offset, end - start_offset);
164  free(buf);
165  return true;
166 }
