56 // Parse the sign.
57 inline bool safe_parse_sign(const char** rest, /*inout*/
58  bool* negative_ptr /*output*/) {
59  const char* start = *rest;
60  const char* end = start + strlen(start);
61 
62  // Consume whitespace.
63  while (start < end && ascii_isspace(start[0])) {
64  ++start;
65  }
66  while (start < end && ascii_isspace(end[-1])) {
67  --end;
68  }
69  if (start >= end) {
70  return false;
71  }
72 
73  // Consume sign.
74  *negative_ptr = (start[0] == '-');
75  if (*negative_ptr || start[0] == '+') {
76  ++start;
77  if (start >= end) {
78  return false;
79  }
80  }
81 
82  *rest = start;
83  return true;
84 }

86 // Consume digits.
87 //
88 // The classic loop:
89 //
90 // for each digit
91 // value = value * base + digit
92 // value *= sign
93 //
94 // The classic loop needs overflow checking. It also fails on the most
95 // negative integer, -2147483648 in 32-bit two's complement representation.
96 //
97 // My improved loop:
98 //
99 // if (!negative)
100 // for each digit
101 // value = value * base
102 // value = value + digit
103 // else
104 // for each digit
105 // value = value * base
106 // value = value - digit
107 //
108 // Overflow checking becomes simple.
109 
110 inline bool safe_parse_positive_int(const char *text, int* value_p) {
111  int value = 0;
112  const int vmax = std::numeric_limits<int>::max();
113  static_assert(vmax > 0, "");
114  const int vmax_over_base = vmax / 10;
115  const char* start = text;
116  const char* end = start + strlen(text);
117  // loop over digits
118  for (; start < end; ++start) {
119  unsigned char c = static_cast<unsigned char>(start[0]);
120  int digit = kAsciiToInt[c];
121  if (digit >= 10) {
122  *value_p = value;
123  return false;
124  }
125  if (value > vmax_over_base) {
126  *value_p = vmax;
127  return false;
128  }
129  value *= 10;
130  if (value > vmax - digit) {
131  *value_p = vmax;
132  return false;
133  }
134  value += digit;
135  }
136  *value_p = value;
137  return true;
138 }

140 inline bool safe_parse_negative_int(const char *text, int* value_p) {
141  int value = 0;
142  const int vmin = std::numeric_limits<int>::min();
143  static_assert(vmin < 0, "");
144  int vmin_over_base = vmin / 10;
145  // 2003 c++ standard [expr.mul]
146  // "... the sign of the remainder is implementation-defined."
147  // Although (vmin/base)*base + vmin%base is always vmin.
148  // 2011 c++ standard tightens the spec but we cannot rely on it.
149  if (vmin % 10 > 0) {
150  vmin_over_base += 1;
151  }
152  const char* start = text;
153  const char* end = start + strlen(text);
154  // loop over digits
155  for (; start < end; ++start) {
156  unsigned char c = static_cast<unsigned char>(start[0]);
157  int digit = kAsciiToInt[c];
158  if (digit >= 10) {
159  *value_p = value;
160  return false;
161  }
162  if (value < vmin_over_base) {
163  *value_p = vmin;
164  return false;
165  }
166  value *= 10;
167  if (value < vmin + digit) {
168  *value_p = vmin;
169  return false;
170  }
171  value -= digit;
172  }
173  *value_p = value;
174  return true;
175 }

177 bool safe_strto32(const string &text, int *value_p) {
178  *value_p = 0;
179  const char* rest = text.c_str();
180  bool negative;
181  if (!safe_parse_sign(&rest, &negative)) {
182  return false;
183  }
184  if (!negative) {

186  } else {
187  return safe_parse_negative_int(rest, value_p);

191 int32_t strto32(const char *str, char **endptr, int base) {
192  if (sizeof(int32_t) == sizeof(long)) { // NOLINT
193  return static_cast<int32_t>(strtol(str, endptr, base)); // NOLINT
194  }
195  const int saved_errno = errno;
196  errno = 0;
197  const long result = strtol(str, endptr, base); // NOLINT
198  if (errno == ERANGE && result == LONG_MIN) {
199  return kint32min;
200  } else if (errno == ERANGE && result == LONG_MAX) {
201  return kint32max;
202  } else if (errno == 0 && result < kint32min) {
203  errno = ERANGE;
204  return kint32min;
205  } else if (errno == 0 && result > kint32max) {
206  errno = ERANGE;
207  return kint32max;
208  }
209  if (errno == 0)
210  errno = saved_errno;
211  return static_cast<int32_t>(result);
212 }
