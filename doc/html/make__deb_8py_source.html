94  for field in DEBIAN_FIELDS:

105  flags.mark_flag_as_required(fieldname)

115 def AddArFileEntry(fileobj, filename,
116  content='', content_len=-1, timestamp=0,
117  owner_id=0, group_id=0, mode=0o644):
118  """Add a AR file entry to fileobj."""
119  # If we got the content as a string, turn it into a file like thing.
120  if isinstance(content, (str, bytes)):
121  content_len, content = ConvertToFileLike(content, content_len, BytesIO)
122  inputs = [
123  (filename + '/').ljust(16), # filename (SysV)
124  str(timestamp).ljust(12), # timestamp
125  str(owner_id).ljust(6), # owner id
126  str(group_id).ljust(6), # group id
127  str(oct(mode)).replace('0o', '0').ljust(8), # mode
128  str(content_len).ljust(10), # size
129  '\x60\x0a', # end of file entry
130  ]
131  for i in inputs:
132  fileobj.write(i.encode('ascii'))
133  size = 0
134  while True:
135  data = content.read(_COPY_CHUNK_SIZE)
136  if not data:
137  break
138  size += len(data)
139  fileobj.write(data)
140  if size % 2 != 0:
141  fileobj.write(b'\n') # 2-byte alignment padding

144 def MakeDebianControlField(name, value, wrap=False):
145  """Add a field to a debian control file."""
146  result = six.ensure_str(name) + ': '
147  if isinstance(value, str):
148  value = six.ensure_str(value, 'utf-8')
149  if isinstance(value, list):
150  value = ', '.join(six.ensure_str(v) for v in value)
151  if wrap:
152  result += ' '.join(six.ensure_str(value).split('\n'))
153  result = textwrap.fill(result,
154  break_on_hyphens=False,
155  break_long_words=False)
156  else:
157  result += value
158  return six.ensure_str(result).replace('\n', '\n ') + '\n'

161 def CreateDebControl(extrafiles=None, **kwargs):
162  """Create the control.tar.gz file."""
163  # create the control file
164  controlfile = ''
165  for values in DEBIAN_FIELDS:
166  fieldname = values[0]
167  key = fieldname[0].lower() + fieldname[1:].replace('-', '')
168  if values[1] or (key in kwargs and kwargs[key]):
169  controlfile += MakeDebianControlField(fieldname, kwargs[key], values[2])
170  # Create the control.tar file
171  tar = BytesIO()
172  with gzip.GzipFile('control.tar.gz', mode='w', fileobj=tar, mtime=0) as gz:
173  with tarfile.open('control.tar.gz', mode='w', fileobj=gz) as f:
174  tarinfo = tarfile.TarInfo('control')
175  controlfile_utf8 = six.ensure_binary(controlfile, 'utf-8')
176  # Don't discard unicode characters when computing the size
177  tarinfo.size = len(controlfile_utf8)
178  f.addfile(tarinfo, fileobj=BytesIO(controlfile_utf8))
179  if extrafiles:
180  for name, (data, mode) in extrafiles.items():
181  tarinfo = tarfile.TarInfo(name)
182  tarinfo.size = len(data)
183  tarinfo.mode = mode
184  f.addfile(tarinfo, fileobj=BytesIO(six.ensure_binary(data, 'utf-8')))
185  control = tar.getvalue()
186  tar.close()
187  return control

190 def CreateDeb(output,
191  data,
192  preinst=None,
193  postinst=None,
194  prerm=None,
195  postrm=None,
196  config=None,
197  templates=None,
198  conffiles=None,
199  **kwargs):
200  """Create a full debian package."""
201  extrafiles = {}
202  if preinst:
203  extrafiles['preinst'] = (preinst, 0o755)
204  if postinst:
205  extrafiles['postinst'] = (postinst, 0o755)
206  if prerm:
207  extrafiles['prerm'] = (prerm, 0o755)
208  if postrm:
209  extrafiles['postrm'] = (postrm, 0o755)
210  if config:
211  extrafiles['config'] = (config, 0o755)
212  if templates:
213  extrafiles['templates'] = (templates, 0o755)
214  if conffiles:
215  extrafiles['conffiles'] = ('\n'.join(conffiles) + '\n', 0o644)
216  control = CreateDebControl(extrafiles=extrafiles, **kwargs)
217 
218  # Write the final AR archive (the deb package)
219  with open(output, 'wb') as f:
220  f.write(b'!<arch>\n') # Magic AR header
221  AddArFileEntry(f, 'debian-binary', b'2.0\n')
222  AddArFileEntry(f, 'control.tar.gz', control)
223  # Tries to preserve the extension name
224  ext = six.ensure_str(os.path.basename(data)).split('.')[-2:]
225  if len(ext) < 2:
226  ext = 'tar'
227  elif ext[1] == 'tgz':
228  ext = 'tar.gz'
229  elif ext[1] == 'tar.bzip2':
230  ext = 'tar.bz2'
231  else:
232  ext = '.'.join(ext)
233  if ext not in ['tar.bz2', 'tar.gz', 'tar.xz', 'tar.lzma']:
234  ext = 'tar'
235  data_size = os.stat(data).st_size
236  with open(data, 'rb') as datafile:
237  AddArFileEntry(f, 'data.' + ext, datafile, content_len=data_size)

240 def GetChecksumsFromFile(filename, hash_fns=None):
241  """Computes MD5 and/or other checksums of a file.
242 
243  Args:
244  filename: Name of the file.
245  hash_fns: Mapping of hash functions.
246  Default is {'md5': hashlib.md5}
247 
248  Returns:
249  Mapping of hash names to hexdigest strings.
250  { <hashname>: <hexdigest>, ... }
251  """
252  hash_fns = hash_fns or {'md5': hashlib.md5}
253  checksums = {k: fn() for (k, fn) in hash_fns.items()}
254 
255  with open(filename, 'rb') as file_handle:
256  while True:
257  buf = file_handle.read(1048576) # 1 MiB
258  if not buf:
259  break
260  for hashfn in checksums.values():
261  hashfn.update(buf)
262 
263  return {k: fn.hexdigest() for (k, fn) in checksums.items()}

266 def CreateChanges(output,
267  deb_file,
268  architecture,
269  short_description,
270  maintainer,
271  package,
272  version,
273  section,
274  priority,
275  distribution,
276  urgency,
277  timestamp=0):
278  """Create the changes file."""
279  checksums = GetChecksumsFromFile(deb_file, {'md5': hashlib.md5,
280  'sha1': hashlib.sha1,
281  'sha256': hashlib.sha256})
282  debsize = str(os.path.getsize(deb_file))
283  deb_basename = os.path.basename(deb_file)
284 
285  changesdata = ''.join([
286  MakeDebianControlField('Format', '1.8'),
287  MakeDebianControlField('Date', time.ctime(timestamp)),
288  MakeDebianControlField('Source', package),
289  MakeDebianControlField('Binary', package),
290  MakeDebianControlField('Architecture', architecture),
291  MakeDebianControlField('Version', version),
292  MakeDebianControlField('Distribution', distribution),
293  MakeDebianControlField('Urgency', urgency),
294  MakeDebianControlField('Maintainer', maintainer),
295  MakeDebianControlField('Changed-By', maintainer),
296  MakeDebianControlField('Description',
297  '\n%s - %s' % (package, short_description)),
298  MakeDebianControlField('Changes',
299  ('\n%s (%s) %s; urgency=%s'
300  '\nChanges are tracked in revision control.') %
301  (package, version, distribution, urgency)),
302  MakeDebianControlField(
303  'Files', '\n' + ' '.join(
304  [checksums['md5'], debsize, section, priority, deb_basename])),
305  MakeDebianControlField(
306  'Checksums-Sha1',
307  '\n' + ' '.join([checksums['sha1'], debsize, deb_basename])),
308  MakeDebianControlField(
309  'Checksums-Sha256',
310  '\n' + ' '.join([checksums['sha256'], debsize, deb_basename]))
311  ])
312  with open(output, 'w') as changes_fh:
313  changes_fh.write(six.ensure_binary(changesdata, 'utf-8'))
