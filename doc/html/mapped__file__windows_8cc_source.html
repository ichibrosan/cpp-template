45 MappedInputFile::MappedInputFile(const char* name) {
46  impl_ = NULL;
47  opened_ = false;
48  errmsg_ = errmsg;
49 
50  wstring wname;
51  string error;
52  if (!blaze_util::AsAbsoluteWindowsPath(name, &wname, &error)) {
53  BAZEL_DIE(255) << "MappedInputFile(" << name
54  << "): AsAbsoluteWindowsPath failed: " << error;
55  }
56  HANDLE file = CreateFileW(wname.c_str(), GENERIC_READ, FILE_SHARE_READ, NULL,
57  OPEN_EXISTING, 0, NULL);
58  if (file == INVALID_HANDLE_VALUE) {
59  string errormsg = blaze_util::GetLastErrorString();
60  BAZEL_DIE(255) << "MappedInputFile(" << name << "): CreateFileW("
61  << blaze_util::WstringToCstring(wname)
62  << ") failed: " << errormsg;
63  }
64 
65  LARGE_INTEGER size;
66  if (!GetFileSizeEx(file, &size)) {
67  string errormsg = blaze_util::GetLastErrorString();
68  BAZEL_DIE(255) << "MappedInputFile(" << name
69  << "): GetFileSizeEx failed: " << errormsg;
70  }
71 
72  HANDLE mapping = CreateFileMapping(file, NULL, PAGE_READONLY,
73  size.HighPart, size.LowPart, NULL);
74  if (mapping == NULL || mapping == INVALID_HANDLE_VALUE) {
75  string errormsg = blaze_util::GetLastErrorString();
76  BAZEL_DIE(255) << "MappedInputFile(" << name
77  << "): CreateFileMapping failed: " << errormsg;
78  }
79 
80  void *view = MapViewOfFileEx(mapping, FILE_MAP_READ, 0, 0, 0, NULL);
81  if (view == NULL) {
82  string errormsg = blaze_util::GetLastErrorString();
83  BAZEL_DIE(255) << "MappedInputFile(" << name
84  << "): MapViewOfFileEx failed: " << errormsg;
85  }
86 
87  impl_ = new MappedInputFileImpl(file, mapping);
88  length_ = size.QuadPart;
89  buffer_ = reinterpret_cast<u1*>(view);
90  opened_ = true;
91 }

135 MappedOutputFile::MappedOutputFile(const char* name, size_t estimated_size) {
136  impl_ = NULL;
137  opened_ = false;
138  errmsg_ = errmsg;
139 
140  wstring wname;
141  string error;
142  if (!blaze_util::AsAbsoluteWindowsPath(name, &wname, &error)) {
143  BAZEL_DIE(255) << "MappedOutputFile(" << name
144  << "): AsAbsoluteWindowsPath failed: " << error;
145  }
146  HANDLE file = CreateFileW(wname.c_str(), GENERIC_READ | GENERIC_WRITE, 0,
147  NULL, CREATE_ALWAYS, 0, NULL);
148  if (file == INVALID_HANDLE_VALUE) {
149  string errormsg = blaze_util::GetLastErrorString();
150  BAZEL_DIE(255) << "MappedOutputFile(" << name << "): CreateFileW("
151  << blaze_util::WstringToCstring(wname)
152  << ") failed: " << errormsg;
153  }
154 
155  HANDLE mapping = CreateFileMapping(file, NULL, PAGE_READWRITE,
156  estimated_size >> 32, estimated_size & 0xffffffffUL, NULL);
157  if (mapping == NULL || mapping == INVALID_HANDLE_VALUE) {
158  BAZEL_DIE(255) << "MappedOutputFile(" << name
159  << "): CreateFileMapping failed";
160  }
161 
162  void *view = MapViewOfFileEx(mapping, FILE_MAP_ALL_ACCESS, 0, 0, 0, NULL);
163  if (view == NULL) {
164  string errormsg = blaze_util::GetLastErrorString();
165  BAZEL_DIE(255) << "MappedOutputFile(" << name
166  << "): MapViewOfFileEx failed: " << errormsg;
167  CloseHandle(mapping);
168  CloseHandle(file);
169  return;
170  }
171 
172  impl_ = new MappedOutputFileImpl(file, mapping);
173  buffer_ = reinterpret_cast<u1*>(view);
174  opened_ = true;
175 }

197

204  BAZEL_DIE(255) << "MappedOutputFile::Close: CloseHandle for file failed: "
205  << blaze_util::GetLastErrorString();
206  }
207 
208  return 0;
209 }
