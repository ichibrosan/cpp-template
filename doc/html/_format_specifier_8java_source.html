168  private void checkGeneral() throws FormatFlagsConversionMismatchException {
169  if ((c == Conversion.BOOLEAN || c == Conversion.HASHCODE)
170  && f.contains(Flags.ALTERNATE))

172  // '-' requires a width

219  private void checkFloat() throws FormatFlagsConversionMismatchException {
220  checkNumeric();
221  if (c == Conversion.DECIMAL_FLOAT) {
222  } else if (c == Conversion.HEXADECIMAL_FLOAT) {
223  checkBadFlags(Flags.PARENTHESES, Flags.GROUP);
224  } else if (c == Conversion.SCIENTIFIC) {

231  private void checkNumeric() {
232  if (width != -1 && width < 0)
233  throw new IllegalFormatWidthException(width);
234 
235  if (precision != -1 && precision < 0)
236  throw new IllegalFormatPrecisionException(precision);
237 
238  // '-' and '0' require a width
239  if (width == -1
240  && (f.contains(Flags.LEFT_JUSTIFY) || f
241  .contains(Flags.ZERO_PAD)))
242  throw new MissingFormatWidthException(toString());
243 
244  // bad combination
245  if ((f.contains(Flags.PLUS) && f.contains(Flags.LEADING_SPACE))
246  || (f.contains(Flags.LEFT_JUSTIFY) && f
247  .contains(Flags.ZERO_PAD)))
248  throw new IllegalFormatFlagsException(f.toString());
249  }
250 
251  private void checkText() {
252  if (precision != -1)
253  throw new IllegalFormatPrecisionException(precision);
254  switch (c) {
255  case Conversion.PERCENT_SIGN:
256  if (f.valueOf() != Flags.LEFT_JUSTIFY.valueOf()
257  && f.valueOf() != Flags.NONE.valueOf())
258  throw new IllegalFormatFlagsException(f.toString());
259  // '-' requires a width
260  if (width == -1 && f.contains(Flags.LEFT_JUSTIFY))
261  throw new MissingFormatWidthException(toString());
262  break;
263  case Conversion.LINE_SEPARATOR:
264  if (width != -1)
265  throw new IllegalFormatWidthException(width);
266  if (f.valueOf() != Flags.NONE.valueOf())
267  throw new IllegalFormatFlagsException(f.toString());
268  break;
269  default:
270  throw new UnknownFormatConversionException(String.valueOf(c));
271  }
272  }

274  public void print(String arg, int argIndex)
275  throws IllegalFormatConversionException,
276  FormatFlagsConversionMismatchException {
277 
278  try {
279  if (arg.charAt(0) == '[')
280 
281  failConversion(arg);
282 
283  if (dt) {
284  printDateTime(arg);
285  return;
286  }
287  switch (c) {
288  case Conversion.DECIMAL_INTEGER:
289  case Conversion.OCTAL_INTEGER:
290  case Conversion.HEXADECIMAL_INTEGER:
291  printInteger(arg);
292  break;
293  case Conversion.SCIENTIFIC:
294  case Conversion.GENERAL:
295  case Conversion.DECIMAL_FLOAT:
296  case Conversion.HEXADECIMAL_FLOAT:
297  printFloat(arg);
298  break;
299  case Conversion.CHARACTER:
300  case Conversion.CHARACTER_UPPER:
301  printCharacter(arg);
302  break;
303  case Conversion.BOOLEAN:
304  printBoolean(arg);
305  break;
306  case Conversion.STRING:
307  case Conversion.HASHCODE:
308  case Conversion.LINE_SEPARATOR:
309  case Conversion.PERCENT_SIGN:
310  break;
311  default:
312  throw new UnknownFormatConversionException(String.valueOf(c));
313  }
314  } catch (IllegalFormatConversionException e) {
315  e.setArgIndex(argIndex);
316  throw e;
317  }
