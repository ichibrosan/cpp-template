151  @SuppressWarnings("unchecked") // Lots of conversions with generic types
152  private void monkeyPatchApplication() {
153  // StubApplication is created by reflection in Application#handleBindApplication() ->
154  // LoadedApk#makeApplication(), and its return value is used to set the Application field in all
155  // sorts of Android internals.
156  //
157  // Fortunately, Application#onCreate() is called quite soon after, so what we do is monkey
158  // patch in the real Application instance in StubApplication#onCreate().
159  //
160  // A few places directly use the created Application instance (as opposed to the fields it is
161  // eventually stored in). Fortunately, it's easy to forward those to the actual real
162  // Application class.
163  try {
164  // Find the ActivityThread instance for the current thread
165  Class<?> activityThread = Class.forName("android.app.ActivityThread");
166  Method m = activityThread.getMethod("currentActivityThread");
167  m.setAccessible(true);
168  Object currentActivityThread = m.invoke(null);
169 
170  // Find the mInitialApplication field of the ActivityThread to the real application
171  Field mInitialApplication = activityThread.getDeclaredField("mInitialApplication");
172  mInitialApplication.setAccessible(true);
173  Application initialApplication = (Application) mInitialApplication.get(currentActivityThread);
174  if (initialApplication == StubApplication.this) {
175  mInitialApplication.set(currentActivityThread, realApplication);
176  }
177 
178  // Replace all instance of the stub application in ActivityThread#mAllApplications with the
179  // real one
180  Field mAllApplications = activityThread.getDeclaredField("mAllApplications");
181  mAllApplications.setAccessible(true);
182  List<Application> allApplications = (List<Application>) mAllApplications
183  .get(currentActivityThread);
184  for (int i = 0; i < allApplications.size(); i++) {
185  if (allApplications.get(i) == StubApplication.this) {
186  allApplications.set(i, realApplication);
187  }
188  }
189 
190  // Figure out how loaded APKs are stored.
191 
192  // API version 8 has PackageInfo, 10 has LoadedApk. 9, I don't know.
193  Class<?> loadedApkClass;
194  try {
195  loadedApkClass = Class.forName("android.app.LoadedApk");
196  } catch (ClassNotFoundException e) {
197  loadedApkClass = Class.forName("android.app.ActivityThread$PackageInfo");
198  }
199  Field mApplication = loadedApkClass.getDeclaredField("mApplication");
200  mApplication.setAccessible(true);
201  Field mResDir = loadedApkClass.getDeclaredField("mResDir");
202  mResDir.setAccessible(true);
203 
204  // 10 doesn't have this field, 14 does. Fortunately, there are not many Honeycomb devices
205  // floating around.
206  Field mLoadedApk = null;
207  try {
208  mLoadedApk = Application.class.getDeclaredField("mLoadedApk");
209  } catch (NoSuchFieldException e) {
210  // According to testing, it's okay to ignore this.
211  }
212 
213  // Enumerate all LoadedApk (or PackageInfo) fields in ActivityThread#mPackages and
214  // ActivityThread#mResourcePackages and do two things:
215  // - Replace the Application instance in its mApplication field with the real one
216  // - Replace mResDir to point to the external resource file instead of the .apk. This is
217  // used as the asset path for new Resources objects.
218  // - Set Application#mLoadedApk to the found LoadedApk instance
219  for (String fieldName : new String[] { "mPackages", "mResourcePackages" }) {
220  Field field = activityThread.getDeclaredField(fieldName);
221  field.setAccessible(true);
222  Object value = field.get(currentActivityThread);
223 
224  for (Map.Entry<String, WeakReference<?>> entry :
225  ((Map<String, WeakReference<?>>) value).entrySet()) {
226  Object loadedApk = entry.getValue().get();
227  if (loadedApk == null) {
228  continue;
229  }
230 
231  if (mApplication.get(loadedApk) == StubApplication.this) {
232  mApplication.set(loadedApk, realApplication);
233  if (externalResourceFile != null) {
234  mResDir.set(loadedApk, externalResourceFile);
235  }
236 
237  if (mLoadedApk != null) {
238  mLoadedApk.set(realApplication, loadedApk);
239  }
240  }
241  }
242  }
243  } catch (IllegalAccessException | NoSuchFieldException | NoSuchMethodException |
244  ClassNotFoundException | InvocationTargetException e) {
245  throw new IllegalStateException(e);
246  }
247  }

312  Field implAssets = mResourceImpl.getClass().getDeclaredField("mAssets");
313  implAssets.setAccessible(true);

315  }

317  }
318  } catch (IllegalAccessException | NoSuchFieldException | NoSuchMethodException |

320  throw new IllegalStateException(e);

356  private String copyNativeLibs(String dataDir) throws IOException {
357  File nativeLibDir = new File(INCREMENTAL_DEPLOYMENT_DIR + "/" + packageName + "/native");
358  File newManifestFile = new File(nativeLibDir, "native_manifest");
359  File incrementalDir = new File(dataDir + "/incrementallib");
360  File installedManifestFile = new File(incrementalDir, "manifest");
361  String defaultNativeLibDir = dataDir + "/lib";
362 
363  if (!newManifestFile.exists()) {
364  // Native libraries are not installed incrementally. Just use the regular directory.
365  return defaultNativeLibDir;
366  }
367 
368  Map<String, String> newManifest = parseManifest(newManifestFile);
369  Map<String, String> installedManifest = new LinkedHashMap<String, String>();
370  Set<String> libsToDelete = new LinkedHashSet<String>();
371  Set<String> libsToUpdate = new LinkedHashSet<String>();
372 
373  String realNativeLibDir = newManifest.isEmpty()
374  ? defaultNativeLibDir : incrementalDir.toString();
375 
376  if (!incrementalDir.exists()) {
377  if (!incrementalDir.mkdirs()) {
378  throw new IOException("Could not mkdir " + incrementalDir);
379  }
380  }
381 
382  if (installedManifestFile.exists()) {
383  installedManifest = parseManifest(installedManifestFile);
384  } else {
385  // Delete old libraries, in case things got out of sync.
386  for (String installed : incrementalDir.list(SO)) {
387  libsToDelete.add(installed);
388  }
389  }
390 
391  for (String installed : installedManifest.keySet()) {
392  if (!newManifest.containsKey(installed)
393  || !newManifest.get(installed).equals(installedManifest.get(installed))) {
394  libsToDelete.add(installed);
395  }
396  }
397 
398  for (String newLib : newManifest.keySet()) {
399  if (!installedManifest.containsKey(newLib)
400  || !installedManifest.get(newLib).equals(newManifest.get(newLib))) {
401  libsToUpdate.add(newLib);
402  }
403  }
404 
405  if (libsToDelete.isEmpty() && libsToUpdate.isEmpty()) {
406  // Nothing to be done. Be lazy.
407  return realNativeLibDir;
408  }
409 
410  // Delete the installed manifest file. If anything below goes wrong, everything will be
411  // reinstalled the next time the app starts up.
412  installedManifestFile.delete();
413 
414  for (String toDelete : libsToDelete) {
415  File fileToDelete = new File(incrementalDir + "/" + toDelete);
416  Log.v("StubApplication", "Deleting " + fileToDelete);
417  if (fileToDelete.exists() && !fileToDelete.delete()) {
418  throw new IOException("Could not delete " + fileToDelete);
419  }
420  }
421 
422  for (String toUpdate : libsToUpdate) {
423  Log.v("StubApplication", "Copying: " + toUpdate);
424  File src = new File(nativeLibDir + "/" + toUpdate);
425  copy(src, new File(incrementalDir + "/" + toUpdate));
426  }
427 
428  try {
429  copy(newManifestFile, installedManifestFile);
430  } finally {
431  // If we can't write the installed manifest file, delete it completely so that the next
432  // time we get here we can start with a clean slate.
433  installedManifestFile.delete();
434  }
435 
436  return realNativeLibDir;
437  }

458  Log.v("StubApplication", "Copying " + src + " -> " + dst);
459  InputStream in = null;
460  OutputStream out = null;
461  try {
462  in = new FileInputStream(src);
463  out = new FileOutputStream(dst);
464 
465  // Transfer bytes from in to out
466  byte[] buf = new byte[1048576];
467  int len;
468  while ((len = in.read(buf)) > 0) {
469  out.write(buf, 0, len);
470  }
471  } finally {
472  if (in != null) {
473  in.close();
474  }
475 
476  if (out != null) {
477  out.close();
478  }
479  }
