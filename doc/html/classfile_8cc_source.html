136 // See sec.4.4 of JVM spec.
137 struct Constant {
138 
139  Constant(u1 tag) :
140  slot_(0),
141  tag_(tag) {}
142 
143  virtual ~Constant() {}
144 
145  // For UTF-8 string constants, returns the encoded string.
146  // Otherwise, returns an undefined string value suitable for debugging.
147  virtual std::string Display() = 0;
148 
149  virtual void Write(u1 *&p) = 0;
150 
151  // Called by slot() when a constant has been identified as required
152  // in the output classfile's constant pool. This is a hook allowing
153  // constants to register their dependency on other constants, by
154  // calling slot() on them in turn.
155  virtual void Keep() {}
156 
157  bool Kept() {
158  return slot_ != 0;
159  }
160 
161  // Returns the index of this constant in the output class's constant
162  // pool, assigning a slot if not already done.
163  u2 slot() {
164  if (slot_ == 0) {
165  Keep();
166  slot_ = const_pool_out.size(); // BugBot's "narrowing" warning
167  // is bogus. The number of
168  // output constants can't exceed
169  // the number of input constants.
170  if (slot_ == 0) {
171  fprintf(stderr, "Constant::slot() called before output phase.\n");
172  abort();
173  }
174  const_pool_out.push_back(this);
175  if (tag_ == CONSTANT_Long || tag_ == CONSTANT_Double) {
176  const_pool_out.push_back(NULL);
177  }
178  }
179  return slot_;
180  }
181 
182  u2 slot_; // zero => "this constant is unreachable garbage"
183  u1 tag_;
184 };

465 // See sec.4.7.6 of JVM spec.
466 struct InnerClassesAttribute : Attribute {
467 
468  struct Entry {
469  Constant *inner_class_info;
470  Constant *outer_class_info;
471  Constant *inner_name;
472  u2 inner_class_access_flags;
473  };
474 
475  virtual ~InnerClassesAttribute() {
476  for (size_t i = 0; i < entries_.size(); i++) {
477  delete entries_[i];
478  }
479  }
480 
481  static InnerClassesAttribute* Read(const u1 *&p, Constant *attribute_name) {
482  InnerClassesAttribute *attr = new InnerClassesAttribute;
483  attr->attribute_name_ = attribute_name;
484 
485  u2 number_of_classes = get_u2be(p);
486  for (int ii = 0; ii < number_of_classes; ++ii) {
487  Entry *entry = new Entry;
488  entry->inner_class_info = constant(get_u2be(p));
489  entry->outer_class_info = constant(get_u2be(p));
490  entry->inner_name = constant(get_u2be(p));
491  entry->inner_class_access_flags = get_u2be(p);
492 
493  attr->entries_.push_back(entry);
494  }
495  return attr;
496  }
497 
498  void Write(u1 *&p) {
499  std::set<int> kept_entries;
500  // We keep an entry if the constant referring to the inner class is already
501  // kept. Then we mark its outer class and its class name as kept, too, then
502  // iterate until a fixed point is reached.
503  int entry_count;
504  int iteration = 0;
505 
506  do {
507  entry_count = kept_entries.size();
508  for (int i_entry = 0; i_entry < static_cast<int>(entries_.size());
509  ++i_entry) {
510  Entry* entry = entries_[i_entry];
511  if (entry->inner_class_info->Kept() ||
512  used_class_names.find(entry->inner_class_info->Display()) !=
513  used_class_names.end() ||
514  entry->outer_class_info == class_name) {
515  if (entry->inner_name == NULL) {
516  // JVMS 4.7.6: inner_name_index is zero iff the class is anonymous
517  continue;
518  }
519 
520  kept_entries.insert(i_entry);
521 
522  // JVMS 4.7.6: outer_class_info_index is zero for top-level classes
523  if (entry->outer_class_info != NULL) {
524  entry->outer_class_info->slot();
525  }
526 
527  entry->inner_name->slot();
528  }
529  }
530  iteration += 1;
531  } while (entry_count != static_cast<int>(kept_entries.size()));
532 
533  if (kept_entries.empty()) {
534  return;
535  }
536 
537  WriteProlog(p, 2 + kept_entries.size() * 8);
538  put_u2be(p, kept_entries.size());
539 
540  for (std::set<int>::iterator it = kept_entries.begin();
541  it != kept_entries.end();
542  ++it) {
543  Entry *entry = entries_[*it];
544  put_u2be(p, entry->inner_class_info == NULL
545  ? 0
546  : entry->inner_class_info->slot());
547  put_u2be(p, entry->outer_class_info == NULL
548  ? 0
549  : entry->outer_class_info->slot());
550  put_u2be(p, entry->inner_name == NULL
551  ? 0
552  : entry->inner_name->slot());
553  put_u2be(p, entry->inner_class_access_flags);
554  }
555  }
556 
557  std::vector<Entry*> entries_;
558 };
