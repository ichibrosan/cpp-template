75 void LogMessage::Finish() {
76  std::string message(message_.str());
77  if (log_handler_ != nullptr) {
78  log_handler_->HandleMessage(level_, filename_, line_, message, exit_code_);
79  } else {
80  // If no custom handler was provided, never print INFO messages,
81  // but USER should always be printed, as should warnings and errors.
82  if (level_ == LOGLEVEL_USER) {
83  std::cerr << message << std::endl;
84  } else if (level_ > LOGLEVEL_USER) {
85  std::cerr << LogLevelName(level_) << ": " << message << std::endl;
86  }
87 
88  if (level_ == LOGLEVEL_FATAL) {
89  // Exit for fatal calls after handling the message.
90  std::exit(exit_code_);
91  }
92  }
93 }
