121  def add_file(self, f, destfile, mode=None, ids=None, names=None):
122  """Add a file to the tar file.
123 
124  Args:
125  f: the file to add to the layer
126  destfile: the name of the file in the layer
127  mode: force to set the specified mode, by default the value from the
128  source is taken.
129  ids: (uid, gid) for the file to set ownership
130  names: (username, groupname) for the file to set ownership. `f` will be
131  copied to `self.directory/destfile` in the layer.
132  """
133  dest = destfile.lstrip('/') # Remove leading slashes
134  if self.directory and self.directory != '/':
135  dest = self.directory.lstrip('/') + '/' + dest
136  # If mode is unspecified, derive the mode from the file's mode.
137  if mode is None:
138  mode = 0o755 if os.access(f, os.X_OK) else 0o644
139  if ids is None:
140  ids = (0, 0)
141  if names is None:
142  names = ('', '')
143  dest = os.path.normpath(dest)
144  self.tarfile.add_file(
145  dest,
146  file_content=f,
147  mode=mode,
148  uid=ids[0],
149  gid=ids[1],
150  uname=names[0],
151  gname=names[1])

241  def add_deb(self, deb):
242  """Extract a debian package in the output tar.
243 
244  All files presents in that debian package will be added to the
245  output tar under the same paths. No user name nor group names will
246  be added to the output.
247 
248  Args:
249  deb: the tar file to add
250 
251  Raises:
252  DebError: if the format of the deb archive is incorrect.
253  """
254  with archive.SimpleArFile(deb) as arfile:
255  current = arfile.next()
256  while current and not current.filename.startswith('data.'):
257  current = arfile.next()
258  if not current:
259  raise self.DebError(deb + ' does not contains a data file!')
260  tmpfile = tempfile.mkstemp(suffix=os.path.splitext(current.filename)[-1])
261  with open(tmpfile[1], 'wb') as f:
262  f.write(current.data)
263  self.add_tar(tmpfile[1])
264  os.remove(tmpfile[1])

267 def unquote_and_split(arg, c):
268  """Split a string at the first unquoted occurrence of a character.
269 
270  Split the string arg at the first unquoted occurrence of the character c.
271  Here, in the first part of arg, the backslash is considered the
272  quoting character indicating that the next character is to be
273  added literally to the first part, even if it is the split character.
274 
275  Args:
276  arg: the string to be split
277  c: the character at which to split
278 
279  Returns:
280  The unquoted string before the separator and the string after the
281  separator.
282  """
283  head = ''
284  i = 0
285  while i < len(arg):
286  if arg[i] == c:
287  return (head, arg[i + 1:])
288  elif arg[i] == '\\':
289  i += 1
290  if i == len(arg):
291  # dangling quotation symbol
292  return (head, '')
293  else:
294  head += arg[i]
295  else:
296  head += arg[i]
297  i += 1
298  # if we leave the loop, the character c was not found unquoted
299  return (head, '')

302 def main(unused_argv):
303  # Parse modes arguments
304  default_mode = None
305  if FLAGS.mode:
306  # Convert from octal
307  default_mode = int(FLAGS.mode, 8)
308 
309  mode_map = {}
310  if FLAGS.modes:
311  for filemode in FLAGS.modes:
312  (f, mode) = unquote_and_split(filemode, '=')
313  if f[0] == '/':
314  f = f[1:]
315  mode_map[f] = int(mode, 8)
316 
317  default_ownername = ('', '')
318  if FLAGS.owner_name:
319  default_ownername = FLAGS.owner_name.split('.', 1)
320  names_map = {}
321  if FLAGS.owner_names:
322  for file_owner in FLAGS.owner_names:
323  (f, owner) = unquote_and_split(file_owner, '=')
324  (user, group) = owner.split('.', 1)
325  if f[0] == '/':
326  f = f[1:]
327  names_map[f] = (user, group)
328 
329  default_ids = FLAGS.owner.split('.', 1)
330  default_ids = (int(default_ids[0]), int(default_ids[1]))
331  ids_map = {}
332  if FLAGS.owners:
333  for file_owner in FLAGS.owners:
334  (f, owner) = unquote_and_split(file_owner, '=')
335  (user, group) = owner.split('.', 1)
336  if f[0] == '/':
337  f = f[1:]
338  ids_map[f] = (int(user), int(group))
339 
340  # Add objects to the tar file
341  with TarFile(FLAGS.output, FLAGS.directory, FLAGS.compression,
342  FLAGS.root_directory, FLAGS.mtime) as output:
343 
344  def file_attributes(filename):
345  if filename.startswith('/'):
346  filename = filename[1:]
347  return {
348  'mode': mode_map.get(filename, default_mode),
349  'ids': ids_map.get(filename, default_ids),
350  'names': names_map.get(filename, default_ownername),
351  }
352 
353  if FLAGS.manifest:
354  with open(FLAGS.manifest, 'r') as manifest_fp:
355  manifest = json.load(manifest_fp)
356  for f in manifest.get('files', []):
357  output.add_file(f['src'], f['dst'], **file_attributes(f['dst']))
358  for f in manifest.get('empty_files', []):
359  output.add_empty_file(f, **file_attributes(f))
360  for d in manifest.get('empty_dirs', []):
361  output.add_empty_dir(d, **file_attributes(d))
362  for d in manifest.get('empty_root_dirs', []):
363  output.add_empty_root_dir(d, **file_attributes(d))
364  for f in manifest.get('symlinks', []):
365  output.add_link(f['linkname'], f['target'])
366  for tar in manifest.get('tars', []):
367  output.add_tar(tar)
368  for deb in manifest.get('debs', []):
369  output.add_deb(deb)
370 
371  for f in FLAGS.file:
372  (inf, tof) = unquote_and_split(f, '=')
373  output.add_file(inf, tof, **file_attributes(tof))
374  for f in FLAGS.empty_file:
375  output.add_empty_file(f, **file_attributes(f))
376  for f in FLAGS.empty_dir:
377  output.add_empty_dir(f, **file_attributes(f))
378  for f in FLAGS.empty_root_dir:
379  output.add_empty_root_dir(f, **file_attributes(f))
380  for tar in FLAGS.tar:
381  output.add_tar(tar)
382  for deb in FLAGS.deb:
383  output.add_deb(deb)
384  for link in FLAGS.link:
385  l = unquote_and_split(link, ':')
386  output.add_link(l[0], l[1])
