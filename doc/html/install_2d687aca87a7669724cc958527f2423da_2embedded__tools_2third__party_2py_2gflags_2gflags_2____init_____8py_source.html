37 """This package is used to define and parse command line flags.
38 
39 This package defines a *distributed* flag-definition policy: rather than
40 an application having to define all flags in or near main(), each python
41 module defines flags that are useful to it. When one python module
42 imports another, it gains access to the other's flags. (This is
43 implemented by having all modules share a common, global registry object
44 containing all the flag information.)
45 
46 Flags are defined through the use of one of the DEFINE_xxx functions.
47 The specific function used determines how the flag is parsed, checked,
48 and optionally type-converted, when it's seen on the command line.
49 """

127 def register_validator(flag_name,
128  checker,
129  message='Flag validation failed',
130  flag_values=FLAGS):
131  """Adds a constraint, which will be enforced during program execution.
132 
133  The constraint is validated when flags are initially parsed, and after each
134  change of the corresponding flag's value.
135  Args:
136  flag_name: str, Name of the flag to be checked.
137  checker: callable, A function to validate the flag.
138  input - A single positional argument: The value of the corresponding
139  flag (string, boolean, etc. This value will be passed to checker
140  by the library).
141  output - Boolean.
142  Must return True if validator constraint is satisfied.
143  If constraint is not satisfied, it should either return False or
144  raise gflags.ValidationError(desired_error_message).
145  message: Error text to be shown to the user if checker returns False.
146  If checker raises gflags.ValidationError, message from the raised
147  Error will be shown.
148  flag_values: An optional FlagValues instance to validate against.
149  Raises:
150  AttributeError: If flag_name is not registered as a valid flag name.
151  """
152  v = gflags_validators.SingleFlagValidator(flag_name, checker, message)
153  _add_validator(flag_values, v)

284  'command line!' % flag_name)

395 def DEFINE_flag(flag, flag_values=FLAGS, module_name=None): # pylint: disable=g-bad-name
396  """Registers a 'Flag' object with a 'FlagValues' object.
397 
398  By default, the global FLAGS 'FlagValue' object is used.
399 
400  Typical users will use one of the more specialized DEFINE_xxx
401  functions, such as DEFINE_string or DEFINE_integer. But developers
402  who need to create Flag objects themselves should use this function
403  to register their flags.
404 
405  Args:
406  flag: A Flag object, a flag that is key to the module.
407  flag_values: FlagValues object with which the flag will be registered.
408  module_name: A string, the name of the Python module declaring this flag.
409  If not provided, it will be computed using the stack trace of this call.
410  """
411  # copying the reference to flag_values prevents pychecker warnings
412  fv = flag_values
413  fv[flag.name] = flag
414  # Tell flag_values who's defining the flag.
415  if isinstance(flag_values, FlagValues):
416  # Regarding the above isinstance test: some users pass funny
417  # values of flag_values (e.g., {}) in order to avoid the flag
418  # registration (in the past, there used to be a flag_values ==
419  # FLAGS test here) and redefine flags with the same name (e.g.,
420  # debug). To avoid breaking their code, we perform the
421  # registration only if flag_values is a real FlagValues object.
422  if module_name:
423  module = sys.modules.get(module_name)
424  else:
425  module, module_name = _helpers.GetCallingModuleObjectAndName()

466 def DECLARE_key_flag( # pylint: disable=g-bad-name
467  flag_name, flag_values=FLAGS):
468  """Declares one flag as key to the current module.
469 
470  Key flags are flags that are deemed really important for a module.
471  They are important when listing help messages; e.g., if the
472  --helpshort command-line flag is used, then only the key flags of the
473  main module are listed (instead of all flags, as in the case of

490  # specially during flag parsing, taking precedence over the
491  # user-defined flags.
492  _internal_declare_key_flags([flag_name],

531 def DISCLAIM_key_flags(): # pylint: disable=g-bad-name
532  """Declares that the current module will not define any more key flags.
533 
534  Normally, the module that calls the DEFINE_xxx functions claims the
535  flag to be its key flag. This is undesirable for modules that
536  define additional DEFINE_yyy functions with its own flag parsers and
