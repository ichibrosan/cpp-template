36 bool MappedFile::Open(const std::string& path) {
37  if (is_open()) {
38  diag_errx(1, "%s:%d: This instance is already open", __FILE__, __LINE__);
39  }
40  if ((fd_ = open(path.c_str(), O_RDONLY)) < 0) {
41  diag_warn("%s:%d: open %s:", __FILE__, __LINE__, path.c_str());
42  return false;
43  }
44  // Map the file, even if it is empty (in which case allocate 1 byte to it).
45  struct stat st;
46  if (fstat(fd_, &st) ||
47  (mapped_start_ = static_cast<unsigned char *>(
48  mmap(nullptr, st.st_size ? st.st_size : 1, PROT_READ, MAP_PRIVATE,
49  fd_, 0))) == MAP_FAILED) {
50  if (S_ISDIR(st.st_mode)) {
51  diag_warn("%s:%d: %s is a directory", __FILE__, __LINE__, path.c_str());
52  } else {
53  diag_warn("%s:%d: mmap %s:", __FILE__, __LINE__, path.c_str());
54  }
55  close(fd_);
56  fd_ = -1;
57  return false;
58  }
59  mapped_end_ = mapped_start_ + st.st_size;
60  return true;
61 }
62 
63 void MappedFile::Close() {
64  if (is_open()) {
65  munmap(mapped_start_, mapped_end_ - mapped_start_);
66  mapped_start_ = mapped_end_ = nullptr;
67  close(fd_);
68  fd_ = -1;
69  }
70 }
