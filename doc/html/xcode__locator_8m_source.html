55 // Given an entry, insert it into a dictionary that is keyed by versions.
56 //
57 // For an entry that is 6.4.1:/Applications/Xcode.app, add it for 6.4.1 and
58 // optionally add it for 6.4 and 6 if it is "better" than any entry that may
59 // already be there, where "better" is defined as:
60 //
61 // 1. Under /Applications/. (This avoids mounted xcode versions taking
62 // precedence over installed versions.)
63 //
64 // 2. Not older (at least as high version number).
65 static void AddEntryToDictionary(
66  XcodeVersionEntry *entry,
67  NSMutableDictionary<NSString *, XcodeVersionEntry *> *dict) {
68  BOOL inApplications =
69  [entry.url.path rangeOfString:@"/Applications/"].location != NSNotFound;
70  NSString *entryVersion = entry.version;
71  NSString *subversion = entryVersion;
72  if (dict[entryVersion] && !inApplications) {
73  return;
74  }
75  dict[entryVersion] = entry;
76  while (YES) {
77  NSRange range = [subversion rangeOfString:@"." options:NSBackwardsSearch];
78  if (range.length == 0 || range.location == 0) {
79  break;
80  }
81  subversion = [subversion substringToIndex:range.location];
82  XcodeVersionEntry *subversionEntry = dict[subversion];
83  if (subversionEntry) {
84  BOOL atLeastAsLarge = ([subversionEntry.version compare:entry.version]
85  == NSOrderedDescending);
86  if (inApplications && atLeastAsLarge) {
87  dict[subversion] = entry;
88  }
89  } else {
90  dict[subversion] = entry;
91  }
92  }
93 }

95 // Given a "version", expand it to at least 3 components by adding .0 as
96 // necessary.
97 static NSString *ExpandVersion(NSString *version) {
98  NSArray *components = [version componentsSeparatedByString:@"."];
99  NSString *appendage = nil;
100  if (components.count == 2) {
101  appendage = @".0";
102  } else if (components.count == 1) {
103  appendage = @".0.0";
104  }
105  if (appendage) {
106  version = [version stringByAppendingString:appendage];
107  }
108  return version;
109 }

111 // Searches for all available Xcodes in the system and returns a dictionary that
112 // maps version identifiers of any form (X, X.Y, and X.Y.Z) to the directory
113 // where the Xcode bundle lives.
114 //
115 // If there is a problem locating the Xcodes, prints one or more error messages
116 // and returns nil.
117 static NSMutableDictionary<NSString *, XcodeVersionEntry *> *FindXcodes()
118  __attribute((ns_returns_retained)) {
119  CFStringRef cfBundleID = CFSTR("com.apple.dt.Xcode");
120  NSString *bundleID = (__bridge NSString *)cfBundleID;
121 
122  NSMutableDictionary<NSString *, XcodeVersionEntry *> *dict =
123  [[NSMutableDictionary alloc] init];
124  CFErrorRef cfError;
125  NSArray *array = CFBridgingRelease(LSCopyApplicationURLsForBundleIdentifier(
126  cfBundleID, &cfError));
127  if (array == nil) {
128  NSError *nsError = (__bridge NSError *)cfError;
129  fprintf(stderr, "error: %s\n", nsError.description.UTF8String);
130  return nil;
131  }
132 
133  // Scan all bundles but delay returning in case of errors until we are
134  // done. This is to let us log details about all the bundles that were
135  // processed so that a faulty bundle doesn't hide useful information about
136  // other bundles that were found.
137  BOOL errors = NO;
138  for (NSURL *url in array) {
139  NSArray *contents = [
140  [NSFileManager defaultManager] contentsOfDirectoryAtURL:url
141  includingPropertiesForKeys:nil
142  options:0
143  error:nil];
144  NSLog(@"Found bundle %@ in %@; contents on disk: %@",
145  bundleID, url, contents);
146 
147  NSBundle *bundle = [NSBundle bundleWithURL:url];
148  if (bundle == nil) {
149  NSLog(@"ERROR: Unable to open bundle at URL: %@\n", url);
150  errors = YES;
151  continue;
152  }
153 
154  // LSCopyApplicationURLsForBundleIdentifier seems to sometimes return
155  // invalid bundles (e.g. an arbitrary folder), which we should ignore (but
156  // don't treat as an error).
157  //
158  // To work around this issue, we double check to make sure the NSBundle's
159  // bundleIdentifier is that of Xcode's, as invalid bundles won't match.
160  if (![bundle.bundleIdentifier isEqualToString:bundleID]) {
161  NSLog(@"WARNING: Ignoring bundle %@ due to bundleID mismatch "
162  @"(got \"%@\" but expected \"%@\"); info: %@",
163  url, bundle.bundleIdentifier, bundleID, bundle.infoDictionary);
164  continue;
165  }
166 
167  NSString *versionKey = @"CFBundleShortVersionString";
168  NSString *version = [bundle.infoDictionary objectForKey:versionKey];
169  if (version == nil) {
170  NSLog(@"ERROR: Cannot find %@ in info for bundle %@; info: %@\n",
171  versionKey, url, bundle.infoDictionary);
172  errors = YES;
173  continue;
174  }
175  NSString *expandedVersion = ExpandVersion(version);
176  NSLog(@"Version strings for %@: short=%@, expanded=%@",
177  url, version, expandedVersion);
178 
179  NSURL *versionPlistUrl = [url URLByAppendingPathComponent:@"Contents/version.plist"];
180  NSDictionary *versionPlistContents =
181  [[NSDictionary alloc] initWithContentsOfURL:versionPlistUrl];
182  NSString *productVersion = [versionPlistContents objectForKey:@"ProductBuildVersion"];
183  if (productVersion) {
184  expandedVersion = [expandedVersion stringByAppendingFormat:@".%@", productVersion];
185  }
186 
187  NSURL *developerDir =
188  [url URLByAppendingPathComponent:@"Contents/Developer"];
189  XcodeVersionEntry *entry =
190  [[XcodeVersionEntry alloc] initWithVersion:expandedVersion
191  url:developerDir];
192  AddEntryToDictionary(entry, dict);
193  }
194  return errors ? nil : dict;
195 }

197 // Prints out the located Xcodes as a set of lines where each line contains the
198 // list of versions for a given Xcode and its location on disk.
199 static void DumpAsVersionsOnly(
200  FILE *output,
201  NSMutableDictionary<NSString *, XcodeVersionEntry *> *dict) {
202  NSMutableDictionary<NSString *, NSMutableSet <NSString *> *> *aliasDict =
203  [[NSMutableDictionary alloc] init];
204  [dict enumerateKeysAndObjectsUsingBlock:^(NSString *aliasVersion,
205  XcodeVersionEntry *entry,
206  BOOL *stop) {
207  NSString *versionString = entry.version;
208  if (aliasDict[versionString] == nil) {
209  aliasDict[versionString] = [[NSMutableSet alloc] init];
210  }
211  [aliasDict[versionString] addObject:aliasVersion];
212  }];
213  for (NSString *version in aliasDict) {
214  XcodeVersionEntry *entry = dict[version];
215  fprintf(output, "%s:%s:%s\n",
216  version.UTF8String,
217  [[aliasDict[version] allObjects]
218  componentsJoinedByString: @","].UTF8String,
219  entry.url.fileSystemRepresentation);
220  }
221 }

257 int main(int argc, const char * argv[]) {
258  @autoreleasepool {
259  NSString *versionArg = nil;
260  BOOL versionsOnly = NO;
261  if (argc == 1) {
262  versionArg = @"";
263  } else if (argc == 2) {
264  NSString *firstArg = [NSString stringWithUTF8String:argv[1]];
265  if ([@"-v" isEqualToString:firstArg]) {
266  versionsOnly = YES;
267  versionArg = @"";
268  } else {
269  versionArg = firstArg;
270  }
271  }
272  if (versionArg == nil) {
273  usage(stderr);
274  return 1;
275  }
276 
277  NSMutableDictionary<NSString *, XcodeVersionEntry *> *dict = FindXcodes();
278  if (dict == nil) {
279  return 1;
280  }
281 
282  XcodeVersionEntry *entry = [dict objectForKey:versionArg];
283  if (entry) {
284  printf("%s\n", entry.url.fileSystemRepresentation);
285  return 0;
286  }
287 
288  if (versionsOnly) {
289  DumpAsVersionsOnly(stdout, dict);
290  } else {
291  DumpAsJson(stdout, dict);
292  }
293  return ([@"" isEqualToString:versionArg] ? 0 : 1);
294  }
