44 class Validator(object):
45  """Base class for flags validators.
46 
47  Users should NOT overload these classes, and use flags.Register...
48  methods instead.

54  def __init__(self, checker, message):
55  """Constructor to create all validators.
56 
57  Args:
58  checker: function to verify the constraint.
59  Input of this method varies, see SingleFlagValidator and
60  multi_flags_validator for a detailed description.
61  message: str, error message to be shown to the user.
62  """
63  self.checker = checker
64  self.message = message
65  Validator.validators_count += 1
66  # Used to assert validators in the order they were registered.
67  self.insertion_index = Validator.validators_count

76  Raises:
77  Error: Raised if constraint is not satisfied.
78  """
79  param = self._get_input_to_checker_function(flag_values)
80  if not self.checker(param):
81  raise _exceptions.ValidationError(self.message)

106 class SingleFlagValidator(Validator):
107  """Validator behind register_validator() method.
108 
109  Validates that a single flag passes its checker function. The checker function
110  takes the flag value and returns True (if value looks fine) or, if flag value
111  is not valid, either returns False or raises an Exception.
112  """
113 
114  def __init__(self, flag_name, checker, message):
115  """Constructor.
116 
117  Args:
118  flag_name: string, name of the flag.
119  checker: function to verify the validator.
120  input - value of the corresponding flag (string, boolean, etc).
121  output - bool, True if validator constraint is satisfied.
122  If constraint is not satisfied, it should either return False or
123  raise flags.ValidationError(desired_error_message).
124  message: str, error message to be shown to the user if validator's
125  condition is not satisfied.
126  """
127  super(SingleFlagValidator, self).__init__(checker, message)
128  self.flag_name = flag_name
129 
130  def get_flags_names(self):
131  return [self.flag_name]
132 
133  def print_flags_with_values(self, flag_values):
134  return 'flag --%s=%s' % (self.flag_name, flag_values[self.flag_name].value)

139  Args:
140  flag_values: flags.FlagValues, the FlagValues instance to get flags from.

195 def register_validator(flag_name,
196  checker,
197  message='Flag validation failed',
198  flag_values=_flagvalues.FLAGS):
199  """Adds a constraint, which will be enforced during program execution.
200 
201  The constraint is validated when flags are initially parsed, and after each
202  change of the corresponding flag's value.
203  Args:
204  flag_name: str, name of the flag to be checked.
205  checker: callable, a function to validate the flag.
206  input - A single positional argument: The value of the corresponding
207  flag (string, boolean, etc. This value will be passed to checker
208  by the library).
209  output - bool, True if validator constraint is satisfied.
210  If constraint is not satisfied, it should either return False or
211  raise flags.ValidationError(desired_error_message).
212  message: str, error text to be shown to the user if checker returns False.
213  If checker raises flags.ValidationError, message from the raised
214  error will be shown.
215  flag_values: flags.FlagValues, optional FlagValues instance to validate
216  against.
217  Raises:
218  AttributeError: Raised when flag_name is not registered as a valid flag
219  name.
220  """
221  v = SingleFlagValidator(flag_name, checker, message)
222  _add_validator(flag_values, v)

329 def mark_flag_as_required(flag_name, flag_values=_flagvalues.FLAGS):
330  """Ensures that flag is not None during program execution.
331 
332  Registers a flag validator, which will follow usual validator rules.
333  Important note: validator will pass for any non-None value, such as False,
334  0 (zero), '' (empty string) and so on.
335 
336  It is recommended to call this method like this:
337 
338  if __name__ == '__main__':
339  flags.mark_flag_as_required('your_flag_name')
340  app.run()
341 
342  Because validation happens at app.run() we want to ensure required-ness
343  is enforced at that time. However, you generally do not want to force
344  users who import your code to have additional required flags for their
345  own binaries or tests.
346 
347  Args:
348  flag_name: str, name of the flag
349  flag_values: flags.FlagValues, optional FlagValues instance where the flag
350  is defined.
351  Raises:
352  AttributeError: Raised when flag_name is not registered as a valid flag
353  name.
354  """
355  if flag_values[flag_name].default is not None:
356  warnings.warn(
357  'Flag --%s has a non-None default value; therefore, '
358  'mark_flag_as_required will pass even if flag is not specified in the '
359  'command line!' % flag_name)
360  register_validator(flag_name,
361  lambda value: value is not None,
362  message='Flag --%s must be specified.' % flag_name,
363  flag_values=flag_values)

368 
369  Recommended usage:
370 
371  if __name__ == '__main__':
372  flags.mark_flags_as_required(['flag1', 'flag2', 'flag3'])

378  are defined.
379  Raises:
380  AttributeError: If any of flag name has not already been defined as a flag.
381  """

386 def mark_flags_as_mutual_exclusive(flag_names, required=False,
387  flag_values=_flagvalues.FLAGS):
388  """Ensures that only one flag among flag_names is set.
389 
390  Important note: validator will pass for any non-None value, such as False,
391  0 (zero), '' (empty string) and so on. For multi flags, this means that the
392  default needs to be None not [].
393 
394  Args:
395  flag_names: [str], names of the flags.
396  required: bool, if set, exactly one of the flags must be set.
397  Otherwise, it is also valid for none of the flags to be set.
398  flag_values: flags.FlagValues, optional FlagValues instance where the flags
399  are defined.
400  """
401 
402  def validate_mutual_exclusion(flags_dict):
403  flag_count = sum(1 for val in flags_dict.values() if val is not None)
404  if flag_count == 1 or (not required and flag_count == 0):
405  return True
406  message = ('%s one of (%s) must be specified.' %
407  ('Exactly' if required else 'At most', ', '.join(flag_names)))
408  raise _exceptions.ValidationError(message)
409 
410  register_multi_flags_validator(
411  flag_names, validate_mutual_exclusion, flag_values=flag_values)
