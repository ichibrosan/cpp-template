33 bool ReadFrom(file_handle_type handle, string *content, int max_size) {
34  static const size_t kReadSize = 4096; // read 4K chunks
35  content->clear();
36  char buf[kReadSize];
37  // OPT: This loop generates one spurious read on regular files.
38  int error;
39  while (int r = ReadFromHandle(
40  handle, buf,
41  max_size > 0 ? std::min(static_cast<size_t>(max_size), kReadSize)
42  : kReadSize,
43  &error)) {
44  if (r < 0) {
45  if (error == ReadFileResult::INTERRUPTED ||
46  error == ReadFileResult::AGAIN) {
47  continue;
48  }

63 bool ReadFrom(file_handle_type handle, void *data, size_t size) {
64  static const size_t kReadSize = 4096; // read 4K chunks
65  size_t offset = 0;
66  int error;
67  while (int r = ReadFromHandle(handle, reinterpret_cast<char *>(data) + offset,
68  std::min(kReadSize, size), &error)) {
69  if (r < 0) {
70  if (error == ReadFileResult::INTERRUPTED ||
71  error == ReadFileResult::AGAIN) {
72  continue;
73  }
74  return false;
75  }
76  offset += r;
77  if (size > static_cast<size_t>(r)) {
78  size -= r;
79  } else {
80  break;
81  }
82  }
83  return true;
84 }
