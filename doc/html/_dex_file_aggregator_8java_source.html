79  public DexFileAggregator add(Dex dexFile) {
80  if (multidex.isMultidexAllowed()) {
81  // To determine whether currentShard is "full" we track unique field and method signatures,
82  // which predicts precisely the number of field and method indices.
83  if (tracker.track(dexFile) && !currentShard.isEmpty()) {
84  // For simplicity just start a new shard to fit the given file.
85  // Don't bother with waiting for a later file that might fit the old shard as in the extreme
86  // we'd have to wait until the end to write all shards.
87  rotateDexFile();
88  tracker.track(dexFile);
89  }
90  }
91  currentShard.add(dexFile);
92  return this;
93  }

131  private void writeMergedFile(Dex... dexes) {
132  checkArgument(0 < dexes.length);
133  checkState(multidex.isMultidexAllowed() || nextDexFileIndex == 0);
134  String filename = getDexFileName(nextDexFileIndex++);
135  ListenableFuture<Dex> merged =
136  dexes.length == 1 && !forceJumbo
137  ? Futures.immediateFuture(dexes[0])
138  : executor.submit(new RunDexMerger(dexes));
139  lastWriter =
140  Futures.whenAllSucceed(lastWriter, merged)
141  .call(new WriteFile(filename, merged, dest), executor);
142  }

144  private Dex merge(Dex... dexes) throws IOException {
145  switch (dexes.length) {
146  case 0:
147  return new Dex(0);
148  case 1:
149  // Need to actually process the single given file for forceJumbo :(
150  return forceJumbo ? merge(dexes[0], new Dex(0)) : dexes[0];
151  default: // fall out
152  }
153  DexMerger dexMerger = new DexMerger(dexes, CollisionPolicy.FAIL, context);
154  dexMerger.setCompactWasteThreshold(wasteThresholdPerDex);
155  if (forceJumbo) {
156  try {
157  DexMerger.class.getMethod("setForceJumbo", Boolean.TYPE).invoke(dexMerger, true);
158  } catch (ReflectiveOperationException e) {
159  throw new IllegalStateException("--forceJumbo flag not supported", e);
160  }
161  }
162 
163  try {
164  return dexMerger.merge();
165  } catch (BufferOverflowException e) {
166  if (dexes.length <= 2) {
167  throw e;
168  }
169  // Bug in dx can cause this for ~1500 or more classes
170  Dex[] left = Arrays.copyOf(dexes, dexes.length / 2);
171  Dex[] right = Arrays.copyOfRange(dexes, left.length, dexes.length);
172  System.err.printf("Couldn't merge %d classes, trying %d%n", dexes.length, left.length);
173  try {
174  return merge(merge(left), merge(right));
175  } catch (RuntimeException e2) {
176  e2.addSuppressed(e);
177  throw e2;
178  }
179  }
180  }
