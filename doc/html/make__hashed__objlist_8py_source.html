16 """Creates symbolic links for .o files with hashcode.
17 
18 This script reads the file list containing the input files, creates
19 symbolic links with a path-hash appended to their original name (foo.o
20 becomes foo_{md5sum}.o), then saves the list of symbolic links to another
21 file.
22 
23 The symbolic links are created into the given temporary directory. There is
24 no guarantee that we can write to the directory that contained the inputs to
25 this script.
26 
27 This is to circumvent a bug in the original libtool that arises when two
28 input files have the same base name (even if they are in different
29 directories).
30 """
31 
32 import hashlib
33 import os
34 import sys
35 
36 
37 def main():
38  outdir = sys.argv[3]
39  with open(sys.argv[1]) as obj_file_list:
40  with open(sys.argv[2], 'w') as hashed_obj_file_list:
41  for line in obj_file_list:
42  obj_file_path = line.rstrip('\n')
43 
44  hashed_obj_file_name = '%s_%s.o' % (
45  os.path.basename(os.path.splitext(obj_file_path)[0]),
46  hashlib.md5(obj_file_path.encode('utf-8')).hexdigest())
47  hashed_obj_file_path = os.path.join(outdir, hashed_obj_file_name)
48 
49  hashed_obj_file_list.write(hashed_obj_file_path + '\n')
50 
51  # Create symlink only if the symlink doesn't exist.
52  if not os.path.exists(hashed_obj_file_path):
53  os.symlink(os.path.abspath(obj_file_path),
54  hashed_obj_file_path)
55 
56 
57 if __name__ == '__main__':
58  main()
