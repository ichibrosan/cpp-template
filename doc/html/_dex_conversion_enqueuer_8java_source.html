56  // Rate-limit to 30000 files in flight at once, which is about what we've tested. Theoretically,
57  // an unbounded queue can lead to OutOfMemoryErrors, and any limit is helpful so that one can
58  // set -Xmx to handle even large files. The "downside" is that this callable can theoretically
59  // block trying to add more elements to the queue, wasting the thread during that time.
60  private final BlockingQueue<Future<ZipEntryContent>> files = new ArrayBlockingQueue<>(30000);
61 
62  public DexConversionEnqueuer(ZipFile in, ExecutorService executor, DexConverter dexer,
63  @Nullable Cache<DexingKey, byte[]> dexCache) {
64  this.in = in;
65  this.executor = executor;
66  this.dexer = dexer;
67  this.dexCache = dexCache;

70  @Override
71  public Void call() throws InterruptedException, IOException {
72  try {
73  Enumeration<? extends ZipEntry> entries = in.entries();
74  while (entries.hasMoreElements()) {
75  ZipEntry entry = entries.nextElement();
76  // Since these entries come from an existing zip file, they should always know their size
77  // (meaning, never return -1). We also can't handle files that don't fit into a byte array.
78  checkArgument(entry.getSize() >= 0, "Cannot process entry with unknown size: %s", entry);
79  checkArgument(entry.getSize() <= Integer.MAX_VALUE, "Entry too large: %s", entry);
80  byte[] content;
81  if (entry.getSize() == 0L) {
82  content = EMPTY; // this in particular covers directory entries
83  } else {
84  try (InputStream entryStream = in.getInputStream(entry)) {
85  // Read all the content at once, which avoids temporary arrays and extra array copies
86  content = new byte[(int) entry.getSize()];
87  ByteStreams.readFully(entryStream, content); // throws if file is smaller than expected
88  checkState(entryStream.read() == -1,
89  "Too many bytes in jar entry %s, expected %s", entry, entry.getSize());
90  }
91  }
92  if (!entry.isDirectory() && entry.getName().endsWith(".class")) {
93  files.put(toDex(entry, content));
94  } else {
95  // Copy other files and directory entries
96  if (entry.getCompressedSize() != 0) {
97  entry.setCompressedSize(-1L); // We may compress differently from source Zip
98  }
99  files.put(immediateFuture(new ZipEntryContent(entry, content)));
100  }
101  }
102  } finally {
103  // Use try-finally to make absolutely sure we do this, otherwise the reader might deadlock
104  files.put(immediateFuture((ZipEntryContent) null)); // "end of stream" marker
105  }
106  return null;
107  }

109  private Future<ZipEntryContent> toDex(ZipEntry entry, byte[] content) {
110  byte[] cached = dexCache != null ? dexCache.getIfPresent(dexer.getDexingKey(content)) : null;
111  return cached != null
112  ? immediateFuture(storedDexEntry(entry, cached))
113  : executor.submit(new ClassToDex(entry, content, dexer, dexCache));
114  }
115 
126  public BlockingQueue<Future<ZipEntryContent>> getFiles() {

130  private static ZipEntryContent storedDexEntry(ZipEntry classfile, byte[] dexed) {
131  return new ZipEntryContent(
132  storedEntry(classfile.getName() + ".dex", classfile.getTime(), dexed),

136  private static ZipEntry storedEntry(String filename, long time, byte[] content) {
137  // Need to pre-compute checksum for STORED (uncompressed) entries)
138  CRC32 checksum = new CRC32();
139  checksum.update(content);
140 
141  ZipEntry result = new ZipEntry(filename);
142  result.setTime(time);
143  result.setCrc(checksum.getValue());
144  result.setSize(content.length);
145  result.setCompressedSize(content.length);
146  // Write uncompressed, since this is just an intermediary artifact that
147  // we will convert to .dex
148  result.setMethod(ZipEntry.STORED);
149  return result;
150  }
