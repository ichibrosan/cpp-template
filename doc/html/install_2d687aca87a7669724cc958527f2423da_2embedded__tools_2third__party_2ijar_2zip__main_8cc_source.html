82 // Concatene 2 path, path1 and path2, using / as a directory separator and
83 // puting the result in "out". "size" specify the size of the output buffer
84 void concat_path(char* out, const size_t size,
85  const char *path1, const char *path2) {
86  int len1 = strlen(path1);
87  size_t l = len1;
88  strncpy(out, path1, size - 1);
89  out[size-1] = 0;
90  if (l < size - 1 && path1[len1] != '/' && path2[0] != '/') {
91  out[l] = '/';
92  l++;
93  out[l] = 0;
94  }
95  if (l < size - 1) {
96  strncat(out, path2, size - 1 - l);
97  }
98 }

100 void UnzipProcessor::Process(const char* filename, const u4 attr,
101  const u1* data, const size_t size) {
102  mode_t perm = zipattr_to_perm(attr);
103  bool isdir = zipattr_is_dir(attr);
104  const char *output_file_name = filename;
105  if (attr == 0) {
106  // Fallback when the external attribute is not set.
107  isdir = filename[strlen(filename)-1] == '/';
108  perm = 0777;
109  }
110 
111  if (flatten_) {
112  if (isdir) {
113  return;
114  }
115  const char *p = strrchr(filename, '/');
116  if (p != NULL) {
117  output_file_name = p + 1;
118  }
119  }
120 
121  if (verbose_) {
122  printf("%c %o %s\n", isdir ? 'd' : 'f', perm, output_file_name);
123  }
124  if (extract_) {
125  char path[PATH_MAX];
126  concat_path(path, PATH_MAX, output_root_, output_file_name);
127  if (!make_dirs(path, perm) ||
128  (!isdir && !write_file(path, perm, data, size))) {
129  abort();
130  }
131  }
132 }

147 // Execute the extraction (or just listing if just v is provided)
148 int extract(char *zipfile, char *exdir, char **files, bool verbose,
149  bool extract, bool flatten) {
150  std::string cwd = get_cwd();
151  if (cwd.empty()) {
152  return -1;
153  }
154 
155  char output_root[PATH_MAX];
156  if (exdir != NULL) {
157  concat_path(output_root, PATH_MAX, cwd.c_str(), exdir);
158  } else {
159  strncpy(output_root, cwd.c_str(), PATH_MAX);
160  }
161 
162  UnzipProcessor processor(output_root, files, verbose, extract, flatten);
163  std::unique_ptr<ZipExtractor> extractor(ZipExtractor::Create(zipfile,
164  &processor));
165  if (extractor == NULL) {
166  fprintf(stderr, "Unable to open zip file %s: %s.\n", zipfile,
167  strerror(errno));
168  return -1;
169  }
170 
171  if (extractor->ProcessAll() < 0) {
172  fprintf(stderr, "%s.\n", extractor->GetError());
173  return -1;
174  }
175  return 0;
176 }

178 // add a file to the zip
179 int add_file(std::unique_ptr<ZipBuilder> const &builder, char *file,
180  char *zip_path, bool flatten, bool verbose, bool compress) {
181  Stat file_stat = {0, 0666, false};
182  if (file != NULL) {
183  if (!stat_file(file, &file_stat)) {
184  fprintf(stderr, "Cannot stat file %s: %s\n", file, strerror(errno));
185  return -1;
186  }
187  }
188  char *final_path = zip_path != NULL ? zip_path : file;
189 
190  bool isdir = file_stat.is_directory;
191 
192  if (flatten && isdir) {
193  return 0;
194  }
195 
196  // Compute the path, flattening it if requested
197  char path[PATH_MAX];
198  size_t len = strlen(final_path);
199  if (len > PATH_MAX) {
200  fprintf(stderr, "Path too long: %s.\n", final_path);
201  return -1;
202  }
203  if (flatten) {
204  basename(final_path, path, PATH_MAX);
205  } else {
206  strncpy(path, final_path, PATH_MAX);
207  path[PATH_MAX - 1] = 0;
208  if (isdir && len < PATH_MAX - 1) {
209  // Add the trailing slash for folders
210  path[len] = '/';
211  path[len + 1] = 0;
212  }
213  }
214 
215  if (verbose) {
216  mode_t perm = file_stat.file_mode & 0777;
217  printf("%c %o %s\n", isdir ? 'd' : 'f', perm, path);
218  }
219 
220  u1 *buffer = builder->NewFile(path, stat_to_zipattr(file_stat));
221  if (isdir || file_stat.total_size == 0) {
222  builder->FinishFile(0);
223  } else {
224  if (!read_file(file, buffer, file_stat.total_size)) {
225  return -1;
226  }
227  builder->FinishFile(file_stat.total_size, compress, true);
228  }
229  return 0;
230 }

232 // Read a list of files separated by newlines. The resulting array can be
233 // freed using the free method.
234 char **read_filelist(char *filename) {
235  Stat file_stat;
236  if (!stat_file(filename, &file_stat)) {
237  fprintf(stderr, "Cannot stat file %s: %s\n", filename, strerror(errno));
238  return NULL;
239  }
240 
241  char *data = static_cast<char *>(malloc(file_stat.total_size));
242  if (!read_file(filename, data, file_stat.total_size)) {
243  return NULL;
244  }
245 
246  int nb_entries = 1;
247  for (int i = 0; i < file_stat.total_size; i++) {
248  if (data[i] == '\n') {
249  nb_entries++;
250  }
251  }
252 
253  size_t sizeof_array = sizeof(char *) * (nb_entries + 1);
254  void *result = malloc(sizeof_array + file_stat.total_size);
255  // copy the content
256  char **filelist = static_cast<char **>(result);
257  char *content = static_cast<char *>(result) + sizeof_array;
258  memcpy(content, data, file_stat.total_size);
259  free(data);
260  // Create the corresponding array
261  int j = 1;
262  filelist[0] = content;
263  for (int i = 0; i < file_stat.total_size; i++) {
264  if (content[i] == '\n') {
265  content[i] = 0;
266  if (i + 1 < file_stat.total_size) {
267  filelist[j] = content + i + 1;
268  j++;
269  }
270  }
271  }
272  filelist[j] = NULL;
273  return filelist;
274 }

276 // return real paths of the files
277 char **parse_filelist(char *zipfile, char **file_entries, int nb_entries,
278  bool flatten) {
279  // no need to free since the path lists should live until the end of the
280  // program
281  char **files = static_cast<char **>(malloc(sizeof(char *) * nb_entries));
282  char **zip_paths = file_entries;
283  for (int i = 0; i < nb_entries; i++) {
284  char *p_eq = strchr(file_entries[i], '=');
285  if (p_eq != NULL) {
286  if (flatten) {
287  fprintf(stderr, "Unable to create zip file %s: %s.\n", zipfile,
288  "= can't be used with flatten");
289  free(files);
290  return NULL;
291  }
292  if (p_eq == file_entries[i]) {
293  fprintf(stderr, "Unable to create zip file %s: %s.\n", zipfile,
294  "A zip path should be given before =");
295  free(files);
296  return NULL;
297  }
298  *p_eq = '\0';
299  files[i] = p_eq + 1;
300  if (files[i][0] == '\0') {
301  files[i] = NULL;
302  }
303  } else {
304  files[i] = file_entries[i];
305  zip_paths[i] = NULL;
306  }
307  }
308  return files;
309 }

311 // Execute the create operation
312 int create(char *zipfile, char **file_entries, bool flatten, bool verbose,
313  bool compress) {
314  int nb_entries = 0;
315  while (file_entries[nb_entries] != NULL) {
316  nb_entries++;
317  }
318  char **zip_paths = file_entries;
319  char **files = parse_filelist(zipfile, file_entries, nb_entries, flatten);
320  if (files == NULL) {
321  return -1;
322  }
323 
324  u8 size = ZipBuilder::EstimateSize(files, zip_paths, nb_entries);
325  if (size == 0) {
326  return -1;
327  }
328  std::unique_ptr<ZipBuilder> builder(ZipBuilder::Create(zipfile, size));
329  if (builder == NULL) {
330  fprintf(stderr, "Unable to create zip file %s: %s.\n",
331  zipfile, strerror(errno));
332  return -1;
333  }
334 
335  for (int i = 0; i < nb_entries; i++) {
336  if (add_file(builder, files[i], zip_paths[i], flatten, verbose, compress) <
337  0) {
338  return -1;
339  }
340  }
341  if (builder->Finish() < 0) {
342  fprintf(stderr, "%s\n", builder->GetError());
343  return -1;
344  }
345  return 0;
346 }

378  fprintf(stderr,
379  "\nIf the zip path is not specified, it is assumed to be the file "

384 int main(int argc, char **argv) {
385  bool extract = false;
386  bool verbose = false;
387  bool create = false;
388  bool compress = false;
389  bool flatten = false;
390 
391  if (argc < 3) {
392  usage(argv[0]);
393  }
394 
395  for (int i = 0; argv[1][i] != 0; i++) {
396  switch (argv[1][i]) {
397  case 'x':
398  extract = true;
399  break;
400  case 'v':
401  verbose = true;
402  break;
403  case 'c':
404  create = true;
405  break;
406  case 'f':
407  flatten = true;
408  break;
409  case 'C':
410  compress = true;
411  break;
412  default:
413  usage(argv[0]);
414  }
415  }
416 
417  // x and c cannot be used in the same command-line.
418  if (create && extract) {
419  usage(argv[0]);
420  }
421 
422  // Calculate the argument index of the first entry file.
423  int filelist_start_index;
424  if (argc > 3 && strcmp(argv[3], "-d") == 0) {
425  filelist_start_index = 5;
426  } else {
427  filelist_start_index = 3;
428  }
429 
430  char** filelist = NULL;
431 
432  // We have one option file. Read and extract the content.
433  if (argc == filelist_start_index + 1 &&
434  argv[filelist_start_index][0] == '@') {
435  char* filelist_name = argv[filelist_start_index];
436  filelist = devtools_ijar::read_filelist(filelist_name + 1);
437  if (filelist == NULL) {
438  fprintf(stderr, "Can't read file list %s: %s.\n", filelist_name,
439  strerror(errno));
440  return -1;
441  }
442  // We have more than one files. Assume that they are all file entries.
443  } else if (argc >= filelist_start_index + 1) {
444  filelist = argv + filelist_start_index;
445  } else {
446  // There are no entry files specified. This is forbidden if we are creating
447  // a zip file.
448  if (create) {
449  fprintf(stderr, "Can't create zip without input files specified.");
450  return -1;
451  }
452  }
453 
454  if (create) {
455  // Create a zip
456  return devtools_ijar::create(argv[2], filelist, flatten, verbose, compress);
457  } else {
458  char* exdir = NULL;
459  if (argc > 3 && strcmp(argv[3], "-d") == 0) {
460  exdir = argv[4];
461  }
462 
463  // Extraction / list mode
464  return devtools_ijar::extract(argv[2], exdir, filelist, verbose, extract,
465  flatten);
466  }
