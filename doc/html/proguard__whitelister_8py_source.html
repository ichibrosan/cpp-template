16 """Checks for proguard configuration rules that cannot be combined across libs.
17 
18 The only valid proguard arguments for a library are -keep, -assumenosideeffects,
19 -assumevalues and -dontnote and -dontwarn when they are provided with arguments.
20 Limiting libraries to using these flags prevents drastic, sweeping effects
21 (such as obfuscation being disabled) from being inadvertently applied to a
22 binary through a library dependency.
23 """

68  def _Validate(self, config):
69  """Checks the config for illegal arguments."""
70  config = re.sub(PROGUARD_COMMENTS_PATTERN, '', six.ensure_str(config))
71  args = re.compile('(?:^-|\n-)').split(config)
72 
73  invalid_configs = []
74  for arg in args:
75  arg = arg.strip()
76  if not arg or self._ValidateArg(arg):
77  continue
78  invalid_configs.append('-' + arg.split()[0])
79 
80  return invalid_configs
81 
82  def _ValidateArg(self, arg):
83  if arg.startswith(ProguardConfigValidator._VALID_ARGS):
84  return True
85  elif arg.split()[0] in ['dontnote', 'dontwarn']:
86  if len(arg.split()) > 1:
87  return True
88  return False
