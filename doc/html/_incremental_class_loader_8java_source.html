61  private static DelegateClassLoader createDelegateClassLoader(
62  File codeCacheDir, String nativeLibDir, List<String> dexes, ClassLoader original) {
63  StringBuilder pathBuilder = new StringBuilder();
64  boolean first = true;
65  for (String dex : dexes) {
66  if (first) {
67  first = false;
68  } else {
69  pathBuilder.append(File.pathSeparator);
70  }
71 
72  pathBuilder.append(dex);
73  }
74 
75  Log.v("IncrementalClassLoader", "Incremental dex path is " + pathBuilder);
76  Log.v("IncrementalClassLoader", "Native lib dir is " + nativeLibDir);
77  return new DelegateClassLoader(pathBuilder.toString(), codeCacheDir,
78  nativeLibDir, original);
79  }
