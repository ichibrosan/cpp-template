129 bool WindowsFileMtime::IsUntampered(const Path& path) {
130  if (path.IsEmpty() || path.IsNull()) {
131  return false;
132  }
133 
134  // Get attributes, to check if the file exists. (It may still be a dangling
135  // junction.)
136  DWORD attrs = GetFileAttributesW(path.AsNativePath().c_str());
137  if (attrs == INVALID_FILE_ATTRIBUTES) {
138  return false;
139  }
140 
141  bool is_directory = attrs & FILE_ATTRIBUTE_DIRECTORY;
142  AutoHandle handle(CreateFileW(
143  /* lpFileName */ path.AsNativePath().c_str(),
144  /* dwDesiredAccess */ GENERIC_READ,
145  /* dwShareMode */ FILE_SHARE_READ,
146  /* lpSecurityAttributes */ NULL,
147  /* dwCreationDisposition */ OPEN_EXISTING,
148  /* dwFlagsAndAttributes */
149  // Per CreateFile's documentation on MSDN, opening directories requires
150  // the FILE_FLAG_BACKUP_SEMANTICS flag.
151  is_directory ? FILE_FLAG_BACKUP_SEMANTICS : FILE_ATTRIBUTE_NORMAL,
152  /* hTemplateFile */ NULL));
153 
154  if (!handle.IsValid()) {
155  return false;
156  }
157 
158  if (is_directory) {
159  return true;
160  } else {
161  BY_HANDLE_FILE_INFORMATION info;
162  if (!GetFileInformationByHandle(handle, &info)) {
163  return false;
164  }
165 
166  // Compare the mtime with `near_future_`, not with `GetNow()` or
167  // `distant_future_`.
168  // This way we don't need to call GetNow() every time we want to compare
169  // (and thus convert a SYSTEMTIME to FILETIME), and we also don't need to
170  // worry about potentially unreliable FILETIME equality check (in case it
171  // uses floats or something crazy).
172  return CompareFileTime(&near_future_, &info.ftLastWriteTime) == -1;
173  }
174 }

212 FILETIME WindowsFileMtime::GetFuture(WORD years) {
213  FILETIME result;
214  GetSystemTimeAsFileTime(&result);
215 
216  // 1 year in FILETIME.
217  constexpr ULONGLONG kOneYear = 365ULL * 24 * 60 * 60 * 10'000'000;
218 
219  ULARGE_INTEGER result_value;
220  result_value.LowPart = result.dwLowDateTime;
221  result_value.HighPart = result.dwHighDateTime;
222  result_value.QuadPart += kOneYear * years;
223  result.dwLowDateTime = result_value.LowPart;
224  result.dwHighDateTime = result_value.HighPart;

381 static bool UnlinkPathW(const wstring& path) {
382  DWORD attrs = ::GetFileAttributesW(path.c_str());
383  if (attrs == INVALID_FILE_ATTRIBUTES) {
384  // Path does not exist.
385  return false;
386  }
387  if (attrs & FILE_ATTRIBUTE_DIRECTORY) {
388  if (!(attrs & FILE_ATTRIBUTE_REPARSE_POINT)) {
389  // Path is a directory; unlink(2) also cannot remove directories.
390  return false;
391  }
392  // Otherwise it's a junction, remove using RemoveDirectoryW.
393  return ::RemoveDirectoryW(path.c_str()) == TRUE;
394  } else {
395  // Otherwise it's a file, remove using DeleteFileW.
396  return ::DeleteFileW(path.c_str()) == TRUE;
397  }
398 }

409 static bool RealPath(const WCHAR* path, unique_ptr<WCHAR[]>* result = nullptr) {
410  // Attempt opening the path, which may be anything -- a file, a directory, a
411  // symlink, even a dangling symlink is fine.
412  // Follow reparse points, getting us that much closer to the real path.
413  AutoHandle h(CreateFileW(path, 0, kAllShare, NULL, OPEN_EXISTING,
414  FILE_FLAG_BACKUP_SEMANTICS, NULL));
415  if (!h.IsValid()) {
416  // Path does not exist or it's a dangling junction/symlink.
417  return false;
418  }
419 
420  if (!result) {
421  // The caller is only interested in whether the file exists, they aren't
422  // interested in its real path. Since we just successfully opened the file
423  // we already know it exists.
424  // Also, GetFinalPathNameByHandleW is slow so avoid calling it if we can.
425  return true;
426  }
427 
428  // kMaxPath value: according to MSDN, maximum path length is 32767, and with
429  // an extra null terminator that's exactly 0x8000.
430  static constexpr size_t kMaxPath = 0x8000;
431  std::unique_ptr<WCHAR[]> buf(new WCHAR[kMaxPath]);
432  DWORD res = GetFinalPathNameByHandleW(h, buf.get(), kMaxPath, 0);
433  if (res > 0 && res < kMaxPath) {
434  *result = std::move(buf);
435  return true;
436  } else {
437  return false;
438  }
439 }

462 string MakeCanonical(const char* path) {
463  if (IsDevNull(path)) {
464  return "NUL";
465  }
466  if (path == nullptr || path[0] == 0) {
467  return "";
468  }
469 
470  std::wstring wpath;
471  string error;
472  if (!AsAbsoluteWindowsPath(path, &wpath, &error)) {
473  BAZEL_DIE(blaze_exit_code::LOCAL_ENVIRONMENTAL_ERROR)
474  << "MakeCanonical(" << path
475  << "): AsAbsoluteWindowsPath failed: " << error;
476  }
477 
478  std::unique_ptr<WCHAR[]> long_realpath;
479  if (!RealPath(wpath.c_str(), &long_realpath)) {
480  return "";
481  }
482 
483  // Convert the path to lower-case.
484  size_t size =
485  wcslen(long_realpath.get()) - (HasUncPrefix(long_realpath.get()) ? 4 : 0);
486  unique_ptr<WCHAR[]> lcase_realpath(new WCHAR[size + 1]);
487  const WCHAR* p_from = RemoveUncPrefixMaybe(long_realpath.get());
488  WCHAR* p_to = lcase_realpath.get();
489  while (size-- > 0) {
490  *p_to++ = towlower(*p_from++);
491  }
492  *p_to = 0;
493  return WstringToCstring(lcase_realpath.get());
494 }

528 bool CanAccessDirectory(const Path& path) {
529  DWORD attr = ::GetFileAttributesW(path.AsNativePath().c_str());
530  if ((attr == INVALID_FILE_ATTRIBUTES) || !(attr & FILE_ATTRIBUTE_DIRECTORY)) {
531  // The path doesn't exist or is not a directory.
532  return false;
533  }
534 
535  // The only easy way to know if a directory is writable is by attempting to
536  // open a file for writing in it.
537  Path dummy_path = path.GetRelative("bazel_directory_access_test");
538 
539  // Attempt to open the dummy file for read/write access.
540  // If the file happens to exist, no big deal, we won't overwrite it thanks to
541  // OPEN_ALWAYS.
542  HANDLE handle = ::CreateFileW(
543  /* lpFileName */ dummy_path.AsNativePath().c_str(),
544  /* dwDesiredAccess */ GENERIC_WRITE | GENERIC_READ,
545  /* dwShareMode */ kAllShare,
546  /* lpSecurityAttributes */ NULL,
547  /* dwCreationDisposition */ OPEN_ALWAYS,
548  /* dwFlagsAndAttributes */ FILE_ATTRIBUTE_NORMAL,
549  /* hTemplateFile */ NULL);
550  DWORD err = GetLastError();
551  if (handle == INVALID_HANDLE_VALUE && err != ERROR_ALREADY_EXISTS) {
552  // We couldn't open the file, and not because the dummy file already exists.
553  // Consequently it is because `wpath` doesn't exist.
554  return false;
555  }
556  // The fact that we could open the file, regardless of it existing beforehand
557  // or not, means the directory also exists and we can read/write in it.
558  CloseHandle(handle);
559  if (err != ERROR_ALREADY_EXISTS) {
560  // The file didn't exist before, but due to OPEN_ALWAYS we created it just
561  // now, so do delete it.
562  ::DeleteFileW(dummy_path.AsNativePath().c_str());
563  } // Otherwise the file existed before, leave it alone.
564  return true;
565 }

711 void ForEachDirectoryEntryW(const wstring& path,
712  DirectoryEntryConsumerW* consume) {
713  wstring wpath;
714  if (path.empty() || IsDevNull(path.c_str())) {
715  return;
716  }
717  string error;
718  if (!AsWindowsPath(path, &wpath, &error)) {
719  std::string err = GetLastErrorString();
720  BAZEL_DIE(blaze_exit_code::LOCAL_ENVIRONMENTAL_ERROR)
721  << "ForEachDirectoryEntryW(" << WstringToCstring(path)
722  << "): AsWindowsPath failed: " << err;
723  }
724 
725  static const wstring kUncPrefix(L"\\\\?\\");
726  static const wstring kDot(L".");
727  static const wstring kDotDot(L"..");
728  // Always add an UNC prefix to ensure we can work with long paths.
729  if (!HasUncPrefix(wpath.c_str())) {
730  wpath = kUncPrefix + wpath;
731  }
732  // Unconditionally add a trailing backslash. We know `wpath` has no trailing
733  // backslash because it comes from AsWindowsPath whose output is always
734  // normalized (see NormalizeWindowsPath).
735  wpath.append(L"\\");
736  WIN32_FIND_DATAW metadata;
737  HANDLE handle = ::FindFirstFileW((wpath + L"*").c_str(), &metadata);
738  if (handle == INVALID_HANDLE_VALUE) {
739  return; // directory does not exist or is empty
740  }
741 
742  do {
743  if (kDot != metadata.cFileName && kDotDot != metadata.cFileName) {
744  wstring wname = wpath + metadata.cFileName;
745  wstring name(/* omit prefix */ 4 + wname.c_str());
746  bool is_dir = (metadata.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0;
747  bool is_junc =
748  (metadata.dwFileAttributes & FILE_ATTRIBUTE_REPARSE_POINT) != 0;
749  consume->Consume(name, is_dir && !is_junc);
750  }
751  } while (::FindNextFileW(handle, &metadata));
752  ::FindClose(handle);
753 }
