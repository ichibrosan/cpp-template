94  self.assertEqual(_helpers.GetCallingModule(), sys.argv[0])

102  # We execute the following exec statements for their side-effect
103  # (i.e., not raising an error). They emphasize the case that not
104  # all code resides in one of the imported modules: Python is a
105  # really dynamic language, where we can dynamically construct some
106  # code and execute it.
107  code = ('from gflags import _helpers\n'
108  'module_name = _helpers.GetCallingModule()')
109  exec(code) # pylint: disable=exec-used
110 
111  # Next two exec statements executes code with a global environment
112  # that is different from the global environment of any imported
113  # module.
114  exec(code, {}) # pylint: disable=exec-used
115  # vars(self) returns a dictionary corresponding to the symbol
116  # table of the self object. dict(...) makes a distinct copy of
117  # this dictionary, such that any new symbol definition by the
118  # exec-ed code (e.g., import flags, module_name = ...) does not
119  # affect the symbol table of self.
120  exec(code, dict(vars(self))) # pylint: disable=exec-used
121 
122  # Next test is actually more involved: it checks not only that
123  # GetCallingModule does not crash inside exec code, it also checks
124  # that it returns the expected value: the code executed via exec
125  # code is treated as being executed by the current module. We
126  # check it twice: first time by executing exec from the main
127  # module, second time by executing it from module_bar.
128  global_dict = {}
129  exec(code, global_dict) # pylint: disable=exec-used
130  self.assertEqual(global_dict['module_name'],
131  sys.argv[0])
132 
133  global_dict = {}
134  module_bar.ExecuteCode(code, global_dict)
135  self.assertEqual(
136  global_dict['module_name'],
137  'gflags.flags_modules_for_testing.module_bar')
