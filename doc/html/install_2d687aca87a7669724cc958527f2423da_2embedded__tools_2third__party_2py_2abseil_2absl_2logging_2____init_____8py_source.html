15 """Abseil Python logging module implemented on top of standard logging.
16 
17 Simple usage:
18 
19  from absl import logging
20 
21  logging.info('Interesting Stuff')
22  logging.info('Interesting Stuff with Arguments: %d', 42)
23 
24  logging.set_verbosity(logging.INFO)
25  logging.log(logging.DEBUG, 'This will *not* be printed')
26  logging.set_verbosity(logging.DEBUG)
27  logging.log(logging.DEBUG, 'This will be printed')
28 
29  logging.warn('Worrying Stuff')
30  logging.error('Alarming Stuff')
31  logging.fatal('AAAAHHHHH!!!!') # Process exits.
32 
33 Usage note: Do not pre-format the strings in your program code.
34 Instead, let the logging module perform argument interpolation.
35 This saves cycles because strings that don't need to be printed
36 are never formatted. Note that this module does not attempt to
37 interpolate arguments when no arguments are given. In other words
38 
39  logging.info('Interesting Stuff: %s')
40 
41 does not raise an exception because logging.info() has only one
42 argument, the message string.
43 
44 "Lazy" evaluation for debugging:
45 
46 If you do something like this:
47  logging.debug('Thing: %s', thing.ExpensiveOp())
48 then the ExpensiveOp will be evaluated even if nothing
49 is printed to the log. To avoid this, use the level_debug() function:
50  if logging.level_debug():
51  logging.debug('Thing: %s', thing.ExpensiveOp())
52 
53 Notes on Unicode:
54 
55 The log output is encoded as UTF-8. Don't pass data in other encodings in
56 bytes() instances -- instead pass unicode string instances when you need to
57 (for both the format string and arguments).
58 
59 Note on critical and fatal:
60 Standard logging module defines fatal as an alias to critical, but it's not
61 documented, and it does NOT actually terminate the program.
62 This module only defines fatal but not critical, and it DOES terminate the
63 program.
64 
65 The differences in behavior are historical and unfortunate.
66 """

454 def find_log_dir_and_names(program_name=None, log_dir=None):
455  """Computes the directory and filename prefix for log file.
456 
457  Args:
458  program_name: str|None, the filename part of the path to the program that
459  is running without its extension. e.g: if your program is called
460  'usr/bin/foobar.py' this method should probably be called with
461  program_name='foobar' However, this is just a convention, you can
462  pass in any string you want, and it will be used as part of the
463  log filename. If you don't pass in anything, the default behavior
464  is as described in the example. In python standard logging mode,
465  the program_name will be prepended with py_ if it is the program_name
466  argument is omitted.
467  log_dir: str|None, the desired log directory.
468 
469  Returns:
470  (log_dir, file_prefix, symlink_prefix)
471  """
472  if not program_name:
473  # Strip the extension (foobar.par becomes foobar, and
474  # fubar.py becomes fubar). We do this so that the log
475  # file names are similar to C++ log file names.
476  program_name = os.path.splitext(os.path.basename(sys.argv[0]))[0]
477 
478  # Prepend py_ to files so that python code gets a unique file, and
479  # so that C++ libraries do not try to write to the same log files as us.
480  program_name = 'py_%s' % program_name
481 
482  actual_log_dir = find_log_dir(log_dir=log_dir)
483 
484  username = getpass.getuser()
485  hostname = socket.gethostname()
486  file_prefix = '%s.%s.%s.log' % (program_name, hostname, username)
487 
488  return actual_log_dir, file_prefix, program_name

493 
494  Args:

571  def start_logging_to_file(self, program_name=None, log_dir=None):
572  """Starts logging messages to files instead of standard error."""
573  FLAGS.logtostderr = 0
574 
575  actual_log_dir, file_prefix, symlink_prefix = find_log_dir_and_names(
576  program_name=program_name, log_dir=log_dir)
577 
578  basename = '%s.INFO.%s.%d' % (
579  file_prefix,
580  time.strftime('%Y%m%d-%H%M%S', time.localtime(time.time())),
581  os.getpid())
582  filename = os.path.join(actual_log_dir, basename)
583 
584  if six.PY2:
585  self.stream = open(filename, 'a')
586  else:
587  self.stream = open(filename, 'a', encoding='utf-8')
588 
589  # os.symlink is not available on Windows Python 2.
590  if getattr(os, 'symlink', None):
591  # Create a symlink to the log file with a canonical name.
592  symlink = os.path.join(actual_log_dir, symlink_prefix + '.INFO')
593  try:
594  if os.path.islink(symlink):
595  os.unlink(symlink)
596  os.symlink(os.path.basename(filename), symlink)
597  except EnvironmentError:
598  # If it fails, we're sad but it's no error. Commonly, this
599  # fails because the symlink was created by another user and so
600  # we can't modify it
601  pass

621  def _log_to_stderr(self, record):
622  """Emits the record to stderr.

639  def emit(self, record):
640  """Prints a record out to some streams.
641 
642  If FLAGS.logtostderr is set, it will print to sys.stderr ONLY.
643  If FLAGS.alsologtostderr is set, it will print to sys.stderr.
644  If FLAGS.logtostderr is not set, it will log to the stream
645  associated with the current thread.
646 
647  Args:
648  record: logging.LogRecord, the record to emit.
649  """
650  # People occasionally call logging functions at import time before
651  # our flags may have even been defined yet, let alone even parsed, as we
652  # rely on the C++ side to define some flags for us and app init to
653  # deal with parsing. Match the C++ library behavior of notify and emit
654  # such messages to stderr. It encourages people to clean-up and does
655  # not hide the message.
656  level = record.levelno
657  if not FLAGS.is_parsed(): # Also implies "before flag has been defined".
658  global _warn_preinit_stderr
659  if _warn_preinit_stderr:
660  sys.stderr.write(
661  'WARNING: Logging before flag parsing goes to stderr.\n')
662  _warn_preinit_stderr = False
663  self._log_to_stderr(record)
664  elif FLAGS['logtostderr'].value:
665  self._log_to_stderr(record)
666  else:
667  super(PythonHandler, self).emit(record)

779

788  the fourth item returned instead of None.
