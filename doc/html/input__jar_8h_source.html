31 /*
32  * An input jar. The usage pattern is:
33  * InputJar input_jar;
34  * if (!input_jar.Open("path/to/file")) { fail...}
35  * CDH *dir_entry;
36  * LH *local_header;
37  * while (dir_entry = input_jar.NextEntry(&local_header)) {
38  * // process entry.
39  * }
40  * input_jar.Close(); // actually, called by destructor, too.
41  */

53  // Opens the file, memory maps it and locates Central Directory.
54  bool Open(const std::string& path);
55 
56  // Returns the next Central Directory Header or NULL.
57  const CDH *NextEntry(const LH **local_header_ptr) {
58  if (path_.empty()) {
59  diag_errx(1, "%s:%d: call Open() first!", __FILE__, __LINE__);
60  }
61  if (!cdh_->is()) {
62  return nullptr;
63  }
64  const CDH *current_cdh = cdh_;
65  const uint8_t *new_cdr = ziph::byte_ptr(cdh_) + cdh_->size();
66  if (!mapped_file_.mapped(new_cdr)) {
67  diag_errx(
68  1,
69  "Bad directory record at offset 0x%" PRIx64 " of %s\n"
70  "file name length = %u, extra_field length = %u, comment length = %u",
71  CentralDirectoryRecordOffset(cdh_), path_.c_str(),
72  cdh_->file_name_length(), cdh_->extra_fields_length(),
73  cdh_->comment_length());
74  }
75  cdh_ = reinterpret_cast<const CDH *>(new_cdr);
76  *local_header_ptr = LocalHeader(current_cdh);
77  return current_cdh;
78  }
79 
80  // Closes the file.
81  bool Close();
