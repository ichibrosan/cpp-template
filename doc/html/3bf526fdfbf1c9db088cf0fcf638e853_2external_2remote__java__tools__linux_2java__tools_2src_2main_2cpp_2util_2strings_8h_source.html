43 // Matches a prefix (which must be a char* literal!) against the beginning of
44 // str. Returns a pointer past the prefix, or NULL if the prefix wasn't matched.
45 // (Like the standard strcasecmp(), but for efficiency doesn't call strlen() on
46 // prefix, and returns a pointer rather than an int.)
47 //
48 // The ""'s catch people who don't pass in a literal for "prefix"
49 #ifndef strprefix
50 #define strprefix(str, prefix) \
51  (strncmp(str, prefix, sizeof("" prefix "") - 1) == 0 \
52  ? str + sizeof(prefix) - 1 \
53  : NULL)
54 #endif

56 // Matches a prefix; returns a pointer past the prefix, or NULL if not found.
57 // (Like strprefix() and strcaseprefix() but not restricted to searching for
58 // char* literals). Templated so searching a const char* returns a const char*,
59 // and searching a non-const char* returns a non-const char*.
60 // Matches a prefix; returns a pointer past the prefix, or NULL if not found.
61 // (Like strprefix() and strcaseprefix() but not restricted to searching for
62 // char* literals). Templated so searching a const char* returns a const char*,
63 // and searching a non-const char* returns a non-const char*.
64 template <class CharStar>
65 inline CharStar var_strprefix(CharStar str, const char *prefix) {
66  const int len = strlen(prefix);
67  return strncmp(str, prefix, len) == 0 ? str + len : NULL;
68 }
