50 def ExtractEmbeddedJars(aar,
51  singlejar_param_file,

54  if not output_dir_orig:

56  jar_pattern = re.compile("^(classes|libs/.+)\\.jar$")
57  singlejar_param_file.write(b"--exclude_build_data\n")
58  for name in aar.namelist():

60  singlejar_param_file.write(b"--sources\n")
