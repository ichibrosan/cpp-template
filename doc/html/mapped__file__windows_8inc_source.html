41 bool MappedFile::Open(const std::string& path) {
42  if (is_open()) {
43  diag_errx(1, "%s:%d: This instance is already open", __FILE__, __LINE__);
44  }
45 
46  std::wstring wpath;
47  std::string error;
48  if (!blaze_util::AsAbsoluteWindowsPath(path, &wpath, &error)) {
49  diag_warn("%s:%d: AsAbsoluteWindowsPath failed: %s", __FILE__, __LINE__,
50  error.c_str());
51  return false;
52  }
53 
54  hFile_ = CreateFileW(wpath.c_str(), GENERIC_READ, FILE_SHARE_READ, NULL,
55  OPEN_EXISTING, 0, NULL);
56  if (hFile_ == INVALID_HANDLE_VALUE) {
57  diag_warn("%s:%d: CreateFileW failed for %S", __FILE__, __LINE__,
58  wpath.c_str());
59  return false;
60  }
61 
62  LARGE_INTEGER temp;
63  ::GetFileSizeEx(hFile_, &temp);
64  size_t fileSize = temp.QuadPart;
65 
66  if (fileSize == 0) {
67  // Handle empty files specially, because CreateFileMapping cannot map them:
68  //
69  // From
70  // https://docs.microsoft.com/en-us/windows/desktop/api/winbase/nf-winbase-createfilemappinga
71  //
72  // An attempt to map a file with a length of 0 (zero) fails with an error
73  // code of ERROR_FILE_INVALID. Applications should test for files with a
74  // length of 0 (zero) and reject those files.
75  mapped_start_ = nullptr;
76  mapped_end_ = nullptr;
77  hMapFile_ = INVALID_HANDLE_VALUE;
78  return true;
79  }
80 
81  hMapFile_ = ::CreateFileMapping(
82  hFile_,
83  nullptr, // default security
84  PAGE_READONLY, // read-only permission
85  static_cast<DWORD>(fileSize >> 32), // size of mapping object, high
86  static_cast<DWORD>(fileSize), // size of mapping object, low
87  nullptr); // name of mapping object
88 
89  if (hMapFile_ == nullptr) {
90  diag_warn("%s:%d: CreateFileMapping for %s failed", __FILE__, __LINE__,
91  path.c_str());
92  ::CloseHandle(hFile_);
93  hFile_ = INVALID_HANDLE_VALUE;
94  return false;
95  }
96 
97  mapped_start_ = static_cast<unsigned char*>(
98  MapViewOfFile(hMapFile_,
99  FILE_MAP_READ | FILE_MAP_COPY, // PROT_READ | MAP_PRIVATE
100  0, // file offset, high
101  0, // file offset, low
102  fileSize)); // file size
103  if (mapped_start_ == nullptr) {
104  diag_warn("%s:%d: MapViewOfFile for %s failed", __FILE__, __LINE__,
105  path.c_str());
106  ::CloseHandle(hMapFile_);
107  ::CloseHandle(hFile_);
108  hFile_ = INVALID_HANDLE_VALUE;
109  hMapFile_ = INVALID_HANDLE_VALUE;
110  return false;
111  }
112 
113  mapped_end_ = mapped_start_ + fileSize;
114  return true;
115 }

117 void MappedFile::Close() {
118  if (is_open()) {
119  if (mapped_start_) {
120  ::UnmapViewOfFile(mapped_start_);
121  }
122  if (hMapFile_ != INVALID_HANDLE_VALUE) {
123  ::CloseHandle(hMapFile_);
124  hMapFile_ = INVALID_HANDLE_VALUE;
125  }
126  ::CloseHandle(hFile_);
127  hFile_ = INVALID_HANDLE_VALUE;
128  mapped_start_ = mapped_end_ = nullptr;
129  }
130 }
