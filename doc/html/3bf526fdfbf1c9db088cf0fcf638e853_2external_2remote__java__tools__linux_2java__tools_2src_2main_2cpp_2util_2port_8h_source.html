74 // Linux I/O priorities support is available only in later versions of glibc.
75 // Therefore, we include some of the needed definitions here. May need to
76 // be removed once we switch to a new version of glibc
77 // (As of 10/24/08 it is unclear when glibc support will become available.)
78 enum IOPriorityClass {
79  // No I/O priority value has yet been set. The kernel may assign I/O
80  // priority based on the process nice value.
81  IOPRIO_CLASS_NONE,
82 
83  // Real-time, highest priority. Given first access to the disk at
84  // every opportunity. Use with care: one such process can STARVE
85  // THE ENTIRE SYSTEM. Has 8 priority levels (0-7).
86  IOPRIO_CLASS_RT,
87 
88  // Best-effort, default for any process. Has 8 priority levels (0-7).
89  IOPRIO_CLASS_BE,
90 
91  // Idle, lowest priority. Processes running at this level only get
92  // I/O time when no one else needs the disk, and MAY BECOME
93  // STARVED if higher priority processes are constantly accessing
94  // the disk. With the "anticipatory" I/O scheduler, mapped to
95  // IOPRIO_CLASS_BE, level 3.
96  IOPRIO_CLASS_IDLE,
97 };
