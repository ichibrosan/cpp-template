135  protected void writeEntry(JarOutputStream out, String name, byte[] content) throws IOException {
136  if (names.add(name)) {
137  // Create a new entry
138  JarEntry entry = new JarEntry(name);
139  entry.setTime(newEntryTimeMillis(name));
140  int size = content.length;
141  entry.setSize(size);
142  if (size == 0) {
143  entry.setMethod(JarEntry.STORED);
144  entry.setCrc(0);
145  out.putNextEntry(entry);
146  } else {
147  entry.setMethod(storageMethod);
148  if (storageMethod == JarEntry.STORED) {
149  CRC32 crc = new CRC32();
150  crc.update(content);
151  entry.setCrc(crc.getValue());
152  }
153  out.putNextEntry(entry);
154  out.write(content);
155  }
156  out.closeEntry();
157  }
158  }

167  protected void writeManifestEntry(JarOutputStream out, byte[] content) throws IOException {
168  int oldStorageMethod = storageMethod;
169  // Do not compress small manifest files, the compressed one is frequently
170  // larger than the original. The threshold of 256 bytes is somewhat arbitrary.
171  if (content.length < 256) {
172  storageMethod = JarEntry.STORED;
173  }
174  try {
175  writeEntry(out, MANIFEST_DIR, new byte[] {});
176  writeEntry(out, MANIFEST_NAME, content);
177  } finally {
178  storageMethod = oldStorageMethod;
179  }
180  }

186  protected void copyEntry(JarOutputStream out, String name, Path path) throws IOException {
187  if (!names.contains(name)) {
188  if (!Files.exists(path)) {
189  throw new FileNotFoundException(path.toAbsolutePath() + " (No such file or directory)");
190  }
191  boolean isDirectory = Files.isDirectory(path);
192  if (isDirectory && !name.endsWith("/")) {
193  name = name + '/'; // always normalize directory names before checking set
194  }
195  if (names.add(name)) {
196  if (verbose) {
197  System.err.println("adding " + path);
198  }
199  // Create a new entry
200  long size = isDirectory ? 0 : Files.size(path);
201  JarEntry outEntry = new JarEntry(name);
202  long newtime =
203  normalize ? normalizedTimestamp(name) : Files.getLastModifiedTime(path).toMillis();
204  outEntry.setTime(newtime);
205  outEntry.setSize(size);
206  if (size == 0L) {
207  outEntry.setMethod(JarEntry.STORED);
208  outEntry.setCrc(0);
209  out.putNextEntry(outEntry);
210  } else {
211  outEntry.setMethod(storageMethod);
212  if (storageMethod == JarEntry.STORED) {
213  // ZipFile requires us to calculate the CRC-32 for any STORED entry.
214  // It would be nicer to do this via DigestInputStream, but
215  // the architecture of ZipOutputStream requires us to know the CRC-32
216  // before we write the data to the stream.
217  byte[] bytes = Files.readAllBytes(path);
218  CRC32 crc = new CRC32();
219  crc.update(bytes);
220  outEntry.setCrc(crc.getValue());
221  out.putNextEntry(outEntry);
222  out.write(bytes);
223  } else {
224  out.putNextEntry(outEntry);
225  Files.copy(path, out);
226  }
227  }
228  out.closeEntry();
229  }
230  }
