44 static wstring GetRunfilesDir(const wchar_t* argv0) {
45  wstring runfiles_dir;
46  // If RUNFILES_DIR is already set (probably we are either in a test or in a
47  // data dependency) then use it.
48  if (GetEnv(L"RUNFILES_DIR", &runfiles_dir)) {
49  return runfiles_dir;
50  }
51  // Otherwise this is probably a top-level non-test binary (e.g. a genrule
52  // tool) and should look for its runfiles beside the executable.
53  return GetBinaryPathWithExtension(argv0) + L".runfiles";
54 }

56 BinaryLauncherBase::BinaryLauncherBase(
57  const LaunchDataParser::LaunchInfo& _launch_info, int argc, wchar_t* argv[])
58  : launch_info(_launch_info),

72  }

75 static bool FindManifestFileImpl(const wchar_t* argv0, wstring* result) {
76  // If this binary X runs as the data-dependency of some other binary Y, then
77  // X has no runfiles manifest/directory and should use Y's.
78  if (GetEnv(L"RUNFILES_MANIFEST_FILE", result) &&
79  DoesFilePathExist(result->c_str())) {
80  return true;
81  }
82 
83  wstring directory;
84  if (GetEnv(L"RUNFILES_DIR", &directory)) {
85  *result = directory + L"/MANIFEST";
86  if (DoesFilePathExist(result->c_str())) {
87  return true;
88  }
89  }
90 
91  // If this binary X runs by itself (not as a data-dependency of another
92  // binary), then look for the manifest in a runfiles directory next to the
93  // main binary, then look for it (the manifest) next to the main binary.
94  directory = GetBinaryPathWithExtension(argv0) + L".runfiles";
95  *result = directory + L"/MANIFEST";
96  if (DoesFilePathExist(result->c_str())) {
97  return true;
98  }
99 
100  *result = directory + L"_manifest";
101  if (DoesFilePathExist(result->c_str())) {
102  return true;
103  }
104 
105  return false;
106 }

108 wstring BinaryLauncherBase::FindManifestFile(const wchar_t* argv0) {
109  wstring manifest_file;
110  if (!FindManifestFileImpl(argv0, &manifest_file)) {
111  return L"";
112  }
113  // The path will be set as the RUNFILES_MANIFEST_FILE envvar and used by the
114  // shell script, so let's convert backslashes to forward slashes.
115  std::replace(manifest_file.begin(), manifest_file.end(), '\\', '/');
116  return manifest_file;
117 }

126 void BinaryLauncherBase::ParseManifestFile(ManifestFileMap* manifest_file_map,
127  const wstring& manifest_path) {
128  ifstream manifest_file(AsAbsoluteWindowsPath(manifest_path.c_str()).c_str());
129 
130  if (!manifest_file) {
131  die(L"Couldn't open MANIFEST file: %s", manifest_path.c_str());
132  }
133 
134  string line;
135  while (getline(manifest_file, line)) {
136  size_t space_pos = line.find_first_of(' ');
137  if (space_pos == string::npos) {
138  die(L"Wrong MANIFEST format at line: %s", line.c_str());
139  }
140  wstring wline = blaze_util::CstringToWstring(line);
141  wstring key = wline.substr(0, space_pos);
142  wstring value = wline.substr(space_pos + 1);
143  manifest_file_map->insert(make_pair(key, value));
144  }
145 }

147 wstring BinaryLauncherBase::Rlocation(wstring path,
148  bool has_workspace_name) const {
149  // No need to do rlocation if the path is absolute.
150  if (blaze_util::IsAbsolute(path)) {
151  return path;
152  }
153 
154  if (path.find(L"external/") == 0) {
155  // Ignore 'has_workspace_name' when the path is under "external/". Such
156  // paths already have a workspace name in the next path component.
157  path = path.substr(9);
158  } else if (!has_workspace_name) {
159  path = this->workspace_name + L"/" + path;
160  }
161 
162  // If the manifest file map is empty, then we're using the runfiles directory
163  // instead.
164  if (manifest_file_map.empty()) {
165  return runfiles_dir + L"/" + path;
166  }
167 
168  auto entry = manifest_file_map.find(path);
169  if (entry == manifest_file_map.end()) {
170  die(L"Rlocation failed on %s, path doesn't exist in MANIFEST file",
171  path.c_str());
172  }
173  return entry->second;
174 }

188 void BinaryLauncherBase::CreateCommandLine(
189  CmdLine* result, const wstring& executable,
190  const vector<wstring>& arguments) const {
191  wostringstream cmdline;
192  cmdline << L'\"' << executable << L'\"';
193  for (const auto& s : arguments) {
194  cmdline << L' ' << s;
195  }
196 
197  wstring cmdline_str = cmdline.str();
198  if (cmdline_str.size() >= MAX_CMDLINE_LENGTH) {
199  die(L"Command line too long: %s", cmdline_str.c_str());
200  }
201 
202  // Copy command line into a mutable buffer.
203  // CreateProcess is allowed to mutate its command line argument.
204  wcsncpy(result->cmdline, cmdline_str.c_str(), MAX_CMDLINE_LENGTH - 1);
205  result->cmdline[MAX_CMDLINE_LENGTH - 1] = 0;
206 }

223 ExitCode BinaryLauncherBase::LaunchProcess(const wstring& executable,
224  const vector<wstring>& arguments,
225  bool suppressOutput) const {
226  if (PrintLauncherCommandLine(executable, arguments)) {
227  return 0;
228  }
229  // Set RUNFILES_DIR if:
230  // 1. Symlink runfiles tree is enabled, or
231  // 2. We couldn't find manifest file (which probably means we are running
232  // remotely).
233  // Otherwise, set RUNFILES_MANIFEST_ONLY and RUNFILES_MANIFEST_FILE
234  if (symlink_runfiles_enabled || manifest_file.empty()) {
235  SetEnv(L"RUNFILES_DIR", runfiles_dir);
236  } else {
237  SetEnv(L"RUNFILES_MANIFEST_ONLY", L"1");
238  SetEnv(L"RUNFILES_MANIFEST_FILE", manifest_file);
239  }
240  CmdLine cmdline;
241  CreateCommandLine(&cmdline, executable, arguments);
242  PROCESS_INFORMATION processInfo = {0};
243  STARTUPINFOW startupInfo = {0};
244  startupInfo.cb = sizeof(startupInfo);
245  BOOL ok = CreateProcessW(
246  /* lpApplicationName */ NULL,
247  /* lpCommandLine */ cmdline.cmdline,
248  /* lpProcessAttributes */ NULL,
249  /* lpThreadAttributes */ NULL,
250  /* bInheritHandles */ FALSE,
251  /* dwCreationFlags */
252  suppressOutput ? CREATE_NO_WINDOW // no console window => no output
253  : 0,
254  /* lpEnvironment */ NULL,
255  /* lpCurrentDirectory */ NULL,
256  /* lpStartupInfo */ &startupInfo,
257  /* lpProcessInformation */ &processInfo);
258  if (!ok) {
259  PrintError(L"Cannot launch process: %s\nReason: %hs", cmdline.cmdline,
260  GetLastErrorString().c_str());
261  return GetLastError();
262  }
263  WaitForSingleObject(processInfo.hProcess, INFINITE);
264  ExitCode exit_code;
265  GetExitCodeProcess(processInfo.hProcess,
266  reinterpret_cast<LPDWORD>(&exit_code));
267  CloseHandle(processInfo.hProcess);
268  CloseHandle(processInfo.hThread);
269  return exit_code;
270 }
