56  class SimpleArFileEntry(object):
57  """Represent one entry in a AR archive.
58 
59  Attributes:
60  filename: the filename of the entry, as described in the archive.
61  timestamp: the timestamp of the file entry.
62  owner_id, group_id: numeric id of the user and group owning the file.
63  mode: unix permission mode of the file
64  size: size of the file
65  data: the content of the file.
66  """
67 
68  def __init__(self, f):
69  self.filename = f.read(16).decode('utf-8').strip()
70  if self.filename.endswith('/'): # SysV variant
71  self.filename = self.filename[:-1]
72  self.timestamp = int(f.read(12).strip())
73  self.owner_id = int(f.read(6).strip())
74  self.group_id = int(f.read(6).strip())
75  self.mode = int(f.read(8).strip(), 8)
76  self.size = int(f.read(10).strip())
77  pad = f.read(2)
78  if pad != b'\x60\x0a':
79  raise SimpleArFile.ArError('Invalid AR file header')
80  self.data = f.read(self.size)
81 
82  MAGIC_STRING = b'!<arch>\n'

87  def __enter__(self):

89  if self.f.read(len(self.MAGIC_STRING)) != self.MAGIC_STRING:
90  raise self.ArError('Not a ar file: ' + six.ensure_str(self.filename))

96  def next(self):
97  """Read the next file. Returns None when reaching the end of file."""
98  # AR sections are two bit aligned using new lines.
99  if self.f.tell() % 2 != 0:
100  self.f.read(1)
101  # An AR sections is at least 60 bytes. Some file might contains garbage
102  # bytes at the end of the archive, ignore them.
103  if self.f.tell() > os.fstat(self.f.fileno()).st_size - 60:
104  return None
105  return self.SimpleArFileEntry(self.f)

161  def add_dir(self,
162  name,
163  path,
164  uid=0,
165  gid=0,
166  uname='',
167  gname='',
168  mtime=None,
169  mode=None,
170  depth=100):
171  """Recursively add a directory.
172 
173  Args:
174  name: the destination path of the directory to add.
175  path: the path of the directory to add.
176  uid: owner user identifier.
177  gid: owner group identifier.
178  uname: owner user names.
179  gname: owner group names.
180  mtime: modification time to put in the archive.
181  mode: unix permission mode of the file, default 0644 (0755).
182  depth: maximum depth to recurse in to avoid infinite loops
183  with cyclic mounts.
184 
185  Raises:
186  TarFileWriter.Error: when the recursion depth has exceeded the
187  `depth` argument.
188  """
189  if not (name == self.root_directory or six.ensure_str(name).startswith('/')
190  or name.startswith(self.root_directory + '/')):
191  name = os.path.join(self.root_directory, name)
192  if mtime is None:
193  mtime = self.default_mtime
194  if os.path.isdir(path):
195  # Remove trailing '/' (index -1 => last character)
196  if name[-1] == '/':
197  name = name[:-1]
198  # Add the x bit to directories to prevent non-traversable directories.
199  # The x bit is set only to if the read bit is set.
200  dirmode = (mode | ((0o444 & mode) >> 2)) if mode else mode
201  self.add_file(
202  six.ensure_str(name) + '/',
203  tarfile.DIRTYPE,
204  uid=uid,
205  gid=gid,
206  uname=uname,
207  gname=gname,
208  mtime=mtime,
209  mode=dirmode)
210  if depth <= 0:
211  raise self.Error('Recursion depth exceeded, probably in '
212  'an infinite directory loop.')
213  # Iterate over the sorted list of file so we get a deterministic result.
214  filelist = os.listdir(path)
215  filelist.sort()
216  for f in filelist:
217  new_name = os.path.join(name, f)
218  new_path = os.path.join(path, f)
219  self.add_dir(new_name, new_path, uid, gid, uname, gname, mtime, mode,
220  depth - 1)
221  else:
222  self.add_file(name,
223  tarfile.REGTYPE,
224  file_content=path,
225  uid=uid,
226  gid=gid,
227  uname=uname,
228  gname=gname,
229  mtime=mtime,
230  mode=mode)

232  def _addfile(self, info, fileobj=None):
233  """Add a file in the tar file if there is no conflict."""
234  if not six.ensure_str(
235  info.name).endswith('/') and info.type == tarfile.DIRTYPE:
236  # Enforce the ending / for directories so we correctly deduplicate.
237  info.name += '/'
238  if info.name not in self.members:
239  self.tar.addfile(info, fileobj)
240  self.members.add(info.name)
241  elif info.type != tarfile.DIRTYPE:
242  print(('Duplicate file in archive: %s, '
243  'picking first occurrence' % info.name))

245  def add_file(self,
246  name,
247  kind=tarfile.REGTYPE,
248  content=None,
249  link=None,
250  file_content=None,
251  uid=0,
252  gid=0,
253  uname='',
254  gname='',
255  mtime=None,
256  mode=None):
257  """Add a file to the current tar.
258 
259  Args:
260  name: the name of the file to add.
261  kind: the type of the file to add, see tarfile.*TYPE.
262  content: a textual content to put in the file.
263  link: if the file is a link, the destination of the link.
264  file_content: file to read the content from. Provide either this
265  one or `content` to specifies a content for the file.
266  uid: owner user identifier.
267  gid: owner group identifier.
268  uname: owner user names.
269  gname: owner group names.
270  mtime: modification time to put in the archive.
271  mode: unix permission mode of the file, default 0644 (0755).
272  """
273  if file_content and os.path.isdir(file_content):
274  # Recurse into directory
275  self.add_dir(name, file_content, uid, gid, uname, gname, mtime, mode)
276  return
277  if not (name == self.root_directory or name.startswith('/') or
278  name.startswith(self.root_directory + '/')):
279  name = os.path.join(self.root_directory, name)
280  if kind == tarfile.DIRTYPE:
281  name = name.rstrip('/')
282  if name in self.directories:
283  return
284  if mtime is None:
285  mtime = self.default_mtime
286 
287  components = name.rsplit('/', 1)
288  if len(components) > 1:
289  d = components[0]
290  self.add_file(d,
291  tarfile.DIRTYPE,
292  uid=uid,
293  gid=gid,
294  uname=uname,
295  gname=gname,
296  mtime=mtime,
297  mode=0o755)
298  tarinfo = tarfile.TarInfo(name)
299  tarinfo.mtime = mtime
300  tarinfo.uid = uid
301  tarinfo.gid = gid
302  tarinfo.uname = uname
303  tarinfo.gname = gname
304  tarinfo.type = kind
305  if mode is None:
306  tarinfo.mode = 0o644 if kind == tarfile.REGTYPE else 0o755
307  else:
308  tarinfo.mode = mode
309  if link:
310  tarinfo.linkname = link
311  if content:
312  content_bytes = six.ensure_binary(content, 'utf-8')
313  tarinfo.size = len(content_bytes)
314  self._addfile(tarinfo, io.BytesIO(content_bytes))
315  elif file_content:
316  with open(file_content, 'rb') as f:
317  tarinfo.size = os.fstat(f.fileno()).st_size
318  self._addfile(tarinfo, f)
319  else:
320  if kind == tarfile.DIRTYPE:
321  self.directories.add(name)
322  self._addfile(tarinfo)

324  def add_tar(self,
325  tar,
326  rootuid=None,
327  rootgid=None,
328  numeric=False,
329  name_filter=None,
330  root=None):
331  """Merge a tar content into the current tar, stripping timestamp.
332 
333  Args:
334  tar: the name of tar to extract and put content into the current tar.
335  rootuid: user id that we will pretend is root (replaced by uid 0).
336  rootgid: group id that we will pretend is root (replaced by gid 0).
337  numeric: set to true to strip out name of owners (and just use the
338  numeric values).
339  name_filter: filter out file by names. If not none, this method will be
340  called for each file to add, given the name and should return true if
341  the file is to be added to the final tar and false otherwise.
342  root: place all non-absolute content under given root directory, if not
343  None.
344 
345  Raises:
346  TarFileWriter.Error: if an error happens when uncompressing the tar file.
347  """
348  if root and root[0] not in ['/', '.']:
349  # Root prefix should start with a '/', adds it if missing
350  root = '/' + six.ensure_str(root)
351  compression = os.path.splitext(tar)[-1][1:]
352  if compression == 'tgz':
353  compression = 'gz'
354  elif compression == 'bzip2':
355  compression = 'bz2'
356  elif compression == 'lzma':
357  compression = 'xz'
358  elif compression not in ['gz', 'bz2', 'xz']:
359  compression = ''
360  if compression == 'xz':
361  # Python 2 does not support lzma, our py3 support is terrible so let's
362  # just hack around.
363  # Note that we buffer the file in memory and it can have an important
364  # memory footprint but it's probably fine as we don't use them for really
365  # large files.
366  # TODO(dmarting): once our py3 support gets better, compile this tools
367  # with py3 for proper lzma support.
368  if subprocess.call('which xzcat', shell=True, stdout=subprocess.PIPE):
369  raise self.Error('Cannot handle .xz and .lzma compression: '
370  'xzcat not found.')
371  p = subprocess.Popen('cat %s | xzcat' % tar,
372  shell=True,
373  stdout=subprocess.PIPE)
374  f = io.BytesIO(p.stdout.read())
375  p.wait()
376  intar = tarfile.open(fileobj=f, mode='r:')
377  else:
378  if compression in ['gz', 'bz2']:
379  # prevent performance issues due to accidentally-introduced seeks
380  # during intar traversal by opening in "streaming" mode. gz, bz2
381  # are supported natively by python 2.7 and 3.x
382  inmode = 'r|' + six.ensure_str(compression)
383  else:
384  inmode = 'r:' + six.ensure_str(compression)
385  intar = tarfile.open(name=tar, mode=inmode)
386  for tarinfo in intar:
387  if name_filter is None or name_filter(tarinfo.name):
388  tarinfo.mtime = self.default_mtime
389  if rootuid is not None and tarinfo.uid == rootuid:
390  tarinfo.uid = 0
391  tarinfo.uname = 'root'
392  if rootgid is not None and tarinfo.gid == rootgid:
393  tarinfo.gid = 0
394  tarinfo.gname = 'root'
395  if numeric:
396  tarinfo.uname = ''
397  tarinfo.gname = ''
398 
399  name = tarinfo.name
400  if (not name.startswith('/') and
401  not name.startswith(self.root_directory)):
402  name = os.path.join(self.root_directory, name)
403  if root is not None:
404  if name.startswith('.'):
405  name = '.' + root + name.lstrip('.')
406  # Add root dir with same permissions if missing. Note that
407  # add_file deduplicates directories and is safe to call here.
408  self.add_file('.' + root,
409  tarfile.DIRTYPE,
410  uid=tarinfo.uid,
411  gid=tarinfo.gid,
412  uname=tarinfo.uname,
413  gname=tarinfo.gname,
414  mtime=tarinfo.mtime,
415  mode=0o755)
416  # Relocate internal hardlinks as well to avoid breaking them.
417  link = tarinfo.linkname
418  if link.startswith('.') and tarinfo.type == tarfile.LNKTYPE:
419  tarinfo.linkname = '.' + root + link.lstrip('.')
420  tarinfo.name = name
421 
422  if tarinfo.isfile():
423  # use extractfile(tarinfo) instead of tarinfo.name to preserve
424  # seek position in intar
425  self._addfile(tarinfo, intar.extractfile(tarinfo))
426  else:
427  self._addfile(tarinfo)
428  intar.close()
