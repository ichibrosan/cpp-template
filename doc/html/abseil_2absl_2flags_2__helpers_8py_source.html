108 def get_calling_module_object_and_name():
109  """Returns the module that's calling into this module.
110 
111  We generally use this function to get the name of the module calling a
112  DEFINE_foo... function.
113 
114  Returns:
115  The module object that called into this one.
116 
117  Raises:
118  AssertionError: Raised when no calling module could be identified.
119  """

121  # sys._getframe is the right thing to use here, as it's the best
122  # way to walk up the call stack.

126  return _ModuleObjectAndName(module, module_name)

135 def str_or_unicode(value):
136  """Converts a value to a python string.
137 
138  Behavior of this function is intentionally different in Python2/3.
139 
140  In Python2, the given value is attempted to convert to a str (byte string).
141  If it contains non-ASCII characters, it is converted to a unicode instead.
142 
143  In Python3, the given value is always converted to a str (unicode string).
144 
145  This behavior reflects the (bad) practice in Python2 to try to represent
146  a string as str as long as it contains ASCII characters only.
147 
148  Args:
149  value: An object to be converted to a string.
150 
151  Returns:
152  A string representation of the given value. See the description above
153  for its type.
154  """
155  try:
156  return str(value)
157  except UnicodeEncodeError:
158  return unicode(value) # Python3 should never come here

161 def create_xml_dom_element(doc, name, value):
162  """Returns an XML DOM element with name and text value.
163 
164  Args:
165  doc: minidom.Document, the DOM document it should create nodes from.
166  name: str, the tag of XML element.
167  value: object, whose string representation will be used
168  as the value of the XML element. Illegal or highly discouraged xml 1.0
169  characters are stripped.
170 
171  Returns:
172  An instance of minidom.Element.
173  """
174  s = str_or_unicode(value)
175  if six.PY2 and not isinstance(s, unicode):
176  # Get a valid unicode string.
177  s = s.decode('utf-8', 'ignore')
178  if isinstance(value, bool):
179  # Display boolean values as the C++ flag library does: no caps.
180  s = s.lower()
181  # Remove illegal xml characters.
182  s = _ILLEGAL_XML_CHARS_REGEX.sub(u'', s)

185  e.appendChild(doc.createTextNode(s))
186  return e

189 def get_help_width():
190  """Returns the integer width of help lines that is used in TextWrap."""
191  if not sys.stdout.isatty() or termios is None or fcntl is None:
192  return _DEFAULT_HELP_WIDTH
193  try:
194  data = fcntl.ioctl(sys.stdout, termios.TIOCGWINSZ, '1234')
195  columns = struct.unpack('hh', data)[1]
196  # Emacs mode returns 0.
197  # Here we assume that any value below 40 is unreasonable.
198  if columns >= _MIN_HELP_WIDTH:
199  return columns
200  # Returning an int as default is fine, int(int) just return the int.
201  return int(os.getenv('COLUMNS', _DEFAULT_HELP_WIDTH))
202 
203  except (TypeError, IOError, struct.error):
204  return _DEFAULT_HELP_WIDTH

207 def get_flag_suggestions(attempt, longopt_list):
208  """Returns helpful similar matches for an invalid flag."""
209  # Don't suggest on very short strings, or if no longopts are specified.
210  if len(attempt) <= 2 or not longopt_list:
211  return []
212 
213  option_names = [v.split('=')[0] for v in longopt_list]
214 
215  # Find close approximations in flag prefixes.
216  # This also handles the case where the flag is spelled right but ambiguous.
217  distances = [(_damerau_levenshtein(attempt, option[0:len(attempt)]), option)
218  for option in option_names]
219  distances.sort(key=lambda t: t[0])
220 
221  least_errors, _ = distances[0]
222  # Don't suggest excessively bad matches.
223  if least_errors >= _SUGGESTION_ERROR_RATE_THRESHOLD * len(attempt):
224  return []
225 
226  suggestions = []
227  for errors, name in distances:
228  if errors == least_errors:
229  suggestions.append(name)
230  else:
231  break
232  return suggestions

235 def _damerau_levenshtein(a, b):
236  """Returns Damerau-Levenshtein edit distance from a to b."""
237  memo = {}
238 
239  def distance(x, y):
240  """Recursively defined string distance with memoization."""
241  if (x, y) in memo:
242  return memo[x, y]
243  if not x:
244  d = len(y)
245  elif not y:
246  d = len(x)
247  else:
248  d = min(
249  distance(x[1:], y) + 1, # correct an insertion error
250  distance(x, y[1:]) + 1, # correct a deletion error
251  distance(x[1:], y[1:]) + (x[0] != y[0])) # correct a wrong character
252  if len(x) >= 2 and len(y) >= 2 and x[0] == y[1] and x[1] == y[0]:
253  # Correct a transposition.
254  t = distance(x[2:], y[2:]) + 1
255  if d > t:
256  d = t
257 
258  memo[x, y] = d
259  return d
260  return distance(a, b)

263 def text_wrap(text, length=None, indent='', firstline_indent=None):
264  """Wraps a given text to a maximum line length and returns it.
265 
266  It turns lines that only contain whitespace into empty lines, keeps new lines,
267  and expands tabs using 4 spaces.
268 
269  Args:
270  text: str, text to wrap.
271  length: int, maximum length of a line, includes indentation.
272  If this is None then use get_help_width()
273  indent: str, indent for all but first line.
274  firstline_indent: str, indent for first line; if None, fall back to indent.
275 
276  Returns:
277  str, the wrapped text.
278 
279  Raises:
280  ValueError: Raised if indent or firstline_indent not shorter than length.
281  """
282  # Get defaults where callee used None
283  if length is None:
284  length = get_help_width()
285  if indent is None:
286  indent = ''
287  if firstline_indent is None:
288  firstline_indent = indent
289 
290  if len(indent) >= length:
291  raise ValueError('Length of indent exceeds length')
292  if len(firstline_indent) >= length:
293  raise ValueError('Length of first line indent exceeds length')
294 
295  text = text.expandtabs(4)
296 
297  result = []
298  # Create one wrapper for the first paragraph and one for subsequent
299  # paragraphs that does not have the initial wrapping.
300  wrapper = textwrap.TextWrapper(
301  width=length, initial_indent=firstline_indent, subsequent_indent=indent)
302  subsequent_wrapper = textwrap.TextWrapper(
303  width=length, initial_indent=indent, subsequent_indent=indent)
304 
305  # textwrap does not have any special treatment for newlines. From the docs:
306  # "...newlines may appear in the middle of a line and cause strange output.
307  # For this reason, text should be split into paragraphs (using
308  # str.splitlines() or similar) which are wrapped separately."
309  for paragraph in (p.strip() for p in text.splitlines()):
310  if paragraph:
311  result.extend(wrapper.wrap(paragraph))
312  else:
313  result.append('') # Keep empty lines.
314  # Replace initial wrapper with wrapper for subsequent paragraphs.
315  wrapper = subsequent_wrapper
316 
317  return '\n'.join(result)

320 def flag_dict_to_args(flag_map):
321  """Convert a dict of values into process call parameters.
322 
323  This method is used to convert a dictionary into a sequence of parameters
324  for a binary that parses arguments using this module.
325 
326  Args:
327  flag_map: dict, a mapping where the keys are flag names (strings).
328  values are treated according to their type:
329  * If value is None, then only the name is emitted.
330  * If value is True, then only the name is emitted.
331  * If value is False, then only the name prepended with 'no' is emitted.
332  * If value is a string then --name=value is emitted.
333  * If value is a collection, this will emit --name=value1,value2,value3.
334  * Everything else is converted to string an passed as such.
335  Yields:
336  sequence of string suitable for a subprocess execution.
337  """
338  for key, value in six.iteritems(flag_map):
339  if value is None:
340  yield '--%s' % key
341  elif isinstance(value, bool):
342  if value:
343  yield '--%s' % key
344  else:
345  yield '--no%s' % key
346  elif isinstance(value, (bytes, type(u''))):
347  # We don't want strings to be handled like python collections.
348  yield '--%s=%s' % (key, value)
349  else:
350  # Now we attempt to deal with collections.
351  try:
352  yield '--%s=%s' % (key, ','.join(str(item) for item in value))
353  except TypeError:
354  # Default case.
355  yield '--%s=%s' % (key, value)

358 def trim_docstring(docstring):
359  """Removes indentation from triple-quoted strings.
360 
361  This is the function specified in PEP 257 to handle docstrings:
362  https://www.python.org/dev/peps/pep-0257/.
363 
364  Args:
365  docstring: str, a python docstring.
366 
367  Returns:
368  str, docstring with indentation removed.
369  """
370  if not docstring:
371  return ''
372 
373  # If you've got a line longer than this you have other problems...
374  max_indent = 1 << 29
375 
376  # Convert tabs to spaces (following the normal Python rules)
377  # and split into a list of lines:
378  lines = docstring.expandtabs().splitlines()
379 
380  # Determine minimum indentation (first line doesn't count):
381  indent = max_indent
382  for line in lines[1:]:
383  stripped = line.lstrip()
384  if stripped:
385  indent = min(indent, len(line) - len(stripped))
386  # Remove indentation (first line is special):
387  trimmed = [lines[0].strip()]
388  if indent < max_indent:
389  for line in lines[1:]:
390  trimmed.append(line[indent:].rstrip())
391  # Strip off trailing and leading blank lines:
392  while trimmed and not trimmed[-1]:
393  trimmed.pop()
394  while trimmed and not trimmed[0]:
395  trimmed.pop(0)
396  # Return a single string:
397  return '\n'.join(trimmed)

400 def doc_to_help(doc):
401  """Takes a __doc__ string and reformats it as help."""
402 
403  # Get rid of starting and ending white space. Using lstrip() or even
404  # strip() could drop more than maximum of first line and right space
405  # of last line.
406  doc = doc.strip()
407 
408  # Get rid of all empty lines.
409  whitespace_only_line = re.compile('^[ \t]+$', re.M)
410  doc = whitespace_only_line.sub('', doc)
411 
412  # Cut out common space at line beginnings.
413  doc = trim_docstring(doc)
414 
415  # Just like this module's comment, comments tend to be aligned somehow.
416  # In other words they all start with the same amount of white space.
417  # 1) keep double new lines;
418  # 2) keep ws after new lines if not empty line;
419  # 3) all other new lines shall be changed to a space;
420  # Solution: Match new lines between non white space and replace with space.
421  doc = re.sub(r'(?<=\S)\n(?=\S)', ' ', doc, flags=re.M)
422 
423  return doc
