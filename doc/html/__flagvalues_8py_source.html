41 class FlagValues(object):
42  """Registry of 'Flag' objects.
43 
44  A 'FlagValues' can then scan command line arguments, passing flag
45  arguments through to the 'Flag' objects that it owns. It also
46  provides easy access to the flag values. Typically only one
47  'FlagValues' object is needed by an application: flags.FLAGS
48 
49  This class is heavily overloaded:
50 
51  'Flag' objects are registered via __setitem__:
52  FLAGS['longname'] = x # register a new flag
53 
54  The .value attribute of the registered 'Flag' objects can be accessed
55  as attributes of this 'FlagValues' object, through __getattr__. Both
56  the long and short name of the original 'Flag' objects can be used to
57  access its value:
58  FLAGS.longname # parsed flag value
59  FLAGS.x # parsed flag value (short name)
60 
61  Command line arguments are scanned and passed to the registered 'Flag'
62  objects through the __call__ method. Unparsed arguments, including
63  argv[0] (e.g. the program name) are returned.
64  argv = FLAGS(sys.argv) # scan command line arguments
65 
66  The original registered Flag objects can be retrieved through the use
67  of the dictionary-like operator, __getitem__:
68  x = FLAGS['longname'] # access the registered Flag object
69 
70  The str() operator of a 'FlagValues' object provides help for all of
71  the registered 'Flag' objects.
72  """

208  def _flag_is_registered(self, flag_obj):
209  """Checks whether a Flag object is registered under long name or short name.
210 
211  Args:
212  flag_obj: Flag, the Flag instance to check for.
213 
214  Returns:
215  bool, True iff flag_obj is registered under long name or short name.
216  """
217  flag_dict = self._flags()
218  # Check whether flag_obj is registered under its long name.
219  name = flag_obj.name
220  if flag_dict.get(name, None) == flag_obj:
221  return True
222  # Check whether flag_obj is registered under its short name.
223  short_name = flag_obj.short_name
224  if (short_name is not None and
225  flag_dict.get(short_name, None) == flag_obj):
226  return True
227  return False

229  def _cleanup_unregistered_flag_from_module_dicts(self, flag_obj):
230  """Cleans up unregistered flags from all module -> [flags] dictionaries.
231 
232  If flag_obj is registered under either its long name or short name, it
233  won't be removed from the dictionaries.
234 
235  Args:
236  flag_obj: Flag, the Flag instance to clean up for.
237  """

245  # flag in the list for the same module.

290  def find_module_defining_flag(self, flagname, default=None):
291  """Return the name of the module defining this flag, or default.
292 
293  Args:
294  flagname: str, name of the flag to lookup.
295  default: Value to return if flagname is not defined. Defaults
296  to None.
297 
298  Returns:
299  The name of the module which registered the flag with this name.
300  If no such module exists (i.e. no flag with this name exists),
301  we return default.
302  """
303  registered_flag = self._flags().get(flagname)
304  if registered_flag is None:
305  return default
306  for module, flags in six.iteritems(self.flags_by_module_dict()):
307  for flag in flags:
308  # It must compare the flag with the one in _flags. This is because a
309  # flag might be overridden only for its long name (or short name),
310  # and only its short name (or long name) is considered registered.
311  if (flag.name == registered_flag.name and
312  flag.short_name == registered_flag.short_name):
313  return module
314  return default

377  def append_flag_values(self, flag_values):

406  def __setitem__(self, name, flag):
407  """Registers a new flag variable."""
408  fl = self._flags()
409  if not isinstance(flag, _flag.Flag):
410  raise _exceptions.IllegalFlagValueError(flag)
411  if str is bytes and isinstance(name, unicode):
412  # When using Python 2 with unicode_literals, allow it but encode it
413  # into the bytes type we require.
414  name = name.encode('utf-8')
415  if not isinstance(name, type('')):
416  raise _exceptions.Error('Flag name must be a string')
417  if not name:
418  raise _exceptions.Error('Flag name cannot be empty')
419  self._check_method_name_conflicts(name, flag)
420  if name in fl and not flag.allow_override and not fl[name].allow_override:
421  module, module_name = _helpers.get_calling_module_object_and_name()
422  if (self.find_module_defining_flag(name) == module_name and
423  id(module) != self.find_module_id_defining_flag(name)):
424  # If the flag has already been defined by a module with the same name,
425  # but a different ID, we can stop here because it indicates that the
426  # module is simply being imported a subsequent time.
427  return
428  raise _exceptions.DuplicateFlagError.from_flag(name, self)
429  short_name = flag.short_name
430  # If a new flag overrides an old one, we need to cleanup the old flag's
431  # modules if it's not registered.
432  flags_to_cleanup = set()
433  if short_name is not None:
434  if (short_name in fl and not flag.allow_override and
435  not fl[short_name].allow_override):
436  raise _exceptions.DuplicateFlagError.from_flag(short_name, self)
437  if short_name in fl and fl[short_name] != flag:
438  flags_to_cleanup.add(fl[short_name])
439  fl[short_name] = flag
440  if (name not in fl # new flag
441  or fl[name].using_default_value
442  or not flag.using_default_value):
443  if name in fl and fl[name] != flag:
444  flags_to_cleanup.add(fl[name])
445  fl[name] = flag
446  for f in flags_to_cleanup:
447  self._cleanup_unregistered_flag_from_module_dicts(f)

473  def __getattr__(self, name):
474  """Retrieves the 'value' attribute of the flag --name."""
475  fl = self._flags()
476  if name not in fl:
477  raise AttributeError(name)
478  if name in self.__dict__['__hiddenflags']:
479  raise AttributeError(name)
480 
481  if self.__dict__['__flags_parsed'] or fl[name].present:
482  return fl[name].value
483  else:
484  error_message = (
485  'Trying to access flag --%s before flags were parsed.' % name)
486  if self._is_unparsed_flag_access_allowed(name):
487  # Print warning to stderr. Messages in logs are often ignored/unnoticed.
488  warnings.warn(
489  error_message + ' This will raise an exception in the future.',
490  RuntimeWarning,
491  stacklevel=2)
492  # Force logging.exception() to behave realistically, but don't propagate
493  # exception up. Allow flag value to be returned (for now).
494  try:
495  raise _exceptions.UnparsedFlagAccessError(error_message)
496  except _exceptions.UnparsedFlagAccessError:
497  logging.exception(error_message)
498  return fl[name].value
499  else:
500  if six.PY2:
501  # In Python 2, hasattr returns False if getattr raises any exception.
502  # That means if someone calls hasattr(FLAGS, 'flag'), it returns False
503  # instead of raises UnparsedFlagAccessError even if --flag is already
504  # defined. To make the error more visible, the best we can do is to
505  # log an error message before raising the exception.
506  # Don't log a full stacktrace here since that makes other callers
507  # get too much noise.
508  logging.error(error_message)
509  raise _exceptions.UnparsedFlagAccessError(error_message)

530  def _assert_validators(self, validators):
531  """Asserts if all validators in the list are satisfied.
532 
533  It asserts validators in the order they were created.
534 
535  Args:
536  validators: Iterable(validators.Validator), validators to be
537  verified.
538  Raises:
539  AttributeError: Raised if validators work with a non-existing flag.
540  IllegalFlagValueError: Raised if validation fails for at least one
541  validator.

544  validators, key=lambda validator: validator.insertion_index):
545  try:
546  validator.verify(self)
547  except _exceptions.ValidationError as e:
548  message = validator.print_flags_with_values(self)
549  raise _exceptions.IllegalFlagValueError('%s: %s' % (message, str(e)))

613  def __call__(self, argv):
614  """Parses flags from argv; stores parsed flags into this FlagValues object.
615 
616  All unparsed arguments are returned.
617 
618  Args:
619  argv: a tuple/list of strings.
620 
621  Returns:
622  The list of arguments not parsed as options, including argv[0].
623 
624  Raises:
625  Error: Raised on any parsing error.
626  TypeError: Raised on passing wrong type of arguments.
627  ValueError: Raised on flag value parsing error.
628  """
629  if _helpers.is_bytes_or_string(argv):
630  raise TypeError(
631  'argv should be a tuple/list of strings, not bytes or string.')
632  if not argv:
633  raise ValueError(
634  'argv cannot be an empty list, and must contain the program name as '
635  'the first element.')
636 
637  # This pre parses the argv list for --flagfile=<> options.
638  program_name = argv[0]
639  args = self.read_flags_from_files(argv[1:], force_gnu=False)
640 
641  # Parse the arguments.
642  unknown_flags, unparsed_args, undefok = self._parse_args(args)
643 
644  # Handle unknown flags by raising UnrecognizedFlagError.
645  # Note some users depend on us raising this particular error.
646  for name, value in unknown_flags:
647  if name in undefok:
648  continue
649 
650  suggestions = _helpers.get_flag_suggestions(name, list(self))
651  raise _exceptions.UnrecognizedFlagError(
652  name, value, suggestions=suggestions)
