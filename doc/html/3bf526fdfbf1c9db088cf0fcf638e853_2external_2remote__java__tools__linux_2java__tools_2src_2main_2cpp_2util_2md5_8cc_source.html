75 // This is a templated function so that T can be either a char*
76 // or a string. This works because we use the [] operator to access
77 // individual characters at a time.
78 template <typename T>
79 static void b2a_hex_t(const unsigned char* b, T a, int num) {
80  for (int i = 0; i < num; i++) {
81  a[i * 2 + 0] = hex_char[b[i] >> 4];
82  a[i * 2 + 1] = hex_char[b[i] & 0xf];
83  }
84 }

86 // ----------------------------------------------------------------------
87 // b2a_hex()
88 // Description: Binary-to-Ascii hex conversion. This converts
89 // 'num' bytes of binary to a 2*'num'-character hexadecimal representation
90 // Return value: 2*'num' characters of ascii text (via the 'to' argument)
91 // ----------------------------------------------------------------------
92 static void b2a_hex(const unsigned char* from, string* to, int num) {
93  to->resize(num << 1);
94  b2a_hex_t<string&>(from, *to, num);
95 }

118 void Md5Digest::Update(const void *buf, unsigned int length) {
119  const unsigned char *input = reinterpret_cast<const unsigned char*>(buf);
120  unsigned int buffer_space_len;
121 
122  buffer_space_len = k8Bytes - ctx_buffer_len;
123 
124  // Transform as many times as possible.
125  if (length >= buffer_space_len) {
126  if (buffer_space_len != 0 && ctx_buffer_len != 0) {
127  // Copy more bytes to fill the complete buffer
128  memcpy(ctx_buffer + ctx_buffer_len, input, buffer_space_len);
129  Transform(ctx_buffer, k8Bytes);
130  input += buffer_space_len;
131  length -= buffer_space_len;
132  ctx_buffer_len = 0;
133  }
134 
135  if (UNALIGNED_P(input)) {
136  while (length >= k8Bytes) {
137  memcpy(ctx_buffer, input, k8Bytes);
138  Transform(ctx_buffer, k8Bytes);
139  input += k8Bytes;
140  length -= k8Bytes;
141  }
142  } else if (length >= k8Bytes) {
143  Transform(input, length & ~k8ByteMask);
144  input += length & ~k8ByteMask;
145  length &= k8ByteMask;
146  }
147  }
148 
149  // Buffer remaining input
150  memcpy(ctx_buffer + ctx_buffer_len, input, length);
151  ctx_buffer_len += length;
152 }

154 void Md5Digest::Finish(unsigned char digest[16]) {
155  count[0] += ctx_buffer_len;
156  if (count[0] < ctx_buffer_len) {
157  ++count[1];
158  }
159 
160  /* Put the 64-bit file length in *bits* at the end of the buffer. */
161  unsigned int size = (ctx_buffer_len < 56 ? 64 : 128);
162  uint32_t words[2] = { count[0] << 3, (count[1] << 3) | (count[0] >> 29) };
163  memcpy(ctx_buffer + size - 8, words, 8);
164 
165  memcpy(ctx_buffer + ctx_buffer_len, kPadding, size - 8 - ctx_buffer_len);
166 
167  Transform(ctx_buffer, size);
168 
169  uint32_t* r = reinterpret_cast<uint32_t*>(digest);
170  r[0] = state[0];
171  r[1] = state[1];
172  r[2] = state[2];
173  r[3] = state[3];
174 }

196  // F, G, H and I are basic MD5 functions.
197 /* These are the four functions used in the four steps of the MD5 algorithm
198  and defined in the RFC 1321. The first function is a little bit optimized
199  (as found in Colin Plumbs public domain implementation). */
200 /* #define F(b, c, d) ((b & c) | (~b & d)) */
201 #define F(x, y, z) (z ^ (x & (y ^ z)))
202 #define G(x, y, z) F (z, x, y)
203 #define H(x, y, z) ((x) ^ (y) ^ (z))
204 #define I(x, y, z) ((y) ^ ((x) | (~z)))
205 
206  // ROTATE_LEFT rotates x left n bits.
207 #define ROTATE_LEFT(x, n) (((x) << (n)) | ((x) >> (32-(n))))
208 
209  // FF, GG, HH, and II transformations for rounds 1, 2, 3, and 4.
210  // Rotation is separate from addition to prevent recomputation.
211 #define FF(a, b, c, d, s, ac) { \
212  (a) += F((b), (c), (d)) + ((*x_pos++ = *cur_word++)) + \
213  static_cast<uint32_t>(ac); \
214  (a) = ROTATE_LEFT((a), (s)); \
215  (a) += (b); \
216  }
217 
218 #define GG(a, b, c, d, x, s, ac) { \
219  (a) += G((b), (c), (d)) + (x) + static_cast<uint32_t>(ac); \
220  (a) = ROTATE_LEFT((a), (s)); \
221  (a) += (b); \
222  }
223 #define HH(a, b, c, d, x, s, ac) { \
224  (a) += H((b), (c), (d)) + (x) + static_cast<uint32_t>(ac); \
225  (a) = ROTATE_LEFT((a), (s)); \
226  (a) += (b); \
227  }
228 #define II(a, b, c, d, x, s, ac) { \
229  (a) += I((b), (c), (d)) + (x) + static_cast<uint32_t>(ac); \
230  (a) = ROTATE_LEFT((a), (s)); \
231  (a) += (b); \
232  }

234  count[0] += len;
235  if (count[0] < len) {
236  ++count[1];
237  }

240  uint32_t b = state[1];
241  uint32_t c = state[2];

244 
245  const uint32_t *cur_word = reinterpret_cast<const uint32_t*>(buffer);
246  const uint32_t *end_word = cur_word + (len / sizeof(uint32_t));
