207 def _run_main(main, argv):
208  """Calls main, optionally with pdb or profiler."""
209  if FLAGS.run_with_pdb:
210  sys.exit(pdb.runcall(main, argv))
211  elif FLAGS.run_with_profiling or FLAGS.profile_file:
212  # Avoid import overhead since most apps (including performance-sensitive
213  # ones) won't be run with profiling.
214  import atexit
215  if FLAGS.use_cprofile_for_profiling:
216  import cProfile as profile
217  else:
218  import profile
219  profiler = profile.Profile()
220  if FLAGS.profile_file:
221  atexit.register(profiler.dump_stats, FLAGS.profile_file)
222  else:
223  atexit.register(profiler.print_stats)
224  retval = profiler.runcall(main, argv)
225  sys.exit(retval)
226  else:
227  sys.exit(main(argv))

247  """Begins executing the program.
248 
249  Args:
250  main: The main function to execute. It takes an single argument "argv",
251  which is a list of command line arguments with parsed flags removed.
252  argv: A non-empty list of the command line arguments including program name,
253  sys.argv is used if None.
254  - Parses command line flags with the flag module.
255  - If there are any errors, prints usage().

297 def usage(shorthelp=False, writeto_stdout=False, detailed_error=None,
298  exitcode=None):
299  """Writes __main__'s docstring to stderr with some help text.
300 
301  Args:
302  shorthelp: bool, if True, prints only flags from this module,
303  rather than all flags.
304  writeto_stdout: bool, if True, writes help message to stdout,
305  rather than to stderr.
306  detailed_error: str, additional detail about why usage info was presented.
307  exitcode: optional integer, if set, exits with this status code after
308  writing help.
309  """
310  if writeto_stdout:
311  stdfile = sys.stdout
312  else:
313  stdfile = sys.stderr
314 
315  doc = sys.modules['__main__'].__doc__
316  if not doc:
317  doc = '\nUSAGE: %s [flags]\n' % sys.argv[0]
318  doc = flags.text_wrap(doc, indent=' ', firstline_indent='')
319  else:
320  # Replace all '%s' with sys.argv[0], and all '%%' with '%'.
321  num_specifiers = doc.count('%') - 2 * doc.count('%%')
322  try:
323  doc %= (sys.argv[0],) * num_specifiers
324  except (OverflowError, TypeError, ValueError):
325  # Just display the docstring as-is.
326  pass
327  if shorthelp:
328  flag_str = FLAGS.main_module_help()
329  else:
330  flag_str = str(FLAGS)
331  try:
332  stdfile.write(doc)
333  if flag_str:
334  stdfile.write('\nflags:\n')
335  stdfile.write(flag_str)
336  stdfile.write('\n')
337  if detailed_error is not None:
338  stdfile.write('\n%s\n' % detailed_error)
339  except IOError as e:
340  # We avoid printing a huge backtrace if we get EPIPE, because
341  # "foo.par --help | less" is a frequent use case.
342  if e.errno != errno.EPIPE:
343  raise
344  if exitcode is not None:
345  sys.exit(exitcode)
