36 public final class WindowsJunctionCreator implements JunctionCreator {
37  private final Path dir;
38  private Map<Path, Path> paths; // allocated lazily, but semantically final
39  private int junctionIndex = 0;

41  public WindowsJunctionCreator(Path dir) {
42  this.dir = Preconditions.checkNotNull(dir);
43  }
44 
45  @Nullable
46  public Path create(@Nullable Path path) throws IOException {
47  if (path == null) {
48  return null;
49  }
50 
51  if (paths == null) {
52  paths = new LinkedHashMap<>();
53  }
54  path = path.toAbsolutePath();
55  if (path.toFile().isDirectory()) {
56  Path link = paths.get(path);
57  if (link == null) {
58  link = dir.resolve(Integer.toString(junctionIndex++));
59  WindowsFileOperations.createJunction(link.toString(), path.toString());
60  paths.put(path, link);
61  }
62  return link;
63  }
64 
65  Path parent = path.getParent();
66  return (parent == null) ? path : create(parent).resolve(path.getFileName());
67  }
68 
69  @Override
70  public void close() throws IOException {
71  // Delete all junctions, otherwise the temp directory deleter would follow them and delete files
72  // from directories they point to.
73  if (paths != null) {
74  for (Path link : paths.values()) {
75  link.toFile().delete();
76  }
77  }
78  dir.toFile().delete();
79  }
80 }
