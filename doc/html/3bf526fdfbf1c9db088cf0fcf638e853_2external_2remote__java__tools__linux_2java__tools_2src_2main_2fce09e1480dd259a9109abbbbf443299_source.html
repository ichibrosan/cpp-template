53 // Messages intended for the user (level USER, along with WARNINGs an ERRORs)
54 // should be printed even if debug level logging was not requested.
55 void PrintUserLevelMessageToStream(std::ostream* stream, LogLevel level,
56  const std::string& message) {
57  if (level == LOGLEVEL_USER) {
58  (*stream) << message << std::endl;
59  } else if (level > LOGLEVEL_USER) {
60  (*stream) << LogLevelName(level) << ": " << message << std::endl;
61  }
62  // If level < USER, this is an INFO message. It's useful for debugging but
63  // should not be printed to the user unless the user has asked for debugging
64  // output. We ignore it here.
65 }

67 static std::string Timestamp() {
68  auto now = std::chrono::system_clock::now();
69  time_t s = std::chrono::system_clock::to_time_t(now);
70  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
71  now.time_since_epoch());
72  struct tm tmbuf = {};
73 #ifdef _WIN32
74  tmbuf = *localtime(&s); // NOLINT -- threadsafe on windows
75 #else
76  localtime_r(&s, &tmbuf);
77 #endif
78  char buf[16];
79  int r = strftime(buf, sizeof buf - 5, "%H:%M:%S", &tmbuf);
80  r += snprintf(buf + r, +5, ".%03d", static_cast<int>(ms.count() % 1000));
81  return std::string(buf, r);
82 }

94 void BazelLogHandler::HandleMessage(LogLevel level, const std::string& filename,
95  int line, const std::string& message,
96  int exit_code) {
97  if (logging_deactivated_) {
98  // If the output stream was explicitly deactivated, never print INFO
99  // messages, but messages of level USER and above should always be printed,
100  // as should warnings and errors. Omit the debug-level file and line number
101  // information, though.
102  PrintUserLevelMessageToStream(&std::cerr, level, message);
103  if (level == LOGLEVEL_FATAL) {
104  std::exit(exit_code);
105  }
106  return;
107  }
108  if (output_stream_ == nullptr) {
109  // If we haven't decided whether messages should be logged to debug levels
110  // or not, buffer each version. This is redundant for USER levels and above,
111  // but is to make sure we can provide the right output to the user once we
112  // know that they do or do not want debug level information.
113  PrintUserLevelMessageToStream(user_buffer_stream_.get(), level, message);
114  PrintDebugLevelMessageToStream(debug_buffer_stream_.get(), filename, line,
115  level, message);
116  } else {
117  // If an output stream has been specifically set, it is for the full suite
118  // of log messages. We don't print the user messages separately here as they
119  // are included.
120  PrintDebugLevelMessageToStream(output_stream_, filename, line, level,
121  message);
122  }
123 
124  // If we have a fatal message, exit with the provided error code.
125  if (level == LOGLEVEL_FATAL) {
126  if (owned_output_stream_ != nullptr) {
127  // If this is is not being printed to stderr but to a custom stream,
128  // also print the error message to stderr.
129  PrintUserLevelMessageToStream(&std::cerr, level, message);
130  }
131  std::exit(exit_code);
132  }
133 }

135 void BazelLogHandler::SetOutputStreamToStderr() {
136  // Disallow second calls to this, we only intend to support setting the output
137  // once, otherwise the buffering will not work as intended and the log will be
138  // fragmented.
139  BAZEL_CHECK(!output_stream_set_) << "Tried to set log output a second time";
140  output_stream_set_ = true;
141 
142  FlushBufferToNewStreamAndSet(debug_buffer_stream_.get(), &std::cerr);
143  debug_buffer_stream_ = nullptr;
144  // The user asked for debug level information, which includes the user
145  // messages. We can discard the separate buffer at this point.
146  user_buffer_stream_ = nullptr;
147 }

149 void BazelLogHandler::SetOutputStream(
150  std::unique_ptr<std::ostream> new_output_stream) {
151  // Disallow second calls to this, we only intend to support setting the output
152  // once, otherwise the buffering will not work as intended and the log will be
153  // fragmented.
154  BAZEL_CHECK(!output_stream_set_) << "Tried to set log output a second time";
155  output_stream_set_ = true;
156 
157  if (new_output_stream == nullptr) {
158  logging_deactivated_ = true;
159  // Flush the buffered user-level messages to stderr - these are messages
160  // that are meant for the user even when debug logging is not set.
161  FlushBufferToNewStreamAndSet(user_buffer_stream_.get(), &std::cerr);
162 
163  user_buffer_stream_ = nullptr;
164  // We discard the debug level logs, the user level ones were enough to
165  // inform the user and debug logging was not requested.
166  debug_buffer_stream_ = nullptr;
167  return;
168  }
169  owned_output_stream_ = std::move(new_output_stream);
170  if (owned_output_stream_->fail()) {
171  // If opening the stream failed, continue buffering and have the logs
172  // dump to stderr at shutdown.
173  BAZEL_LOG(ERROR) << "Provided stream failed.";
174  return;
175  }
176  FlushBufferToNewStreamAndSet(debug_buffer_stream_.get(),
177  owned_output_stream_.get());
178  debug_buffer_stream_ = nullptr;
179  // The user asked for debug level information, which includes the user
180  // messages. We can discard the separate buffer at this point.
181  user_buffer_stream_ = nullptr;
182 }

184 void BazelLogHandler::FlushBufferToNewStreamAndSet(
185  std::stringstream* buffer, std::ostream* new_output_stream) {
186  // Flush the buffer to the new stream, and print new log lines to it.
187  output_stream_ = new_output_stream;
188  // Transfer the contents of the buffer to the new stream, then remove the
189  // buffer.
190  (*output_stream_) << buffer->str();
191  output_stream_->flush();
192 }
