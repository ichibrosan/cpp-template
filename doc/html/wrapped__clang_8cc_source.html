15 // wrapped_clang.cc: Pass args to 'xcrun clang' and zip dsym files.
16 //
17 // wrapped_clang passes its args to clang, but also supports a separate set of
18 // invocations to generate dSYM files. If "DSYM_HINT" flags are passed in, they
19 // are used to construct that separate set of invocations (instead of being
20 // passed to clang).
21 // The following "DSYM_HINT" flags control dsym generation. If any one if these
22 // are passed in, then they all must be passed in.
23 // "DSYM_HINT_LINKED_BINARY": Workspace-relative path to binary output of the
24 // link action generating the dsym file.
25 // "DSYM_HINT_DSYM_PATH": Workspace-relative path to dSYM dwarf file.
26 //
27 // Likewise, this wrapper also contains a workaround for a bug in ld that causes
28 // flaky builds when using Bitcode symbol maps. ld allows the
29 // -bitcode_symbol_map to be either a directory (into which the file will be
30 // written) or a file, but the return value of the call to ::stat is never
31 // checked so examining the S_ISDIR bit of the struct afterwards returns
32 // true/false randomly depending on what data happened to be in memory at the
33 // time it was called:
34 // https://github.com/michaelweiser/ld64/blob/9c3700b64ed03e2d55ba094176bf6a172bf2bc6b/src/ld/Options.cpp#L3261
35 // To address this, we prepend a special "BITCODE_TOUCH_SYMBOL_MAP=" flag to the
36 // symbol map filename and touch it before passing it along to clang, forcing
37 // the file to exist.

62 // Returns the base name of the given filepath. For example, given

69 // Converts an array of string arguments to char *arguments.
70 // The first arg is reduced to its basename as per execve conventions.
71 // Note that the lifetime of the char* arguments in the returned array
72 // are controlled by the lifetime of the strings in args.
73 std::vector<const char *> ConvertToCArgs(const std::vector<std::string> &args) {
74  std::vector<const char *> c_args;
75  c_args.push_back(Basename(args[0].c_str()));
76  for (int i = 1; i < args.size(); i++) {
77  c_args.push_back(args[i].c_str());
78  }
79  c_args.push_back(nullptr);
80  return c_args;
81 }

83 // Turn our current process into a new process. Avoids fork overhead.
84 // Never returns.
85 void ExecProcess(const std::vector<std::string> &args) {
86  std::vector<const char *> exec_argv = ConvertToCArgs(args);
87  execv(args[0].c_str(), const_cast<char **>(exec_argv.data()));
88  std::cerr << "Error executing child process.'" << args[0] << "'. "
89  << strerror(errno) << "\n";
90  abort();
91 }

93 // Spawns a subprocess for given arguments args. The first argument is used
94 // for the executable path.
95 void RunSubProcess(const std::vector<std::string> &args) {
96  std::vector<const char *> exec_argv = ConvertToCArgs(args);
97  pid_t pid;
98  int status = posix_spawn(&pid, args[0].c_str(), NULL, NULL,
99  const_cast<char **>(exec_argv.data()), environ);
100  if (status == 0) {
101  int wait_status;
102  do {
103  wait_status = waitpid(pid, &status, 0);
104  } while ((wait_status == -1) && (errno == EINTR));
105  if (wait_status < 0) {
106  std::cerr << "Error waiting on child process '" << args[0] << "'. "
107  << strerror(errno) << "\n";
108  abort();
109  }
110  if (WEXITSTATUS(status) != 0) {
111  std::cerr << "Error in child process '" << args[0] << "'. "
112  << WEXITSTATUS(status) << "\n";
113  abort();
114  }
115  } else {
116  std::cerr << "Error forking process '" << args[0] << "'. "
117  << strerror(status) << "\n";
118  abort();
119  }
120 }

122 // Finds and replaces all instances of oldsub with newsub, in-place on str.
123 void FindAndReplace(const std::string &oldsub, const std::string &newsub,
124  std::string *str) {
125  int start = 0;
126  while ((start = str->find(oldsub, start)) != std::string::npos) {
127  str->replace(start, oldsub.length(), newsub);
128  start += newsub.length();
129  }
130 }

132 // If arg is of the classic flag form "foo=bar", and flagname is 'foo', sets
133 // str to point to a new std::string 'bar' and returns true.

135 bool SetArgIfFlagPresent(const std::string &arg, const std::string &flagname,
136  std::string *str) {
137  std::string prefix_string = flagname + "=";
138  if (arg.compare(0, prefix_string.length(), prefix_string) == 0) {
139  *str = arg.substr(prefix_string.length());
140  return true;

173 int main(int argc, char *argv[]) {
174  std::string tool_name;
175 
176  std::string binary_name = Basename(argv[0]);
177  if (binary_name == "wrapped_clang_pp") {
178  tool_name = "clang++";
179  } else if (binary_name == "wrapped_clang") {
180  tool_name = "clang";
181  } else {
182  std::cerr << "Binary must either be named 'wrapped_clang' or "
183  "'wrapped_clang_pp', not "
184  << binary_name << "\n";
185  abort();
186  }
187 
188  std::string developer_dir = GetMandatoryEnvVar("DEVELOPER_DIR");
189  std::string sdk_root = GetMandatoryEnvVar("SDKROOT");
190 
191  std::vector<std::string> processed_args = {"/usr/bin/xcrun", tool_name};
192 
193  std::string linked_binary, dsym_path, bitcode_symbol_map;
194  std::string dest_dir;
195 
196  std::unique_ptr<char, decltype(std::free) *> cwd{getcwd(nullptr, 0),
197  std::free};
198  if (cwd == nullptr) {
199  std::cerr << "Error determining current working directory\n";
200  abort();
201  }
202 
203  for (int i = 1; i < argc; i++) {
204  std::string arg(argv[i]);
205 
206  if (SetArgIfFlagPresent(arg, "DSYM_HINT_LINKED_BINARY", &linked_binary)) {
207  continue;
208  }
209  if (SetArgIfFlagPresent(arg, "DSYM_HINT_DSYM_PATH", &dsym_path)) {
210  continue;
211  }
212  if (SetArgIfFlagPresent(arg, "BITCODE_TOUCH_SYMBOL_MAP",
213  &bitcode_symbol_map)) {
214  // Touch bitcode_symbol_map.
215  std::ofstream bitcode_symbol_map_file(bitcode_symbol_map);
216  arg = bitcode_symbol_map;
217  }
218  if (SetArgIfFlagPresent(arg, "DEBUG_PREFIX_MAP_PWD", &dest_dir)) {
219  arg = "-fdebug-prefix-map=" + std::string(cwd.get()) + "=" + dest_dir;
220  }
221  FindAndReplace("__BAZEL_XCODE_DEVELOPER_DIR__", developer_dir, &arg);
222  FindAndReplace("__BAZEL_XCODE_SDKROOT__", sdk_root, &arg);
223 
224  // Make the `add_ast_path` options used to embed Swift module references
225  // absolute to enable Swift debugging without dSYMs: see
226  // https://forums.swift.org/t/improving-swift-lldb-support-for-path-remappings/22694
227  if (StripPrefixStringIfPresent(&arg, kAddASTPathPrefix)) {
228  // Only modify relative paths.
229  if (!StartsWith(arg, "/")) {
230  arg = std::string(kAddASTPathPrefix) +
231  std::string(cwd.get()) + "/" + arg;
232  } else {
233  arg = std::string(kAddASTPathPrefix) + arg;
234  }
235  }
236 
237  processed_args.push_back(arg);
238  }
239 
240  // Special mode that only prints the command. Used for testing.
241  if (getenv("__WRAPPED_CLANG_LOG_ONLY")) {
242  for (const std::string &arg : processed_args)
243  std::cout << arg << ' ';
244  std::cout << "\n";
245  return 0;
246  }
247 
248  // Check to see if we should postprocess with dsymutil.
249  bool postprocess = false;
250  if ((!linked_binary.empty()) || (!dsym_path.empty())) {
251  if ((linked_binary.empty()) || (dsym_path.empty())) {
252  const char *missing_dsym_flag;
253  if (linked_binary.empty()) {
254  missing_dsym_flag = "DSYM_HINT_LINKED_BINARY";
255  } else {
256  missing_dsym_flag = "DSYM_HINT_DSYM_PATH";
257  }
258  std::cerr << "Error in clang wrapper: If any dsym "
259  "hint is defined, then "
260  << missing_dsym_flag << " must be defined\n";
261  abort();
262  } else {
263  postprocess = true;
264  }
265  }
266 
267  if (!postprocess) {
268  ExecProcess(processed_args);
269  std::cerr << "ExecProcess should not return. Please fix!\n";
270  abort();
271  }
272 
273  RunSubProcess(processed_args);
274 
275  std::vector<std::string> dsymutil_args = {"/usr/bin/xcrun", "dsymutil",
276  linked_binary, "-o", dsym_path,
277  "--flat"};
278  ExecProcess(dsymutil_args);
279  std::cerr << "ExecProcess should not return. Please fix!\n";
280  abort();
281 
282  return 0;
283 }
