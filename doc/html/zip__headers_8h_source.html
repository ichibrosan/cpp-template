60  /* Utility functions to handle Zip64 extensions. Size and position fields in
61  * the Zip headers are 32-bit wide. If field's value does not fit into 32
62  * bits (more precisely, it is >= 0xFFFFFFFF), the field contains 0xFFFFFFFF
63  * and the actual value is saved in the corresponding 64-bit extension field.

66  * extension.

96 class ExtraField {
97  public:
98  static const ExtraField *find(uint16_t tag, const uint8_t *start,
99  const uint8_t *end) {
100  while (start < end) {
101  auto extra_field = reinterpret_cast<const ExtraField *>(start);
102  if (extra_field->is(tag)) {
103  return extra_field;
104  }
105  start = ziph::byte_ptr(start) + extra_field->size();
106  }
107  return nullptr;
108  }

131 /* Zip64 Extra Field (section 4.5.3 of the .ZIP format spec)
132  *
133  * It is present if a value of a uncompressed_size/compressed_size/file_offset
134  * exceeds 32 bits. It consists of a 4-byte header followed by
135  * [64-bit uncompressed_size] [64-bit compressed_size] [64-bit file_offset]
136  * Only the entities whose value exceed 32 bits are present, and the present
137  * ones are always in the order shown above. The originating 32-bit field
138  * contains 0xFFFFFFFF to indicate that the value is 64-bit and is in
139  * Zip64 Extra Field. Section 4.5.3 of the spec mentions that Zip64 extra field
140  * of the Local Header MUST have both uncompressed and compressed sizes present.
141  */

145  return reinterpret_cast<const Zip64ExtraField *>(
146  ExtraField::find(1, start, end));
147  }
148 
149  bool is() const { return is_zip64(); }
150  void signature() { ExtraField::signature(1); }
151 
152  // The value of i-th attribute

332 /* Data descriptor Record:
333  * 4.3.9 Data descriptor:
334  *
335  * crc-32 4 bytes
336  * compressed size 4 bytes
337  * uncompressed size 4 bytes
338  *
339  * 4.3.9.1 This descriptor MUST exist if bit 3 of the general purpose bit
340  * flag is set (see below). It is byte aligned and immediately follows the
341  * last byte of compressed data. This descriptor SHOULD be used only when it
342  * was not possible to seek in the output .ZIP file, e.g., when the output
343  * .ZIP file was standard output or a non-seekable device. For ZIP64(tm)
344  * format archives, the compressed and uncompressed sizes are 8 bytes each.
345  *
346  * 4.3.9.2 When compressing files, compressed and uncompressed sizes should
347  * be stored in ZIP64 format (as 8 byte values) when a file's size exceeds
348  * 0xFFFFFFFF. However ZIP64 format may be used regardless of the size of a
349  * file. When extracting, if the zip64 extended information extra field is
350  * present for the file the compressed and uncompressed sizes will be 8 byte
351  * values.
352  *
353  * 4.3.9.3 Although not originally assigned a signature, the value 0x08074b50
354  * has commonly been adopted as a signature value for the data descriptor
355  * record. Implementers should be aware that ZIP files may be encountered
356  * with or without this signature marking data descriptors and SHOULD account
357  * for either case when reading ZIP files to ensure compatibility.
358  */

363  return (0x08074b50 == le32toh(optional_signature_) ? 8 : 4) +
364  (compressed_size_is_64bits ? 8 : 4) +
365  (original_size_is_64bits ? 8 : 4);
366  }

370 } attr_packed;
