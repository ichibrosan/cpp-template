15 """Decorator and context manager for saving and restoring flag values.
16 
17 There are many ways to save and restore. Always use the most convenient method
18 for a given use case.
19 
20 Here are examples of each method. They all call do_stuff() while FLAGS.someflag
21 is temporarily set to 'foo'.
22 
23  # Use a decorator which can optionally override flags via arguments.
24  @flagsaver.flagsaver(someflag='foo')
25  def some_func():
26  do_stuff()
27 
28  # Use a decorator which does not override flags itself.
29  @flagsaver.flagsaver
30  def some_func():
31  FLAGS.someflag = 'foo'
32  do_stuff()
33 
34  # Use a context manager which can optionally override flags via arguments.
35  with flagsaver.flagsaver(someflag='foo'):
36  do_stuff()
37 
38  # Save and restore the flag values yourself.
39  saved_flag_values = flagsaver.save_flag_values()
40  try:
41  FLAGS.someflag = 'foo'
42  do_stuff()
43  finally:
44  flagsaver.restore_flag_values(saved_flag_values)
45 
46 We save and restore a shallow copy of each Flag object's __dict__ attribute.
47 This preserves all attributes of the flag, such as whether or not it was
48 overridden from its default value.
49 
50 WARNING: Currently a flag that is saved and then deleted cannot be restored. An
51 exception will be raised. However if you *add* a flag after saving flag values,
52 and then restore flag values, the added flag will be deleted with no errors.
53 """

85 def save_flag_values(flag_values=FLAGS):
86  """Returns copy of flag values as a dict.
87 
88  Args:
89  flag_values: FlagValues, the FlagValues instance with which the flag will
90  be saved. This should almost never need to be overridden.
91  Returns:
92  Dictionary mapping keys to values. Keys are flag names, values are
93  corresponding __dict__ members. E.g. {'key': value_dict, ...}.
94  """
95  return {name: _copy_flag_dict(flag_values[name]) for name in flag_values}

98 def restore_flag_values(saved_flag_values, flag_values=FLAGS):
99  """Restores flag values based on the dictionary of flag values.
100 
101  Args:
102  saved_flag_values: {'flag_name': value_dict, ...}
103  flag_values: FlagValues, the FlagValues instance from which the flag will
104  be restored. This should almost never need to be overridden.
105  """
106  new_flag_names = list(flag_values)
107  for name in new_flag_names:
108  saved = saved_flag_values.get(name)
109  if saved is None:
110  # If __dict__ was not saved delete "new" flag.
111  delattr(flag_values, name)
112  else:
113  if flag_values[name].value != saved['_value']:
114  flag_values[name].value = saved['_value'] # Ensure C++ value is set.
115  flag_values[name].__dict__ = saved

118 def _wrap(func, overrides):
119  """Creates a wrapper function that saves/restores flag values.
120 
121  Args:
122  func: function object - This will be called between saving flags and
123  restoring flags.
124  overrides: {str: object} - Flag names mapped to their values. These flags
125  will be set after saving the original flag state.
126 
127  Returns:
128  return value from func()
129  """
130  @functools.wraps(func)
131  def _flagsaver_wrapper(*args, **kwargs):
132  """Wrapper function that saves and restores flags."""
133  with _FlagOverrider(**overrides):
134  return func(*args, **kwargs)
135  return _flagsaver_wrapper

138 class _FlagOverrider(object):
139  """Overrides flags for the duration of the decorated function call.
140 
141  It also restores all original values of flags after decorated method
142  completes.
143  """
144 
145  def __init__(self, **overrides):
146  self._overrides = overrides
147  self._saved_flag_values = None
148 
149  def __call__(self, func):
150  if inspect.isclass(func):
151  raise TypeError('flagsaver cannot be applied to a class.')
152  return _wrap(func, self._overrides)
153 
154  def __enter__(self):
155  self._saved_flag_values = save_flag_values(FLAGS)
156  try:
157  for name, value in six.iteritems(self._overrides):
158  setattr(FLAGS, name, value)
159  except:
160  # It may fail because of flag validators.
161  restore_flag_values(self._saved_flag_values, FLAGS)
162  raise
163 
164  def __exit__(self, exc_type, exc_value, traceback):
165  restore_flag_values(self._saved_flag_values, FLAGS)

168 def _copy_flag_dict(flag):
169  """Returns a copy of the flag object's __dict__.
170 
171  It's mostly a shallow copy of the __dict__, except it also does a shallow
172  copy of the validator list.
173 
174  Args:
175  flag: flags.Flag, the flag to copy.
176 
177  Returns:
178  A copy of the flag object's __dict__.
179  """
180  copy = flag.__dict__.copy()
181  copy['_value'] = flag.value # Ensure correct restore for C++ flags.
182  copy['validators'] = list(flag.validators)
183  return copy
