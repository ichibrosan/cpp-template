68  """Raised if there is a flag naming conflict."""

70  @classmethod
71  def from_flag(cls, flagname, flag_values, other_flag_values=None):
72  """Create a DuplicateFlagError by providing flag name and values.
73 
74  Args:
75  flagname: Name of the flag being redefined.
76  flag_values: FlagValues object containing the first definition of
77  flagname.
78  other_flag_values: If this argument is not None, it should be the
79  FlagValues object where the second definition of flagname occurs.
80  If it is None, we assume that we're being called when attempting
81  to create the flag a second time, and we use the module calling
82  this one as the source of the second definition.
83 
84  Returns:
85  An instance of DuplicateFlagError.
86  """
87  first_module = flag_values.FindModuleDefiningFlag(
88  flagname, default='<unknown>')
89  if other_flag_values is None:
90  second_module = _helpers.GetCallingModule()
91  else:
92  second_module = other_flag_values.FindModuleDefiningFlag(
93  flagname, default='<unknown>')
94  flag_summary = flag_values[flagname].help
95  msg = ("The flag '%s' is defined twice. First from %s, Second from %s. "
96  "Description from first occurrence: %s") % (
97  flagname, first_module, second_module, flag_summary)
98  return cls(msg)
