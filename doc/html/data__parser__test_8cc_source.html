57  static void WriteBinaryFileWithList(const string& binary_file,
58  const vector<string>& launch_info) {
59  ofstream binary_file_stream(binary_file, ios::out | ios::binary);
60 
61  int64_t data_size = 0;

63  binary_file_stream << entry;
64  binary_file_stream << '\0';

100  string GetLaunchInfo(const string& key) const {
101  auto item = parsed_launch_info->find(key);
102  if (item == parsed_launch_info->end()) {
103  return "Cannot find key: " + key;
104  }
105  return blaze_util::WstringToCstring(item->second);

112 TEST_F(LaunchDataParserTest, GetLaunchInfoTest) {
