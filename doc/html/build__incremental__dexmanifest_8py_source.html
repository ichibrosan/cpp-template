58  def Checksum(self, filename):
59  """Compute the SHA-256 checksum of a file."""
60  h = hashlib.sha256()
61  with open(filename, "rb") as f:
62  while True:
63  data = f.read(65536)
64  if not data:
65  break
66 
67  h.update(data)

71  def AddDex(self, input_dex_or_zip, zippath, dex):
72  """Adds a dex file to the output.
73 
74  Args:
75  input_dex_or_zip: the input file written to the manifest
76  zippath: the zip path written to the manifest or None if the input file

78  dex: the dex file to be added

84  fs_checksum = self.Checksum(dex)

94  def Run(self, argv):
95  """Creates a dex manifest."""
96  if len(argv) < 1:
97  raise Exception("At least one argument expected")
98 
99  if argv[0][0] == "@":
100  if len(argv) != 1:
101  raise IOError("A parameter file should be the only argument")
102  with open(argv[0][1:]) as param_file:
103  argv = [a.strip() for a in param_file.readlines()]
104 
105  for input_filename in argv[1:]:
106  input_filename = input_filename.strip()
107  if input_filename.endswith(".zip"):
108  with zipfile.ZipFile(input_filename, "r") as input_dex_zip:
109  input_dex_dir = os.path.join(self.tmpdir, str(self.dir_counter))
110  os.makedirs(input_dex_dir)
111  self.dir_counter += 1
112 
113  for input_dex_dex in input_dex_zip.namelist():
114  if not input_dex_dex.endswith(".dex"):
115  continue
116 
117  input_dex_zip.extract(input_dex_dex, input_dex_dir)
118  fs_dex = input_dex_dir + "/" + input_dex_dex
119  self.AddDex(input_filename, input_dex_dex, fs_dex)
120  elif input_filename.endswith(".dex"):
121  self.AddDex(input_filename, None, input_filename)
122 
123  with open(argv[0], "wb") as manifest:
