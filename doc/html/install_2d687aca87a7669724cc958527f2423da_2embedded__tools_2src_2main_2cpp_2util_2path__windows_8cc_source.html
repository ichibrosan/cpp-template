94 std::string MakeAbsoluteAndResolveEnvvars(const std::string& path) {
95  // Get the size of the expanded string, so we know how big of a buffer to
96  // provide. The returned size includes the null terminator.
97  std::unique_ptr<CHAR[]> resolved(new CHAR[MAX_PATH]);
98  DWORD size =
99  ::ExpandEnvironmentStrings(path.c_str(), resolved.get(), MAX_PATH);
100  if (size == 0) {
101  BAZEL_DIE(blaze_exit_code::LOCAL_ENVIRONMENTAL_ERROR)
102  << "MakeAbsoluteAndResolveWindowsEnvvars(" << path
103  << "): ExpandEnvironmentStrings failed: " << GetLastErrorString();
104  } else if (size > MAX_PATH) {
105  // Try again with a buffer bigger than MAX_PATH.
106  resolved.reset(new CHAR[size]);
107  DWORD second_size =
108  ::ExpandEnvironmentStrings(path.c_str(), resolved.get(), size);
109  if (second_size == 0) {
110  BAZEL_DIE(blaze_exit_code::LOCAL_ENVIRONMENTAL_ERROR)
111  << "MakeAbsoluteAndResolveWindowsEnvvars(" << path
112  << "): ExpandEnvironmentStrings failed with second buffer: "
113  << GetLastErrorString();
114  }
115  assert(second_size <= size);
116  }
117  return MakeAbsolute(std::string(resolved.get()));
118 }

124 std::string PathAsJvmFlag(const std::string& path) {
125  std::string cpath;
126  std::string error;
127  if (!AsWindowsPath(path, &cpath, &error)) {
128  BAZEL_DIE(blaze_exit_code::LOCAL_ENVIRONMENTAL_ERROR)
129  << "PathAsJvmFlag(" << path << "): AsWindowsPath failed: " << error;
130  }
131  // Convert forward slashes and backslashes to double (escaped) backslashes, so
132  // they are safe to pass on the command line to the JVM and the JVM won't
133  // misinterpret them.
134  // See https://github.com/bazelbuild/bazel/issues/2576 and
135  // https://github.com/bazelbuild/bazel/issues/6098
136  size_t separators = 0;
137  for (const auto& c : cpath) {
138  if (c == '/' || c == '\\') {
139  separators++;
140  }
141  }
142  // In the result we replace each '/' and '\' with "\\", i.e. the total size
143  // *increases* by `separators`.
144  // Create a string of that size, filled with zeroes.
145  std::string result(/* count */ cpath.size() + separators, '\0');
146  std::string::size_type i = 0;
147  for (const auto& c : cpath) {
148  if (c == '/' || c == '\\') {
149  result[i++] = '\\';
150  result[i++] = '\\';
151  } else {
152  result[i++] = c;
153  }
154  }
155  return result;
156 }

168 // Checks if the path is absolute and/or is a root path.
169 //
170 // If `must_be_root` is true, then in addition to being absolute, the path must
171 // also be just the root part, no other components, e.g. "c:\" is both absolute
172 // and root, but "c:\foo" is just absolute.
173 template <typename char_type>
174 static bool IsRootOrAbsolute(const std::basic_string<char_type>& path,
175  bool must_be_root) {
176  // An absolute path is one that starts with "/", "\", "c:/", "c:\",
177  // "\\?\c:\", or rarely "\??\c:\" or "\\.\c:\".
178  //
179  // It is unclear whether the UNC prefix is just "\\?\" or is "\??\" also
180  // valid (in some cases it seems to be, though MSDN doesn't mention it).
181  return
182  // path is (or starts with) "/" or "\"
183  ((must_be_root ? path.size() == 1 : !path.empty()) &&
184  IsPathSeparator(path[0])) ||
185  // path is (or starts with) "c:/" or "c:\" or similar
186  ((must_be_root ? path.size() == 3 : path.size() >= 3) &&
187  HasDriveSpecifierPrefix(path.c_str()) && IsPathSeparator(path[2])) ||
188  // path is (or starts with) "\\?\c:\" or "\??\c:\" or similar
189  ((must_be_root ? path.size() == 7 : path.size() >= 7) &&
190  HasUncPrefix(path.c_str()) &&
191  HasDriveSpecifierPrefix(path.c_str() + 4) && IsPathSeparator(path[6]));
192 }

194 template <typename char_type>
195 static std::pair<std::basic_string<char_type>, std::basic_string<char_type> >
196 SplitPathImpl(const std::basic_string<char_type>& path) {
197  if (path.empty()) {
198  return std::make_pair(std::basic_string<char_type>(),
199  std::basic_string<char_type>());
200  }
201 
202  size_t pos = path.size() - 1;
203  for (auto it = path.crbegin(); it != path.crend(); ++it, --pos) {
204  if (IsPathSeparator(*it)) {
205  if ((pos == 2 || pos == 6) &&
206  IsRootOrAbsolute(path.substr(0, pos + 1), /* must_be_root */ true)) {
207  // Windows path, top-level directory, e.g. "c:\foo",
208  // result is ("c:\", "foo").
209  // Or UNC path, top-level directory, e.g. "\\?\c:\foo"
210  // result is ("\\?\c:\", "foo").
211  return std::make_pair(
212  // Include the "/" or "\" in the drive specifier.
213  path.substr(0, pos + 1), path.substr(pos + 1));
214  } else {
215  // Windows path (neither top-level nor drive root), Unix path, or
216  // relative path.
217  return std::make_pair(
218  // If the only "/" is the leading one, then that shall be the first
219  // pair element, otherwise the substring up to the rightmost "/".
220  pos == 0 ? path.substr(0, 1) : path.substr(0, pos),
221  // If the rightmost "/" is the tail, then the second pair element
222  // should be empty.
223  pos == path.size() - 1 ? std::basic_string<char_type>()
224  : path.substr(pos + 1));
225  }
226  }
227  }
228  // Handle the case with no '/' or '\' in `path`.
229  return std::make_pair(std::basic_string<char_type>(), path);
230 }

244 template <typename char_type>
245 static bool AsWindowsPathImpl(const std::basic_string<char_type>& path,
246  std::basic_string<char_type>* result,
247  std::string* error) {
248  if (path.empty()) {
249  result->clear();
250  return true;
251  }
252  if (IsDevNull(path.c_str())) {
253  assignNUL(result);
254  return true;
255  }
256  if (HasUncPrefix(path.c_str())) {
257  // Path has "\\?\" prefix --> assume it's already Windows-style.
258  *result = path.c_str();
259  return true;
260  }
261  if (IsPathSeparator(path[0]) && path.size() > 1 && IsPathSeparator(path[1])) {
262  // Unsupported path: "\\" or "\\server\path", or some degenerate form of
263  // these, such as "//foo".
264  if (error) {
265  *error = "network paths are unsupported";
266  }
267  return false;
268  }
269  if (HasDriveSpecifierPrefix(path.c_str()) &&
270  (path.size() < 3 || !IsPathSeparator(path[2]))) {
271  // Unsupported path: "c:" or "c:foo"
272  if (error) {
273  *error = "working-directory relative paths are unsupported";
274  }
275  return false;
276  }
277 
278  std::basic_string<char_type> mutable_path = path;
279  if (path[0] == '/') {
280  if (error) {
281  *error = "Unix-style paths are unsupported";
282  }
283  return false;
284  }
285 
286  if (path[0] == '\\') {
287  // This is an absolute Windows path on the current drive, e.g. "\foo\bar".
288  std::basic_string<char_type> drive(1, GetCurrentDrive());
289  drive.push_back(':');
290  mutable_path = drive + path;
291  } // otherwise this is a relative path, or absolute Windows path.
292 
293  *result = bazel::windows::Normalize(mutable_path);
294  return true;
295 }

312 static bool AsAbsoluteWindowsPathImpl(const std::wstring& path,
313  std::wstring* result,
314  std::string* error) {
315  if (path.empty()) {
316  result->clear();
317  return true;
318  }

320  result->assign(L"NUL");

324  return false;

332  }

335  return true;

359 bool AsShortWindowsPath(const std::wstring& path, std::wstring* result,
360  std::string* error) {
361  if (IsDevNull(path.c_str())) {
362  *result = L"NUL";
363  return true;
364  }
365 
366  result->clear();
367  std::wstring wpath;
368  std::wstring wsuffix;
369  if (!AsAbsoluteWindowsPath(path, &wpath, error)) {
370  return false;
371  }
372  DWORD size = ::GetShortPathNameW(wpath.c_str(), nullptr, 0);
373  if (size == 0) {
374  // GetShortPathNameW can fail if `wpath` does not exist. This is expected
375  // when we are about to create a file at that path, so instead of failing,
376  // walk up in the path until we find a prefix that exists and can be
377  // shortened, or is a root directory. Save the non-existent tail in
378  // `wsuffix`, we'll add it back later.
379  std::vector<std::wstring> segments;
380  while (size == 0 && !IsRootDirectoryW(wpath)) {
381  std::pair<std::wstring, std::wstring> split = SplitPathW(wpath);
382  wpath = split.first;
383  segments.push_back(split.second);
384  size = ::GetShortPathNameW(wpath.c_str(), nullptr, 0);
385  }
386 
387  // Join all segments.
388  std::wostringstream builder;
389  bool first = true;
390  for (auto it = segments.crbegin(); it != segments.crend(); ++it) {
391  if (!first || !IsRootDirectoryW(wpath)) {
392  builder << L'\\' << *it;
393  } else {
394  builder << *it;
395  }
396  first = false;
397  }
398  wsuffix = builder.str();
399  }
400 
401  std::wstring wresult;
402  if (IsRootDirectoryW(wpath)) {
403  // Strip the UNC prefix from `wpath`, and the leading "\" from `wsuffix`.
404  wresult = std::wstring(RemoveUncPrefixMaybe(wpath.c_str())) + wsuffix;
405  } else {
406  std::unique_ptr<WCHAR[]> wshort(
407  new WCHAR[size]); // size includes null-terminator
408  if (size - 1 != ::GetShortPathNameW(wpath.c_str(), wshort.get(), size)) {
409  if (error) {
410  std::string last_error = GetLastErrorString();
411  std::stringstream msg;
412  msg << "AsShortWindowsPath(" << WstringToCstring(path)
413  << "): GetShortPathNameW(" << WstringToCstring(wpath)
414  << ") failed: " << last_error;
415  *error = msg.str();
416  }
417  return false;
418  }
419  // GetShortPathNameW may preserve the UNC prefix in the result, so strip it.
420  wresult = std::wstring(RemoveUncPrefixMaybe(wshort.get())) + wsuffix;
421  }
422 
423  std::transform(wresult.begin(), wresult.end(), wresult.begin(), towlower);
424  *result = wresult;
425  return true;
426 }

464 static char GetCurrentDrive() {
465  std::wstring cwd = GetCwdW();
466  wchar_t wdrive = RemoveUncPrefixMaybe(cwd.c_str())[0];
467  wchar_t offset = wdrive >= L'A' && wdrive <= L'Z' ? L'A' : L'a';
468  return 'a' + wdrive - offset;
