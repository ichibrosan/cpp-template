65 def _escape_cdata(s):
66  """Escapes a string to be used as XML CDATA.
67 
68  CDATA characters are treated strictly as character data, not as XML markup,
69  but there are still certain restrictions on them.

92 class _TestCaseResult(object):
93  """Private helper for _TextAndXMLTestResult that represents a test result.
94 
95  Attributes:
96  test: A TestCase instance of an individual test method.
97  name: The name of the individual test method.
98  full_class_name: The full name of the test class.
99  run_time: The duration (in seconds) it took to run the test.
100  errors: A list of error 4-tuples. Error tuple entries are
101  1) a string identifier of either "failure" or "error"
102  2) an exception_type
103  3) an exception_message
104  4) a string version of a sys.exc_info()-style tuple of values
105  ('error', err[0], err[1], self._exc_info_to_string(err))
106  If the length of errors is 0, then the test is either passed or
107  skipped.
108  skip_reason: A string explaining why the test was skipped.
109  """

111  def __init__(self, test):

113  self.skip_reason = None
114  self.errors = []

117  # Parse the test id to get its test name and full class path.
118  # Unfortunately there is no better way of knowning the test and class.
119  # Worse, unittest uses _ErrorHandler instances to represent class / module
120  # level failures.
121  test_desc = test.id() or str(test)
122  # Check if it's something like "setUpClass (__main__.TestCase)".
123  match = _CLASS_OR_MODULE_LEVEL_TEST_DESC_REGEX.match(test_desc)
124  if match:
125  name = match.group(1)
126  full_class_name = match.group(2)
127  else:
128  class_name = unittest.util.strclass(test.__class__)
129  if test_desc.startswith(class_name + '.'):
130  # In a typical unittest.TestCase scenario, test.id() returns with
131  # a class name formatted using unittest.util.strclass.
132  name = test_desc[len(class_name)+1:]
133  full_class_name = class_name
134  else:
135  # Otherwise make a best effort to guess the test name and full class
136  # path.
137  parts = test_desc.rsplit('.', 1)
138  name = parts[-1]
139  full_class_name = parts[0] if len(parts) == 2 else ''
140  self.name = _escape_xml_attr(name)
141  self.full_class_name = _escape_xml_attr(full_class_name)

146  def print_xml_summary(self, stream):
147  """Prints an XML Summary of a TestCase.
148 
149  Status and result are populated as per JUnit XML test result reporter.
150  A test that has been skipped will always have a skip reason,
151  as every skip method in Python's unittest requires the reason arg to be
152  passed.
153 
154  Args:
155  stream: output stream to write test report XML to
156  """
157 
158  if self.skip_reason is None:
159  status = 'run'
160  result = 'completed'
161  else:
162  status = 'notrun'
163  result = 'suppressed'
164 
165  stream.write(
166  ' <testcase name="%s" status="%s" result="%s" time="%.1f" '
167  'classname="%s">\n' % (
168  self.name, status, result, self.run_time, self.full_class_name))
169  self._print_testcase_details(stream)
170  stream.write(' </testcase>\n')

172  def _print_testcase_details(self, stream):
173  for error in self.errors:
174  outcome, exception_type, message, error_msg = error # pylint: disable=unpacking-non-sequence
175  message = _escape_xml_attr(_safe_str(message))
176  exception_type = _escape_xml_attr(str(exception_type))
177  error_msg = _escape_cdata(error_msg)
178  stream.write(' <%s message="%s" type="%s"><![CDATA[%s]]></%s>\n'
179  % (outcome, message, exception_type, error_msg, outcome))

182 class _TestSuiteResult(object):
183  """Private helper for _TextAndXMLTestResult."""
184 
185  def __init__(self):
186  self.suites = {}
187  self.failure_counts = {}
188  self.error_counts = {}

190  def add_test_case_result(self, test_case_result):
191  suite_name = type(test_case_result.test).__name__
192  if suite_name == '_ErrorHolder':
193  # _ErrorHolder is a special case created by unittest for class / module
194  # level functions.
195  suite_name = test_case_result.full_class_name.rsplit('.')[-1]
196  self._setup_test_suite(suite_name)
197  self.suites[suite_name].append(test_case_result)
198  for error in test_case_result.errors:
199  # Only count the first failure or error so that the sum is equal to the
200  # total number of *testcases* that have failures or errors.
201  if error[0] == 'failure':
202  self.failure_counts[suite_name] += 1
203  break
204  elif error[0] == 'error':
205  self.error_counts[suite_name] += 1
206  break
207 
208  def print_xml_summary(self, stream):
209  overall_test_count = sum([len(x) for x in self.suites.values()])
210  overall_failures = sum(self.failure_counts.values())
211  overall_errors = sum(self.error_counts.values())
212  overall_time = 0
213  for tests in self.suites.values():
214  overall_time += sum([x.run_time for x in tests])
215  overall_args = (overall_test_count, overall_failures, overall_errors,
216  overall_time)
217  stream.write('<testsuites name="" tests="%d" failures="%d" '
218  'errors="%d" time="%.1f">\n' % overall_args)
219  for suite_name in self.suites:
220  suite = self.suites[suite_name]
221  suite_time = sum([x.run_time for x in suite])
222  failures = self.failure_counts[suite_name]
223  errors = self.error_counts[suite_name]
224  args = (suite_name, len(suite), failures, errors, suite_time)
225  stream.write('<testsuite name="%s" tests="%d" failures="%d" '
226  'errors="%d" time="%.1f">\n' % args)
227  for test_case_result in suite:
228  test_case_result.print_xml_summary(stream)
229  stream.write('</testsuite>\n')
230  stream.write('</testsuites>\n')

284  def stopTestRun(self):
285  # All pending_test_case_results will be added to the suite and removed from
286  # the pending_test_case_results dictionary. Grabing the write lock to avoid
287  # results from being added during this process to avoid duplicating adds or
288  # accidentally erasing newly appended pending results.
289  with self._pending_test_case_results_lock:
290  # Errors in the test fixture (setUpModule, tearDownModule,
291  # setUpClass, tearDownClass) can leave a pending result which
292  # never gets added to the suite. The runner calls stopTestRun
293  # which gives us an opportunity to add these errors for
294  # reporting here.
295  for test_id in self.pending_test_case_results:
296  result = self.pending_test_case_results[test_id]
297  if hasattr(self, 'start_time'):
298  run_time = self.time_getter() - self.start_time
299  result.set_run_time(run_time)
300  self.suite.add_test_case_result(result)
301  self.pending_test_case_results.clear()

320  def add_pending_test_case_result(self, test, error_summary=None,
321  skip_reason=None):
322  """Adds result information to a test case result which may still be running.
323 
324  If a result entry for the test already exists, add_pending_test_case_result
325  will add error summary tuples and/or overwrite skip_reason for the result.
326  If it does not yet exist, a result entry will be created.
327  Note that a test result is considered to have been run and passed
328  only if there are no errors or skip_reason.
329 
330  Args:
331  test: A test method as defined by unittest
332  error_summary: A 4-tuple with the following entries:
333  1) a string identifier of either "failure" or "error"
334  2) an exception_type
335  3) an exception_message
336  4) a string version of a sys.exc_info()-style tuple of values
337  ('error', err[0], err[1], self._exc_info_to_string(err))
338  If the length of errors is 0, then the test is either passed or
339  skipped.
340  skip_reason: a string explaining why the test was skipped
341  """
342  with self._pending_test_case_results_lock:
343  test_id = id(test)
344  if test_id not in self.pending_test_case_results:
345  self.pending_test_case_results[test_id] = self._TEST_CASE_RESULT_CLASS(
346  test)
347  if error_summary:
348  self.pending_test_case_results[test_id].errors.append(error_summary)
349  if skip_reason:
350  self.pending_test_case_results[test_id].skip_reason = skip_reason
351 
352  def delete_pending_test_case_result(self, test):
353  with self._pending_test_case_results_lock:
354  test_id = id(test)
355  del self.pending_test_case_results[test_id]
