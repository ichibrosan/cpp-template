33 bool HasUncPrefix(const char_type* path) {
34  // Return true iff `path` starts with "\\?\", "\\.\", or "\??\".
35  return path[0] == '\\' &&
36  ((path[1] == '\\' && (path[2] == '?' || path[2] == '.')) ||
37  (path[1] == '?' && path[2] == '?')) &&
38  path[3] == '\\';
39 }

160 // Returns a normalized form of the input `path`.
161 //
162 // Normalization:
163 // Normalization means removing "." references, resolving ".." references,
164 // and deduplicating "/" characters while converting them to "\\". For
165 // example if `path` is "foo/../bar/.//qux", the result is "bar\\qux".
166 //
167 // Uplevel references ("..") that cannot go any higher in the directory tree
168 // are preserved if the path is relative, and ignored if the path is
169 // absolute, e.g. "../../foo" is normalized to "..\\..\\foo" but "c:/.." is
170 // normalized to "c:\\".
171 //
172 // This method does not check the semantics of the `path` beyond checking if
173 // it starts with a directory separator. Illegal paths such as one with a
174 // drive specifier in the middle (e.g. "foo/c:/bar") are accepted -- it's the
175 // caller's responsibility to pass a path that, when normalized, will be
176 // semantically correct.
177 //
178 // Current directory references (".") are preserved if and only if they are
179 // the only path segment, so "./" becomes "." but "./foo" becomes "foo".
180 //
181 // Arguments:
182 // `path` must be a relative or absolute Windows path, it may use "/" instead
183 // of "\\". The path should not start with "/" or "\\".
184 //
185 // Result:
186 // Returns false if and only if the path starts with a directory separator.
187 //
188 // The result won't have a UNC prefix, even if `path` did. The result won't
189 // have a trailing "\\" except when and only when the path is normalized to
190 // just a drive specifier (e.g. when `path` is "c:/" or "c:/foo/.."). The
191 // result will preserve the casing of the input, so "D:/Bar" becomes
192 // "D:\\Bar".
193 std::string Normalize(const std::string& p);
