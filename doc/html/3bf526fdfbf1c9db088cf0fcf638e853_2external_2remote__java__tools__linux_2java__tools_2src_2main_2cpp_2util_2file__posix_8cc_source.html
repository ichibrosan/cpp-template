42 // Runs "stat" on `path`. Returns -1 and sets errno if stat fails or
43 // `path` isn't a directory. If check_perms is true, this will also
44 // make sure that `path` is owned by the current user and has `mode`
45 // permissions (observing the umask). It attempts to run chmod to
46 // correct the mode if necessary. If `path` is a symlink, this will
47 // check ownership of the link, not the underlying directory.
48 static bool GetDirectoryStat(const string &path, mode_t mode,
49  bool check_perms) {
50  struct stat filestat = {};
51  if (stat(path.c_str(), &filestat) == -1) {
52  return false;
53  }
54 
55  if (!S_ISDIR(filestat.st_mode)) {
56  errno = ENOTDIR;
57  return false;
58  }
59 
60  if (check_perms) {
61  // If this is a symlink, run checks on the link. (If we did lstat above
62  // then it would return false for ISDIR).
63  struct stat linkstat = {};
64  if (lstat(path.c_str(), &linkstat) != 0) {
65  return false;
66  }
67  if (linkstat.st_uid != geteuid()) {
68  // The directory isn't owned by me.
69  errno = EACCES;
70  return false;
71  }
72 
73  mode_t mask = umask(022);
74  umask(mask);
75  mode = (mode & ~mask);
76  if ((filestat.st_mode & 0777) != mode && chmod(path.c_str(), mode) == -1) {
77  // errno set by chmod.
78  return false;
79  }
80  }
81  return true;
82 }

84 static bool MakeDirectories(const string &path, mode_t mode, bool childmost) {
85  if (path.empty() || IsRootDirectory(path)) {
86  errno = EACCES;
87  return false;
88  }
89 
90  bool stat_succeeded = GetDirectoryStat(path, mode, childmost);
91  if (stat_succeeded) {
92  return true;
93  }
94 
95  if (errno == ENOENT) {
96  // Path does not exist, attempt to create its parents, then it.
97  string parent = Dirname(path);
98  if (!MakeDirectories(parent, mode, false)) {
99  // errno set by stat.
100  return false;
101  }
102 
103  if (mkdir(path.c_str(), mode) == -1) {
104  if (errno == EEXIST) {
105  if (childmost) {
106  // If there are multiple bazel calls at the same time then the
107  // directory could be created between the MakeDirectories and mkdir
108  // calls. This is okay, but we still have to check the permissions.
109  return GetDirectoryStat(path, mode, childmost);
110  } else {
111  // If this isn't the childmost directory, we don't care what the
112  // permissions were. If it's not even a directory then that error will
113  // get caught when we attempt to create the next directory down the
114  // chain.
115  return true;
116  }
117  }
118  // errno set by mkdir.
119  return false;
120  }
121  return true;
122  }
123 
124  return stat_succeeded;
125 }

454 void ForEachDirectoryEntry(const string &path,
455  DirectoryEntryConsumer *consume) {
456  DIR *dir;
457  struct dirent *ent;
458 
459  if ((dir = opendir(path.c_str())) == NULL) {
460  // This is not a directory or it cannot be opened.
461  return;
462  }
463 
464  while ((ent = readdir(dir)) != NULL) {
465  if (!strcmp(ent->d_name, ".") || !strcmp(ent->d_name, "..")) {
466  continue;
467  }
468 
469  string filename(blaze_util::JoinPath(path, ent->d_name));
470  bool is_directory;
471 // 'd_type' field isn't part of the POSIX spec.
472 #ifdef _DIRENT_HAVE_D_TYPE
473  if (ent->d_type != DT_UNKNOWN) {
474  is_directory = (ent->d_type == DT_DIR);
475  } else // NOLINT (the brace is on the next line)
476 #endif
477  {
478  struct stat buf;
479  if (lstat(filename.c_str(), &buf) == -1) {
480  BAZEL_DIE(blaze_exit_code::INTERNAL_ERROR)
481  << "stat failed for filename '" << filename
482  << "': " << GetLastErrorString();
483  }
484  is_directory = S_ISDIR(buf.st_mode);
485  }
486 
487  consume->Consume(filename, is_directory);
488  }
489 
490  closedir(dir);
491  }
