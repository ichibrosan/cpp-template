52 def Cd(newdir, cleanup=lambda: True):
53  """Change the current working directory.
54 
55  This will run the provided cleanup function when the context exits and the
56  previous working directory is restored.
57 
58  Args:
59  newdir: The directory to change to. This must already exist.
60  cleanup: An optional cleanup function to be executed when the context exits.
61 
62  Yields:
63  Nothing.
64  """
65 
66  prevdir = os.getcwd()

70  finally:

116 def CopyAndRewrite(input_file, output_file, replacements=None):
117  """Copies the given file and optionally rewrites with replacements.
118 
119  Args:
120  input_file: The file to copy.
121  output_file: The file to write to.
122  replacements: A dictionary of replacements.
123  Keys are prefixes scan for, values are the replacements to write after
124  the prefix.
125  """
126  with open(output_file, 'w') as output:
127  for line in fileinput.input(input_file):
128  if replacements:
129  for prefix, text in replacements.items():
130  if six.ensure_str(line).startswith(prefix):
131  line = prefix + ' ' + six.ensure_str(text) + '\n'
132  break
133  output.write(six.ensure_str(line))

195  def AddFiles(self, paths, root=''):
196  """Add a set of files to the current RPM.
197 
198  If an item in paths is a directory, its files are recursively added.
199 
200  Args:
201  paths: The files to add.
202  root: The root of the filesystem to search for files. Defaults to ''.
203  """
204  for path in paths:
205  full_path = os.path.join(root, path)
206  if os.path.isdir(full_path):
207  self.AddFiles(os.listdir(full_path), full_path)
208  else:
209  self.files.append(full_path)

211  def SetupWorkdir(self, spec_file, original_dir):
212  """Create the needed structure in the workdir."""
213 
214  # Create directory structure.
215  for name in RpmBuilder.DIRS:
216  if not os.path.exists(name):
217  os.makedirs(name, 0o777)
218 
219  # Copy the files.
220  for f in self.files:
221  dst_dir = os.path.join(RpmBuilder.BUILD_DIR, os.path.dirname(f))
222  if not os.path.exists(dst_dir):
223  os.makedirs(dst_dir, 0o777)
224  shutil.copy(os.path.join(original_dir, f), dst_dir)
225 
226  # Copy the spec file, updating with the correct version.
227  spec_origin = os.path.join(original_dir, spec_file)
228  self.spec_file = os.path.basename(spec_file)
229  replacements = {}
230  if self.version:
231  replacements['Version:'] = self.version
232  if self.release:
233  replacements['Release:'] = self.release
234  CopyAndRewrite(spec_origin, self.spec_file, replacements)

236  def CallRpmBuild(self, dirname):
237  """Call rpmbuild with the correct arguments."""
238 
239  args = [
240  self.rpmbuild_path,
241  '--define',
242  '_topdir %s' % dirname,
243  '--define',
244  '_tmppath %s/TMP' % dirname,
245  '--bb',
246  '--buildroot',
247  os.path.join(dirname, 'BUILDROOT'),
248  self.spec_file,
249  ]
250  p = subprocess.Popen(
251  args,
252  stdout=subprocess.PIPE,
253  stderr=subprocess.STDOUT,
254  env={'LANG': 'C'})
255  output = p.communicate()[0].decode()
256 
257  if p.returncode == 0:
258  # Find the created file.
259  self.rpm_path = FindOutputFile(output)
260 
261  if p.returncode != 0 or not self.rpm_path:
262  print('Error calling rpmbuild:')
263  print(output)
264

278  def Build(self, spec_file, out_file):
279  """Build the RPM described by the spec_file."""
