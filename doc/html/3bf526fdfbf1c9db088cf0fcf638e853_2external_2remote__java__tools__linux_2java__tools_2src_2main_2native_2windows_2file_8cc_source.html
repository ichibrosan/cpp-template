48 bool IsAbsoluteNormalizedWindowsPath(const wstring& p) {
49  if (p.empty()) {
50  return false;
51  }
52  if (IsDevNull(p.c_str())) {
53  return true;
54  }
55  if (p.find_first_of('/') != wstring::npos) {
56  return false;
57  }
58 
59  return HasDriveSpecifierPrefix(p.c_str()) && p.find(L".\\") != 0 &&
60  p.find(L"\\.\\") == wstring::npos && p.find(L"\\.") != p.size() - 2 &&
61  p.find(L"..\\") != 0 && p.find(L"\\..\\") == wstring::npos &&
62  p.find(L"\\..") != p.size() - 3;
63 }

65 static wstring uint32asHexString(uint32_t value) {
66  WCHAR attr_str[8];
67  for (int i = 0; i < 8; ++i) {
68  attr_str[7 - i] = L"0123456789abcdef"[value & 0xF];
69  value >>= 4;
70  }
71  return wstring(attr_str, 8);
72 }

149 int CreateJunction(const wstring& junction_name, const wstring& junction_target,
150  wstring* error) {
151  if (!IsAbsoluteNormalizedWindowsPath(junction_name)) {
152  if (error) {
153  *error = MakeErrorMessage(
154  WSTR(__FILE__), __LINE__, L"CreateJunction", junction_name,
155  L"expected an absolute Windows path for junction_name");
156  }
157  return CreateJunctionResult::kError;
158  }
159  if (!IsAbsoluteNormalizedWindowsPath(junction_target)) {
160  if (error) {
161  *error = MakeErrorMessage(
162  WSTR(__FILE__), __LINE__, L"CreateJunction", junction_target,
163  L"expected an absolute Windows path for junction_target");
164  }
165  return CreateJunctionResult::kError;
166  }
167 
168  const WCHAR* target = HasUncPrefix(junction_target.c_str())
169  ? junction_target.c_str() + 4
170  : junction_target.c_str();
171  const size_t target_size = HasUncPrefix(junction_target.c_str())
172  ? junction_target.size() - 4
173  : junction_target.size();
174  // The entire REPARSE_DATA_BUFFER cannot be larger than
175  // MAXIMUM_REPARSE_DATA_BUFFER_SIZE bytes.
176  //
177  // The structure's layout is:
178  // [8 bytes] : ReparseTag, ReparseDataLength, Reserved
179  // [8 bytes] : MountPointReparseBuffer members before PathBuffer
180  // ---- start of MountPointReparseBuffer.PathBuffer ----
181  // [4 WCHARs] : "\??\" prefix
182  // [target.size() WCHARs] : junction target name
183  // [1 WCHAR] : null-terminator
184  // [target.size() WCHARs] : junction target displayed name
185  // [1 WCHAR] : null-terminator
186  // The sum of these must not exceed MAXIMUM_REPARSE_DATA_BUFFER_SIZE.
187  // We can rearrange this to get the limit for target.size().
188  static const size_t kMaxJunctionTargetLen =
189  ((MAXIMUM_REPARSE_DATA_BUFFER_SIZE -
190  offsetof(REPARSE_DATA_BUFFER, MountPointReparseBuffer.PathBuffer)) /
191  sizeof(WCHAR) -
192  /* one "\??\" prefix */ 4 -
193  /* two null terminators */ 2) /
194  /* two copies of the string are stored */ 2;
195  if (target_size > kMaxJunctionTargetLen) {
196  if (error) {
197  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__, L"CreateJunction",
198  target, L"target path is too long");
199  }
200  return CreateJunctionResult::kTargetNameTooLong;
201  }
202  const wstring name = HasUncPrefix(junction_name.c_str())
203  ? junction_name
204  : (wstring(L"\\\\?\\") + junction_name);
205 
206  // Junctions are directories, so create a directory.
207  // If CreateDirectoryW succeeds, we'll try to set the junction's target.
208  // If CreateDirectoryW fails, we don't care about the exact reason -- could be
209  // that the directory already exists, or we have no access to create a
210  // directory, or the path was invalid to begin with. Either way set `create`
211  // to false, meaning we'll just attempt to open the path for metadata-reading
212  // and check if it's a junction pointing to the desired target.
213  bool create = CreateDirectoryW(name.c_str(), NULL) != 0;
214 
215  AutoHandle handle;
216  if (create) {
217  handle = CreateFileW(
218  name.c_str(), GENERIC_READ | GENERIC_WRITE, FILE_SHARE_READ, NULL,
219  OPEN_EXISTING,
220  FILE_FLAG_OPEN_REPARSE_POINT | FILE_FLAG_BACKUP_SEMANTICS, NULL);
221  }
222 
223  if (!handle.IsValid()) {
224  // We can't open the directory for writing: either we didn't even try to
225  // (`create` was false), or the path disappeared, or it turned into a file,
226  // or another process holds it open without write-sharing.
227  // Either way, don't try to create the junction, just try opening it without
228  // any read or write access (we can still read its metadata) and maximum
229  // sharing, and check its target.
230  create = false;
231  handle = CreateFileW(
232  name.c_str(), 0, FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE,
233  NULL, OPEN_EXISTING,
234  FILE_FLAG_OPEN_REPARSE_POINT | FILE_FLAG_BACKUP_SEMANTICS, NULL);
235  if (!handle.IsValid()) {
236  // We can't open the directory at all: either it disappeared, or it turned
237  // into a file, or the path is invalid, or another process holds it open
238  // without any sharing. Give up.
239  DWORD err = GetLastError();
240  if (err == ERROR_SHARING_VIOLATION) {
241  // The junction is held open by another process.
242  return CreateJunctionResult::kAccessDenied;
243  } else if (err == ERROR_FILE_NOT_FOUND || err == ERROR_PATH_NOT_FOUND) {
244  // Meanwhile the directory disappeared or one of its parent directories
245  // disappeared.
246  return CreateJunctionResult::kDisappeared;
247  }
248 
249  // The path seems to exist yet we cannot open it for metadata-reading.
250  // Report as much information as we have, then give up.
251  if (error) {
252  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__, L"CreateFileW",
253  name, err);
254  }
255  return CreateJunctionResult::kError;
256  }
257  }
258 
259  // We have an open handle to the file! It may still be other than a junction,
260  // so check its attributes.
261  BY_HANDLE_FILE_INFORMATION info;
262  if (!GetFileInformationByHandle(handle, &info)) {
263  DWORD err = GetLastError();
264  // Some unknown error occurred.
265  if (error) {
266  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
267  L"GetFileInformationByHandle", name, err);
268  }
269  return CreateJunctionResult::kError;
270  }
271 
272  if (info.dwFileAttributes == INVALID_FILE_ATTRIBUTES) {
273  DWORD err = GetLastError();
274  // Some unknown error occurred.
275  if (error) {
276  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
277  L"GetFileInformationByHandle", name, err);
278  }
279  return CreateJunctionResult::kError;
280  }
281 
282  if (info.dwFileAttributes & FILE_ATTRIBUTE_REPARSE_POINT) {
283  // The path already exists and it's a junction. Do not overwrite, just check
284  // its target.
285  create = false;
286  }
287 
288  if (create) {
289  if (!(info.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY)) {
290  // Even though we managed to create the directory and it didn't exist
291  // before, another process changed it in the meantime so it's no longer a
292  // directory.
293  create = false;
294  if (!(info.dwFileAttributes & FILE_ATTRIBUTE_REPARSE_POINT)) {
295  // The path is no longer a directory, and it's not a junction either.
296  // Though this is a case for kAlreadyExistsButNotJunction, let's instead
297  // print the attributes and return kError, to give more information to
298  // the user.
299  if (error) {
300  *error = MakeErrorMessage(
301  WSTR(__FILE__), __LINE__, L"GetFileInformationByHandle", name,
302  wstring(L"attrs=0x") + uint32asHexString(info.dwFileAttributes));
303  }
304  return CreateJunctionResult::kError;
305  }
306  }
307  }
308 
309  if (!create) {
310  // The path already exists. Check if it's a junction.
311  if (!(info.dwFileAttributes & FILE_ATTRIBUTE_REPARSE_POINT)) {
312  return CreateJunctionResult::kAlreadyExistsButNotJunction;
313  }
314  }
315 
316  uint8_t reparse_buffer_bytes[MAXIMUM_REPARSE_DATA_BUFFER_SIZE];
317  PREPARSE_DATA_BUFFER reparse_buffer =
318  reinterpret_cast<PREPARSE_DATA_BUFFER>(reparse_buffer_bytes);
319  if (create) {
320  // The junction doesn't exist yet, and we have an open handle to the
321  // candidate directory with write access and no sharing. Proceed to turn the
322  // directory into a junction.
323 
324  memset(reparse_buffer_bytes, 0, MAXIMUM_REPARSE_DATA_BUFFER_SIZE);
325 
326  reparse_buffer->MountPointReparseBuffer.SubstituteNameOffset = 0;
327  reparse_buffer->MountPointReparseBuffer.SubstituteNameLength =
328  (4 + target_size) * sizeof(WCHAR);
329  reparse_buffer->MountPointReparseBuffer.PrintNameOffset =
330  reparse_buffer->MountPointReparseBuffer.SubstituteNameLength +
331  /* null-terminator */ sizeof(WCHAR);
332  reparse_buffer->MountPointReparseBuffer.PrintNameLength =
333  target_size * sizeof(WCHAR);
334 
335  reparse_buffer->ReparseTag = IO_REPARSE_TAG_MOUNT_POINT;
336  reparse_buffer->ReparseDataLength =
337  4 * sizeof(USHORT) +
338  reparse_buffer->MountPointReparseBuffer.SubstituteNameLength +
339  reparse_buffer->MountPointReparseBuffer.PrintNameLength +
340  /* 2 null-terminators */ (2 * sizeof(WCHAR));
341  reparse_buffer->Reserved = 0;
342 
343  // "\??\" is meaningful to the kernel, it's a synomym for the "\DosDevices\"
344  // object path. (NOT to be confused with "\\?\" which is meaningful for the
345  // Win32 API.) We need to use this prefix to tell the kernel where the
346  // reparse point is pointing to.
347  memcpy((uint8_t*)reparse_buffer->MountPointReparseBuffer.PathBuffer +
348  reparse_buffer->MountPointReparseBuffer.SubstituteNameOffset,
349  L"\\??\\", 4 * sizeof(WCHAR));
350  memcpy((uint8_t*)reparse_buffer->MountPointReparseBuffer.PathBuffer +
351  reparse_buffer->MountPointReparseBuffer.SubstituteNameOffset +
352  4 * sizeof(WCHAR),
353  target,
354  reparse_buffer->MountPointReparseBuffer.SubstituteNameLength -
355  4 * sizeof(WCHAR));
356 
357  // In addition to their target, junctions also have another string which is
358  // a user-visible name of where the junction points, as listed by "dir".
359  // This can be any string and won't affect the usability of the junction.
360  // MKLINK uses the target path without the "\??\" prefix as the display
361  // name, so let's do that here too. This is also in line with how UNIX
362  // behaves. Using a dummy or fake display name would be misleading, it would
363  // make the output of `dir` look like:
364  // 2017-01-18 01:37 PM <JUNCTION> juncname [dummy string]
365  memcpy((uint8_t*)reparse_buffer->MountPointReparseBuffer.PathBuffer +
366  reparse_buffer->MountPointReparseBuffer.PrintNameOffset,
367  target, reparse_buffer->MountPointReparseBuffer.PrintNameLength);
368 
369  DWORD bytes_returned;
370  if (!::DeviceIoControl(
371  handle, FSCTL_SET_REPARSE_POINT, reparse_buffer,
372  reparse_buffer->ReparseDataLength +
373  offsetof(REPARSE_DATA_BUFFER, GenericReparseBuffer.DataBuffer),
374  NULL, 0, &bytes_returned, NULL)) {
375  DWORD err = GetLastError();
376  if (err == ERROR_DIR_NOT_EMPTY) {
377  return CreateJunctionResult::kAlreadyExistsButNotJunction;
378  }
379  // Some unknown error occurred.
380  if (error) {
381  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__, L"DeviceIoControl",
382  name, err);
383  }
384  return CreateJunctionResult::kError;
385  }
386  } else {
387  // The junction already exists. Check if it points to the right target.
388 
389  DWORD bytes_returned;
390  if (!::DeviceIoControl(handle, FSCTL_GET_REPARSE_POINT, NULL, 0,
391  reparse_buffer, MAXIMUM_REPARSE_DATA_BUFFER_SIZE,
392  &bytes_returned, NULL)) {
393  DWORD err = GetLastError();
394  // Some unknown error occurred.
395  if (error) {
396  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__, L"DeviceIoControl",
397  name, err);
398  }
399  return CreateJunctionResult::kError;
400  }
401 
402  WCHAR* actual_target =
403  reparse_buffer->MountPointReparseBuffer.PathBuffer +
404  reparse_buffer->MountPointReparseBuffer.SubstituteNameOffset +
405  /* "\??\" prefix */ 4;
406  if (reparse_buffer->MountPointReparseBuffer.SubstituteNameLength !=
407  (/* "\??\" prefix */ 4 + target_size) * sizeof(WCHAR) ||
408  _wcsnicmp(actual_target, target, target_size) != 0) {
409  return CreateJunctionResult::kAlreadyExistsWithDifferentTarget;
410  }
411  }
412 
413  return CreateJunctionResult::kSuccess;
414 }

416 int ReadSymlinkOrJunction(const wstring& path, wstring* result,
417  wstring* error) {
418  if (!IsAbsoluteNormalizedWindowsPath(path)) {
419  if (error) {
420  *error = MakeErrorMessage(
421  WSTR(__FILE__), __LINE__, L"ReadSymlinkOrJunction", path,
422  L"expected an absolute Windows path for 'path'");
423  }
424  return ReadSymlinkOrJunctionResult::kError;
425  }
426 
427  AutoHandle handle(CreateFileW(
428  AddUncPrefixMaybe(path).c_str(), 0,
429  FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE, NULL,
430  OPEN_EXISTING, FILE_FLAG_OPEN_REPARSE_POINT | FILE_FLAG_BACKUP_SEMANTICS,
431  NULL));
432  if (!handle.IsValid()) {
433  DWORD err = GetLastError();
434  if (err == ERROR_SHARING_VIOLATION) {
435  // The path is held open by another process.
436  return ReadSymlinkOrJunctionResult::kAccessDenied;
437  } else if (err == ERROR_FILE_NOT_FOUND || err == ERROR_PATH_NOT_FOUND) {
438  // Path or a parent directory does not exist.
439  return ReadSymlinkOrJunctionResult::kDoesNotExist;
440  }
441 
442  // The path seems to exist yet we cannot open it for metadata-reading.
443  // Report as much information as we have, then give up.
444  if (error) {
445  *error =
446  MakeErrorMessage(WSTR(__FILE__), __LINE__, L"CreateFileW", path, err);
447  }
448  return ReadSymlinkOrJunctionResult::kError;
449  }
450 
451  uint8_t raw_buf[MAXIMUM_REPARSE_DATA_BUFFER_SIZE];
452  PREPARSE_DATA_BUFFER buf = reinterpret_cast<PREPARSE_DATA_BUFFER>(raw_buf);
453  DWORD bytes_returned;
454  if (!::DeviceIoControl(handle, FSCTL_GET_REPARSE_POINT, NULL, 0, buf,
455  MAXIMUM_REPARSE_DATA_BUFFER_SIZE, &bytes_returned,
456  NULL)) {
457  DWORD err = GetLastError();
458  if (err == ERROR_NOT_A_REPARSE_POINT) {
459  return ReadSymlinkOrJunctionResult::kNotALink;
460  }
461 
462  // Some unknown error occurred.
463  if (error) {
464  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__, L"DeviceIoControl",
465  path, err);
466  }
467  return ReadSymlinkOrJunctionResult::kError;
468  }
469 
470  switch (buf->ReparseTag) {
471  case IO_REPARSE_TAG_SYMLINK: {
472  wchar_t* p =
473  (wchar_t*)(((uint8_t*)buf->SymbolicLinkReparseBuffer.PathBuffer) +
474  buf->SymbolicLinkReparseBuffer.SubstituteNameOffset);
475  *result = wstring(p, buf->SymbolicLinkReparseBuffer.SubstituteNameLength /
476  sizeof(WCHAR));
477  return ReadSymlinkOrJunctionResult::kSuccess;
478  }
479  case IO_REPARSE_TAG_MOUNT_POINT: {
480  wchar_t* p =
481  (wchar_t*)(((uint8_t*)buf->MountPointReparseBuffer.PathBuffer) +
482  buf->MountPointReparseBuffer.SubstituteNameOffset);
483  *result = wstring(
484  p, buf->MountPointReparseBuffer.SubstituteNameLength / sizeof(WCHAR));
485  return ReadSymlinkOrJunctionResult::kSuccess;
486  }
487  default:
488  return ReadSymlinkOrJunctionResult::kUnknownLinkType;
489  }

501 // Check whether the directory and its child elements truly exist, or are marked
502 // for deletion. The result could be:
503 // 1. The give path doesn't exist
504 // 2. The directory is empty
505 // 3. The directory contains valid files or dirs, so not empty
506 // 4. The directory contains only files or dirs marked for deletion.
507 int CheckDirectoryStatus(const wstring& path) {
508  static const wstring kDot(L".");
509  static const wstring kDotDot(L"..");
510  bool found_valid_file = false;
511  bool found_child_marked_for_deletion = false;
512  WIN32_FIND_DATAW metadata;
513  HANDLE handle = ::FindFirstFileW((path + L"\\*").c_str(), &metadata);
514  if (handle == INVALID_HANDLE_VALUE) {
515  return DirectoryStatus::kDoesNotExist;
516  }
517  do {
518  if (kDot != metadata.cFileName && kDotDot != metadata.cFileName) {
519  std::wstring child = path + L"\\" + metadata.cFileName;
520  DWORD attributes = GetFileAttributesW(child.c_str());
521  if (attributes != INVALID_FILE_ATTRIBUTES) {
522  // If there is a valid file under the directory,
523  // then the directory is truely not empty.
524  // We should just return kDirectoryNotEmpty.
525  found_valid_file = true;
526  break;
527  } else {
528  DWORD error_code = GetLastError();
529  // If the file or directory is in deleting process,
530  // GetFileAttributesW returns ERROR_ACCESS_DENIED,
531  // If it's already deleted at the time we check,
532  // GetFileAttributesW returns ERROR_FILE_NOT_FOUND.
533  // If GetFileAttributesW fails with other reason, we consider there is a
534  // valid file that we cannot open, thus return kDirectoryNotEmpty
535  if (error_code != ERROR_ACCESS_DENIED &&
536  error_code != ERROR_FILE_NOT_FOUND) {
537  found_valid_file = true;
538  break;
539  } else if (error_code == ERROR_ACCESS_DENIED) {
540  found_child_marked_for_deletion = true;
541  }
542  }
543  }
544  } while (::FindNextFileW(handle, &metadata));
545  ::FindClose(handle);
546  if (found_valid_file) {
547  return DirectoryStatus::kDirectoryNotEmpty;
548  }
549  if (found_child_marked_for_deletion) {
550  return DirectoryStatus::kChildMarkedForDeletionExists;
551  }
552  return DirectoryStatus::kDirectoryEmpty;
553 }

573 int DeletePath(const wstring& path, wstring* error) {
574  if (!IsAbsoluteNormalizedWindowsPath(path)) {
575  if (error) {
576  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__, L"DeletePath", path,
577  L"expected an absolute Windows path");
578  }
579  return DeletePathResult::kError;
580  }
581 
582  const std::wstring winpath(AddUncPrefixMaybe(path));
583  const wchar_t* wpath = winpath.c_str();
584 
585  DWORD attr = GetFileAttributesW(wpath);
586  DWORD err;
587  if (attr == INVALID_FILE_ATTRIBUTES) {
588  return GetResultFromErrorCode(L"GetFileAttributesW", path,
589  GetLastError(), error);
590  }
591 
592  if (attr & FILE_ATTRIBUTE_DIRECTORY) {
593  // It's a directory or a junction, RemoveDirectoryW should be used.
594  //
595  // Sometimes a deleted directory lingers in its parent dir
596  // after the deleting handle has already been closed.
597  // In this case we check the content of the parent directory,
598  // if we don't find any valid file, we try to delete it again after 5 ms.
599  // But we don't want to hang infinitely because another application
600  // can hold the handle for a long time. So try at most 20 times,
601  // which means a process time of 100-120ms.
602  // Inspired by
603  // https://github.com/Alexpux/Cygwin/commit/28fa2a72f810670a0562ea061461552840f5eb70
604  // Useful link: https://stackoverflow.com/questions/31606978
605  int count;
606  for (count = 0; count < 20 && !RemoveDirectoryW(wpath); ++count) {
607  // Failed to delete the directory.
608  err = GetLastError();
609  if (err == ERROR_SHARING_VIOLATION || err == ERROR_ACCESS_DENIED) {
610  // The junction or directory is in use by another process, or we have
611  // no permission to delete it.
612  return DeletePathResult::kAccessDenied;
613  } else if (err == ERROR_FILE_NOT_FOUND || err == ERROR_PATH_NOT_FOUND) {
614  // The directory or one of its parent directories disappeared or is no
615  // longer a directory.
616  return DeletePathResult::kDoesNotExist;
617  } else if (err == ERROR_DIR_NOT_EMPTY) {
618  // We got ERROR_DIR_NOT_EMPTY error, but maybe the child files and
619  // dirs are already marked for deletion, let's check the status of the
620  // child elements to see if we should retry the delete operation.
621  switch (CheckDirectoryStatus(winpath)) {
622  case DirectoryStatus::kDirectoryNotEmpty:
623  // The directory is truely not empty.
624  return DeletePathResult::kDirectoryNotEmpty;
625  case DirectoryStatus::kDirectoryEmpty:
626  // If no children are pending deletion then the directory is now
627  // empty. We can try deleting it again without waiting.
628  continue;
629  case DirectoryStatus::kChildMarkedForDeletionExists:
630  // If all child elements are marked for deletion, then wait 5 ms for
631  // the system to delete the files and try deleting the directory
632  // again.
633  Sleep(5L);
634  continue;
635  case DirectoryStatus::kDoesNotExist:
636  // This case should never happen, because ERROR_DIR_NOT_EMPTY
637  // means the directory exists. But if it does happen, return an
638  // error message.
639  if (error) {
640  *error =
641  MakeErrorMessage(WSTR(__FILE__), __LINE__,
642  L"RemoveDirectoryW", path, GetLastError());
643  }
644  return DeletePathResult::kError;
645  }
646  }
647 
648  // Some unknown error occurred.
649  if (error) {
650  *error = MakeErrorMessage(WSTR(__FILE__), __LINE__,
651  L"RemoveDirectoryW", path, err);
652  }
653  return DeletePathResult::kError;
654  }
655 
656  if (count == 20) {
657  // After trying 20 times, the "deleted" sub-directories or files still
658  // won't go away, so just return kDirectoryNotEmpty error.
659  return DeletePathResult::kDirectoryNotEmpty;
660  }
661  } else {
662  // It's a regular file or symlink, DeleteFileW should be used.
663  if (attr & FILE_ATTRIBUTE_READONLY) {
664  // Remove the read-only attribute.
665  attr &= ~FILE_ATTRIBUTE_READONLY;
666  if (!SetFileAttributesW(wpath, attr)) {
667  return GetResultFromErrorCode(L"SetFileAttributesW", path,
668  GetLastError(), error);
669  }
670  }
671  if (!DeleteFileW(wpath)) {
672  // Failed to delete the file or symlink.
673  return GetResultFromErrorCode(L"DeleteFileW", path,
674  GetLastError(), error);
675  }
676  }
677 
678  return DeletePathResult::kSuccess;
679 }

681 template <typename C>
682 std::basic_string<C> NormalizeImpl(const std::basic_string<C>& p) {
683  if (p.empty()) {
684  return p;
685  }
686  typedef std::basic_string<C> Str;
687  static const Str kDot(1, '.');
688  static const Str kDotDot(2, '.');
689  std::vector<std::pair<typename Str::size_type, typename Str::size_type> >
690  segments;
691  typename Str::size_type seg_start = Str::npos;
692  bool first = true;
693  bool abs = false;
694  bool starts_with_dot = false;
695  for (typename Str::size_type i = HasUncPrefix(p.c_str()) ? 4 : 0;
696  i <= p.size(); ++i) {
697  if (seg_start == Str::npos) {
698  if (i < p.size() && p[i] != '/' && p[i] != '\\') {
699  seg_start = i;
700  }
701  } else {
702  if (i == p.size() || (p[i] == '/' || p[i] == '\\')) {
703  // The current character ends a segment.
704  typename Str::size_type len = i - seg_start;
705  if (first) {
706  first = false;
707  abs = len == 2 &&
708  ((p[seg_start] >= 'A' && p[seg_start] <= 'Z') ||
709  (p[seg_start] >= 'a' && p[seg_start] <= 'z')) &&
710  p[seg_start + 1] == ':';
711  segments.push_back(std::make_pair(seg_start, len));
712  starts_with_dot = !abs && p.compare(seg_start, len, kDot) == 0;
713  } else {
714  if (p.compare(seg_start, len, kDot) == 0) {
715  if (segments.empty()) {
716  // Retain "." if that is the first (and possibly only segment).
717  segments.push_back(std::make_pair(seg_start, len));
718  starts_with_dot = true;
719  }
720  } else {
721  if (starts_with_dot) {
722  // Delete the existing "." if that was the only path segment.
723  segments.clear();
724  starts_with_dot = false;
725  }
726  if (p.compare(seg_start, len, kDotDot) == 0) {
727  if (segments.empty() ||
728  p.compare(segments.back().first, segments.back().second,
729  kDotDot) == 0) {
730  // Preserve ".." if the path is relative and there are only ".."
731  // segment(s) at the front.
732  segments.push_back(std::make_pair(seg_start, len));
733  } else if (!abs || segments.size() > 1) {
734  // Remove the last segment unless the path is already at the
735  // root directory.
736  segments.pop_back();
737  } // Ignore ".." otherwise.
738  } else {
739  // This is a normal path segment, i.e. neither "." nor ".."
740  segments.push_back(std::make_pair(seg_start, len));
741  }
742  }
743  }
744  // Indicate that there's no segment started.
745  seg_start = Str::npos;
746  }
747  }
748  }
749  std::basic_stringstream<C> res;
750  first = true;
751  for (const auto& i : segments) {
752  Str s = p.substr(i.first, i.second);
753  if (first) {
754  first = false;
755  } else {
756  res << '\\';
757  }
758  res << s;
759  }
760  if (abs && segments.size() == 1) {
761  res << '\\';
762  }
763  return res.str();
764 }
