89 int OutputJar::Doit(Options *options) {
90  if (nullptr != options_) {
91  diag_errx(1, "%s:%d: Doit() can be called only once.", __FILE__, __LINE__);
92  }
93  options_ = options;
94 
95  // Register the handler for the build-data.properties file unless
96  // --exclude_build_data is present. Otherwise we do not generate this file,
97  // and it will be copied from the first source archive containing it.
98  if (!options_->exclude_build_data) {
99  known_members_.emplace(build_properties_.filename(),
100  EntryInfo{&build_properties_});
101  }
102 
103  // TODO(b/28294322): do we need to resolve the path to be absolute or
104  // canonical?
105  build_properties_.AddProperty("build.target", options_->output_jar.c_str());
106  if (options_->verbose) {
107  fprintf(stderr, "combined_file_name=%s\n", options_->output_jar.c_str());
108  if (!options_->main_class.empty()) {
109  fprintf(stderr, "main_class=%s\n", options_->main_class.c_str());
110  }
111  if (!options_->java_launcher.empty()) {
112  fprintf(stderr, "java_launcher_file=%s\n",
113  options_->java_launcher.c_str());
114  }
115  fprintf(stderr, "%zu source files\n", options_->input_jars.size());
116  fprintf(stderr, "%zu manifest lines\n", options_->manifest_lines.size());
117  }
118 
119  if (!Open()) {
120  exit(1);
121  }
122 
123  // Copy launcher if it is set.
124  if (!options_->java_launcher.empty()) {
125  const char *const launcher_path = options_->java_launcher.c_str();
126  int in_fd = open(launcher_path, O_RDONLY);
127  struct stat statbuf;
128  if (file_ == nullptr || fstat(in_fd, &statbuf)) {
129  diag_err(1, "%s", launcher_path);
130  }
131  // TODO(asmundak): Consider going back to sendfile() or reflink
132  // (BTRFS_IOC_CLONE/XFS_IOC_CLONE) here. The launcher preamble can
133  // be very large for targets with many native deps.
134  ssize_t byte_count = AppendFile(in_fd, 0, statbuf.st_size);
135  if (byte_count < 0) {
136  diag_err(1, "%s:%d: Cannot copy %s to %s", __FILE__, __LINE__,
137  launcher_path, options_->output_jar.c_str());
138  } else if (byte_count != statbuf.st_size) {
139  diag_err(1, "%s:%d: Copied only %zu bytes out of %" PRIu64 " from %s",
140  __FILE__, __LINE__, byte_count, statbuf.st_size, launcher_path);
141  }
142  close(in_fd);
143  if (options_->verbose) {
144  fprintf(stderr, "Prepended %s (%" PRIu64 " bytes)\n", launcher_path,
145  statbuf.st_size);
146  }
147  }
148 
149  if (!options_->main_class.empty()) {
150  build_properties_.AddProperty("main.class", options_->main_class);
151  manifest_.Append("Main-Class: ");
152  manifest_.Append(options_->main_class);
153  manifest_.Append("\r\n");
154  }
155 
156  for (auto &manifest_line : options_->manifest_lines) {
157  if (!manifest_line.empty()) {
158  manifest_.Append(manifest_line);
159  if (manifest_line[manifest_line.size() - 1] != '\n') {
160  manifest_.Append("\r\n");
161  }
162  }
163  }
164 
165  for (auto &build_info_line : options_->build_info_lines) {
166  build_properties_.Append(build_info_line);
167  build_properties_.Append("\n");
168  }
169 
170  for (auto &build_info_file : options_->build_info_files) {
171  MappedFile mapped_file;
172  if (!mapped_file.Open(build_info_file)) {
173  diag_err(1, "%s:%d: Bad build info file %s", __FILE__, __LINE__,
174  build_info_file.c_str());
175  }
176  const char *data = reinterpret_cast<const char *>(mapped_file.start());
177  const char *data_end = reinterpret_cast<const char *>(mapped_file.end());
178  // TODO(asmundak): this isn't right, we should parse properties file.
179  while (data < data_end) {
180  const char *next_data = strchr(static_cast<const char *>(data), '\n');
181  if (next_data) {
182  ++next_data;
183  } else {
184  next_data = data_end;
185  }
186  build_properties_.Append(data, next_data - data);
187  data = next_data;
188  }
189  mapped_file.Close();
190  }
191 
192  for (auto &rpath : options_->classpath_resources) {
193  ClasspathResource(Basename(rpath), rpath);
194  }
195 
196  for (auto &rdesc : options_->resources) {
197  // A resource description is either NAME or PATH:NAME
198  // Find the last ':' instead of the first because Windows uses ':' as volume
199  // separator in absolute path.
200  std::size_t colon = rdesc.find_last_of(':');
201  if (0 == colon) {
202  diag_errx(1, "%s:%d: Bad resource description %s", __FILE__, __LINE__,
203  rdesc.c_str());
204  }
205  bool shouldSplit = colon != std::string::npos;
206 #ifdef _WIN32
207  // If colon points to volume separator, don't split.
208  if (colon == 1 && blaze_util::IsAbsolute(rdesc)) {
209  shouldSplit = false;
210  }
211 #endif
212  if (shouldSplit) {
213  ClasspathResource(rdesc.substr(colon + 1), rdesc.substr(0, colon));
214  } else {
215  ClasspathResource(rdesc, rdesc);
216  }
217  }
218 
219  // Ready to write zip entries. Decide whether created entries should be
220  // compressed.
221  bool compress = options_->force_compression || options_->preserve_compression;
222  // First, write a directory entry for the META-INF, followed by the manifest
223  // file, followed by the build properties file.
224  WriteMetaInf();
225  manifest_.Append("\r\n");
226  WriteEntry(manifest_.OutputEntry(compress));
227  if (!options_->exclude_build_data) {
228  WriteEntry(build_properties_.OutputEntry(compress));
229  }
230 
231  // Then classpath resources.
232  for (auto &classpath_resource : classpath_resources_) {
233  bool do_compress = compress;
234  if (do_compress && !options_->nocompress_suffixes.empty()) {
235  for (auto &suffix : options_->nocompress_suffixes) {
236  auto entry_name = classpath_resource->filename();
237  if (entry_name.length() >= suffix.size() &&
238  !entry_name.compare(entry_name.length() - suffix.size(),
239  suffix.size(), suffix)) {
240  do_compress = false;
241  break;
242  }
243  }
244  }
245 
246  // Add parent directory entries.
247  size_t pos = classpath_resource->filename().find('/');
248  while (pos != std::string::npos) {
249  std::string dir(classpath_resource->filename(), 0, pos + 1);
250  if (NewEntry(dir)) {
251  WriteDirEntry(dir, nullptr, 0);
252  }
253  pos = classpath_resource->filename().find('/', pos + 1);
254  }
255 
256  WriteEntry(classpath_resource->OutputEntry(do_compress));
257  }
258 
259  // Then copy source files' contents.
260  for (size_t ix = 0; ix < options_->input_jars.size(); ++ix) {
261  if (!AddJar(ix)) {
262  exit(1);
263  }
264  }
265 
266  // All entries written, write Central Directory and close.
267  Close();
268  return 0;
269 }

278 // (128KB is the default max request size for fuse filesystems.)
279 static constexpr size_t kBufferSize = 128 << 10;

281 bool OutputJar::Open() {
282  if (file_) {
283  diag_errx(1, "%s:%d: Cannot open output archive twice", __FILE__, __LINE__);
284  }
285 
286  int mode = O_CREAT | O_WRONLY | O_TRUNC;
287 
288 #ifdef _WIN32
289  std::wstring wpath;
290  std::string error;
291  if (!blaze_util::AsAbsoluteWindowsPath(path(), &wpath, &error)) {
292  diag_warn("%s:%d: AsAbsoluteWindowsPath failed: %s", __FILE__, __LINE__,
293  error.c_str());
294  return false;
295  }
296 
297  HANDLE hFile = CreateFileW(wpath.c_str(), GENERIC_READ | GENERIC_WRITE,
298  // Must share for reading, otherwise
299  // symlink-following file existence checks (e.g.
300  // java.nio.file.Files.exists()) fail.
301  FILE_SHARE_READ, NULL, CREATE_ALWAYS, 0, NULL);
302  if (hFile == INVALID_HANDLE_VALUE) {
303  diag_warn("%s:%d: CreateFileW failed for %S", __FILE__, __LINE__,
304  wpath.c_str());
305  return false;
306  }
307 
308  // Make sure output file is in binary mode, or \r\n will be converted to \n.
309  mode |= _O_BINARY;
310  int fd = _open_osfhandle(reinterpret_cast<intptr_t>(hFile), mode);
311 #else
312  // Set execute bits since we may produce an executable output file.
313  int fd = open(path(), mode, 0777);
314 #endif
315 
316  if (fd < 0) {
317  diag_warn("%s:%d: %s", __FILE__, __LINE__, path());
318  return false;
319  }
320  file_ = fdopen(fd, "w");
321  if (file_ == nullptr) {
322  diag_warn("%s:%d: fdopen of %s", __FILE__, __LINE__, path());
323  close(fd);
324  return false;
325  }
326  outpos_ = 0;
327  buffer_.reset(new char[kBufferSize]);
328  setvbuf(file_, buffer_.get(), _IOFBF, kBufferSize);
329  if (options_->verbose) {
330  fprintf(stderr, "Writing to %s\n", path());
331  }
332  return true;
333 }

338 bool OutputJar::AddJar(int jar_path_index) {
339  const std::string &input_jar_path =
340  options_->input_jars[jar_path_index].first;
341  const std::string &input_jar_aux_label =
342  options_->input_jars[jar_path_index].second;
343 
344  InputJar input_jar;
345  if (!input_jar.Open(input_jar_path)) {
346  return false;
347  }
348  const CDH *jar_entry;
349  const LH *lh;
350  while ((jar_entry = input_jar.NextEntry(&lh))) {
351  const char *file_name = jar_entry->file_name();
352  auto file_name_length = jar_entry->file_name_length();
353  if (!file_name_length) {
354  diag_errx(
355  1, "%s:%d: Bad central directory record in %s at offset 0x%" PRIx64,
356  __FILE__, __LINE__, input_jar_path.c_str(),
357  input_jar.CentralDirectoryRecordOffset(jar_entry));
358  }
359  // Special files that cannot be handled by looking up known_members_ map:
360  // * ignore *.SF, *.RSA, *.DSA
361  // (TODO(asmundak): should this be done only in META-INF?
362  //
363  if (ends_with(file_name, file_name_length, ".SF") ||
364  ends_with(file_name, file_name_length, ".RSA") ||
365  ends_with(file_name, file_name_length, ".DSA")) {
366  continue;
367  }
368 
369  bool include_entry = true;
370  if (!options_->include_prefixes.empty()) {
371  for (auto &prefix : options_->include_prefixes) {
372  if ((include_entry =
373  (prefix.size() <= file_name_length &&
374  0 == strncmp(file_name, prefix.c_str(), prefix.size())))) {
375  break;
376  }
377  }
378  }
379  if (!include_entry) {
380  continue;
381  }
382 
383  bool is_file = (file_name[file_name_length - 1] != '/');
384  if (is_file &&
385  begins_with(file_name, file_name_length, "META-INF/services/")) {
386  // The contents of the META-INF/services/<SERVICE> on the output is the
387  // concatenation of the META-INF/services/<SERVICE> files from all inputs.
388  std::string service_path(file_name, file_name_length);
389  if (NewEntry(service_path)) {
390  // Create a concatenator and add it to the known_members_ map.
391  // The call to Merge() below will then take care of the rest.
392  Concatenator *service_handler = new Concatenator(service_path);
393  service_handlers_.emplace_back(service_handler);
394  known_members_.emplace(service_path, EntryInfo{service_handler});
395  }
396  } else {
397  ExtraHandler(input_jar_path, jar_entry, &input_jar_aux_label);
398  }
399 
400  if (options_->check_desugar_deps &&
401  begins_with(file_name, file_name_length, "j$/")) {
402  diag_errx(1, "%s:%d: desugar_jdk_libs file %.*s unexpectedly found in %s",
403  __FILE__, __LINE__, file_name_length, file_name,
404  input_jar_path.c_str());
405  }
406 
407  // Install a new entry unless it is already present. All the plain (non-dir)
408  // entries that require a combiner have been already installed, so the call
409  // will add either a directory entry whose handler will ignore subsequent
410  // duplicates, or an ordinary plain entry, for which we save the index of
411  // the first input jar (in order to provide diagnostics on duplicate).
412  auto got =
413  known_members_.emplace(std::string(file_name, file_name_length),
414  EntryInfo{is_file ? nullptr : &null_combiner_,
415  is_file ? jar_path_index : -1});
416  if (!got.second) {
417  auto &entry_info = got.first->second;
418  // Handle special entries (the ones that have a combiner).
419  if (entry_info.combiner_ != nullptr) {
420  // TODO(kmb,asmundak): Should be checking Merge() return value but fails
421  // for build-data.properties when merging deploy jars into deploy jars.
422  entry_info.combiner_->Merge(jar_entry, lh);
423  continue;
424  }
425 
426  // Plain file entry. If duplicates are not allowed, bail out. Otherwise
427  // just ignore this entry.
428  if (options_->no_duplicates ||
429  (options_->no_duplicate_classes &&
430  ends_with(file_name, file_name_length, ".class"))) {
431  diag_errx(
432  1, "%s:%d: %.*s is present both in %s and %s", __FILE__, __LINE__,
433  file_name_length, file_name,
434  options_->input_jars[entry_info.input_jar_index_].first.c_str(),
435  input_jar_path.c_str());
436  } else {
437  duplicate_entries_++;
438  continue;
439  }
440  }
441 
442  // Add any missing parent directory entries (first) if requested.
443  if (options_->add_missing_directories) {
444  // Ignore very last character in case this entry is a directory itself.
445  for (size_t pos = 0; pos < static_cast<size_t>(file_name_length - 1);
446  ++pos) {
447  if (file_name[pos] == '/') {
448  std::string dir(file_name, 0, pos + 1);
449  if (NewEntry(dir)) {
450  WriteDirEntry(dir, nullptr, 0);
451  }
452  }
453  }
454  }
455 
456  // For the file entries, decide whether output should be compressed.
457  if (is_file) {
458  bool input_compressed =
459  jar_entry->compression_method() != Z_NO_COMPRESSION;
460  bool output_compressed =
461  options_->force_compression ||
462  (options_->preserve_compression && input_compressed);
463  if (output_compressed && !options_->nocompress_suffixes.empty()) {
464  for (auto &suffix : options_->nocompress_suffixes) {
465  if (file_name_length >= suffix.size() &&
466  !strncmp(file_name + file_name_length - suffix.size(),
467  suffix.c_str(), suffix.size())) {
468  output_compressed = false;
469  break;
470  }
471  }
472  }
473  if (input_compressed != output_compressed) {
474  Concatenator combiner(jar_entry->file_name_string());
475  if (!combiner.Merge(jar_entry, lh)) {
476  diag_err(1, "%s:%d: cannot add %.*s", __FILE__, __LINE__,
477  jar_entry->file_name_length(), jar_entry->file_name());
478  }
479  WriteEntry(combiner.OutputEntry(output_compressed));
480  continue;
481  }
482  }
483 
484  // Now we have to copy:
485  // local header
486  // file data
487  // data descriptor, if present.
488  off64_t copy_from = jar_entry->local_header_offset();
489  size_t num_bytes = lh->size();
490  if (jar_entry->no_size_in_local_header()) {
491  const DDR *ddr = reinterpret_cast<const DDR *>(
492  lh->data() + jar_entry->compressed_file_size());
493  num_bytes +=
494  jar_entry->compressed_file_size() +
495  ddr->size(
496  ziph::zfield_has_ext64(jar_entry->compressed_file_size32()),
497  ziph::zfield_has_ext64(jar_entry->uncompressed_file_size32()));
498  } else {
499  num_bytes += lh->compressed_file_size();
500  }
501  off64_t local_header_offset = Position();
502 
503  // When normalize_timestamps is set, entry's timestamp is to be set to
504  // 01/01/2010 00:00:00 (or to 01/01/2010 00:00:02, if an entry is a .class
505  // file). This is somewhat expensive because we have to copy the local
506  // header to memory as input jar is memory mapped as read-only. Try to copy
507  // as little as possible.
508  uint16_t normalized_time = 0;
509  const UnixTimeExtraField *lh_field_to_remove = nullptr;
510  bool fix_timestamp = false;
511  if (options_->normalize_timestamps) {
512  if (ends_with(file_name, file_name_length, ".class")) {
513  normalized_time = 1;
514  }
515  lh_field_to_remove = lh->unix_time_extra_field();
516  fix_timestamp = jar_entry->last_mod_file_date() != kDefaultDate ||
517  jar_entry->last_mod_file_time() != normalized_time ||
518  lh_field_to_remove != nullptr;
519  }
520  if (fix_timestamp) {
521  uint8_t lh_buffer[512];
522  size_t lh_size = lh->size();
523  LH *lh_new = lh_size > sizeof(lh_buffer)
524  ? reinterpret_cast<LH *>(malloc(lh_size))
525  : reinterpret_cast<LH *>(lh_buffer);
526  // Remove Unix timestamp field.
527  if (lh_field_to_remove != nullptr) {
528  auto from_end = ziph::byte_ptr(lh) + lh->size();
529  size_t removed_size = lh_field_to_remove->size();
530  size_t chunk1_size =
531  ziph::byte_ptr(lh_field_to_remove) - ziph::byte_ptr(lh);
532  size_t chunk2_size = lh->size() - (chunk1_size + removed_size);
533  memcpy(lh_new, lh, chunk1_size);
534  if (chunk2_size) {
535  memcpy(reinterpret_cast<uint8_t *>(lh_new) + chunk1_size,
536  from_end - chunk2_size, chunk2_size);
537  }
538  lh_new->extra_fields(lh_new->extra_fields(),
539  lh->extra_fields_length() - removed_size);
540  } else {
541  memcpy(lh_new, lh, lh_size);
542  }
543  lh_new->last_mod_file_date(kDefaultDate);
544  lh_new->last_mod_file_time(normalized_time);
545  // Now write these few bytes and adjust read/write positions accordingly.
546  if (!WriteBytes(lh_new, lh_new->size())) {
547  diag_err(1, "%s:%d: Cannot copy modified local header for %.*s",
548  __FILE__, __LINE__, file_name_length, file_name);
549  }
550  copy_from += lh_size;
551  num_bytes -= lh_size;
552  if (reinterpret_cast<uint8_t *>(lh_new) != lh_buffer) {
553  free(lh_new);
554  }
555  }
556 
557  // Do the actual copy.
558  if (!WriteBytes(input_jar.mapped_start() + copy_from, num_bytes)) {
559  diag_err(1, "%s:%d: Cannot write %zu bytes of %.*s from %s", __FILE__,
560  __LINE__, num_bytes, file_name_length, file_name,
561  input_jar_path.c_str());
562  }
563 
564  AppendToDirectoryBuffer(jar_entry, local_header_offset, normalized_time,
565  fix_timestamp);
566  ++entries_;
567  }
568  return input_jar.Close();
569 }

581 // Writes an entry. The argument is the pointer to the contiguous block of
582 // memory containing Local Header for the entry, immediately followed by
583 // the data. The memory is freed after the data has been written.
584 void OutputJar::WriteEntry(void *buffer) {
585  if (buffer == nullptr) {
586  return;
587  }
588  LH *entry = reinterpret_cast<LH *>(buffer);
589  if (options_->verbose) {
590  fprintf(stderr, "%-.*s combiner has %zu bytes, %s to %zu\n",
591  entry->file_name_length(), entry->file_name(),
592  entry->uncompressed_file_size(),
593  entry->compression_method() == Z_NO_COMPRESSION ? "copied"
594  : "compressed",
595  entry->compressed_file_size());
596  }
597 
598  // Set this entry's timestamp.
599  // MSDOS file timestamp format that Zip uses is described here:
600  // https://msdn.microsoft.com/en-us/library/9kkf9tah.aspx
601  // ("32-Bit Windows Time/Date Formats")
602  if (options_->normalize_timestamps) {
603  // Regular "normalized" timestamp is 01/01/2010 00:00:00, while for the
604  // .class file it is 01/01/2010 00:00:02
605  entry->last_mod_file_date(kDefaultDate);
606  entry->last_mod_file_time(
607  ends_with(entry->file_name(), entry->file_name_length(), ".class") ? 1
608  : 0);
609  } else {
610  struct tm tm;
611  // Time has 2-second resolution, so round up:
612  time_t t_adjusted = (time(nullptr) + 1) & ~1;
613  localtime_r(&t_adjusted, &tm);
614  uint16_t dos_date =
615  ((tm.tm_year - 80) << 9) | ((tm.tm_mon + 1) << 5) | tm.tm_mday;
616  uint16_t dos_time =
617  (tm.tm_hour << 11) | (tm.tm_min << 5) | (tm.tm_sec >> 1);
618  entry->last_mod_file_time(dos_time);
619  entry->last_mod_file_date(dos_date);
620  }
621 
622  uint8_t *data = reinterpret_cast<uint8_t *>(entry);
623  off64_t output_position = Position();
624  if (!WriteBytes(data, entry->data() + entry->in_zip_size() - data)) {
625  diag_err(1, "%s:%d: write", __FILE__, __LINE__);
626  }
627  // Data written, allocate CDH space and populate CDH.
628  // Space needed for the CDH varies depending on whether output position field
629  // fits into 32 bits (we do not handle compressed/uncompressed entry sizes
630  // exceeding 32 bits at the moment).
631  uint16_t zip64_size = ziph::zfield_needs_ext64(output_position)
632  ? Zip64ExtraField::space_needed(1)
633  : 0;
634  CDH *cdh = reinterpret_cast<CDH *>(
635  ReserveCdh(sizeof(CDH) + entry->file_name_length() +
636  entry->extra_fields_length() + zip64_size));
637  cdh->signature();
638  // Note: do not set the version to Unix 3.0 spec, otherwise
639  // unzip will think that 'external_attributes' field contains access mode
640  cdh->version(20);
641  cdh->version_to_extract(20); // 2.0
642  cdh->bit_flag(0x0);
643  cdh->compression_method(entry->compression_method());
644  cdh->last_mod_file_time(entry->last_mod_file_time());
645  cdh->last_mod_file_date(entry->last_mod_file_date());
646  cdh->crc32(entry->crc32());
647  TODO(entry->compressed_file_size32() != 0xFFFFFFFF, "Handle Zip64");
648  cdh->compressed_file_size32(entry->compressed_file_size32());
649  TODO(entry->uncompressed_file_size32() != 0xFFFFFFFF, "Handle Zip64");
650  cdh->uncompressed_file_size32(entry->uncompressed_file_size32());
651  cdh->file_name(entry->file_name(), entry->file_name_length());
652  cdh->extra_fields(entry->extra_fields(), entry->extra_fields_length());
653  if (zip64_size > 0) {
654  Zip64ExtraField *zip64_ef = reinterpret_cast<Zip64ExtraField *>(
655  cdh->extra_fields() + cdh->extra_fields_length());
656  zip64_ef->signature();
657  zip64_ef->attr_count(1);
658  zip64_ef->attr64(0, output_position);
659  cdh->local_header_offset32(0xFFFFFFFF);
660  // Field address argument points to the already existing field,
661  // so the call just updates the length.
662  cdh->extra_fields(cdh->extra_fields(),
663  cdh->extra_fields_length() + zip64_size);
664  } else {
665  cdh->local_header_offset32(output_position);
666  }
667  cdh->comment_length(0);
668  cdh->start_disk_nr(0);
669  cdh->internal_attributes(0);
670  cdh->external_attributes(0);
671  ++entries_;
672  free(reinterpret_cast<void *>(entry));
673 }

707 // Create output Central Directory entry for the input jar entry.
708 void OutputJar::AppendToDirectoryBuffer(const CDH *cdh, off64_t lh_pos,
709  uint16_t normalized_time,
710  bool fix_timestamp) {
711  // While copying from the input CDH pointed to by 'cdh', we may need to drop
712  // Unix timestamp extra field, and we might need to change the number of
713  // attributes of the Zip64 extra field, or create it, or destroy it if entry's
714  // position relative to 4G boundary changes.
715  // The rest of the input CDH is copied.
716 
717  // 1. Decide if we need to drop UnixTime.
718  size_t removed_unix_time_field_size = 0;
719  if (fix_timestamp) {
720  auto unix_time_field = cdh->unix_time_extra_field();
721  if (unix_time_field != nullptr) {
722  removed_unix_time_field_size = unix_time_field->size();
723  }
724  }
725 
726  // 2. Figure out how many attributes input entry has and how many
727  // the output entry is going to have.
728  const Zip64ExtraField *zip64_ef = cdh->zip64_extra_field();
729  const int zip64_attr_count = zip64_ef == nullptr ? 0 : zip64_ef->attr_count();
730  const bool lh_pos_needs64 = ziph::zfield_needs_ext64(lh_pos);
731  int out_zip64_attr_count;
732  if (zip64_attr_count > 0) {
733  out_zip64_attr_count = zip64_attr_count;
734  // The number of attributes may remain the same, or it may increase or
735  // decrease by 1, depending on local_header_offset value.
736  if (ziph::zfield_has_ext64(cdh->local_header_offset32()) !=
737  lh_pos_needs64) {
738  if (lh_pos_needs64) {
739  out_zip64_attr_count += 1;
740  } else {
741  out_zip64_attr_count -= 1;
742  }
743  }
744  } else {
745  out_zip64_attr_count = lh_pos_needs64 ? 1 : 0;
746  }
747  const uint16_t zip64_size = Zip64ExtraField::space_needed(zip64_attr_count);
748  const uint16_t out_zip64_size =
749  Zip64ExtraField::space_needed(out_zip64_attr_count);
750 
751  // Allocate output CDH and copy everything but extra fields.
752  const uint16_t ef_size = cdh->extra_fields_length();
753  const uint16_t out_ef_size =
754  (ef_size + out_zip64_size) - (removed_unix_time_field_size + zip64_size);
755 
756  const size_t out_cdh_size = cdh->size() + out_ef_size - ef_size;
757  CDH *out_cdh = reinterpret_cast<CDH *>(ReserveCdr(out_cdh_size));
758 
759  // Calculate ExtraFields boundaries in the input and output entries.
760  auto ef_begin = reinterpret_cast<const ExtraField *>(cdh->extra_fields());
761  auto ef_end =
762  reinterpret_cast<const ExtraField *>(ziph::byte_ptr(ef_begin) + ef_size);
763  // Copy [cdh..ef_begin) -> [out_cdh..out_ef_begin)
764  memcpy(out_cdh, cdh, ziph::byte_ptr(ef_begin) - ziph::byte_ptr(cdh));
765 
766  auto out_ef_begin = reinterpret_cast<ExtraField *>(
767  const_cast<uint8_t *>(out_cdh->extra_fields()));
768  auto out_ef_end = reinterpret_cast<ExtraField *>(
769  reinterpret_cast<uint8_t *>(out_ef_begin) + out_ef_size);
770 
771  // Copy [ef_end..cdh_end) -> [out_ef_end..out_cdh_end)
772  memcpy(out_ef_end, ef_end,
773  ziph::byte_ptr(cdh) + cdh->size() - ziph::byte_ptr(ef_end));
774 
775  // Copy extra fields, dropping Zip64 and possibly UnixTime fields.
776  ExtraField *out_ef = out_ef_begin;
777  for (const ExtraField *ef = ef_begin; ef < ef_end; ef = ef->next()) {
778  if ((fix_timestamp && ef->is_unix_time()) || ef->is_zip64()) {
779  // Skip this one.
780  } else {
781  memcpy(out_ef, ef, ef->size());
782  out_ef = reinterpret_cast<ExtraField *>(
783  reinterpret_cast<uint8_t *>(out_ef) + ef->size());
784  }
785  }
786 
787  // Set up Zip64 extra field if necessary.
788  if (out_zip64_size > 0) {
789  Zip64ExtraField *out_zip64_ef = reinterpret_cast<Zip64ExtraField *>(out_ef);
790  out_zip64_ef->signature();
791  out_zip64_ef->attr_count(out_zip64_attr_count);
792  int copy_count = out_zip64_attr_count < zip64_attr_count
793  ? out_zip64_attr_count
794  : zip64_attr_count;
795  if (copy_count > 0) {
796  out_zip64_ef->attr64(0, zip64_ef->attr64(0));
797  if (copy_count > 1) {
798  out_zip64_ef->attr64(1, zip64_ef->attr64(1));
799  }
800  }
801  // Set 64-bit local_header_offset if necessary. It's always the last
802  // attribute.
803  if (lh_pos_needs64) {
804  out_zip64_ef->attr64(out_zip64_attr_count - 1, lh_pos);
805  }
806  }
807  out_cdh->extra_fields(ziph::byte_ptr(out_ef_begin), out_ef_size);
808  out_cdh->local_header_offset32(lh_pos_needs64 ? 0xFFFFFFFF : lh_pos);
809  if (fix_timestamp) {
810  out_cdh->last_mod_file_time(normalized_time);
811  out_cdh->last_mod_file_date(kDefaultDate);
812  }
813 }

834 bool OutputJar::Close() {
835  if (file_ == nullptr) {
836  return true;
837  }
838 
839  for (auto &service_handler : service_handlers_) {
840  WriteEntry(service_handler->OutputEntry(options_->force_compression));
841  }
842  for (auto &extra_combiner : extra_combiners_) {
843  WriteEntry(extra_combiner->OutputEntry(options_->force_compression));
844  }
845  WriteEntry(spring_handlers_.OutputEntry(options_->force_compression));
846  WriteEntry(spring_schemas_.OutputEntry(options_->force_compression));
847  WriteEntry(protobuf_meta_handler_.OutputEntry(options_->force_compression));
848  // TODO(asmundak): handle manifest;
849  off64_t output_position = Position();
850  bool write_zip64_ecd = output_position >= 0xFFFFFFFF || entries_ >= 0xFFFF ||
851  cen_size_ >= 0xFFFFFFFF;
852 
853  size_t cen_size = cen_size_; // Save it before ReserveCdh updates it.
854  if (write_zip64_ecd) {
855  {
856  ECD64 *ecd64 = reinterpret_cast<ECD64 *>(ReserveCdh(sizeof(ECD64)));
857  ecd64->signature();
858  ecd64->remaining_size(sizeof(ECD64) - 12);
859  ecd64->version(0x031E); // Unix, version 3.0
860  ecd64->version_to_extract(45); // 4.5 (Zip64 support)
861  ecd64->this_disk_entries(entries_);
862  ecd64->total_entries(entries_);
863  ecd64->cen_size(cen_size);
864  ecd64->cen_offset(output_position);
865  }
866  {
867  ECD64Locator *ecd64_locator =
868  reinterpret_cast<ECD64Locator *>(ReserveCdh(sizeof(ECD64Locator)));
869  ecd64_locator->signature();
870  ecd64_locator->ecd64_offset(output_position + cen_size);
871  ecd64_locator->total_disks(1);
872  }
873  {
874  ECD *ecd = reinterpret_cast<ECD *>(ReserveCdh(sizeof(ECD)));
875  ecd->signature();
876  ecd->this_disk_entries16(0xFFFF);
877  ecd->total_entries16(0xFFFF);
878  // Java Compiler (javac) uses its own "optimized" Zip handler (see
879  // https://bugs.openjdk.java.net/browse/JDK-7018859) which may fail
880  // to handle 0xFFFFFFFF in the CEN size and CEN offset fields. Try
881  // to use 32-bit values here, too. Hopefully by the time we need to
882  // handle really large archives, this is fixes upstream. Note that this
883  // affects javac and javah only, 'jar' experiences no problems.
884  ecd->cen_size32(std::min(cen_size, static_cast<size_t>(0xFFFFFFFFUL)));
885  ecd->cen_offset32(
886  std::min(output_position, static_cast<off64_t>(0x0FFFFFFFFL)));
887  }
888  } else {
889  ECD *ecd = reinterpret_cast<ECD *>(ReserveCdh(sizeof(ECD)));
890  ecd->signature();
891  ecd->this_disk_entries16((uint16_t)entries_);
892  ecd->total_entries16((uint16_t)entries_);
893  ecd->cen_size32(cen_size);
894  ecd->cen_offset32(output_position);
895  }
896 
897  // Save Central Directory and wrap up.
898  if (!WriteBytes(cen_, cen_size_)) {
899  diag_err(1, "%s:%d: Cannot write central directory", __FILE__, __LINE__);
900  }
901  free(cen_);
902 
903  if (fclose(file_)) {
904  diag_err(1, "%s:%d: %s", __FILE__, __LINE__, path());
905  }
906  file_ = nullptr;
907  // Free the buffer only after fclose(); stdio may flush data from the
908  // buffer on close.
909  buffer_.reset();
910 
911  if (options_->verbose) {
912  fprintf(stderr, "Wrote %s with %d entries", path(), entries_);
913  if (duplicate_entries_) {
914  fprintf(stderr, ", skipped %d entries", duplicate_entries_);
915  }
916  fprintf(stderr, "\n");
917  }
918  return true;
919 }
