40  """Metaclass used to cache and share argument parsers among flags."""

42  _instances = {}

44  def __call__(cls, *args, **kwargs):
45  """Returns an instance of the argument parser cls.
46 
47  This method overrides behavior of the __new__ methods in
48  all subclasses of ArgumentParser (inclusive). If an instance
49  for cls with the same set of arguments exists, this instance is
50  returned, otherwise a new instance is created.
51 
52  If any keyword arguments are defined, or the values in args
53  are not hashable, this method always returns a new instance of
54  cls.
55 
56  Args:
57  *args: Positional initializer arguments.
58  **kwargs: Initializer keyword arguments.
59 
60  Returns:
61  An instance of cls, shared or new.
62  """
63  if kwargs:
64  return type.__call__(cls, *args, **kwargs)
65  else:
66  instances = cls._instances
67  key = (cls,) + tuple(args)
68  try:
69  return instances[key]
70  except KeyError:
71  # No cache entry for key exists, create a new one.
72  return instances.setdefault(key, type.__call__(cls, *args))
73  except TypeError:
74  # An object in args cannot be hashed, always return
75  # a new instance.
76  return type.__call__(cls, *args)

79 class ArgumentParser(six.with_metaclass(_ArgumentParserCache, object)):
80  """Base class used to parse and convert arguments.
81 
82  The parse() method checks to make sure that the string argument is a
83  legal value and convert it to a native type. If the value cannot be
84  converted, it should throw a 'ValueError' exception with a human
85  readable explanation of why the value is illegal.
86 
87  Subclasses should also define a syntactic_help string which may be
88  presented to the user to describe the form of the legal values.
89 
90  Argument parser classes must be stateless, since instances are cached
91  and shared between flags. Initializer arguments are allowed, but all
92  member variables must be derived from initializer arguments only.
93  """

279  """See base class."""

281  if argument.lower() in ('true', 't', '1'):
282  return True

327 
328  Raises:

366  """Serializes a list as a CSV string or unicode."""
367  if six.PY2:
368  # In Python2 csv.writer doesn't accept unicode, so we convert to UTF-8.
369  output = io.BytesIO()
370  csv.writer(output).writerow([unicode(x).encode('utf-8') for x in value])
371  serialized_value = output.getvalue().decode('utf-8').strip()
372  else:
373  # In Python3 csv.writer expects a text stream.

375  csv.writer(output).writerow([str(x) for x in value])
376  serialized_value = output.getvalue().strip()

379  # when the xml help is generated they are usefully encodable.
380  return _helpers.str_or_unicode(serialized_value)
