68  public boolean addEntry(String entryName, Path path) {
69  if (entryName.startsWith("/")) {
70  entryName = entryName.substring(1);
71  } else if (entryName.length() >= 3
72  && Character.isLetter(entryName.charAt(0))
73  && entryName.charAt(1) == ':'
74  && (entryName.charAt(2) == '\\' || entryName.charAt(2) == '/')) {
75  // Windows absolute path, e.g. "D:\foo" or "e:/blah".
76  // Windows paths are case-insensitive, and support both backslashes and forward slashes.
77  entryName = entryName.substring(3);
78  } else if (entryName.startsWith("./")) {
79  entryName = entryName.substring(2);
80  }
81  return jarEntries.put(entryName, path) == null;
82  }

140  void addEntry(Path path, boolean isDirectory) {
141  StringBuilder sb = new StringBuilder();
142  boolean first = true;
143  for (Path entry : directory.relativize(path)) {
144  if (!first) {
145  // use `/` as the directory separator for jar paths, even on Windows
146  sb.append('/');
147  }
148  sb.append(entry.getFileName());
149  first = false;
150  }
151  if (isDirectory) {
152  sb.append('/');
153  }
154  jarEntries.put(sb.toString(), path);
155  }
156  });
157  } catch (IOException e) {
158  throw new UncheckedIOException(e);
159  }
160  }

202  private byte[] manifestContent() throws IOException {
203  if (manifestFile != null) {
204  try (FileInputStream in = new FileInputStream(manifestFile)) {
205  return manifestContentImpl(new Manifest(in));
206  }
207  } else {
208  return manifestContentImpl(new Manifest());

212  private byte[] manifestContentImpl(Manifest manifest) throws IOException {
213  Attributes attributes = manifest.getMainAttributes();
214  attributes.put(Attributes.Name.MANIFEST_VERSION, "1.0");
215  Attributes.Name createdBy = new Attributes.Name("Created-By");
216  if (attributes.getValue(createdBy) == null) {
217  attributes.put(createdBy, "bazel");
218  }
219  if (mainClass != null) {
220  attributes.put(Attributes.Name.MAIN_CLASS, mainClass);
221  }
222  if (targetLabel != null) {
223  attributes.put(JarHelper.TARGET_LABEL, targetLabel);
224  }
225  if (injectingRuleKind != null) {
226  attributes.put(JarHelper.INJECTING_RULE_KIND, injectingRuleKind);
227  }
228  ByteArrayOutputStream out = new ByteArrayOutputStream();
229  manifest.write(out);
230  return out.toByteArray();
231  }

238  public void execute() throws IOException {
239  try (OutputStream os = Files.newOutputStream(jarPath);
240  BufferedOutputStream bos = new BufferedOutputStream(os);
241  JarOutputStream out = new JarOutputStream(bos)) {
242 
243  // Create the manifest entry in the Jar file
244  writeManifestEntry(out, manifestContent());
245 
246  for (Map.Entry<String, Path> entry : jarEntries.entrySet()) {
247  copyEntry(out, entry.getKey(), entry.getValue());
248  }
249  }
