75 std::string ResolveEnvvars(const std::string &path) {
76  std::string result = path;
77  size_t start = 0;
78  while ((start = result.find("${", start)) != std::string::npos) {
79  // Just match to the next }
80  size_t end = result.find("}", start + 1);
81  if (end == std::string::npos) {
82  BAZEL_DIE(blaze_exit_code::LOCAL_ENVIRONMENTAL_ERROR)
83  << "ResolveEnvvars(" << path << "): incomplete variable at position "
84  << start;
85  }
86  // Extract the variable name
87  const std::string name = result.substr(start + 2, end - start - 2);
88  // Get the value from the environment
89  const char *c_value = getenv(name.c_str());
90  const std::string value = std::string(c_value ? c_value : "");
91  result.replace(start, end - start + 1, value);
92  start += value.length();
93  }
94  return result;
95 }

101 static std::string NormalizeAbsPath(const std::string &p) {
102  if (p.empty() || p[0] != '/') {
103  return "";
104  }
105  typedef std::string::size_type index;
106  std::vector<std::pair<index, index> > segments;
107  for (index s = 0; s < p.size();) {
108  index e = p.find_first_of('/', s);
109  if (e == std::string::npos) {
110  e = p.size();
111  }
112  if (e > s) {
113  if (p.compare(s, e - s, "..") == 0) {
114  if (!segments.empty()) {
115  segments.pop_back();
116  }
117  } else if (p.compare(s, e - s, ".") != 0) {
118  segments.push_back(std::make_pair(s, e - s));
119  }
120  }
121  s = e + 1;
122  }
123  if (segments.empty()) {
124  return "/";
125  } else {
126  std::stringstream r;
127  for (const auto &s : segments) {
128  r << "/" << p.substr(s.first, s.second);
129  }
130  if (p[p.size() - 1] == '/') {
131  r << "/";
132  }
133  return r.str();
134  }
135 }
