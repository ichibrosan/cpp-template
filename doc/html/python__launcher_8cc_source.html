32 ExitCode PythonBinaryLauncher::Launch() {
33  wstring python_binary = this->GetLaunchInfoByKey(PYTHON_BIN_PATH);
34 
35  // There are three kinds of values for `python_binary`:
36  // 1. An absolute path to a system interpreter. This is the case if
37  // `--python_path` is set by the
38  // user, or if a `py_runtime` is used that has `interpreter_path` set.
39  // 2. A runfile path to an in-workspace interpreter. This is the case if a
40  // `py_runtime` is used that has `interpreter` set.
41  // 3. The special constant, "python". This is the default case if neither of
42  // the above apply. Rlocation resolves runfiles paths to absolute paths, and
43  // if given an absolute path it leaves it alone, so it's suitable for cases 1
44  // and 2.
45  if (GetBinaryPathWithoutExtension(python_binary) != L"python") {
46  // Rlocation returns the original path if python_binary is an absolute path.
47  python_binary = this->Rlocation(python_binary, true);
48  }
49 
50  // If specified python binary path doesn't exist, then fall back to
51  // python.exe and hope it's in PATH.
52  if (!DoesFilePathExist(python_binary.c_str())) {
53  python_binary = L"python.exe";
54  }
55 
56  vector<wstring> args = this->GetCommandlineArguments();
57  wstring use_zip_file = this->GetLaunchInfoByKey(USE_ZIP_FILE);
58  wstring python_file;
59  if (use_zip_file == L"1") {
60  python_file = GetBinaryPathWithoutExtension(args[0]) + L".zip";
61  } else {
62  python_file = GetBinaryPathWithoutExtension(args[0]);
63  }
64 
65  // Replace the first argument with python file path
66  args[0] = python_file;
67 
68  for (int i = 1; i < args.size(); i++) {
69  args[i] = bazel::windows::WindowsEscapeArg(args[i]);
70  }
71 
72  return this->LaunchProcess(python_binary, args);
73 }
