31 ExitCode BashBinaryLauncher::Launch() {
32  wstring bash_binary = this->GetLaunchInfoByKey(BASH_BIN_PATH);
33 
34  // If bash_binary is already "bash" or "bash.exe", that means we want to
35  // rely on the shell binary in PATH, no need to do Rlocation.
36  if (GetBinaryPathWithoutExtension(bash_binary) != L"bash") {
37  // Rlocation returns the original path if bash_binary is an absolute path.
38  bash_binary = this->Rlocation(bash_binary, true);
39  }
40 
41  if (DoesFilePathExist(bash_binary.c_str())) {
42  wstring bash_bin_dir = GetParentDirFromPath(bash_binary);
43  wstring path_env;
44  GetEnv(L"PATH", &path_env);
45  path_env = bash_bin_dir + L";" + path_env;
46  SetEnv(L"PATH", path_env);
47  } else {
48  // If specified bash binary path doesn't exist, then fall back to
49  // bash.exe and hope it's in PATH.
50  bash_binary = L"bash.exe";
51  }
52 
53  vector<wstring> origin_args = this->GetCommandlineArguments();
54  wostringstream bash_command;
55  wstring bash_main_file = GetBinaryPathWithoutExtension(origin_args[0]);
56  bash_command << BashEscapeArg(bash_main_file);
57  for (int i = 1; i < origin_args.size(); i++) {
58  bash_command << L' ';
59  bash_command << BashEscapeArg(origin_args[i]);
60  }
61 
62  vector<wstring> args;
63  args.push_back(L"-c");
64  args.push_back(BashEscapeArg(bash_command.str()));
65  return this->LaunchProcess(bash_binary, args);
66 }
