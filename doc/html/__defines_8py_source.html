146 def declare_key_flag(flag_name, flag_values=_flagvalues.FLAGS):
147  """Declares one flag as key to the current module.
148 
149  Key flags are flags that are deemed really important for a module.
150  They are important when listing help messages; e.g., if the
151  --helpshort command-line flag is used, then only the key flags of the
152  main module are listed (instead of all flags, as in the case of
153  --helpfull).
154 
155  Sample usage:
156 
157  flags.declare_key_flag('flag_1')
158 
159  Args:
160  flag_name: str, the name of an already declared flag.
161  (Redeclaring flags as key, including flags implicitly key
162  because they were declared in this module, is a no-op.)
163  flag_values: FlagValues, the FlagValues instance in which the flag will
164  be declared as a key flag. This should almost never need to be
165  overridden.
166 
167  Raises:
168  ValueError: Raised if flag_name not defined as a Python flag.
169  """
170  if flag_name in _helpers.SPECIAL_FLAGS:
171  # Take care of the special flags, e.g., --flagfile, --undefok.
172  # These flags are defined in _SPECIAL_FLAGS, and are treated
173  # specially during flag parsing, taking precedence over the
174  # user-defined flags.
175  _internal_declare_key_flags([flag_name],
176  flag_values=_helpers.SPECIAL_FLAGS,
177  key_flag_values=flag_values)
178  return
179  try:
180  _internal_declare_key_flags([flag_name], flag_values=flag_values)
181  except KeyError:
182  raise ValueError(
183  'Key flag --%s needs to be defined in Python' % flag_name)

186 def adopt_module_key_flags(module, flag_values=_flagvalues.FLAGS):
187  """Declares that all flags key to a module are key to the current module.
188 
189  Args:
190  module: module, the module object from which all key flags will be declared
191  as key flags to the current module.
192  flag_values: FlagValues, the FlagValues instance in which the flags will
193  be declared as key flags. This should almost never need to be
194  overridden.

197  Error: Raised when given an argument that is a module name (a string),

218 def disclaim_key_flags():
219  """Declares that the current module will not define any more key flags.
220 
221  Normally, the module that calls the DEFINE_xxx functions claims the
222  flag to be its key flag. This is undesirable for modules that
223  define additional DEFINE_yyy functions with its own flag parsers and
224  serializers, since that module will accidentally claim flags defined
225  by DEFINE_yyy as its key flags. After calling this function, the
226  module disclaims flag definitions thereafter, so the key flags will
227  be correctly attributed to the caller of DEFINE_yyy.
228 
229  After calling this function, the module will not be able to define
230  any more flags. This function will affect all FlagValues objects.
231  """
232  globals_for_caller = sys._getframe(1).f_globals # pylint: disable=protected-access
233  module, _ = _helpers.get_module_object_and_name(globals_for_caller)
234  _helpers.disclaim_module_ids.add(id(module))
