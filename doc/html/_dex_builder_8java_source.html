127  @VisibleForTesting
128  static void buildDexArchive(Options options, Dexing dexing) throws Exception {
129  checkArgument(options.maxThreads > 0,
130  "--max_threads must be strictly positive, was: %s", options.maxThreads);
131  try (ZipFile in = new ZipFile(options.inputJar.toFile())) {
132  // Heuristic: use at most 1 thread per 1000 files in the input Jar
133  int threads = Math.min(options.maxThreads, in.size() / 1000 + 1);
134  ExecutorService executor = newFixedThreadPool(threads);
135  try (ZipOutputStream out = createZipOutputStream(options.outputZip)) {
136  produceDexArchive(in, out, executor, threads <= 1, dexing, null);
137  } finally {
138  executor.shutdown();
139  }
140  }

146  private static void runPersistentWorker() throws IOException {
147  ExecutorService executor = newFixedThreadPool(Runtime.getRuntime().availableProcessors());
148  Cache<DexingKey, byte[]> dexCache = CacheBuilder.newBuilder()
149  // Use at most 200 MB for cache and leave at least 25 MB of heap space alone. For reference:
150  // .class & class.dex files are around 1-5 KB, so this fits ~30K-35K class-dex pairs.
151  .maximumWeight(Math.min(Runtime.getRuntime().maxMemory() - 25 * ONE_MEG, 200 * ONE_MEG))
152  .weigher(new Weigher<DexingKey, byte[]>() {
153  @Override
154  public int weigh(DexingKey key, byte[] value) {
155  return key.classfileContent().length + value.length;
156  }
157  })
158  .build();
159  try {
160  while (true) {
161  WorkRequest request = WorkRequest.parseDelimitedFrom(System.in);
162  if (request == null) {
163  return;
164  }
165 
166  // Redirect dx's output so we can return it in response
167  ByteArrayOutputStream baos = new ByteArrayOutputStream();
168  PrintStream ps = new PrintStream(baos, /*autoFlush*/ true);
169  DxContext context = new DxContext(ps, ps);
170  // Make sure that we exit nonzero in case uncaught errors occur during processRequest.
171  int exitCode = 1;
172  try {
173  processRequest(executor, dexCache, context, request.getArgumentsList());
174  exitCode = 0; // success!
175  } catch (Exception e) {
176  // Deliberate catch-all so we can capture a stack trace.
177  // TODO(bazel-team): Consider canceling any outstanding futures created for this request
178  e.printStackTrace(ps);
179  } catch (Error e) {
180  e.printStackTrace();
181  e.printStackTrace(ps); // try capturing the error, may fail if out of memory
182  throw e; // rethrow to kill the worker
183  } finally {
184  // Try sending a response no matter what
185  String output;
186  try {
187  output = baos.toString();
188  } catch (Throwable t) { // most likely out of memory, so log with minimal memory needs
189  t.printStackTrace();
190  output = "check worker log for exceptions";
191  }
192  WorkResponse.newBuilder()
193  .setOutput(output)
194  .setExitCode(exitCode)
195  .build()
196  .writeDelimitedTo(System.out);
197  System.out.flush();
198  }
199  }
200  } finally {
201  executor.shutdown();
202  }
203  }

234  private static void produceDexArchive(
235  ZipFile in,
236  ZipOutputStream out,
237  ExecutorService executor,
238  boolean convertOnReaderThread,
239  Dexing dexing,
240  @Nullable Cache<DexingKey, byte[]> dexCache)
241  throws InterruptedException, ExecutionException, IOException {
242  // If we only have one thread in executor, we give a "direct" executor to the stuffer, which
243  // will convert .class files to .dex inline on the same thread that reads the input jar.
244  // This is an optimization that makes sure we can start writing the output file below while
245  // the stuffer is still working its way through the input.
246  DexConversionEnqueuer enqueuer = new DexConversionEnqueuer(in,
247  convertOnReaderThread ? MoreExecutors.newDirectExecutorService() : executor,
248  new DexConverter(dexing),
249  dexCache);
250  Future<?> enqueuerTask = executor.submit(enqueuer);
251  while (true) {
252  // Wait for next future in the queue *and* for that future to finish. To guarantee
253  // deterministic output we just write out the files in the order they appear, which is
254  // the same order as in the input zip.
255  ZipEntryContent file = enqueuer.getFiles().take().get();
256  if (file == null) {
257  // "done" marker indicating no more files coming.
258  // Make sure enqueuer terminates normally (any wait should be minimal). This in
259  // particular surfaces any exceptions thrown in the enqueuer.
260  enqueuerTask.get();
261  break;
262  }
263  out.putNextEntry(file.getEntry());
264  out.write(file.getContent());
265  out.closeEntry();
266  }
267  }
