116 def GetCallingModuleObjectAndName():
117  """Returns the module that's calling into this module.
118 
119  We generally use this function to get the name of the module calling a
120  DEFINE_foo... function.
121 
122  Returns:
123  The module object that called into this one.
124

126  AssertionError: if no calling module could be identified.

216 def GetFlagSuggestions(attempt, longopt_list):
217  """Get helpful similar matches for an invalid flag."""
218  # Don't suggest on very short strings, or if no longopts are specified.
219  if len(attempt) <= 2 or not longopt_list:
220  return []
221 
222  option_names = [v.split('=')[0] for v in longopt_list]
223 
224  # Find close approximations in flag prefixes.
225  # This also handles the case where the flag is spelled right but ambiguous.
226  distances = [(_DamerauLevenshtein(attempt, option[0:len(attempt)]), option)
227  for option in option_names]
228  distances.sort(key=lambda t: t[0])
229 
230  least_errors, _ = distances[0]
231  # Don't suggest excessively bad matches.
232  if least_errors >= _SUGGESTION_ERROR_RATE_THRESHOLD * len(attempt):
233  return []
234 
235  suggestions = []
236  for errors, name in distances:
237  if errors == least_errors:
238  suggestions.append(name)
239  else:
240  break
241  return suggestions

244 def _DamerauLevenshtein(a, b):
245  """Damerau-Levenshtein edit distance from a to b."""
246  memo = {}
247 
248  def Distance(x, y):
249  """Recursively defined string distance with memoization."""
250  if (x, y) in memo:
251  return memo[x, y]
252  if not x:
253  d = len(y)
254  elif not y:
255  d = len(x)
256  else:
257  d = min(
258  Distance(x[1:], y) + 1, # correct an insertion error
259  Distance(x, y[1:]) + 1, # correct a deletion error
260  Distance(x[1:], y[1:]) + (x[0] != y[0])) # correct a wrong character
261  if len(x) >= 2 and len(y) >= 2 and x[0] == y[1] and x[1] == y[0]:
262  # Correct a transposition.
263  t = Distance(x[2:], y[2:]) + 1
264  if d > t:
265  d = t
266 
267  memo[x, y] = d
268  return d
269  return Distance(a, b)

272 def TextWrap(text, length=None, indent='', firstline_indent=None):
273  """Wraps a given text to a maximum line length and returns it.
274 
275  It turns lines that only contain whitespace into empty lines, keeps new lines,
276  and expands tabs using 4 spaces.
277 
278  Args:
279  text: str, Text to wrap.
280  length: int, Maximum length of a line, includes indentation.
281  If this is None then use GetHelpWidth()
282  indent: str, Indent for all but first line.
283  firstline_indent: str, Indent for first line; if None, fall back to indent.
284 
285  Returns:
286  Wrapped text.
287 
288  Raises:
289  ValueError: if indent or firstline_indent not shorter than length.
290  """
291  # Get defaults where callee used None
292  if length is None:
293  length = GetHelpWidth()
294  if indent is None:
295  indent = ''
296  if firstline_indent is None:
297  firstline_indent = indent
298 
299  if len(indent) >= length:
300  raise ValueError('Length of indent exceeds length')
301  if len(firstline_indent) >= length:
302  raise ValueError('Length of first line indent exceeds length')
303 
304  text = text.expandtabs(4)
305 
306  result = []
307  # Create one wrapper for the first paragraph and one for subsequent
308  # paragraphs that does not have the initial wrapping.
309  wrapper = textwrap.TextWrapper(
310  width=length, initial_indent=firstline_indent, subsequent_indent=indent)
311  subsequent_wrapper = textwrap.TextWrapper(
312  width=length, initial_indent=indent, subsequent_indent=indent)
313 
314  # textwrap does not have any special treatment for newlines. From the docs:
315  # "...newlines may appear in the middle of a line and cause strange output.
316  # For this reason, text should be split into paragraphs (using
317  # str.splitlines() or similar) which are wrapped separately."
318  for paragraph in (p.strip() for p in text.splitlines()):
319  if paragraph:
320  result.extend(wrapper.wrap(paragraph))
321  else:
322  result.append('') # Keep empty lines.
323  # Replace initial wrapper with wrapper for subsequent paragraphs.
324  wrapper = subsequent_wrapper
325 
326  return '\n'.join(result)

329 def FlagDictToArgs(flag_map):
330  """Convert a dict of values into process call parameters.
331 
332  This method is used to convert a dictionary into a sequence of parameters
333  for a binary that parses arguments using this module.
334 
335  Args:
336  flag_map: a mapping where the keys are flag names (strings).
337  values are treated according to their type:
338  * If value is None, then only the name is emitted.
339  * If value is True, then only the name is emitted.
340  * If value is False, then only the name prepended with 'no' is emitted.
341  * If value is a string then --name=value is emitted.
342  * If value is a collection, this will emit --name=value1,value2,value3.
343  * Everything else is converted to string an passed as such.
344  Yields:
345  sequence of string suitable for a subprocess execution.
346  """
347  for key, value in six.iteritems(flag_map):
348  if value is None:
349  yield '--%s' % key
350  elif isinstance(value, bool):
351  if value:
352  yield '--%s' % key
353  else:
354  yield '--no%s' % key
355  elif isinstance(value, (bytes, type(u''))):
356  # We don't want strings to be handled like python collections.
357  yield '--%s=%s' % (key, value)
358  else:
359  # Now we attempt to deal with collections.
360  try:
361  yield '--%s=%s' % (key, ','.join(str(item) for item in value))
362  except TypeError:
363  # Default case.
364  yield '--%s=%s' % (key, value)

367 def DocToHelp(doc):
368  """Takes a __doc__ string and reformats it as help."""
369 
370  # Get rid of starting and ending white space. Using lstrip() or even
371  # strip() could drop more than maximum of first line and right space
372  # of last line.
373  doc = doc.strip()
374 
375  # Get rid of all empty lines.
376  whitespace_only_line = re.compile('^[ \t]+$', re.M)
377  doc = whitespace_only_line.sub('', doc)
378 
379  # Cut out common space at line beginnings.
380  doc = pep257.trim(doc)
381 
382  # Just like this module's comment, comments tend to be aligned somehow.
383  # In other words they all start with the same amount of white space.
384  # 1) keep double new lines;
385  # 2) keep ws after new lines if not empty line;
386  # 3) all other new lines shall be changed to a space;
387  # Solution: Match new lines between non white space and replace with space.
388  doc = re.sub(r'(?<=\S)\n(?=\S)', ' ', doc, flags=re.M)
389 
390  return doc
