23 // An interface to zlib's inflater. Usage:
24 // Inflater inflater;
25 // inflater.DataToInflate(data, data_size);
26 // for (;;) {
27 // int rc = inflater.Inflate(out_buffer, out_buffer_size);
28 // if (rc == Z_STREAM_END) {
29 // break;
30 // }
31 // // If we ran out of out_buffer, create a new one
32 // }
33 // inflater.reset();
34 // NOTE that the sizes of the input/output buffers in zlib are 32-bit entities.
35 // Call Inflater::DataToInflate multiple times if 'data_size' in the usage
36 // example exceeds 4GB-1.
37 class Inflater {
38  public:
39  Inflater() {
40  zstream_.zalloc = Z_NULL;
41  zstream_.zfree = Z_NULL;
42  zstream_.opaque = Z_NULL;
43  zstream_.avail_in = 0;
44  zstream_.next_in = nullptr;
45  int ret = inflateInit2(&zstream_, -MAX_WBITS);
46  if (ret != Z_OK) {
47  diag_errx(2, "inflateInit2 returned %d\n", ret);
48  }
49  }
50 
51  ~Inflater() { inflateEnd(&zstream_); }
52 
53  void reset() { inflateReset(&zstream_); }
54 
55  void DataToInflate(const uint8_t *in_buffer, uint32_t in_buffer_length) {
56  zstream_.next_in = const_cast<uint8_t *>(in_buffer);
57  zstream_.avail_in = in_buffer_length;
58  }
59 
60  int Inflate(uint8_t *out_buffer, uint32_t out_buffer_length) {
61  zstream_.next_out = out_buffer;
62  zstream_.avail_out = out_buffer_length;
63  return inflate(&zstream_, Z_SYNC_FLUSH);
64  }
65 
66  const uint8_t *next_in() const { return zstream_.next_in; }
67  uint64_t total_in() const { return zstream_.total_in; }
68 
69  uint32_t available_out() const { return zstream_.avail_out; }
70  uint64_t total_out() const { return zstream_.total_out; }
71 
72  const char *error_message() const { return zstream_.msg; }
73 
74  private:
75  z_stream zstream_;
76 };
