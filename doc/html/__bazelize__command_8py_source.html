24 def get_executable_path(py_binary_path):
25  """Returns the executable path of a py_binary.
26 
27  This returns the executable path of a py_binary that is in another Bazel
28  target's data dependencies.
29 
30  On Linux/macOS, it's the same as the py_binary_path.
31  On Windows, the py_binary_path points to a zip file, and Bazel 0.5.3+
32  generates a .cmd file that can be used to execute the py_binary.
33 
34  Args:
35  py_binary_path: string, the path of a py_binary that is in another Bazel
36  target's data dependencies.
37  """
38  if os.name == 'nt':
39  executable_path = py_binary_path + '.cmd'
40  if executable_path.startswith('\\\\?\\'):
41  # In Bazel 0.5.3 and Python 3, the paths starts with "\\?\".
42  # However, Python subprocess doesn't support those paths well.
43  # Strip them as we don't need the prefix.
44  # See this page for more informaton about "\\?\":
45  # https://msdn.microsoft.com/en-us/library/windows/desktop/aa365247.
46  executable_path = executable_path[4:]
47  return executable_path
48  else:
49  return py_binary_path
