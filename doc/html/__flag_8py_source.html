33 class Flag(object):
34  """Information about a command-line flag.
35 
36  'Flag' objects define the following fields:
37  .name - the name for this flag;
38  .default - the default value for this flag;
39  .default_unparsed - the unparsed default value for this flag.
40  .default_as_str - default value as repr'd string, e.g., "'true'" (or None);
41  .value - the most recent parsed value of this flag; set by parse();
42  .help - a help string or None if no help is available;
43  .short_name - the single letter alias for this flag (or None);
44  .boolean - if 'true', this flag does not accept arguments;
45  .present - true if this flag was parsed from command line flags;
46  .parser - an ArgumentParser object;
47  .serializer - an ArgumentSerializer object;
48  .allow_override - the flag may be redefined without raising an error, and
49  newly defined flag overrides the old one.
50  .allow_override_cpp - the flag may be redefined in C++ without raising an
51  error, value "transferred" to C++, and the flag is
52  replaced by the C++ flag after init;
53  .allow_hide_cpp - the flag may be redefined despite hiding a C++ flag with
54  the same name;
55  .using_default_value - the flag value has not been set by user;
56  .allow_overwrite - the flag may be parsed more than once without raising
57  an error, the last set value will be used;
58  .allow_using_method_names - whether this flag can be defined even if it has
59  a name that conflicts with a FlagValues method.
60 
61  The only public method of a 'Flag' object is parse(), but it is
62  typically only called by a 'FlagValues' object. The parse() method is
63  a thin wrapper around the 'ArgumentParser' parse() method. The parsed
64  value is saved in .value, and the .present attribute is updated. If
65  this flag was already present, an Error is raised.
66 
67  parse() is also called during __init__ to parse the default value and
68  initialize the .value attribute. This enables other python modules to
69  safely use flags even if the __main__ module neglects to parse the
70  command line arguments. The .present attribute is cleared after
71  __init__ parsing. If the default value is set to None, then the
72  __init__ parsing step is skipped and the .value attribute is
73  initialized to None.
74 
75  Note: The default value is also presented to the user in the help
76  string, so it is important that it be a legal value for this flag.
77  """

211  def _create_xml_dom_element(self, doc, module_name, is_key=False):
212  """Returns an XML element that contains this flag's information.
213 
214  This is information that is relevant to all flags (e.g., name,
215  meaning, etc.). If you defined a flag that has some other pieces of
216  info, then please override _ExtraXMLInfo.
217 
218  Please do NOT override this method.
219 
220  Args:
221  doc: minidom.Document, the DOM document it should create nodes from.
222  module_name: str,, the name of the module that defines this flag.
223  is_key: boolean, True iff this flag is key for main module.
224 
225  Returns:
226  A minidom.Element instance.
227  """
228  element = doc.createElement('flag')
229  if is_key:
230  element.appendChild(_helpers.create_xml_dom_element(doc, 'key', 'yes'))
231  element.appendChild(_helpers.create_xml_dom_element(
232  doc, 'file', module_name))
233  # Adds flag features that are relevant for all flags.
234  element.appendChild(_helpers.create_xml_dom_element(doc, 'name', self.name))
235  if self.short_name:
236  element.appendChild(_helpers.create_xml_dom_element(
237  doc, 'short_name', self.short_name))
238  if self.help:
239  element.appendChild(_helpers.create_xml_dom_element(
240  doc, 'meaning', self.help))
241  # The default flag value can either be represented as a string like on the
242  # command line, or as a Python object. We serialize this value in the
243  # latter case in order to remain consistent.
244  if self.serializer and not isinstance(self.default, str):
245  if self.default is not None:
246  default_serialized = self.serializer.serialize(self.default)
247  else:
248  default_serialized = ''
249  else:
250  default_serialized = self.default
251  element.appendChild(_helpers.create_xml_dom_element(
252  doc, 'default', default_serialized))
253  element.appendChild(_helpers.create_xml_dom_element(
254  doc, 'current', self.value))
255  element.appendChild(_helpers.create_xml_dom_element(
256  doc, 'type', self.flag_type()))
257  # Adds extra flag features this flag may have.
258  for e in self._extra_xml_dom_elements(doc):
259  element.appendChild(e)
260  return element

278 class BooleanFlag(Flag):

285

316 class MultiFlag(Flag):
317  """A flag that can appear multiple time on the command-line.
318 
319  The value of such a flag is a list that contains the individual values
320  from all the appearances of that flag on the command-line.
321 
322  See the __doc__ for Flag for most behavior of this class. Only
323  differences in behavior are described here:
324 
325  * The default value may be either a single value or a list of values.
326  A single value is interpreted as the [value] singleton list.
327 
328  * The value of the flag is always a list, even if the option was
329  only supplied once, and even if the default value is a single
330  value
331  """
332 
333  def __init__(self, *args, **kwargs):
334  super(MultiFlag, self).__init__(*args, **kwargs)
335  self.help += ';\n repeat this option to specify a list of values'
336 
337  def parse(self, arguments):
338  """Parses one or more arguments with the installed parser.
339 
340  Args:
341  arguments: a single argument or a list of arguments (typically a
342  list of default values); a single argument is converted
343  internally into a list containing one item.
344  """
345  new_values = self._parse(arguments)
346  if self.present:
347  self.value.extend(new_values)
348  else:
349  self.value = new_values
350  self.present += len(new_values)
351 
352  def _parse(self, arguments):
353  if not isinstance(arguments, list):
354  # Default value may be a list of values. Most other arguments
355  # will not be, so convert them into a single-item list to make
356  # processing simpler below.
357  arguments = [arguments]
358 
359  return [super(MultiFlag, self)._parse(item) for item in arguments]
