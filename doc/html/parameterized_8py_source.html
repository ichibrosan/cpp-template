15 """Adds support for parameterized tests to Python's unittest TestCase class.
16 
17 A parameterized test is a method in a test case that is invoked with different
18 argument tuples.
19 
20 A simple example:
21 
22  class AdditionExample(parameterized.TestCase):
23  @parameterized.parameters(
24  (1, 2, 3),
25  (4, 5, 9),
26  (1, 1, 3))
27  def testAddition(self, op1, op2, result):
28  self.assertEqual(result, op1 + op2)
29 
30 
31 Each invocation is a separate test case and properly isolated just
32 like a normal test method, with its own setUp/tearDown cycle. In the
33 example above, there are three separate testcases, one of which will
34 fail due to an assertion error (1 + 1 != 3).
35 
36 Parameters for invididual test cases can be tuples (with positional parameters)
37 or dictionaries (with named parameters):
38 
39  class AdditionExample(parameterized.TestCase):
40  @parameterized.parameters(
41  {'op1': 1, 'op2': 2, 'result': 3},
42  {'op1': 4, 'op2': 5, 'result': 9},
43  )
44  def testAddition(self, op1, op2, result):
45  self.assertEqual(result, op1 + op2)
46 
47 If a parameterized test fails, the error message will show the
48 original test name (which is modified internally) and the arguments
49 for the specific invocation, which are part of the string returned by
50 the shortDescription() method on test cases.
51 
52 The id method of the test, used internally by the unittest framework,
53 is also modified to show the arguments. To make sure that test names
54 stay the same across several invocations, object representations like
55 
56  >>> class Foo(object):
57  ... pass
58  >>> repr(Foo())
59  '<__main__.Foo object at 0x23d8610>'
60 
61 are turned into '<__main__.Foo>'. For even more descriptive names,
62 especially in test logs, you can use the named_parameters decorator. In
63 this case, only tuples or dicts are supported. For tuples, the first parameters
64 has to be a string (or an object that returns an apt name when converted via
65 str()). For dicts, a value for the key 'testcase_name' must be present and must
66 be a string (or an object that returns an apt name when converted via str()):
67 
68  class NamedExample(parameterized.TestCase):
69  @parameterized.named_parameters(
70  ('Normal', 'aa', 'aaa', True),
71  ('EmptyPrefix', '', 'abc', True),
72  ('BothEmpty', '', '', True))
73  def testStartsWith(self, prefix, string, result):
74  self.assertEqual(result, string.startswith(prefix))
75 
76  class NamedExample(parameterized.TestCase):
77  @parameterized.named_parameters(
78  {'testcase_name': 'Normal',
79  'result': True, 'string': 'aaa', 'prefix': 'aa'},
80  {'testcase_name': 'EmptyPrefix',
81  'result': True, 'string: 'abc', 'prefix': ''},
82  {'testcase_name': 'BothEmpty',
83  'result': True, 'string': '', 'prefix': ''})
84  def testStartsWith(self, prefix, string, result):
85  self.assertEqual(result, string.startswith(prefix))
86 
87 Named tests also have the benefit that they can be run individually
88 from the command line:
89 
90  $ testmodule.py NamedExample.testStartsWithNormal
91  .

97 Parameterized Classes
98 =====================
99 If invocation arguments are shared across test methods in a single
100 TestCase class, instead of decorating all test methods
101 individually, the class itself can be decorated:
102 
103  @parameterized.parameters(
104  (1, 2, 3),
105  (4, 5, 9))
106  class ArithmeticTest(parameterized.TestCase):
107  def testAdd(self, arg1, arg2, result):
108  self.assertEqual(arg1 + arg2, result)
109 
110  def testSubtract(self, arg1, arg2, result):
111  self.assertEqual(result - arg1, arg2)
112 
113 Inputs from Iterables
114 =====================
115 If parameters should be shared across several test cases, or are dynamically
116 created from other sources, a single non-tuple iterable can be passed into
117 the decorator. This iterable will be used to obtain the test cases:
118 
119  class AdditionExample(parameterized.TestCase):
120  @parameterized.parameters(
121  c.op1, c.op2, c.result for c in testcases
122  )
123  def testAddition(self, op1, op2, result):
124  self.assertEqual(result, op1 + op2)
125

127 Single-Argument Test Methods
128 ============================
129 If a test method takes only one argument, the single arguments must not be
130 wrapped into a tuple:
131 
132  class NegativeNumberExample(parameterized.TestCase):
133  @parameterized.parameters(
134  -1, -3, -4, -5
135  )
136  def testIsNegative(self, arg):
137  self.assertTrue(IsNegative(arg))
138 
139 
140 List/tuple as a Single Argument
141 ===============================
142 If a test method takes a single argument of a list/tuple, it must be wrapped
143 inside a tuple:
144 
145  class ZeroSumExample(parameterized.TestCase):
146  @parameterized.parameters(
147  ([-1, 0, 1], ),
148  ([-2, 0, 2], ),
149  )
150  def testSumIsZero(self, arg):
151  self.assertEqual(0, sum(arg))

200  def __init__(self, test_method, testcases, naming_type):
201  """Returns concrete test functions for a test and a list of parameters.
202 
203  The naming_type is used to determine the name of the concrete
204  functions as reported by the unittest framework. If naming_type is
205  _FIRST_ARG, the testcases must be tuples, and the first element must
206  have a string representation that is a valid Python identifier.
207 
208  Args:
209  test_method: The decorated test method.
210  testcases: (list of tuple/dict) A list of parameter
211  tuples/dicts for individual test invocations.
212  naming_type: The test naming type, either _NAMED or _ARGUMENT_REPR.
213  """

215  self.testcases = testcases
216  self._naming_type = naming_type

225  'one, in which case you should reverse the order.')

232  def make_bound_param_test(testcase_params):
233  @functools.wraps(test_method)
234  def bound_param_test(self):
235  if isinstance(testcase_params, collections.Mapping):
236  test_method(self, **testcase_params)
237  elif _non_string_or_bytes_iterable(testcase_params):
238  test_method(self, *testcase_params)
239  else:
240  test_method(self, testcase_params)
241 
242  if naming_type is _NAMED:
243  # Signal the metaclass that the name of the test function is unique
244  # and descriptive.
245  bound_param_test.__x_use_name__ = True
246 
247  testcase_name = None
248  if isinstance(testcase_params, collections.Mapping):
249  if _NAMED_DICT_KEY not in testcase_params:
250  raise RuntimeError(
251  'Dict for named tests must contain key "%s"' % _NAMED_DICT_KEY)
252  # Create a new dict to avoid modifying the supplied testcase_params.
253  testcase_name = testcase_params[_NAMED_DICT_KEY]
254  testcase_params = {k: v for k, v in six.iteritems(testcase_params)
255  if k != _NAMED_DICT_KEY}
256  elif _non_string_or_bytes_iterable(testcase_params):
257  testcase_name = testcase_params[0]
258  testcase_params = testcase_params[1:]
259  else:
260  raise RuntimeError(
261  'Named tests must be passed a dict or non-string iterable.')
262 
263  # Support PEP-8 underscore style for test naming if used.
264  if (bound_param_test.__name__.startswith('test_')
265  and testcase_name
266  and not testcase_name.startswith('_')):
267  bound_param_test.__name__ += '_'
268 
269  bound_param_test.__name__ += str(testcase_name)
270  elif naming_type is _ARGUMENT_REPR:
271  # If it's a generator, convert it to a tuple and treat them as
272  # parameters.
273  if isinstance(testcase_params, types.GeneratorType):
274  testcase_params = tuple(testcase_params)
275  # The metaclass creates a unique, but non-descriptive method name for
276  # _ARGUMENT_REPR tests using an indexed suffix.
277  # To keep test names descriptive, only the original method name is used.
278  # To make sure test names are unique, we add a unique descriptive suffix
279  # __x_extra_id__ for every test.
280  extra_id = '(%s)' % (_format_parameter_list(testcase_params),)
281  extra_ids[extra_id] += 1
282  while extra_ids[extra_id] > 1:
283  extra_id = '%s (%d)' % (extra_id, extra_ids[extra_id])
284  extra_ids[extra_id] += 1
285  bound_param_test.__x_extra_id__ = extra_id
286  else:
287  raise RuntimeError('%s is not a valid naming type.' % (naming_type,))
288 
289  bound_param_test.__doc__ = '%s(%s)' % (
290  bound_param_test.__name__, _format_parameter_list(testcase_params))
291  if test_method.__doc__:
292  bound_param_test.__doc__ += '\n%s' % (test_method.__doc__,)
293  return bound_param_test

298 def _modify_class(class_object, testcases, naming_type):
299  assert not getattr(class_object, '_test_method_ids', None), (
300  'Cannot add parameters to %s. Either it already has parameterized '
301  'methods, or its super class is also a parameterized class.' % (
302  class_object,))
303  class_object._test_method_ids = test_method_ids = {}
304  for name, obj in six.iteritems(class_object.__dict__.copy()):
305  if (name.startswith(unittest.TestLoader.testMethodPrefix)
306  and isinstance(obj, types.FunctionType)):
307  delattr(class_object, name)
308  methods = {}
309  _update_class_dict_for_param_test_case(
310  methods, test_method_ids, name,
311  _ParameterizedTestIter(obj, testcases, naming_type))
312  for name, meth in six.iteritems(methods):
313  setattr(class_object, name, meth)

316 def _parameter_decorator(naming_type, testcases):
317  """Implementation of the parameterization decorators.
318 
319  Args:
320  naming_type: The naming type.
321  testcases: Testcase parameters.
322 
323  Raises:
324  NoTestsError: Raised when the decorator generates no tests.
325 
326  Returns:
327  A function for modifying the decorated object.
328  """
329  def _apply(obj):
330  if isinstance(obj, type):
331  _modify_class(obj, testcases, naming_type)
332  return obj
333  else:
334  return _ParameterizedTestIter(obj, testcases, naming_type)
335 
336  if (len(testcases) == 1 and
337  not isinstance(testcases[0], tuple) and
338  not (naming_type == _NAMED and
339  isinstance(testcases[0], collections.Mapping))):
340  # Support using a single non-tuple parameter as a list of test cases.
341  # Note in named parameters case, the single non-tuple parameter can't be
342  # Mapping either, which means a single named parameter case.
343  assert _non_string_or_bytes_iterable(testcases[0]), (
344  'Single parameter argument must be a non-string iterable')
345  testcases = testcases[0]
346 
347  if not isinstance(testcases, collections.Sequence):
348  testcases = list(testcases)
349  if not testcases:
350  raise NoTestsError(
351  'parameterized test decorators did not generate any tests. '
352  'Make sure you specify non-empty parameters, '
353  'and do not reuse generators more than once.')
354 
355  return _apply

399 class TestGeneratorMetaclass(type):
400  """Metaclass for test cases with test generators.
401 
402  A test generator is an iterable in a testcase that produces callables. These
403  callables must be single-argument methods. These methods are injected into
404  the class namespace and the original iterable is removed. If the name of the
405  iterable conforms to the test pattern, the injected methods will be picked
406  up as tests by the unittest framework.
407 
408  In general, it is supposed to be used in conjuction with the
409  parameters decorator.
410  """
411 
412  def __new__(mcs, class_name, bases, dct):
413  test_method_ids = dct.setdefault('_test_method_ids', {})
414  for name, obj in six.iteritems(dct.copy()):
415  if (name.startswith(unittest.TestLoader.testMethodPrefix) and
416  _non_string_or_bytes_iterable(obj)):
417  iterator = iter(obj)
418  dct.pop(name)
419  _update_class_dict_for_param_test_case(
420  dct, test_method_ids, name, iterator)
421  # If the base class is a subclass of parameterized.TestCase, inherit its
422  # _test_method_ids too.
423  for base in bases:
424  # Check if the base has _test_method_ids first, then check if it's a
425  # subclass of parameterized.TestCase. Otherwise when this is called for
426  # the parameterized.TestCase definition itself, this raises because
427  # itself is not defined yet. This works as long as absltest.TestCase does
428  # not define _test_method_ids.
429  if getattr(base, '_test_method_ids', None) and issubclass(base, TestCase):
430  for test_method, test_method_id in six.iteritems(base._test_method_ids):
431  # test_method may both exists in base and this class.
432  # This class's method overrides base class's.
433  # That's why it should only inherit it if it does not exist.
434  test_method_ids.setdefault(test_method, test_method_id)
435 
436  return type.__new__(mcs, class_name, bases, dct)

439 def _update_class_dict_for_param_test_case(
440  dct, test_method_ids, name, iterator):
441  """Adds individual test cases to a dictionary.
442 
443  Args:
444  dct: The target dictionary.
445  test_method_ids: The dictionary for mapping names to test IDs.
446  name: The original name of the test case.
447  iterator: The iterator generating the individual test cases.
448  """
449  for idx, func in enumerate(iterator):
450  assert callable(func), 'Test generators must yield callables, got %r' % (
451  func,)
452  if getattr(func, '__x_use_name__', False):
453  original_name = func.__name__
454  new_name = original_name
455  else:
456  original_name = name
457  new_name = '%s%d' % (original_name, idx)
458  assert new_name not in dct, (
459  'Name of parameterized test case "%s" not unique' % (new_name,))
460  dct[new_name] = func
461  test_method_id = original_name + getattr(func, '__x_extra_id__', '')
462  assert test_method_id not in test_method_ids.values(), (
463  'Id of parameterized test case "%s" not unique' % (test_method_id,))
464  test_method_ids[new_name] = test_method_id
