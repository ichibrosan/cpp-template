77 class NativeOutputStream {
78  public:
79  NativeOutputStream()
80  : handle_(INVALID_HANDLE_VALUE), error_(L""), closed_(false) {}
81 
82  void Close() {
83  closed_.store(true);
84  if (handle_ == INVALID_HANDLE_VALUE) {
85  return;
86  }
87 
88  // CancelIoEx only cancels I/O operations in the current process.
89  // https://msdn.microsoft.com/en-us/library/windows/desktop/aa363792(v=vs.85).aspx
90  //
91  // Therefore if this process bequested `handle_` to a child process, we
92  // cannot cancel I/O in the child process.
93  CancelIoEx(handle_, NULL);
94  CloseHandle(handle_);
95  handle_ = INVALID_HANDLE_VALUE;
96  }
97 
98  void SetHandle(HANDLE handle) { handle_ = handle; }
99 
100  jint ReadStream(JNIEnv* env, jbyteArray java_bytes, jint offset,
101  jint length) {
102  JavaByteArray bytes(env, java_bytes);
103  if (offset < 0 || length <= 0 || offset > bytes.size() - length) {
104  error_ = bazel::windows::MakeErrorMessage(WSTR(__FILE__), __LINE__,
105  L"nativeReadStream", L"",
106  L"Array index out of bounds");
107  return -1;
108  }
109 
110  if (handle_ == INVALID_HANDLE_VALUE || closed_.load()) {
111  error_ = L"";
112  return 0;
113  }
114 
115  DWORD bytes_read;
116  if (!::ReadFile(handle_, bytes.ptr() + offset, length, &bytes_read, NULL)) {
117  // Check if either the other end closed the pipe or we did it with
118  // NativeOutputStream.Close() . In the latter case, we'll get a "system
119  // call interrupted" error.
120  if (GetLastError() == ERROR_BROKEN_PIPE || closed_.load()) {
121  // End of file.
122  error_ = L"";
123  bytes_read = 0;
124  } else {
125  DWORD err_code = GetLastError();
126  error_ = bazel::windows::MakeErrorMessage(
127  WSTR(__FILE__), __LINE__, L"nativeReadStream", L"", err_code);
128  bytes_read = -1;
129  }
130  } else {
131  error_ = L"";
132  }
133 
134  return bytes_read;
135  }
136 
137  // Return the last error as a human-readable string and clear it.
138  jstring GetLastErrorAsString(JNIEnv* env) {
139  jstring result = env->NewString(
140  reinterpret_cast<const jchar*>(error_.c_str()), error_.size());
141  error_ = L"";
142  return result;
143  }
144 
145  private:
146  HANDLE handle_;
147  std::wstring error_;
148  std::atomic<bool> closed_;
149 };

152  public:
153  NativeProcess() : stdout_(), stderr_(), error_(L"") {}

160  jboolean Create(JNIEnv* env, jstring java_argv0, jstring java_argv_rest,
161  jbyteArray java_env, jstring java_cwd,
162  jstring java_stdout_redirect, jstring java_stderr_redirect,
163  jboolean redirectErrorStream) {
164  std::wstring wpath(bazel::windows::GetJavaWpath(env, java_argv0));
165 
166  std::wstring stdout_redirect = bazel::windows::AddUncPrefixMaybe(
167  bazel::windows::GetJavaWpath(env, java_stdout_redirect));
168  std::wstring stderr_redirect = bazel::windows::AddUncPrefixMaybe(
169  bazel::windows::GetJavaWpath(env, java_stderr_redirect));
170 
171  const bool stdout_is_stream = stdout_redirect.empty();
172  const bool stderr_is_stream =
173  redirectErrorStream ? stdout_is_stream : stderr_redirect.empty();
174  const bool stderr_same_handle_as_stdout =
175  redirectErrorStream ||
176  (!stderr_redirect.empty() &&
177  stderr_redirect.size() == stdout_redirect.size() &&
178  _wcsnicmp(stderr_redirect.c_str(), stdout_redirect.c_str(),
179  stderr_redirect.size()) == 0);
180 
181  SECURITY_ATTRIBUTES sa = {0};
182  sa.nLength = sizeof(SECURITY_ATTRIBUTES);
183  sa.bInheritHandle = TRUE;
184 
185  // Standard file handles are closed even if the process was successfully
186  // created. If this was not so, operations on these file handles would not
187  // return immediately if the process is terminated.
188  // Therefore we make these handles auto-closing (by using AutoHandle).
189  bazel::windows::AutoHandle stdin_process;
190  bazel::windows::AutoHandle stdout_process;
191  bazel::windows::AutoHandle stderr_process;
192 
193  JavaByteArray env_map(env, java_env);
194  if (env_map.ptr() != nullptr) {
195  if (env_map.size() < 4) {
196  error_ = bazel::windows::MakeErrorMessage(
197  WSTR(__FILE__), __LINE__, L"nativeCreateProcess", wpath,
198  std::wstring(
199  L"the environment must be at least 4 bytes long, was ") +
200  ToString(env_map.size()) + L" bytes");
201  return false;
202  } else if (env_map.ptr()[env_map.size() - 1] != 0 ||
203  env_map.ptr()[env_map.size() - 2] != 0 ||
204  env_map.ptr()[env_map.size() - 3] != 0 ||
205  env_map.ptr()[env_map.size() - 4] != 0) {
206  error_ = bazel::windows::MakeErrorMessage(
207  WSTR(__FILE__), __LINE__, L"nativeCreateProcess", wpath,
208  L"environment array must end with 4 null bytes");
209  return false;
210  }
211  }
212 
213  {
214  HANDLE pipe_read_h, pipe_write_h;
215  if (!CreatePipe(&pipe_read_h, &pipe_write_h, &sa, 0)) {
216  DWORD err_code = GetLastError();
217  error_ = bazel::windows::MakeErrorMessage(
218  WSTR(__FILE__), __LINE__, L"nativeCreateProcess", wpath, err_code);
219  return false;
220  }
221  stdin_process = pipe_read_h;
222  stdin_ = pipe_write_h;
223  }
224 
225  if (!stdout_is_stream) {
226  stdout_.Close();
227 
228  stdout_process = CreateFileW(
229  /* lpFileName */ stdout_redirect.c_str(),
230  /* dwDesiredAccess */ GENERIC_WRITE,
231  // Must share for reading, otherwise symlink-following file existence
232  // checks (e.g. java.nio.file.Files.exists()) fail.
233  /* dwShareMode */ FILE_SHARE_READ,
234  /* lpSecurityAttributes */ &sa,
235  /* dwCreationDisposition */ OPEN_ALWAYS,
236  /* dwFlagsAndAttributes */ FILE_ATTRIBUTE_NORMAL,
237  /* hTemplateFile */ NULL);
238 
239  if (!stdout_process.IsValid()) {
240  DWORD err_code = GetLastError();
241  error_ = bazel::windows::MakeErrorMessage(WSTR(__FILE__), __LINE__,
242  L"nativeCreateProcess",
243  stdout_redirect, err_code);
244  return false;
245  }
246  if (!SetFilePointerEx(stdout_process, {0}, NULL, FILE_END)) {
247  DWORD err_code = GetLastError();
248  error_ = bazel::windows::MakeErrorMessage(WSTR(__FILE__), __LINE__,
249  L"nativeCreateProcess",
250  stdout_redirect, err_code);
251  return false;
252  }
253  } else {
254  HANDLE pipe_read_h, pipe_write_h;
255  if (!CreatePipe(&pipe_read_h, &pipe_write_h, &sa, 0)) {
256  DWORD err_code = GetLastError();
257  error_ = bazel::windows::MakeErrorMessage(
258  WSTR(__FILE__), __LINE__, L"nativeCreateProcess", wpath, err_code);
259  return false;
260  }
261  stdout_.SetHandle(pipe_read_h);
262  stdout_process = pipe_write_h;
263  }
264 
265  if (stderr_same_handle_as_stdout) {
266  HANDLE stdout_process_dup_h;
267  if (!DuplicateHandle(GetCurrentProcess(), stdout_process,
268  GetCurrentProcess(), &stdout_process_dup_h, 0, TRUE,
269  DUPLICATE_SAME_ACCESS)) {
270  DWORD err_code = GetLastError();
271  error_ = bazel::windows::MakeErrorMessage(
272  WSTR(__FILE__), __LINE__, L"nativeCreateProcess", wpath, err_code);
273  return false;
274  }
275  if (!stderr_is_stream) {
276  stderr_.Close();
277  }
278 
279  stderr_process = stdout_process_dup_h;
280  } else if (!stderr_redirect.empty()) {
281  stderr_.Close();
282  stderr_process = CreateFileW(
283  /* lpFileName */ stderr_redirect.c_str(),
284  /* dwDesiredAccess */ GENERIC_WRITE,
285  // Must share for reading, otherwise symlink-following file existence
286  // checks (e.g. java.nio.file.Files.exists()) fail.
287  /* dwShareMode */ FILE_SHARE_READ,
288  /* lpSecurityAttributes */ &sa,
289  /* dwCreationDisposition */ OPEN_ALWAYS,
290  /* dwFlagsAndAttributes */ FILE_ATTRIBUTE_NORMAL,
291  /* hTemplateFile */ NULL);
292 
293  if (!stderr_process.IsValid()) {
294  DWORD err_code = GetLastError();
295  error_ = bazel::windows::MakeErrorMessage(WSTR(__FILE__), __LINE__,
296  L"nativeCreateProcess",
297  stderr_redirect, err_code);
298  return false;
299  }
300  if (!SetFilePointerEx(stderr_process, {0}, NULL, FILE_END)) {
301  DWORD err_code = GetLastError();
302  error_ = bazel::windows::MakeErrorMessage(WSTR(__FILE__), __LINE__,
303  L"nativeCreateProcess",
304  stderr_redirect, err_code);
305  return false;
306  }
307  } else {
308  HANDLE pipe_read_h, pipe_write_h;
309  if (!CreatePipe(&pipe_read_h, &pipe_write_h, &sa, 0)) {
310  DWORD err_code = GetLastError();
311  error_ = bazel::windows::MakeErrorMessage(
312  WSTR(__FILE__), __LINE__, L"nativeCreateProcess", wpath, err_code);
313  return false;
314  }
315  stderr_.SetHandle(pipe_read_h);
316  stderr_process = pipe_write_h;
317  }
318  return proc_.Create(
319  wpath, bazel::windows::GetJavaWstring(env, java_argv_rest),
320  env_map.ptr(), bazel::windows::GetJavaWpath(env, java_cwd),
321  stdin_process, stdout_process, stderr_process, nullptr, &error_);
322  }

341  jint WriteStdin(JNIEnv* env, jbyteArray java_bytes, jint offset,
342  jint length) {
343  JavaByteArray bytes(env, java_bytes);
344  if (offset < 0 || length <= 0 || offset > bytes.size() - length) {
345  error_ = bazel::windows::MakeErrorMessage(
346  WSTR(__FILE__), __LINE__, L"NativeProcess:WriteStdin",
347  ToString(GetPid()), L"Array index out of bounds");
348  return -1;
349  }
350 
351  DWORD bytes_written;
352 
353  if (!::WriteFile(stdin_, bytes.ptr() + offset, length, &bytes_written,
354  NULL)) {
355  DWORD err_code = GetLastError();
356  error_ = bazel::windows::MakeErrorMessage(WSTR(__FILE__), __LINE__,
357  L"NativeProcess:WriteStdin",
358  ToString(GetPid()), err_code);
359  return -1;
360  }
361 
362  error_ = L"";
363  return bytes_written;
364  }
