54 wstring GetLastErrorString(DWORD error_code) {
55  if (error_code == 0) {
56  return L"";
57  }
58 
59  LPWSTR message = NULL;
60  DWORD size = FormatMessageW(
61  FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS |
62  FORMAT_MESSAGE_ALLOCATE_BUFFER,
63  NULL, error_code, LANG_USER_DEFAULT, (LPWSTR)&message, 0, NULL);
64 
65  if (size == 0) {
66  wstringstream err;
67  DWORD format_message_error = GetLastError();
68  err << L"Error code " << error_code
69  << L"; cannot format message due to error code "
70  << format_message_error;
71  return err.str();
72  }
73 
74  wstring result(message);
75  HeapFree(GetProcessHeap(), LMEM_FIXED, message);
76  return result;

79 bool AutoAttributeList::Create(HANDLE stdin_h, HANDLE stdout_h, HANDLE stderr_h,
80  std::unique_ptr<AutoAttributeList>* result,
81  wstring* error_msg) {
82  if (stdin_h == INVALID_HANDLE_VALUE && stdout_h == INVALID_HANDLE_VALUE &&
83  stderr_h == INVALID_HANDLE_VALUE) {
84  result->reset(new AutoAttributeList());
85  return true;
86  }
87 
88  static constexpr DWORD kAttributeCount = 1;
89 
90  SIZE_T size = 0;
91  // According to MSDN, the first call to InitializeProcThreadAttributeList is
92  // expected to fail.
93  InitializeProcThreadAttributeList(NULL, kAttributeCount, 0, &size);
94  SetLastError(ERROR_SUCCESS);
95 
96  std::unique_ptr<uint8_t[]> data(new uint8_t[size]);
97  LPPROC_THREAD_ATTRIBUTE_LIST attrs =
98  reinterpret_cast<LPPROC_THREAD_ATTRIBUTE_LIST>(data.get());
99  if (!InitializeProcThreadAttributeList(attrs, kAttributeCount, 0, &size)) {
100  if (error_msg) {
101  DWORD err = GetLastError();
102  *error_msg =
103  MakeErrorMessage(WSTR(__FILE__), __LINE__,
104  L"InitializeProcThreadAttributeList", L"", err);
105  }
106  return false;
107  }
108 
109  std::unique_ptr<AutoAttributeList> attr_list(
110  new AutoAttributeList(std::move(data), stdin_h, stdout_h, stderr_h));
111  if (!UpdateProcThreadAttribute(
112  attrs, 0, PROC_THREAD_ATTRIBUTE_HANDLE_LIST,
113  attr_list->handles_.ValidHandles(),
114  attr_list->handles_.ValidHandlesCount() * sizeof(HANDLE), NULL,
115  NULL)) {
116  if (error_msg) {
117  DWORD err = GetLastError();
118  *error_msg = MakeErrorMessage(WSTR(__FILE__), __LINE__,
119  L"UpdateProcThreadAttribute", L"", err);
120  }
121  return false;
122  }
123  *result = std::move(attr_list);
124  return true;
125 }

197 wstring AsShortPath(wstring path, wstring* result) {
198  if (path.empty()) {
199  result->clear();
200  return L"";
201  }
202  if (path[0] == '"') {
203  return MakeErrorMessage(WSTR(__FILE__), __LINE__, L"AsShortPath", path,
204  L"path should not be quoted");
205  }
206  if (IsSeparator(path[0])) {
207  return MakeErrorMessage(WSTR(__FILE__), __LINE__, L"AsShortPath", path,
208  L"path is absolute without a drive letter");
209  }
210  if (Contains(path, L"/./") || Contains(path, L"\\.\\") ||
211  Contains(path, L"/..") || Contains(path, L"\\..")) {
212  return MakeErrorMessage(WSTR(__FILE__), __LINE__, L"AsShortPath", path,
213  L"path is not normalized");
214  }
215  if (path.size() >= MAX_PATH && !HasSeparator(path)) {
216  return MakeErrorMessage(WSTR(__FILE__), __LINE__, L"AsShortPath", path,
217  L"path is just a file name but too long");
218  }
219  if (HasSeparator(path) && !HasDriveSpecifierPrefix(path.c_str())) {
220  return MakeErrorMessage(WSTR(__FILE__), __LINE__, L"AsShortPath", path,
221  L"path is not absolute");
222  }
223  // At this point we know the path is either just a file name (shorter than
224  // MAX_PATH), or an absolute, normalized, Windows-style path (of any length).
225 
226  std::replace(path.begin(), path.end(), '/', '\\');
227  // Fast-track: the path is already short.
228  if (path.size() < MAX_PATH) {
229  *result = path;
230  return L"";
231  }
232  // At this point we know that the path is at least MAX_PATH long and that it's
233  // absolute, normalized, and Windows-style.
234 
235  wstring wlong = wstring(L"\\\\?\\") + path;
236 
237  // Experience shows that:
238  // - GetShortPathNameW's result has a "\\?\" prefix if and only if the input
239  // did too (though this behavior is not documented on MSDN)
240  // - CreateProcess{A,W} only accept an executable of MAX_PATH - 1 length
241  // Therefore for our purposes the acceptable shortened length is
242  // MAX_PATH + 4 (null-terminated). That is, MAX_PATH - 1 for the shortened
243  // path, plus a potential "\\?\" prefix that's only there if `wlong` also had
244  // it and which we'll omit from `result`, plus a null terminator.
245  static const size_t kMaxShortPath = MAX_PATH + 4;
246 
247  WCHAR wshort[kMaxShortPath];
248  DWORD wshort_size = ::GetShortPathNameW(wlong.c_str(), NULL, 0);
249  if (wshort_size == 0) {
250  DWORD err_code = GetLastError();
251  wstring res = MakeErrorMessage(WSTR(__FILE__), __LINE__,
252  L"GetShortPathNameW", wlong, err_code);
253  return res;
254  }
255 
256  if (wshort_size >= kMaxShortPath) {
257  return MakeErrorMessage(WSTR(__FILE__), __LINE__, L"GetShortPathNameW",
258  wlong, L"cannot shorten the path enough");
259  }
260  GetShortPathNameW(wlong.c_str(), wshort, kMaxShortPath);
261  result->assign(wshort + 4);
262  return L"";
263 }

265 wstring AsExecutablePathForCreateProcess(wstring path, wstring* result) {
266  if (path.empty()) {
267  return MakeErrorMessage(WSTR(__FILE__), __LINE__,
268  L"AsExecutablePathForCreateProcess", path,
269  L"path should not be empty");
270  }
271  if (IsSeparator(path[0])) {
272  return MakeErrorMessage(WSTR(__FILE__), __LINE__,
273  L"AsExecutablePathForCreateProcess", path,
274  L"path is absolute without a drive letter");
275  }
276  if (HasSeparator(path) && !HasDriveSpecifierPrefix(path.c_str())) {
277  wstring cwd;
278  DWORD err;
279  if (!GetCwd(&cwd, &err)) {
280  return MakeErrorMessage(WSTR(__FILE__), __LINE__,
281  L"AsExecutablePathForCreateProcess", path, err);
282  }
283  path = cwd + L"\\" + path;
284  }
285  wstring error = AsShortPath(path, result);
286  if (!error.empty()) {
287  return MakeErrorMessage(WSTR(__FILE__), __LINE__,
288  L"AsExecutablePathForCreateProcess", path, error);
289  }
290  // Quote the path in case it's something like "c:\foo\app name.exe".
291  // Do this unconditionally, there's no harm in quoting. Quotes are not
292  // allowed inside paths so we don't need to escape quotes.
293  QuotePath(*result, result);
294  return L"";
295 }
