107  public static Runfiles create(Map<String, String> env) throws IOException {
108  if (isManifestOnly(env)) {
109  // On Windows, Bazel sets RUNFILES_MANIFEST_ONLY=1.
110  // On every platform, Bazel also sets RUNFILES_MANIFEST_FILE, but on Linux and macOS it's
111  // faster to use RUNFILES_DIR.
112  return new ManifestBased(getManifestPath(env));
113  } else {
114  return new DirectoryBased(getRunfilesDir(env));
115  }
116  }

200  private static Map<String, String> loadRunfiles(String path) throws IOException {
201  HashMap<String, String> result = new HashMap<>();
202  try (BufferedReader r =
203  new BufferedReader(
204  new InputStreamReader(new FileInputStream(path), StandardCharsets.UTF_8))) {
205  String line = null;
206  while ((line = r.readLine()) != null) {
207  int index = line.indexOf(' ');
208  String runfile = (index == -1) ? line : line.substring(0, index);
209  String realPath = (index == -1) ? line : line.substring(index + 1);
210  result.put(runfile, realPath);
211  }
212  }
213  return Collections.unmodifiableMap(result);
214  }

216  private static String findRunfilesDir(String manifest) {
217  if (manifest.endsWith("/MANIFEST")
218  || manifest.endsWith("\\MANIFEST")
219  || manifest.endsWith(".runfiles_manifest")) {
220  String path = manifest.substring(0, manifest.length() - 9);
221  if (new File(path).isDirectory()) {
222  return path;
223  }
224  }
225  return "";
226  }
