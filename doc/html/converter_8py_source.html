15 """Module to convert log levels between Abseil Python, C++, and Python standard.
16 
17 This converter has to convert (best effort) between three different
18 logging level schemes:
19  cpp = The C++ logging level scheme used in Abseil C++.
20  absl = The absl.logging level scheme used in Abseil Python.
21  standard = The python standard library logging level scheme.
22 
23 Here is a handy ascii chart for easy mental mapping.
24 
25  LEVEL | cpp | absl | standard |
26  ----------+-----+--------+----------+
27  DEBUG | 0 | 1 | 10 |
28  INFO | 0 | 0 | 20 |
29  WARN(ING) | 1 | -1 | 30 |
30  ERROR | 2 | -2 | 40 |
31  CRITICAL | 3 | -3 | 50 |
32  FATAL | 3 | -3 | 50 |
33 
34 Note: standard logging CRITICAL is mapped to absl/cpp FATAL.
35 However, only CRITICAL logs from the absl logger (or absl.logging.fatal) will
36 terminate the program. CRITICAL logs from non-absl loggers are treated as
37 error logs with a message prefix "CRITICAL - ".
38 
39 Converting from standard to absl or cpp is a lossy conversion.
40 Converting back to standard will lose granularity. For this reason,
41 users should always try to convert to standard, the richest
42 representation, before manipulating the levels, and then only to cpp
43 or absl if those level schemes are absolutely necessary.
44 """

86 def get_initial_for_level(level):
87  """Gets the initial that should start the log line for the given level.
88 
89  It returns:
90  - 'I' when: level < STANDARD_WARNING.
91  - 'W' when: STANDARD_WARNING <= level < STANDARD_ERROR.
92  - 'E' when: STANDARD_ERROR <= level < STANDARD_CRITICAL.
93  - 'F' when: level >= STANDARD_CRITICAL.
94 
95  Args:
96  level: int, a Python standard logging level.
97 
98  Returns:
99  The first initial as it would be logged by the C++ logging module.
100  """
101  if level < STANDARD_WARNING:
102  return 'I'
103  elif level < STANDARD_ERROR:
104  return 'W'
105  elif level < STANDARD_CRITICAL:
106  return 'E'
107  else:
108  return 'F'

132 def absl_to_standard(level):
133  """Converts an integer level from the absl value to the standard value.
134 
135  Args:
136  level: int, an absl.logging level.
137 
138  Raises:
139  TypeError: Raised when level is not an integer.
140 
141  Returns:
142  The corresponding integer level for use in standard logging.
143  """
144  if not isinstance(level, int):
145  raise TypeError('Expect an int level, found {}'.format(type(level)))
146  if level < ABSL_FATAL:
147  level = ABSL_FATAL
148  if level <= ABSL_DEBUG:
149  return ABSL_TO_STANDARD[level]
150  # Maps to vlog levels.
151  return STANDARD_DEBUG - level + 1

169 def standard_to_absl(level):
170  """Converts an integer level from the standard value to the absl value.
171 
172  Args:
173  level: int, a Python standard logging level.
174 
175  Raises:
176  TypeError: Raised when level is not an integer.
177 
178  Returns:
179  The corresponding integer level for use in absl logging.
180  """
181  if not isinstance(level, int):
182  raise TypeError('Expect an int level, found {}'.format(type(level)))
183  if level < 0:
184  level = 0
185  if level < STANDARD_DEBUG:
186  # Maps to vlog levels.
187  return STANDARD_DEBUG - level + 1
188  elif level < STANDARD_INFO:
189  return ABSL_DEBUG
190  elif level < STANDARD_WARNING:
191  return ABSL_INFO
192  elif level < STANDARD_ERROR:
193  return ABSL_WARN
194  elif level < STANDARD_CRITICAL:
195  return ABSL_ERROR
196  else:
197  return ABSL_FATAL
