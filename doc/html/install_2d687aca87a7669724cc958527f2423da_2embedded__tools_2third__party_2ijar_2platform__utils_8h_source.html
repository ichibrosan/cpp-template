38 // Converts a Stat object to ZIP attributes.
39 inline u4 stat_to_zipattr(const Stat& file_stat) {
40  return (((u4)file_stat.file_mode) << 16) |
41  (file_stat.is_directory != 0 ? 0x10 : 0);
