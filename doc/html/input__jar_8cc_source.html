17 bool InputJar::Open(const std::string &path) {
18  if (!path_.empty()) {
19  diag_errx(1, "%s:%d: This instance is already handling %s\n", __FILE__,
20  __LINE__, path_.c_str());
21  }
22  if (!mapped_file_.Open(path)) {
23  diag_warn("%s:%d: Cannot open input jar %s", __FILE__, __LINE__,
24  path.c_str());
25  mapped_file_.Close();
26  return false;
27  }
28  if (mapped_file_.size() < sizeof(ECD)) {
29  diag_warnx(
30  "%s:%d: %s is only 0x%zx"
31  " bytes long, should be at least 0x%zx bytes long",
32  __FILE__, __LINE__, path.c_str(), mapped_file_.size(), sizeof(ECD));
33  mapped_file_.Close();
34  return false;
35  }
36 
37  // Now locate End of Central Directory (ECD) record.
38  auto ecd_min = mapped_file_.end() - 65536 - sizeof(ECD);
39  if (ecd_min < mapped_file_.start()) {
40  ecd_min = mapped_file_.start();
41  }
42 
43  const ECD *ecd = nullptr;
44  for (auto ecd_ptr = mapped_file_.end() - sizeof(ECD); ecd_ptr >= ecd_min;
45  --ecd_ptr) {
46  if (reinterpret_cast<const ECD *>(ecd_ptr)->is()) {
47  ecd = reinterpret_cast<const ECD *>(ecd_ptr);
48  break;
49  }
50  }
51  if (ecd == nullptr) {
52  diag_warnx("%s:%d: Cannot locate ECD record in %s", __FILE__, __LINE__,
53  path.c_str());
54  mapped_file_.Close();
55  return false;
56  }
57 
58  /* Find Central Directory and preamble size. We want to handle the case
59  * where a Jar/Zip file contains a preamble (an arbitrary data before the
60  * first entry) and 'zip -A' was not called to adjust the offsets, so all
61  * the offsets are off by the preamble size. In the 32-bit case (that is,
62  * there is no ECD64Locator+ECD64), ECD immediately follows the last CDH,
63  * ECD immediately follows the Central Directory, and contains its size, so
64  * Central Directory can be found reliably. We then use its stated location,
65  * which ECD contains, too, to calculate the preamble size. In the 64-bit
66  * case, there are ECD64 and ECD64Locator records between the end of the
67  * Central Directory and the ECD, the calculation is similar, with the
68  * exception of the logic to find the actual start of the ECD64.
69  * ECD64Locator contains only its position in the file, which is off by
70  * preamble size, but does not contain the actual size of ECD64, which in
71  * theory is variable (the fixed fields may be followed by some custom data,
72  * with the total size saved in ECD64::remaining_size and thus unavailable
73  * until we find ECD64. We assume that the custom data is missing.
74  */
75 
76  // First, sanity checks.
77  uint32_t cen_position = ecd->cen_offset32();
78  if (!ziph::zfield_has_ext64(cen_position)) {
79  if (!mapped_file_.mapped(mapped_file_.address(cen_position))) {
80  diag_warnx("%s:%d: %s is corrupt: Central Directory location 0x%" PRIx32
81  " is invalid",
82  __FILE__, __LINE__, path.c_str(), cen_position);
83  mapped_file_.Close();
84  return false;
85  }
86  if (mapped_file_.offset(ecd) < cen_position) {
87  diag_warnx("%s:%d: %s is corrupt: End of Central Directory at 0x%" PRIx64
88  " precedes Central Directory at 0x%" PRIx32,
89  __FILE__, __LINE__, path.c_str(), mapped_file_.offset(ecd),
90  cen_position);
91  mapped_file_.Close();
92  return false;
93  }
94  }
95  uint32_t cen_size = ecd->cen_size32();
96  if (!ziph::zfield_has_ext64(cen_size)) {
97  if (cen_size > mapped_file_.offset(ecd)) {
98  diag_warnx("%s:%d: %s is corrupt: Central Directory size 0x%" PRIx32
99  " is too large",
100  __FILE__, __LINE__, path.c_str(), cen_size);
101  mapped_file_.Close();
102  return false;
103  }
104  }
105  if (cen_size == 0) {
106  // Empty archive, let cdh_ point to End of Central Directory.
107  cdh_ = reinterpret_cast<const CDH *>(ecd);
108  preamble_size_ = mapped_file_.offset(cdh_) - cen_position;
109  } else {
110  auto ecd64loc = reinterpret_cast<const ECD64Locator *>(
111  ziph::byte_ptr(ecd) - sizeof(ECD64Locator));
112  if (ecd64loc->is()) {
113  auto ecd64 = reinterpret_cast<const ECD64 *>(ziph::byte_ptr(ecd64loc) -
114  sizeof(ECD64));
115  if (!ecd64->is()) {
116  diag_warnx(
117  "%s:%d: %s is corrupt, expected ECD64 record at offset 0x%" PRIx64
118  " is missing",
119  __FILE__, __LINE__, path.c_str(), mapped_file_.offset(ecd64));
120  mapped_file_.Close();
121  return false;
122  }
123  cdh_ = reinterpret_cast<const CDH *>(ziph::byte_ptr(ecd64) -
124  ecd64->cen_size());
125  preamble_size_ = mapped_file_.offset(cdh_) - ecd64->cen_offset();
126  // Find CEN and preamble size.
127  } else {
128  if (ziph::zfield_has_ext64(cen_size) ||
129  ziph::zfield_has_ext64(cen_position)) {
130  diag_warnx(
131  "%s:%d: %s is corrupt, expected ECD64 locator record at "
132  "offset 0x%" PRIx64 " is missing",
133  __FILE__, __LINE__, path.c_str(), mapped_file_.offset(ecd64loc));
134  return false;
135  }
136  cdh_ = reinterpret_cast<const CDH *>(ziph::byte_ptr(ecd) - cen_size);
137  preamble_size_ = mapped_file_.offset(cdh_) - cen_position;
138  }
139  if (!cdh_->is()) {
140  diag_warnx(
141  "%s:%d: In %s, expected central file header signature at "
142  "offset0x%" PRIx64,
143  __FILE__, __LINE__, path.c_str(), mapped_file_.offset(cdh_));
144  mapped_file_.Close();
145  return false;
146  }
147  }
148  path_ = path;
149  return true;
150 }
