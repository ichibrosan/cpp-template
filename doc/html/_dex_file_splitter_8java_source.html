144  @VisibleForTesting
145  static void splitIntoShards(Options options) throws IOException {
146  checkArgument(
147  !options.minimalMainDex || options.mainDexListFile != null,
148  "--minimal-main-dex not allowed without --main-dex-list");
149 
150  if (!Files.exists(options.outputDirectory)) {
151  Files.createDirectories(options.outputDirectory);
152  }
153 
154  ImmutableSet<String> classesInMainDex =
155  options.mainDexListFile != null
156  ? ImmutableSet.copyOf(Files.readAllLines(options.mainDexListFile, UTF_8))
157  : null;
158  ImmutableSet<String> expected =
159  options.inclusionFilterJar != null ? expectedEntries(options.inclusionFilterJar) : null;
160  try (Closer closer = Closer.create();
161  DexFileSplitter out =
162  new DexFileSplitter(options.outputDirectory, options.maxNumberOfIdxPerDex)) {
163  // 1. Scan inputs in order and keep first occurrence of each class, keeping all zips open.
164  // We don't process anything yet so we can shard in sorted order, which is what dx would do
165  // if presented with a single jar containing all the given inputs.
166  // TODO(kmb): Abandon alphabetic sorting to process each input fully before moving on (still
167  // requires scanning inputs twice for main dex list).
168  Predicate<ZipEntry> inclusionFilter = ZipEntryPredicates.suffixes(".dex", ".class");
169  if (expected != null) {
170  inclusionFilter = inclusionFilter.and(e -> expected.contains(e.getName()));
171  }
172  LinkedHashMap<String, ZipFile> deduped = new LinkedHashMap<>();
173  for (Path inputArchive : options.inputArchives) {
174  ZipFile zip = closer.register(new ZipFile(inputArchive.toFile()));
175  zip.stream()
176  .filter(inclusionFilter)
177  .forEach(e -> deduped.putIfAbsent(e.getName(), zip));
178  }
179  ImmutableList<Map.Entry<String, ZipFile>> files =
180  deduped
181  .entrySet()
182  .stream()
183  .sorted(Comparator.comparing(e -> e.getKey(), ZipEntryComparator::compareClassNames))
184  .collect(ImmutableList.toImmutableList());
185 
186  // 2. Process each class in desired order, rolling from shard to shard as needed.
187  if (classesInMainDex == null || classesInMainDex.isEmpty()) {
188  out.processDexFiles(files, Predicates.alwaysTrue());
189  } else {
190  checkArgument(classesInMainDex.stream().noneMatch(s -> s.startsWith("j$/")),
191  "%s lists classes in package 'j$', which can't be included in classes.dex and can "
192  + "cause runtime errors. Please avoid needing these classes in the main dex file.",
193  options.mainDexListFile);
194  // To honor --main_dex_list make two passes:
195  // 1. process only the classes listed in the given file
196  // 2. process the remaining files
197  Predicate<String> mainDexFilter = ZipEntryPredicates.classFileNameFilter(classesInMainDex);
198  out.processDexFiles(files, mainDexFilter);
199  // Fail if main_dex_list is too big, following dx's example
200  checkState(out.shardsWritten() == 0, "Too many classes listed in main dex list file "
201  + "%s, main dex capacity exceeded", options.mainDexListFile);
202  if (options.minimalMainDex) {
203  out.nextShard(); // Start new .dex file if requested
204  }
205  out.processDexFiles(files, mainDexFilter.negate());
206  }
207  }
208  }

237  private void nextShard() throws IOException {
238  // Eagerly tell the last shard that it's done so it can finish writing the zip file and release
239  // resources as soon as possible, without blocking the start of the next shard.
240  curOut.finishAsync(); // will NPE if called after close()
241  ++curShard;
242  startShard();
243  }
244 
245  private void startShard() throws IOException {
246  tracker = new DexLimitTracker(maxNumberOfIdxPerDex);
247  curOut =
248  closer.register(
249  new AsyncZipOut(
250  outputDirectory.resolve((curShard + 1) + ".shard.zip"),
251  StandardOpenOption.CREATE_NEW,
252  StandardOpenOption.WRITE));
253  }
254 
255  private int shardsWritten() {
256  return curShard;
257  }

259  @Override
260  public void close() throws IOException {
261  if (curOut != null) {
262  curOut.finishAsync();
263  curOut = null;
264  ++curShard;
265  }
266  // Wait for all shards to finish writing. We told them to finish already but need to wait for
267  // any pending writes so we're sure all output was successfully written.
268  closer.close();
269  }

271  private void processDexFiles(
272  ImmutableList<Map.Entry<String, ZipFile>> filesToProcess, Predicate<String> filter)
273  throws IOException {
274  for (Map.Entry<String, ZipFile> entry : filesToProcess) {
275  String filename = entry.getKey();
276  if (filter.test(filename)) {
277  ZipFile zipFile = entry.getValue();
278  processDexEntry(zipFile, zipFile.getEntry(filename));
279  }
280  }
281  }

283  private void processDexEntry(ZipFile zip, ZipEntry entry) throws IOException {
284  String filename = entry.getName();
285  checkState(filename.endsWith(".class.dex"),
286  "%s isn't a dex archive: %s", zip.getName(), filename);
287  checkState(entry.getMethod() == ZipEntry.STORED, "Expect to process STORED: %s", filename);
288  if (inCoreLib == null) {
289  inCoreLib = filename.startsWith("j$/");
290  } else if (inCoreLib != filename.startsWith("j$/")) {
291  // Put j$.xxx classes in separate file. This shouldn't normally happen (b/134705306).
292  nextShard();
293  inCoreLib = !inCoreLib;
294  }
295  if (inCoreLib) {
296  System.err.printf(
297  "WARNING: Unexpected file %s found. Please ensure this only happens in test APKs.%n",
298  filename);
299  }
300 
301  try (InputStream entryStream = zip.getInputStream(entry)) {
302  // We don't want to use the Dex(InputStream) constructor because it closes the stream,
303  // which will break the for loop, and it has its own bespoke way of reading the file into
304  // a byte buffer before effectively calling Dex(byte[]) anyway.
305  // TODO(kmb) since entry is stored, mmap content and give to Dex(ByteBuffer) and output zip
306  byte[] content = new byte[(int) entry.getSize()];
307  ByteStreams.readFully(entryStream, content); // throws if file is smaller than expected
308  checkState(entryStream.read() == -1,
309  "Too many bytes in jar entry %s, expected %s", entry, entry.getSize());
310 
311  Dex dexFile = new Dex(content);
312  if (tracker.track(dexFile)) {
313  nextShard();
314  tracker.track(dexFile);
315  }
316  curOut.writeAsync(entry, content);
317  }
318  }
