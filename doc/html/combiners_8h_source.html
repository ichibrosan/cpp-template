27 // An interface for combining the files.
28 class Combiner {
29  public:
30  virtual ~Combiner();
31  // Merges the contents of the given Zip entry to this instance.
32  virtual bool Merge(const CDH *cdh, const LH *lh) = 0;
33  // Returns a point to the buffer containing Local Header followed by the
34  // payload. The caller is responsible of freeing the buffer. If `compress'
35  // is not set, the payload is a copy of the bytes held by this combiner.
36  // Otherwise the payload is compressed, provided that the compressed data
37  // is smaller than the original.
38  virtual void *OutputEntry(bool compress) = 0;
39 };
