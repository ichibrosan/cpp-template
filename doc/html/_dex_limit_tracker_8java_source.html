58  private void trackFieldsAndMethods(Dex dexFile) {
59  int fieldCount = dexFile.fieldIds().size();
60  for (int fieldIndex = 0; fieldIndex < fieldCount; ++fieldIndex) {
61  fieldsSeen.add(FieldDescriptor.fromDex(dexFile, fieldIndex));

63  int methodCount = dexFile.methodIds().size();
64  for (int methodIndex = 0; methodIndex < methodCount; ++methodIndex) {
65  methodsSeen.add(MethodDescriptor.fromDex(dexFile, methodIndex));
