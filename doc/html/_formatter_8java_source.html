45  public static void check(String format, String... args)
46  throws ExtraFormatArgumentsException,
47  IllegalFormatConversionException, IllegalFormatException,
48  FormatFlagsConversionMismatchException,
49  MissingFormatArgumentException, FormatterNumberFormatException {
50 
51  // index of last argument referenced
52  int last = -1;
53  // last ordinary index
54  int lasto = -1;
55 
56  // last index used
57  int maxIndex = -1;
58 
59  for (FormatSpecifier fs : parse(format)) {
60  int index = fs.index();
61  switch (index) {
62  case -2:
63  // ignore it
64  break;
65  case -1: // relative index
66  if (last < 0 || last > args.length - 1)
67  throw new MissingFormatArgumentException(last,
68  fs.toString());
69  fs.print(args[last], last);
70  break;
71  case 0: // ordinary index
72  lasto++;
73  last = lasto;
74  if (lasto > args.length - 1)
75  throw new MissingFormatArgumentException(lasto,
76  fs.toString());

78  fs.print(args[lasto], lasto);
79  break;
80  default: // explicit index
81  last = index - 1;

83  throw new MissingFormatArgumentException(last,

86  fs.print(args[last], last);
87  break;
88  }
89 
90  }
91  if (maxIndex < args.length - 1) {

96  // %[argument_index$][flags][width][.precision][t]conversion
97  private static final String formatSpecifier = "%(\\d+\\$)?([-#+ 0,(\\<]*)?(\\d+)?(\\.\\d+)?([tT])?([a-zA-Z%])";
98 
99  private static Pattern fsPattern = Pattern.compile(formatSpecifier);
100 
101  // Look for format specifiers in the format string.
102  private static List<FormatSpecifier> parse(String s)
103  throws FormatFlagsConversionMismatchException,
104  FormatterNumberFormatException {
105  ArrayList<FormatSpecifier> al = new ArrayList<FormatSpecifier>();
106  Matcher m = fsPattern.matcher(s);
107  int i = 0;
108  while (i < s.length()) {
109  if (m.find(i)) {
110  // Anything between the start of the string and the beginning
111  // of the format specifier is either fixed text or contains
112  // an invalid format string.
113  if (m.start() != i) {
114  // Make sure we didn't miss any invalid format specifiers
115  checkText(s.substring(i, m.start()));
116  }
117 
118  // Expect 6 groups in regular expression
119  String[] sa = new String[6];
120  for (int j = 0; j < m.groupCount(); j++) {
121  sa[j] = m.group(j + 1);
122  }
123  al.add(new FormatSpecifier(m.group(0), sa));
124  i = m.end();
125  } else {
126  // No more valid format specifiers. Check for possible invalid
127  // format specifiers.
128  checkText(s.substring(i));
129  // The rest of the string is fixed text
130  break;
131 
132  }
133  }
134 
135  return al;
136  }
137 
138  private static void checkText(String s) {
139  int idx;
140  // If there are any '%' in the given string, we got a bad format
141  // specifier.
142  if ((idx = s.indexOf('%')) != -1) {
143  char c = (idx > s.length() - 2 ? '%' : s.charAt(idx + 1));
144  throw new UnknownFormatConversionException(String.valueOf(c));
145  }
146  }
