42 // Accumulates statistics about a function or some C++ scope.
43 //
44 // Usage: see ScopedTask.
45 //
46 // Records the number of times the scope was entered (the function called) and
47 // the total time spent in there. Prints the statistics in the destructor.
48 class Task {
49  const char* name_;
50  uint64_t calls_;
51  Ticks total_;
52 
53  public:
54  Task(const char* name) : name_(name), calls_(0), total_({0}) {}
55  ~Task();
56  void AddCall() { calls_++; }
57  void AddTicks(const Ticks t) { total_.value_ += t.value_; }
58  uint64_t GetCalls() const { return calls_; }
59  Duration GetDuration() const { return Duration::FromTicks(total_); }

62 // Measures elapsed time.
63 //
64 // Example:
65 // void foo() {
66 // StopWatch s;
67 // ...
68 // s.PrintAndReset("marker 1"); // prints elapsed time since creation
69 // ...
70 // s.PrintAndReset("marker 2"); // prints elapsed time since "marker 1"
71 // ...
72 // s.Reset();
73 // ...
74 // Ticks t1 = s.Elapsed(); // time since Reset
75 // ...
76 // Ticks t2 = s.Elapsed(); // time since Reset, not since t1
77 // }
78 //
79 class StopWatch {
80  Ticks start_;
81 
82  public:
83  // Constructor -- it also resets this StopWatch.
84  StopWatch() { start_ = Ticks::Now(); }
85 
86  // Prints elapsed time since last reset, then resets.
87  //
88  // Args:
89  // name: a descriptive name, will be printed in the output
90  void PrintAndReset(const char* name);
91 
92  // Returns the elapsed time since the last reset as `Ticks`.
93  Ticks Elapsed() const {
94  Ticks now = Ticks::Now();
95  return {now.value_ - start_.value_};
96  }
97 
98  // Returns the elapsed time since the last reset as `Duration`.
99  Duration ElapsedTime() const { return Duration::FromTicks(Elapsed()); }
100 
101  // Resets this StopWatch to the current time.
102  void Reset() { start_ = Ticks::Now(); }
103 };

105 // Measures the execution duration of a given C++ scope.
106 //
107 // The constructor records one call of the scope in a `Task` object, and the
108 // destructor records the time spent in the scope in that `Task` object.
109 //
110 // Usage:
111 // create one Task that accumulates the statistics for a given function
112 // or scope, and create one ScopedTask in the beginning of the scope you want
113 // to measure. Every time the scope is entered (the function is called), a
114 // ScopedTask is created, then destructed when the execution leaves the scope.
115 // The destructor then records the statistics in the Task.
116 //
117 // Example:
118 // Task slow_function_stats("slow function"); // d'tor prints stats
119 //
120 // void slow_function() {
121 // ScopedTask prof(&slow_function_stats);
122 // ...
123 // }
124 //
125 class ScopedTask {
126  public:
127  ScopedTask(Task* s) : stat_(s) { stat_->AddCall(); }
128  ~ScopedTask() { stat_->AddTicks(prof_.Elapsed()); }
129 
130  private:
131  Task* stat_;
132  StopWatch prof_;
133 };
