30 /*
31  * Tokenize command line containing indirect command line arguments.
32  * An '@' at the beginning of a command line argument indicates that
33  * the rest of the argument is the name of the file which should be
34  * read and tokenized as Bash does it: tokens are separated by the
35  * whitespace, quotes and double quotes can be used to have whitespace
36  * and the other quote inside the token, and backslash followed by
37  * newline is treated as empty string.
38  */
39 
40 class ArgTokenStream {
41  /* This class is used as follows:
42  *
43  * int main(int argc, char* argv[]) {
44  * ArgTokenStream tokens(argc-1, argv+1);
45  * while (!tokens.AtEnd()) {
46  * if (tokens.MatchAndSet("--opt1", ...) ||
47  * tokens.MatchAndSet("--opt2", ...) ||
48  * ...) {
49  * continue;
50  * }
51  * // Process non-option argument or report an error.
52  * // ArgTokenStream::token() returns the current token.
53  * }
54  * }
55  */

83  // Assign next token to TOKEN, return true on success, false on EOF.
84  bool next_token(std::string *token) {
85  if (!fp_) {
86  return false;
87  }
88  *token = "";
89  while (current_char_ != EOF && isspace(current_char_)) {
90  next_char();
91  }
92  if (current_char_ == EOF) {
93  close();
94  return false;
95  }
96  for (;;) {
97  if (current_char_ == '\'' || current_char_ == '"') {
98  process_quoted(token);
99  if (isspace(current_char_)) {
100  next_char();
101  return true;
102  } else {
103  next_char();
104  }
105  } else if (current_char_ == '\\') {
106  next_char();
107  if ((current_char_ != EOF)) {
108  token->push_back(current_char_);
109  next_char();
110  } else {
111  diag_errx(1, "Expected character after \\, got EOF in %s",
112  filename_.c_str());
113  }
114  } else if (current_char_ == EOF || isspace(current_char_)) {
115  next_char();
116  return true;
117  } else {
118  token->push_back(current_char_);
119  next_char();
120  }
121  }
122  }

133  // Append the quoted string to the TOKEN. The quote character (which can be
134  // single or double quote) is in the current character. Everything up to the
135  // matching quote character is appended.
136  void process_quoted(std::string *token) {
137  char quote = current_char_;
138  next_char();
139  while (current_char_ != quote) {
140  if (current_char_ == '\\' && quote == '"') {
141  // In the "-quoted token, \" stands for ", and \x
142  // is copied literally for any other x.
143  next_char();
144  if (current_char_ == '"') {
145  token->push_back('"');
146  next_char();
147  } else if (current_char_ != EOF) {
148  token->push_back('\\');
149  token->push_back(current_char_);
150  next_char();
151  } else {
152  diag_errx(1, "No closing %c in %s", quote, filename_.c_str());
153  }
154  } else if (current_char_ != EOF) {
155  token->push_back(current_char_);
156  next_char();
157  } else {
158  diag_errx(1, "No closing %c in %s", quote, filename_.c_str());
159  }
160  }
161  }

163  // Get the next character from the input stream. Skip backslash followed
164  // by the newline.
165  void next_char() {
166  if (feof(fp_)) {
167  current_char_ = EOF;
168  return;
169  }
170  current_char_ = getc(fp_);
171  // Eat "\\\n" sequence.
172  while (current_char_ == '\\') {
173  int c = getc(fp_);
174  if (c == '\n') {
175  current_char_ = getc(fp_);
176  } else {
177  if (c != EOF) {
178  ungetc(c, fp_);
179  }
180  break;
181  }
182  }
183  }

197  // Process --OPTION
198  // If the current token is --OPTION, set given FLAG to true, proceed to next
199  // token and return true
200  bool MatchAndSet(const char *option, bool *flag) {

209  // Process --OPTION OPTARG
210  // If the current token is --OPTION, set OPTARG to the next token, proceed to
211  // the next token after it and return true.
212  bool MatchAndSet(const char *option, std::string *optarg) {
213  if (token_.compare(option) != 0) {
214  return false;
215  }
216  next();
217  if (AtEnd()) {
218  diag_errx(1, "%s requires argument", option);
219  }
220  *optarg = token_;
221  next();
222  return true;
223  }

225  // Process --OPTION OPTARG1 OPTARG2 ...
226  // If a current token is --OPTION, push_back all subsequent tokens up to the
227  // next option to the OPTARGS array, proceed to the next option and return
228  // true.
229  bool MatchAndSet(const char *option, std::vector<std::string> *optargs) {
230  if (token_.compare(option) != 0) {
231  return false;
232  }
233  next();
234  while (!AtEnd() && '-' != token_.at(0)) {
235  optargs->push_back(token_);
236  next();
237  }
238  return true;
239  }
240 
241  // Process --OPTION OPTARG1,OPTSUFF1 OPTARG2,OPTSUFF2 ...
242  // If a current token is --OPTION, push_back all subsequent tokens up to the
243  // next option to the OPTARGS array, splitting the OPTARG,OPTSUFF by a comma,
244  // proceed to the next option and return true.
245  bool MatchAndSet(const char *option,
246  std::vector<std::pair<std::string, std::string> > *optargs) {
247  if (token_.compare(option) != 0) {
248  return false;
249  }
250  next();
251  while (!AtEnd() && '-' != token_.at(0)) {
252  size_t commapos = token_.find(',');
253  if (commapos == std::string::npos) {
254  optargs->push_back(std::pair<std::string, std::string>(token_, ""));
255  } else {
256  std::string first = token_.substr(0, commapos);
257  token_.erase(0, commapos + 1);
258  optargs->push_back(std::pair<std::string, std::string>(first, token_));
259  }
260 
261  next();
262  }
263  return true;
264  }

269  // Read the next token.
270  void next() {
271  if (AtEnd()) {
272  return;
273  }
274  if (file_token_stream_.get() && token_from_file()) {
275  return;
276  }
277  while (argv_ < argv_end_) {
278  if (**argv_ != '@') {
279  token_ = *argv_++;
280  return;
281  }
282  file_token_stream_.reset(new FileTokenStream(*(argv_++) + 1));
283  if (token_from_file()) {
284  return;
285  }
286  }
287  argv_++;
288  }
289 
290  // True if there are no more tokens.
291  bool AtEnd() const { return argv_ > argv_end_; }

295  if (file_token_stream_->next_token(&token_)) {
296  return true;

298  file_token_stream_.reset(nullptr);

300  }
