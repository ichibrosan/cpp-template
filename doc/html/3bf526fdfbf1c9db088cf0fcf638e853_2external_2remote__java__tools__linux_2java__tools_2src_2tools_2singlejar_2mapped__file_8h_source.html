22 /*
23  * A mapped read-only file with auto closing.
24  *
25  * MappedFile::Open maps a file with specified name to memory as read-only.
26  * It is assumed that the address space is large enough for that.
27  * MappedFile::Close deletes the mapping. The destructor calls it, too.
28  * A predictable set of methods provide conversion between file offsets and
29  * mapped addresses, returns map size, etc.
30  */
31 class MappedFile {
32  public:
33  MappedFile();
34 
35  ~MappedFile() { Close(); }
36 
37  bool Open(const std::string &path);
38 
39  void Close();
40 
41  bool mapped(const void *addr) const {
42  return mapped_start_ <= addr && addr < mapped_end_;
43  }
44 
45  const unsigned char *start() const { return mapped_start_; }
46  const unsigned char *end() const { return mapped_end_; }
47  const unsigned char *address(off64_t offset) const {
48  return mapped_start_ + offset;
49  }
50  off64_t offset(const void *address) const {
51  return reinterpret_cast<const unsigned char *>(address) - mapped_start_;
52  }
