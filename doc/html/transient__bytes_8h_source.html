29 /*
30  * An instance of this class holds decompressed data in a list of chunks,
31  * to be eventually compressed to the output buffer.
32  * Use DecompressFile() or ReadFile() (depending on whether an entry is
33  * compressed or not) to append the contents of a Zip entry.
34  * Use Append() to append a sequence of bytes or a string.
35  * Use Write() to write out the contents, it will compress the entry if
36  * necessary.
37  */
38 class TransientBytes {

55  // Appends raw bytes.
56  void Append(const uint8_t *data, uint64_t data_size) {
57  uint64_t chunk_size;
58  auto data_end = data + data_size;
59  for (; data < data_end; data += chunk_size) {
60  chunk_size =
61  std::min(static_cast<uint64_t>(data_end - data), ensure_space());
62  copy(data, chunk_size);
63  }
64  }

76  // Appends the contents of the compressed Zip entry. Resets the inflater
77  // used to decompress.
78  void DecompressEntryContents(const CDH *cdh, const LH *lh,
79  Inflater *inflater) {
80  uint64_t old_total_out = inflater->total_out();
81  uint64_t in_bytes;
82  uint64_t out_bytes;
83  const uint8_t *data = lh->data();
84 
85  if (cdh->no_size_in_local_header()) {
86  in_bytes = cdh->compressed_file_size();
87  out_bytes = cdh->uncompressed_file_size();
88  } else {
89  in_bytes = lh->compressed_file_size();
90  out_bytes = lh->uncompressed_file_size();
91  }
92 
93  while (in_bytes > 0) {
94  // A single region to inflate cannot exceed 4GB-1.
95  uint32_t in_bytes_chunk = 0xFFFFFFFF;
96  if (in_bytes_chunk > in_bytes) {
97  in_bytes_chunk = in_bytes;
98  }
99  inflater->DataToInflate(data, in_bytes_chunk);
100  for (;;) {
101  uint32_t available_out = ensure_space();
102  int ret = inflater->Inflate(append_position(), available_out);
103  uint32_t inflated = available_out - inflater->available_out();
104  if (Z_STREAM_END == ret) {
105  // No more data to decompress. Update write position and we are done
106  // for this input chunk.
107  advance(inflated);
108  break;
109  } else if (Z_OK == ret) {
110  // No more space in the output buffer. Advance write position, update
111  // the number of remaining bytes.
112  if (inflater->available_out()) {
113  diag_errx(2,
114  "%s:%d: Internal error inflating %.*s: Inflate reported "
115  "Z_OK but there are still %" PRIu32
116  " bytes available in the output buffer",
117  __FILE__, __LINE__, lh->file_name_length(),
118  lh->file_name(), inflater->available_out());
119  }
120  advance(inflated);
121  } else {
122  diag_errx(2,
123  "%s:%d: Internal error inflating %.*s: inflate() call "
124  "returned %d (%s)",
125  __FILE__, __LINE__, lh->file_name_length(), lh->file_name(),
126  ret, inflater->error_message());
127  }
128  }
129  data += in_bytes_chunk;
130  in_bytes -= in_bytes_chunk;
131  }
132 
133  // Smog check
134  // This check is disabled on Windows because z_stream::total_out is of type
135  // of uLong (unsigned long), which is 64-bit for most 64-bit Unix platforms,
136  // but it is 32-bit even for Win64. This means even though zlib is capable
137  // of compressing data >4GB as long as it is processed by chunks, zlib
138  // cannot report the correct total number of processed bytes >4GB through
139  // z_stream::total_out on Windows.
140 #ifndef _WIN32
141  if (inflater->total_out() - old_total_out != out_bytes) {
142  diag_errx(2,
143  "%s:%d: Internal error inflating %.*s: inflater wrote %" PRIu64
144  " bytes , but the uncompressed entry should be %" PRIu64
145  "bytes long",
146  __FILE__, __LINE__, lh->file_name_length(), lh->file_name(),
147  inflater->total_out() - old_total_out, out_bytes);
148  }
149 #endif
150  inflater->reset();
151  }

153  // Writes the contents bytes to the given buffer in an optimal way, i.e., the
154  // shorter of compressed or uncompressed. Sets the checksum and number of
155  // bytes written and returns Z_DEFLATED if compression took place or
156  // Z_NO_COMPRESSION otherwise.
157  uint16_t CompressOut(uint8_t *buffer, uint32_t *checksum,
158  uint64_t *bytes_written) {
159  *checksum = 0;
160  uint64_t to_compress = data_size();
161  if (to_compress == 0) {
162  *bytes_written = 0;
163  return Z_NO_COMPRESSION;
164  }
165 
166  Deflater deflater;
167  deflater.next_out = buffer;
168  uint16_t compression_method = Z_DEFLATED;
169 
170  // Feed data blocks to the deflater one by one, but break if the compressed
171  // size exceeds the original size.
172  for (auto data_block = first_block_;
173  data_block && compression_method != Z_NO_COMPRESSION;
174  data_block = data_block->next_block_) {
175  // The compressed size should not exceed the original size less the number
176  // of bytes already compressed. And, it should not exceed 4GB-1.
177  deflater.avail_out = std::min(data_size() - deflater.total_out,
178  static_cast<uint64_t>(0xFFFFFFFF));
179  // Out of the total number of bytes that remain to be compressed, we
180  // can compress no more than this block.
181  uint32_t chunk_size = static_cast<uint32_t>(std::min(
182  static_cast<uint64_t>(sizeof(data_block->data_)), to_compress));
183  *checksum = crc32(*checksum, data_block->data_, chunk_size);
184  deflater.avail_in = chunk_size;
185  to_compress -= chunk_size;
186  int ret = deflater.Deflate(data_block->data_, chunk_size,
187  to_compress ? Z_NO_FLUSH : Z_FINISH);
188  if (ret == Z_OK) {
189  if (!deflater.avail_out) {
190  // We ran out of space in the output buffer, which means
191  // that deflated size exceeds original size. Leave the loop
192  // and just copy the data.
193  compression_method = Z_NO_COMPRESSION;
194  }
195  } else if (ret == Z_BUF_ERROR && !deflater.avail_in) {
196  // We ran out of data block, this is not a error.
197  } else if (ret == Z_STREAM_END) {
198  if (data_block->next_block_ || to_compress) {
199  diag_errx(2,
200  "%s:%d: Internal error: deflate() call at the end, but "
201  "there is more data to compress!",
202  __FILE__, __LINE__);
203  }
204  } else {
205  diag_errx(2, "%s:%d: deflate error %d(%s)", __FILE__, __LINE__, ret,
206  deflater.msg);
207  }
208  }
209  if (compression_method != Z_NO_COMPRESSION) {
210  *bytes_written = deflater.total_out;
211  return compression_method;
212  }
213 
214  // Compression does not help, just copy the bytes to the output buffer.
215  CopyOut(buffer, checksum);
216  *bytes_written = data_size();
217  return Z_NO_COMPRESSION;
218  }

220  // Copies the bytes to the buffer and sets the checksum.
221  void CopyOut(uint8_t *buffer, uint32_t *checksum) {
222  uint64_t to_copy = data_size();
223  uint8_t *buffer_end = buffer + to_copy;
224  *checksum = 0;
225  for (auto data_block = first_block_; data_block;
226  data_block = data_block->next_block_) {
227  size_t chunk_size =
228  std::min(static_cast<uint64_t>(sizeof(data_block->data_)), to_copy);
229  *checksum = crc32(*checksum, data_block->data_, chunk_size);
230  memcpy(buffer_end - to_copy, data_block->data_, chunk_size);
231  to_copy -= chunk_size;
232  }
233  }

238  // This is mostly for testing: stream out contents to a Sink instance.
239  // The class Sink has to have
240  // void operator()(const void *chunk, uint64_t chunk_size) const;
241  //
242  template <class Sink>
243  void stream_out(const Sink &sink) const {
244  uint64_t to_copy = data_size();
245  for (auto data_block = first_block_; data_block;
246  data_block = data_block->next_block_) {
247  uint64_t chunk_size = sizeof(data_block->data_);
248  if (chunk_size > to_copy) {
249  chunk_size = to_copy;
250  }
251  sink.operator()(data_block->data_, chunk_size);
252  to_copy -= chunk_size;
253  }
254  }

269  // Ensures there is some space to write to, returns the amount available.
270  uint64_t ensure_space() {
271  if (!free_size()) {
272  auto *data_block = new DataBlock();
273  if (last_block_) {
274  last_block_->next_block_ = data_block;
275  }
276  last_block_ = data_block;
277  if (!first_block_) {

281  }

283  }
